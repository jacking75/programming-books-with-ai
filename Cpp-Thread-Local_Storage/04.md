# 4. 고성능 프로그래밍 활용 사례
Thread-local storage를 활용하면 뮤텍스나 원자연산 없이 멀티스레드 환경에서 고성능을 달성할 수 있습니다. 이 섹션에서는 실제 프로덕션 코드에서 쓰이는 다양한 활용 사례를 살펴봅시다.

## 4.1 락 없는(Lock-free) 메모리 할당자

### 개념
일반적인 메모리 할당자(`malloc`, `new`)는 모든 스레드가 공유하는 메모리 풀을 사용하므로, 멀티스레드 환경에서 내부적으로 뮤텍스를 사용합니다. 이는 경합(contention)을 발생시킵니다.

**스레드별 메모리 풀** 방식은 각 스레드가 자신의 메모리 풀을 가지므로 락이 필요 없습니다.

```
전통적 할당:
Thread 1  ──┐
Thread 2  ──┼──[Mutex]──> Shared Memory Pool
Thread 3  ──┘
           ↑
        경합 발생

스레드별 풀:
Thread 1  ──> Pool 1 (락 없음)
Thread 2  ──> Pool 2 (락 없음)
Thread 3  ──> Pool 3 (락 없음)
```

### 구현 예제

```cpp
// ThreadLocalAllocator.h
#include <memory>
#include <vector>
#include <cstddef>
#include <cstring>

// 각 스레드가 소유할 메모리 풀
class ThreadLocalMemoryPool {
private:
    static constexpr size_t BLOCK_SIZE = 4096;      // 4KB 블록 크기
    static constexpr size_t SMALL_THRESHOLD = 256;  // 작은 할당 임계값
    
    // 작은 할당용 프리 리스트
    struct FreeBlock {
        FreeBlock* next;
        size_t size;
    };
    
    FreeBlock* small_free_list[32];  // 8, 16, 24, ... 256 바이트
    std::vector<void*> large_blocks;  // 큰 할당용
    
    size_t total_allocated = 0;
    size_t total_freed = 0;
    
public:
    ThreadLocalMemoryPool() {
        std::memset(small_free_list, 0, sizeof(small_free_list));
    }
    
    ~ThreadLocalMemoryPool() {
        // 할당된 모든 메모리 해제
        for (auto block : large_blocks) {
            ::free(block);
        }
    }
    
    // 메모리 할당 (락 없음)
    void* allocate(size_t size) {
        if (size == 0) return nullptr;
        
        // 작은 할당: 프리 리스트에서 찾기
        if (size <= SMALL_THRESHOLD) {
            int index = (size + 7) / 8;  // 8바이트 단위로 올림
            if (index >= 32) index = 31;
            
            FreeBlock* block = small_free_list[index];
            if (block != nullptr) {
                small_free_list[index] = block->next;
                total_allocated += size;
                return block;
            }
        }
        
        // 프리 리스트에 없으면 새로 할당
        void* ptr = ::malloc(size + sizeof(size_t));
        if (ptr != nullptr) {
            // 크기를 메모리 앞에 저장 (해제할 때 필요)
            *(size_t*)ptr = size;
            total_allocated += size;
            
            if (size > SMALL_THRESHOLD) {
                large_blocks.push_back(ptr);
            }
            return (char*)ptr + sizeof(size_t);
        }
        return nullptr;
    }
    
    // 메모리 해제 (락 없음)
    void deallocate(void* ptr, size_t size) {
        if (ptr == nullptr) return;
        
        void* real_ptr = (char*)ptr - sizeof(size_t);
        total_freed += size;
        
        // 작은 할당은 프리 리스트에 반환
        if (size <= SMALL_THRESHOLD) {
            int index = (size + 7) / 8;
            if (index >= 32) index = 31;
            
            FreeBlock* block = (FreeBlock*)real_ptr;
            block->size = size;
            block->next = small_free_list[index];
            small_free_list[index] = block;
        } else {
            // 큰 할당도 프리 리스트에 보관 (단순화)
            FreeBlock* block = (FreeBlock*)real_ptr;
            block->size = size;
            // 실제로는 더 복잡한 관리가 필요하지만, 예제 단순화
        }
    }
    
    // 통계
    struct Stats {
        size_t allocated;
        size_t freed;
        size_t net;
    };
    
    Stats get_stats() const {
        return {total_allocated, total_freed, total_allocated - total_freed};
    }
};

// 스레드 로컬 할당자
class ThreadLocalAllocator {
private:
    // 각 스레드가 자신의 풀을 가짐
    static thread_local ThreadLocalMemoryPool pool;
    
public:
    static void* allocate(size_t size) {
        return pool.allocate(size);
    }
    
    static void deallocate(void* ptr, size_t size) {
        pool.deallocate(ptr, size);
    }
    
    static ThreadLocalMemoryPool::Stats get_stats() {
        return pool.get_stats();
    }
};

// 정의
thread_local ThreadLocalMemoryPool ThreadLocalAllocator::pool;
```

### 성능 비교 예제

```cpp
// MemoryAllocatorBenchmark.cpp
#include <iostream>
#include <thread>
#include <vector>
#include <chrono>
#include <mutex>
#include <random>

// 뮤텍스 기반 할당자 (비교용)
class MutexBasedAllocator {
private:
    static std::mutex alloc_mutex;
    
public:
    static void* allocate(size_t size) {
        std::lock_guard<std::mutex> lock(alloc_mutex);
        return ::malloc(size);
    }
    
    static void deallocate(void* ptr, size_t size) {
        std::lock_guard<std::mutex> lock(alloc_mutex);
        ::free(ptr);
    }
};

std::mutex MutexBasedAllocator::alloc_mutex;

// 벤치마크 함수
template<typename Allocator>
void benchmark_allocator(const char* name, int num_threads, int iterations) {
    auto start = std::chrono::high_resolution_clock::now();
    
    std::vector<std::thread> threads;
    
    for (int t = 0; t < num_threads; ++t) {
        threads.emplace_back([iterations]() {
            std::vector<void*> ptrs;
            std::mt19937 gen(std::this_thread::get_id());
            std::uniform_int_distribution<> dis(10, 1000);
            
            for (int i = 0; i < iterations; ++i) {
                size_t size = dis(gen);
                
                // 할당과 해제 반복
                void* ptr = Allocator::allocate(size);
                if (ptr) {
                    ptrs.push_back(ptr);
                    if (ptrs.size() > 100) {
                        Allocator::deallocate(ptrs[0], 0);
                        ptrs.erase(ptrs.begin());
                    }
                }
            }
            
            // 정리
            for (auto ptr : ptrs) {
                Allocator::deallocate(ptr, 0);
            }
        });
    }
    
    for (auto& t : threads) {
        t.join();
    }
    
    auto end = std::chrono::high_resolution_clock::now();
    auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(end - start);
    
    std::cout << name << ": " << duration.count() << "ms\n";
}

int main() {
    std::cout << "=== 메모리 할당 성능 비교 ===\n";
    std::cout << "스레드: 8개, 반복: 10000\n\n";
    
    // Mutex 기반
    benchmark_allocator<MutexBasedAllocator>("Mutex 기반 할당자", 8, 10000);
    
    // Thread-local 기반
    benchmark_allocator<ThreadLocalAllocator>("Thread-local 할당자", 8, 10000);
    
    return 0;
}
```

## 4.2 캐싱 최적화

### False Sharing 문제
멀티코어 CPU는 L1/L2/L3 캐시를 가지며, 캐시는 **캐시 라인**(보통 64바이트)이라는 단위로 데이터를 관리합니다. 여러 스레드가 같은 캐시 라인의 다른 변수에 접근하면 "False Sharing"이 발생하여 성능이 급격히 저하됩니다.

```
메모리 레이아웃:

변수 a(4바이트)     변수 b(4바이트)     변수 c(4바이트)     ...
|───────────────────── 캐시 라인(64바이트) ─────────────────|

Thread 1에서 a 수정 → 캐시 라인 전체가 무효화
                  ↓
Thread 2에서 b 접근 → 메모리에서 다시 로드
                  ↓
심각한 성능 저하!
```

### 해결 방안: Thread-local로 False Sharing 회피

```cpp
// CacheOptimization.h
#include <atomic>
#include <array>
#include <iostream>
#include <thread>
#include <vector>
#include <chrono>

// 1. False Sharing이 발생하는 경우 (나쁜 예)
struct BadCounter {
    std::atomic<long long> counter{0};
};

// 2. Padding으로 해결 (좋은 예)
struct GoodCounter {
    alignas(64) std::atomic<long long> counter{0};
};

// 3. Thread-local로 해결 (최고의 예)
class ThreadLocalCounter {
private:
    static thread_local long long local_count;
    
public:
    static void increment() {
        ++local_count;  // 락 없음, 원자연산 없음!
    }
    
    static long long get_local() {
        return local_count;
    }
    
    // 모든 스레드의 값을 합산 (이건 좀 복잡하지만, 예제를 위해 간단히)
    static long long get_total_approx() {
        return local_count;
    }
};

thread_local long long ThreadLocalCounter::local_count = 0;

// 스레드별 캐시 구현
template<typename T, size_t CACHE_SIZE = 1024>
class ThreadLocalCache {
private:
    struct CacheEntry {
        int key;
        T value;
        bool valid;
    };
    
    static thread_local std::array<CacheEntry, CACHE_SIZE> cache_table;
    static thread_local int cache_hits;
    static thread_local int cache_misses;
    
public:
    static bool get(int key, T& value) {
        size_t index = key % CACHE_SIZE;
        
        if (cache_table[index].valid && cache_table[index].key == key) {
            value = cache_table[index].value;
            ++cache_hits;
            return true;
        }
        
        ++cache_misses;
        return false;
    }
    
    static void put(int key, const T& value) {
        size_t index = key % CACHE_SIZE;
        cache_table[index].key = key;
        cache_table[index].value = value;
        cache_table[index].valid = true;
    }
    
    struct CacheStats {
        int hits;
        int misses;
        double hit_ratio;
    };
    
    static CacheStats get_stats() {
        int total = cache_hits + cache_misses;
        return {
            cache_hits,
            cache_misses,
            total > 0 ? (double)cache_hits / total : 0.0
        };
    }
    
    static void reset() {
        cache_hits = 0;
        cache_misses = 0;
        for (auto& entry : cache_table) {
            entry.valid = false;
        }
    }
};

template<typename T, size_t CACHE_SIZE>
thread_local std::array<typename ThreadLocalCache<T, CACHE_SIZE>::CacheEntry, CACHE_SIZE>
    ThreadLocalCache<T, CACHE_SIZE>::cache_table;

template<typename T, size_t CACHE_SIZE>
thread_local int ThreadLocalCache<T, CACHE_SIZE>::cache_hits = 0;

template<typename T, size_t CACHE_SIZE>
thread_local int ThreadLocalCache<T, CACHE_SIZE>::cache_misses = 0;

// 벤치마크
void benchmark_false_sharing() {
    std::cout << "\n=== False Sharing 벤치마크 ===\n";
    
    constexpr int ITERATIONS = 100000000;  // 1억 번
    constexpr int NUM_THREADS = 8;
    
    // 1. 나쁜 예 - False Sharing 발생
    {
        std::vector<BadCounter> counters(NUM_THREADS);
        auto start = std::chrono::high_resolution_clock::now();
        
        std::vector<std::thread> threads;
        for (int t = 0; t < NUM_THREADS; ++t) {
            threads.emplace_back([&counters, t, ITERATIONS]() {
                for (int i = 0; i < ITERATIONS; ++i) {
                    counters[t].counter++;
                }
            });
        }
        
        for (auto& t : threads) t.join();
        
        auto end = std::chrono::high_resolution_clock::now();
        auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(end - start);
        std::cout << "❌ False Sharing (나쁜 예): " << duration.count() << "ms\n";
    }
    
    // 2. 좋은 예 - Padding
    {
        std::vector<GoodCounter> counters(NUM_THREADS);
        auto start = std::chrono::high_resolution_clock::now();
        
        std::vector<std::thread> threads;
        for (int t = 0; t < NUM_THREADS; ++t) {
            threads.emplace_back([&counters, t, ITERATIONS]() {
                for (int i = 0; i < ITERATIONS; ++i) {
                    counters[t].counter++;
                }
            });
        }
        
        for (auto& t : threads) t.join();
        
        auto end = std::chrono::high_resolution_clock::now();
        auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(end - start);
        std::cout << "✓ Padding (좋은 예): " << duration.count() << "ms\n";
    }
    
    // 3. 최고의 예 - Thread-local
    {
        auto start = std::chrono::high_resolution_clock::now();
        
        std::vector<std::thread> threads;
        for (int t = 0; t < NUM_THREADS; ++t) {
            threads.emplace_back([ITERATIONS]() {
                for (int i = 0; i < ITERATIONS; ++i) {
                    ThreadLocalCounter::increment();
                }
            });
        }
        
        for (auto& t : threads) t.join();
        
        auto end = std::chrono::high_resolution_clock::now();
        auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(end - start);
        std::cout << "⭐ Thread-local (최고의 예): " << duration.count() << "ms\n";
    }
}
```

## 4.3 난수 생성기 최적화

### 문제: 공유 PRNG의 경합
표준 `<random>` 라이브러리의 난수 생성기(예: `mt19937`)를 여러 스레드에서 공유하면, 내부 상태 동기화로 인한 경합이 발생합니다.

```
공유 PRNG:
Thread 1  ──┐
Thread 2  ──┼──[Mutex]──> MT19937 상태
Thread 3  ──┤            (lock_guard)
Thread 4  ──┘
           ↑
        높은 경합

스레드별 PRNG (Thread-local):
Thread 1  ──> MT19937 #1 (시드 = tid + offset)
Thread 2  ──> MT19937 #2 (시드 = tid + offset)
Thread 3  ──> MT19937 #3 (시드 = tid + offset)
Thread 4  ──> MT19937 #4 (시드 = tid + offset)
           ↑
        경합 없음
```

### 구현

```cpp
// ThreadLocalRNG.h
#include <random>
#include <cstdint>
#include <thread>
#include <chrono>

class ThreadLocalRNG {
private:
    // 각 스레드가 자신의 PRNG를 소유
    static thread_local std::mt19937 rng;
    
    // 초기화 플래그
    static thread_local bool initialized;
    
    static void ensure_initialized() {
        if (!initialized) {
            // 스레드별 고유한 시드 생성
            auto seed = std::hash<std::thread::id>{}(std::this_thread::get_id())
                      ^ std::chrono::high_resolution_clock::now().time_since_epoch().count();
            rng.seed(seed);
            initialized = true;
        }
    }
    
public:
    // 정수 난수
    static uint32_t uniform_u32() {
        ensure_initialized();
        static thread_local std::uniform_int_distribution<uint32_t> dist;
        return dist(rng);
    }
    
    // 정수 범위 난수
    static int uniform_int(int min, int max) {
        ensure_initialized();
        std::uniform_int_distribution<int> dist(min, max);
        return dist(rng);
    }
    
    // 실수 난수 [0, 1)
    static double uniform_double() {
        ensure_initialized();
        static thread_local std::uniform_real_distribution<double> dist(0.0, 1.0);
        return dist(rng);
    }
    
    // 정규분포
    static double normal(double mean = 0.0, double stddev = 1.0) {
        ensure_initialized();
        std::normal_distribution<double> dist(mean, stddev);
        return dist(rng);
    }
    
    // 가우시안 난수
    static double gaussian() {
        ensure_initialized();
        static thread_local std::normal_distribution<double> dist(0.0, 1.0);
        return dist(rng);
    }
};

// 정의
thread_local std::mt19937 ThreadLocalRNG::rng;
thread_local bool ThreadLocalRNG::initialized = false;

// ========================================
// 벤치마크
// ========================================

#include <vector>
#include <iostream>
#include <iomanip>
#include <mutex>

// 뮤텍스 기반 PRNG (비교용)
class SharedRNG {
private:
    static std::mutex rng_mutex;
    static std::mt19937 rng;
    
public:
    static uint32_t uniform_u32() {
        std::lock_guard<std::mutex> lock(rng_mutex);
        static std::uniform_int_distribution<uint32_t> dist;
        return dist(rng);
    }
    
    static double uniform_double() {
        std::lock_guard<std::mutex> lock(rng_mutex);
        static std::uniform_real_distribution<double> dist(0.0, 1.0);
        return dist(rng);
    }
};

std::mutex SharedRNG::rng_mutex;
std::mt19937 SharedRNG::rng(42);

// 벤치마크 함수 템플릿
template<typename RNG>
void benchmark_rng(const char* name, int num_threads, int iterations) {
    auto start = std::chrono::high_resolution_clock::now();
    
    std::vector<std::thread> threads;
    std::vector<uint64_t> results(num_threads, 0);
    
    for (int t = 0; t < num_threads; ++t) {
        threads.emplace_back([t, iterations, &results]() {
            uint64_t sum = 0;
            for (int i = 0; i < iterations; ++i) {
                double val = RNG::uniform_double();
                sum += (uint64_t)(val * 1000000);
            }
            results[t] = sum;
        });
    }
    
    for (auto& t : threads) {
        t.join();
    }
    
    auto end = std::chrono::high_resolution_clock::now();
    auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(end - start);
    
    std::cout << std::setw(25) << name 
              << ": " << std::setw(4) << duration.count() << "ms\n";
}

void run_rng_benchmark() {
    std::cout << "\n=== 난수 생성기 성능 비교 ===\n";
    std::cout << "스레드 수: 8, 반복: 10,000,000/스레드\n\n";
    
    benchmark_rng<SharedRNG>("Mutex 기반 (공유)", 8, 10000000);
    benchmark_rng<ThreadLocalRNG>("Thread-local", 8, 10000000);
}
```

## 4.4 에러 핸들링

### errno 스타일의 스레드별 에러 코드
전통적인 `errno`는 C 라이브러리 함수들이 사용하는 전역 에러 변수입니다. 멀티스레드 환경에서는 thread-local로 구현되어야 합니다.

```cpp
// ThreadLocalError.h
#include <string>
#include <cstring>

// 스레드별 에러 상태
class ThreadLocalError {
public:
    enum ErrorCode : int {
        SUCCESS = 0,
        INVALID_ARGUMENT = 1,
        FILE_NOT_FOUND = 2,
        PERMISSION_DENIED = 3,
        OUT_OF_MEMORY = 4,
        TIMEOUT = 5,
        UNKNOWN = 999
    };
    
private:
    static thread_local int error_code;
    static thread_local char error_message[256];
    
public:
    // 에러 설정
    static void set_error(int code, const char* message) {
        error_code = code;
        if (message) {
            strncpy_s(error_message, sizeof(error_message), message, 
                     sizeof(error_message) - 1);
        } else {
            error_message[0] = '\0';
        }
    }
    
    // 에러 조회
    static int get_error() {
        return error_code;
    }
    
    static const char* get_error_message() {
        return error_message;
    }
    
    // 에러 초기화
    static void clear_error() {
        error_code = SUCCESS;
        error_message[0] = '\0';
    }
    
    // 에러 설명 출력
    static void print_error(const char* context = nullptr) {
        if (context) {
            printf("[%s] Error %d: %s\n", context, error_code, error_message);
        } else {
            printf("Error %d: %s\n", error_code, error_message);
        }
    }
};

// 정의
thread_local int ThreadLocalError::error_code = 0;
thread_local char ThreadLocalError::error_message[256] = "";

// 유틸리티 함수
bool process_file(const std::string& filename) {
    ThreadLocalError::clear_error();
    
    if (filename.empty()) {
        ThreadLocalError::set_error(
            ThreadLocalError::INVALID_ARGUMENT,
            "Filename cannot be empty"
        );
        return false;
    }
    
    FILE* file = nullptr;
    errno_t err = fopen_s(&file, filename.c_str(), "r");
    
    if (err != 0) {
        ThreadLocalError::set_error(
            ThreadLocalError::FILE_NOT_FOUND,
            "Cannot open file: " + filename
        );
        return false;
    }
    
    // 파일 처리...
    fclose(file);
    return true;
}

// 사용 예제
void example_error_handling() {
    std::vector<std::thread> threads;
    
    for (int t = 0; t < 4; ++t) {
        threads.emplace_back([t]() {
            if (t % 2 == 0) {
                // 성공 케이스
                process_file("test.txt");
                if (ThreadLocalError::get_error() == ThreadLocalError::SUCCESS) {
                    printf("Thread %d: 파일 처리 성공\n", t);
                }
            } else {
                // 실패 케이스
                process_file("");  // 빈 파일명
                ThreadLocalError::print_error("Thread " + std::to_string(t));
            }
        });
    }
    
    for (auto& t : threads) {
        t.join();
    }
}
```

## 4.5 프로파일링과 통계 수집

### 스레드별 성능 카운터

```cpp
// ThreadLocalProfiler.h
#include <atomic>
#include <chrono>
#include <iostream>
#include <iomanip>
#include <map>

class ThreadLocalProfiler {
public:
    struct Counter {
        long long count = 0;
        long long total_time_us = 0;
        long long min_time_us = LLONG_MAX;
        long long max_time_us = 0;
        
        double avg_time() const {
            return count > 0 ? (double)total_time_us / count : 0.0;
        }
    };
    
private:
    static thread_local std::map<std::string, Counter> counters;
    
public:
    // 카운터 증가
    static void increment(const std::string& name) {
        counters[name].count++;
    }
    
    // 시간 측정 및 기록
    static void record_time(const std::string& name, long long duration_us) {
        auto& counter = counters[name];
        counter.count++;
        counter.total_time_us += duration_us;
        counter.min_time_us = std::min(counter.min_time_us, duration_us);
        counter.max_time_us = std::max(counter.max_time_us, duration_us);
    }
    
    // 현재 스레드의 통계 출력
    static void print_stats() {
        std::cout << "\n===== 스레드 " << std::this_thread::get_id() 
                  << " 통계 =====\n";
        std::cout << std::setw(30) << "카운터"
                  << std::setw(12) << "횟수"
                  << std::setw(12) << "평균(μs)"
                  << std::setw(12) << "최소(μs)"
                  << std::setw(12) << "최대(μs)\n";
        std::cout << std::string(78, '-') << "\n";
        
        for (const auto& [name, counter] : counters) {
            std::cout << std::setw(30) << name
                      << std::setw(12) << counter.count
                      << std::setw(12) << std::fixed << std::setprecision(2) 
                      << counter.avg_time()
                      << std::setw(12) << counter.min_time_us
                      << std::setw(12) << counter.max_time_us << "\n";
        }
    }
};

thread_local std::map<std::string, ThreadLocalProfiler::Counter> 
    ThreadLocalProfiler::counters;

// RAII 스타일 타이머
class ScopedTimer {
private:
    std::string name;
    std::chrono::high_resolution_clock::time_point start;
    
public:
    explicit ScopedTimer(const std::string& timer_name) 
        : name(timer_name), start(std::chrono::high_resolution_clock::now()) {
    }
    
    ~ScopedTimer() {
        auto end = std::chrono::high_resolution_clock::now();
        auto duration_us = std::chrono::duration_cast<std::chrono::microseconds>(
            end - start).count();
        ThreadLocalProfiler::record_time(name, duration_us);
    }
};

// 사용 예제
void example_profiling() {
    {
        ScopedTimer timer("function_a");
        // 함수 A 작업
        std::this_thread::sleep_for(std::chrono::milliseconds(10));
    }
    
    {
        ScopedTimer timer("function_b");
        // 함수 B 작업
        std::this_thread::sleep_for(std::chrono::milliseconds(20));
    }
    
    ThreadLocalProfiler::print_stats();
}
```

```cpp
// 요청 컨텍스트

// 스레드 로컬 컨텍스트 관리자
```

## 4.7 멀티스레드 큐 최적화 (추가 섹션)

### 문제: MPMC 큐의 경합
일반적인 `std::queue`는 뮤텍스로 보호되어 모든 스레드가 경합합니다.

```
문제:
Producer 1 ──┐
Producer 2  ──┼──[Mutex]──> Shared Queue
Producer 3  ──┤
Consumer 1  ──┘
           ↑
        심각한 경합

해결 (Thread-local 버퍼):
Producer 1  ──> Local Buffer 1
Producer 2  ──> Local Buffer 2       ──┐
Producer 3  ──> Local Buffer 3  ──────┼──[배치 플러시]──> Central Queue
...                                  ──┘    (경합 줄임)
```

### 구현

```cpp
// ThreadLocalQueue.h
#include <queue>
#include <mutex>
#include <vector>
#include <memory>

template<typename T>
class ThreadLocalBatchQueue {
private:
    static constexpr size_t BATCH_SIZE = 64;  // 배치 크기
    
    // 로컬 배치 버퍼
    static thread_local std::vector<T> local_buffer;
    static thread_local size_t buffer_index;
    
    // 공유 큐 (중앙 집계)
    static std::queue<T> central_queue;
    static std::mutex central_mutex;
    
    static void flush_local_buffer() {
        if (buffer_index > 0) {
            std::lock_guard<std::mutex> lock(central_mutex);
            for (size_t i = 0; i < buffer_index; ++i) {
                central_queue.push(local_buffer[i]);
            }
            buffer_index = 0;
        }
    }
    
public:
    static void push(const T& value) {
        local_buffer[buffer_index++] = value;
        
        // 로컬 버퍼가 가득 찼을 때만 플러시 (경합 최소화)
        if (buffer_index >= BATCH_SIZE) {
            flush_local_buffer();
        }
    }
    
    static bool try_pop(T& value) {
        {
            std::lock_guard<std::mutex> lock(central_mutex);
            if (!central_queue.empty()) {
                value = central_queue.front();
                central_queue.pop();
                return true;
            }
        }
        return false;
    }
    
    static void finalize() {
        flush_local_buffer();
    }
    
    static size_t size() {
        std::lock_guard<std::mutex> lock(central_mutex);
        return central_queue.size();
    }
};

template<typename T>
thread_local std::vector<T> ThreadLocalBatchQueue<T>::local_buffer(
    ThreadLocalBatchQueue<T>::BATCH_SIZE);

template<typename T>
thread_local size_t ThreadLocalBatchQueue<T>::buffer_index = 0;

template<typename T>
std::queue<T> ThreadLocalBatchQueue<T>::central_queue;

template<typename T>
std::mutex ThreadLocalBatchQueue<T>::central_mutex;

// 벤치마크
void benchmark_queue() {
    std::cout << "\n=== 큐 성능 비교 ===\n";
    
    constexpr int ITEMS_PER_THREAD = 1000000;
    constexpr int NUM_THREADS = 8;
    
    // 1. 표준 큐 (mutex)
    {
        std::queue<int> q;
        std::mutex q_mutex;
        
        auto start = std::chrono::high_resolution_clock::now();
        
        std::vector<std::thread> threads;
        for (int t = 0; t < NUM_THREADS; ++t) {
            threads.emplace_back([&q, &q_mutex, ITEMS_PER_THREAD]() {
                for (int i = 0; i < ITEMS_PER_THREAD; ++i) {
                    std::lock_guard<std::mutex> lock(q_mutex);
                    q.push(i);
                }
            });
        }
        
        for (auto& t : threads) t.join();
        
        auto end = std::chrono::high_resolution_clock::now();
        auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(end - start);
        std::cout << "표준 큐 (Mutex): " << duration.count() << "ms\n";
    }
    
    // 2. Thread-local 배치 큐
    {
        auto start = std::chrono::high_resolution_clock::now();
        
        std::vector<std::thread> threads;
        for (int t = 0; t < NUM_THREADS; ++t) {
            threads.emplace_back([ITEMS_PER_THREAD]() {
                for (int i = 0; i < ITEMS_PER_THREAD; ++i) {
                    ThreadLocalBatchQueue<int>::push(i);
                }
                ThreadLocalBatchQueue<int>::finalize();
            });
        }
        
        for (auto& t : threads) t.join();
        
        auto end = std::chrono::high_resolution_clock::now();
        auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(end - start);
        std::cout << "Thread-local 배치 큐: " << duration.count() << "ms\n";
    }
}
```
