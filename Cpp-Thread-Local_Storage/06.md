## 사례 1: 스레드별 메모리 풀 (Memory Pool)

**왜 TLS인가?** `malloc/free`는 내부적으로 전역 힙에 락을 겁니다. 스레드별로 독립적인 풀을 두면 락 없이 O(1) 할당이 가능합니다. 실제로 `jemalloc`, `tcmalloc`이 이 방식을 사용합니다.

```cpp
#include <iostream>
#include <thread>
#include <vector>
#include <cassert>
#include <cstddef>
#include <cstring>
#include <atomic>

// ============================================================
// 스레드 로컬 메모리 풀
// - 락 없이 O(1) 할당/해제
// - 각 스레드가 자신만의 슬롯 풀을 소유
// ============================================================

template<size_t ObjectSize, size_t PoolCapacity>
class ThreadLocalPool {
private:
    struct Slot {
        alignas(std::max_align_t) char data[ObjectSize];
        bool inUse = false;
    };

    struct PoolBlock {
        Slot slots[PoolCapacity];
        size_t freeCount = PoolCapacity;

        void* Allocate() {
            for (size_t i = 0; i < PoolCapacity; ++i) {
                if (!slots[i].inUse) {
                    slots[i].inUse = true;
                    --freeCount;
                    return slots[i].data;
                }
            }
            return nullptr;  // 풀 소진 시 nullptr
        }

        bool Deallocate(void* ptr) {
            char* p = static_cast<char*>(ptr);
            for (size_t i = 0; i < PoolCapacity; ++i) {
                if (slots[i].data == p) {
                    slots[i].inUse = false;
                    ++freeCount;
                    return true;
                }
            }
            return false;
        }

        size_t FreeCount() const { return freeCount; }
    };

    // 핵심: 스레드마다 독립적인 풀 블록
    static thread_local PoolBlock tlPool;
    static std::atomic<size_t> totalAllocations;

public:
    static void* Allocate() {
        void* ptr = tlPool.Allocate();
        if (ptr) {
            totalAllocations.fetch_add(1, std::memory_order_relaxed);
        }
        return ptr;
    }

    static void Deallocate(void* ptr) {
        tlPool.Deallocate(ptr);
    }

    static size_t GetFreeSlots() {
        return tlPool.FreeCount();
    }

    static size_t GetTotalAllocations() {
        return totalAllocations.load();
    }
};

template<size_t S, size_t C>
thread_local typename ThreadLocalPool<S, C>::PoolBlock 
    ThreadLocalPool<S, C>::tlPool;

template<size_t S, size_t C>
std::atomic<size_t> ThreadLocalPool<S, C>::totalAllocations{0};


// 실제 사용: 256바이트짜리 객체를 128개 풀로 관리
using MyPool = ThreadLocalPool<256, 128>;

struct Task {
    int id;
    double value;
    char name[64];

    // 풀 할당자 연동
    static void* operator new(size_t) {
        void* ptr = MyPool::Allocate();
        if (!ptr) throw std::bad_alloc();
        return ptr;
    }

    static void operator delete(void* ptr) {
        MyPool::Deallocate(ptr);
    }
};

void PoolWorker(int threadId, int count) {
    std::vector<Task*> tasks;
    tasks.reserve(count);

    for (int i = 0; i < count; ++i) {
        auto* t = new Task{threadId * 1000 + i, 3.14 * i, {}};
        std::snprintf(t->name, sizeof(t->name), "task-%d-%d", threadId, i);
        tasks.push_back(t);

        // 절반은 즉시 해제해서 재사용 확인
        if (i % 2 == 0) {
            delete tasks.back();
            tasks.pop_back();
        }
    }

    std::cout << "[Thread " << threadId << "] "
              << "Free slots: " << MyPool::GetFreeSlots() << "\n";

    for (auto* t : tasks) delete t;
}

void RunPoolDemo() {
    std::cout << "=== Thread-Local Memory Pool ===\n";

    std::vector<std::thread> threads;
    for (int i = 0; i < 4; ++i) {
        threads.emplace_back(PoolWorker, i, 50);
    }
    for (auto& t : threads) t.join();

    std::cout << "Total allocations: " 
              << MyPool::GetTotalAllocations() << "\n\n";
}
```

## 사례 2: 스레드별 난수 생성기 (Random Number Generator)

**왜 TLS인가?** `rand()`는 전역 상태를 공유해서 스레드 안전하지 않습니다. `mt19937`에 뮤텍스를 거는 것은 병렬 시뮬레이션에서 병목이 됩니다. TLS로 각 스레드가 독립적인 시드를 가지면 락 없이 완전한 재현성도 확보됩니다.

```cpp
#include <random>
#include <functional>

// ============================================================
// 스레드 로컬 난수 생성기
// - 락 없는 고품질 난수
// - 스레드 ID 기반 시드로 재현 가능한 결과
// ============================================================

class ThreadLocalRNG {
private:
    struct RNGState {
        std::mt19937_64 engine;
        uint64_t threadSeed;
        size_t callCount = 0;

        RNGState() {
            // 스레드 ID + 시간 기반으로 고유 시드 생성
            auto tid = std::hash<std::thread::id>{}(
                std::this_thread::get_id()
            );
            auto tsc = static_cast<uint64_t>(
                std::chrono::high_resolution_clock::now()
                    .time_since_epoch().count()
            );

            // 두 값을 XOR-mix로 결합
            threadSeed = tid ^ (tsc * 0x9e3779b97f4a7c15ULL);
            engine.seed(threadSeed);
        }

        // 재현을 위한 수동 시드 설정
        void SetSeed(uint64_t seed) {
            threadSeed = seed;
            engine.seed(seed);
            callCount = 0;
        }
    };

    static thread_local RNGState tlRNG;

public:
    // [0, max) 정수
    static int RandInt(int min, int max) {
        std::uniform_int_distribution<int> dist(min, max - 1);
        ++tlRNG.callCount;
        return dist(tlRNG.engine);
    }

    // [0.0, 1.0) 실수
    static double RandDouble() {
        std::uniform_real_distribution<double> dist(0.0, 1.0);
        ++tlRNG.callCount;
        return dist(tlRNG.engine);
    }

    // 정규분포
    static double RandNormal(double mean, double stddev) {
        std::normal_distribution<double> dist(mean, stddev);
        ++tlRNG.callCount;
        return dist(tlRNG.engine);
    }

    // 확률 p로 true 반환 (베르누이)
    static bool RandBool(double p = 0.5) {
        std::bernoulli_distribution dist(p);
        ++tlRNG.callCount;
        return dist(tlRNG.engine);
    }

    // 시드 고정 (테스트/재현용)
    static void FixSeed(uint64_t seed) {
        tlRNG.SetSeed(seed);
    }

    static uint64_t GetThreadSeed() { return tlRNG.threadSeed; }
    static size_t GetCallCount()    { return tlRNG.callCount; }
};

thread_local ThreadLocalRNG::RNGState ThreadLocalRNG::tlRNG;


// 활용: 몬테카를로 π 추정
void MonteCarloPi(int threadId, long long samples, 
                  std::atomic<long long>& insideCount) {
    // 재현 가능한 결과를 위해 스레드별 고정 시드
    ThreadLocalRNG::FixSeed(42 + threadId);

    long long localInside = 0;
    for (long long i = 0; i < samples; ++i) {
        double x = ThreadLocalRNG::RandDouble();
        double y = ThreadLocalRNG::RandDouble();
        if (x * x + y * y < 1.0) {
            ++localInside;
        }
    }

    insideCount.fetch_add(localInside, std::memory_order_relaxed);

    std::cout << "[Thread " << threadId << "] "
              << "seed=" << ThreadLocalRNG::GetThreadSeed()
              << " calls=" << ThreadLocalRNG::GetCallCount() << "\n";
}

void RunRNGDemo() {
    std::cout << "=== Thread-Local RNG (Monte Carlo Pi) ===\n";

    const int    NUM_THREADS = 4;
    const long long SAMPLES  = 10'000'000LL;

    std::atomic<long long> inside{0};
    std::vector<std::thread> threads;

    for (int i = 0; i < NUM_THREADS; ++i) {
        threads.emplace_back(MonteCarloPi, i, SAMPLES / NUM_THREADS, 
                             std::ref(inside));
    }
    for (auto& t : threads) t.join();

    double pi = 4.0 * inside.load() / SAMPLES;
    std::cout << "Estimated π = " << std::fixed 
              << std::setprecision(6) << pi << "\n\n";
}
```

## 사례 3: 스레드별 에러 컨텍스트 (errno 패턴)

**왜 TLS인가?** C 표준의 `errno` 자체가 TLS입니다. 같은 패턴으로 **구조화된 에러 컨텍스트**를 만들면, 예외 없이도 호출 스택을 가로질러 에러 정보를 전달할 수 있습니다. 임베디드, 게임엔진처럼 예외를 금지하는 환경에서 특히 유용합니다.

```cpp
#include <optional>
#include <string_view>
#include <array>

// ============================================================
// 스레드 로컬 에러 컨텍스트
// - errno 패턴의 구조화된 버전
// - 예외 없이 에러 전파
// - 호출 스택 추적 포함
// ============================================================

struct ErrorInfo {
    int         code    = 0;
    const char* message = nullptr;
    const char* file    = nullptr;
    int         line    = 0;

    bool HasError() const { return code != 0; }
    void Clear()         { code = 0; message = nullptr; }
};

// 에러 코드 정의
enum class ErrCode : int {
    None            = 0,
    InvalidArgument = 1001,
    OutOfMemory     = 1002,
    IOFailure       = 1003,
    Timeout         = 1004,
    NotFound        = 1005,
};

class ThreadLocalError {
private:
    // 에러 히스토리 (최근 8개)
    static constexpr size_t MAX_STACK = 8;

    struct ErrorContext {
        ErrorInfo current;
        std::array<ErrorInfo, MAX_STACK> history;
        size_t historyCount = 0;

        void Push(ErrorInfo info) {
            current = info;
            // 히스토리에 추가
            if (historyCount < MAX_STACK) {
                history[historyCount++] = info;
            } else {
                // 오래된 것 밀어내기
                for (size_t i = 1; i < MAX_STACK; ++i)
                    history[i - 1] = history[i];
                history[MAX_STACK - 1] = info;
            }
        }

        void Clear() {
            current.Clear();
            historyCount = 0;
        }
    };

    static thread_local ErrorContext tlError;

public:
    // 에러 설정
    static void SetError(ErrCode code, const char* message,
                         const char* file, int line) {
        tlError.Push({
            static_cast<int>(code), message, file, line
        });
    }

    // 에러 확인
    static bool HasError() {
        return tlError.current.HasError();
    }

    static const ErrorInfo& GetError() {
        return tlError.current;
    }

    // 에러 소비 (확인 후 클리어)
    static std::optional<ErrorInfo> ConsumeError() {
        if (!tlError.current.HasError()) return std::nullopt;
        ErrorInfo err = tlError.current;
        tlError.current.Clear();
        return err;
    }

    // 에러 히스토리 출력
    static void PrintHistory() {
        std::cout << "[Error History - Thread "
                  << std::this_thread::get_id() << "]\n";
        for (size_t i = 0; i < tlError.historyCount; ++i) {
            const auto& e = tlError.history[i];
            std::cout << "  [" << i << "] code=" << e.code
                      << " msg=" << (e.message ? e.message : "")
                      << " @ " << (e.file ? e.file : "") 
                      << ":" << e.line << "\n";
        }
    }

    static void Clear() { tlError.Clear(); }
};

thread_local ThreadLocalError::ErrorContext ThreadLocalError::tlError;

// 에러 설정 매크로 (파일/라인 자동 캡처)
#define SET_ERROR(code, msg) \
    ThreadLocalError::SetError(code, msg, __FILE__, __LINE__)

#define CHECK_ERROR(expr)          \
    do {                           \
        if (!(expr)) return false; \
    } while(0)


// 활용: 예외 없는 에러 전파 체인
bool ParseConfig(std::string_view input) {
    if (input.empty()) {
        SET_ERROR(ErrCode::InvalidArgument, "Config input is empty");
        return false;
    }
    if (input.size() > 1024) {
        SET_ERROR(ErrCode::InvalidArgument, "Config too large");
        return false;
    }
    return true;
}

bool LoadConfig(std::string_view path) {
    // path가 비어있으면 에러 전파
    if (!ParseConfig(path)) {
        // 에러는 이미 설정됨, 그냥 전파
        return false;
    }
    // ... 실제 파일 로드 로직
    return true;
}

bool InitSystem(std::string_view configPath) {
    CHECK_ERROR(LoadConfig(configPath));  // 실패 시 즉시 false 반환
    // ... 이후 초기화
    return true;
}

void ErrorWorker(int id) {
    ThreadLocalError::Clear();

    bool ok = InitSystem(id % 2 == 0 ? "" : "config.ini");

    if (!ok) {
        auto err = ThreadLocalError::ConsumeError();
        if (err) {
            std::cout << "[Thread " << id << "] Error "
                      << err->code << ": " << err->message
                      << " (" << err->file << ":" << err->line << ")\n";
        }
    } else {
        std::cout << "[Thread " << id << "] OK\n";
    }

    ThreadLocalError::PrintHistory();
}

void RunErrorDemo() {
    std::cout << "=== Thread-Local Error Context ===\n";

    std::vector<std::thread> threads;
    for (int i = 0; i < 4; ++i) {
        threads.emplace_back(ErrorWorker, i);
    }
    for (auto& t : threads) t.join();
    std::cout << "\n";
}
```

## 사례 4: 스레드별 문자열 포맷 버퍼

**왜 TLS인가?** 고빈도 로깅, 직렬화, 네트워크 패킷 생성에서 매 호출마다 `std::string`을 힙 할당하는 비용이 누적됩니다. TLS 버퍼를 재사용하면 핫패스에서 **힙 할당을 완전히 제거**할 수 있습니다.

```cpp
#include <cstdarg>

// ============================================================
// 스레드 로컬 재사용 포맷 버퍼
// - 힙 할당 제로
// - printf 스타일 + 빌더 패턴 지원
// ============================================================

class ThreadLocalFormatter {
public:
    static constexpr size_t CAPACITY = 64 * 1024;  // 64KB

private:
    struct FormatBuffer {
        char   buf[CAPACITY];
        size_t len = 0;

        void Reset() { len = 0; }

        // printf 스타일 추가
        bool Appendf(const char* fmt, ...) {
            va_list args;
            va_start(args, fmt);
            int written = std::vsnprintf(
                buf + len, CAPACITY - len, fmt, args
            );
            va_end(args);

            if (written > 0) {
                len += std::min(
                    static_cast<size_t>(written), CAPACITY - len - 1
                );
                return true;
            }
            return false;
        }

        // 문자열 추가
        bool Append(std::string_view sv) {
            size_t space = CAPACITY - len - 1;
            size_t toCopy = std::min(sv.size(), space);
            std::memcpy(buf + len, sv.data(), toCopy);
            len += toCopy;
            buf[len] = '\0';
            return toCopy == sv.size();
        }

        std::string_view View() const {
            return { buf, len };
        }
    };

    static thread_local FormatBuffer tlFmt;

public:
    // RAII 스코프: 스코프 종료 시 자동 리셋
    struct Scope {
        size_t savedLen;

        Scope() : savedLen(tlFmt.len) {}
        ~Scope()                      { tlFmt.len = savedLen; }

        Scope(const Scope&)            = delete;
        Scope& operator=(const Scope&) = delete;
    };

    static void Reset() { tlFmt.Reset(); }

    static bool Appendf(const char* fmt, ...) {
        va_list args;
        va_start(args, fmt);
        int written = std::vsnprintf(
            tlFmt.buf + tlFmt.len, 
            CAPACITY - tlFmt.len, 
            fmt, args
        );
        va_end(args);

        if (written > 0) {
            tlFmt.len += std::min(
                static_cast<size_t>(written), 
                CAPACITY - tlFmt.len - 1
            );
            return true;
        }
        return false;
    }

    static bool Append(std::string_view sv) {
        return tlFmt.Append(sv);
    }

    static std::string_view View() { return tlFmt.View(); }

    // 완성된 문자열이 필요한 경우에만 std::string 생성
    static std::string ToString() { 
        return std::string(tlFmt.View()); 
    }
};

thread_local ThreadLocalFormatter::FormatBuffer ThreadLocalFormatter::tlFmt;


// 활용: 고빈도 JSON 직렬화
std::string_view SerializeEvent(int userId, const char* action, 
                                double value) {
    // 매 호출마다 힙 할당 없이 포맷
    ThreadLocalFormatter::Reset();
    ThreadLocalFormatter::Appendf(
        R"({"user":%d,"action":"%s","value":%.3f,"ts":%lld})",
        userId, action, value,
        std::chrono::system_clock::now().time_since_epoch().count()
    );
    return ThreadLocalFormatter::View();
    // 반환된 string_view는 다음 Reset() 전까지 유효
}

void FormatterWorker(int threadId, int count) {
    for (int i = 0; i < count; ++i) {
        // string_view: 힙 할당 없음
        auto json = SerializeEvent(
            threadId * 1000 + i, "click", 3.14 * i
        );

        // 실제로 전송/기록이 필요할 때만 std::string 생성
        if (i == 0) {
            std::cout << "[Thread " << threadId << "] "
                      << json << "\n";
        }
    }
}

void RunFormatterDemo() {
    std::cout << "=== Thread-Local Format Buffer ===\n";

    std::vector<std::thread> threads;
    for (int i = 0; i < 4; ++i) {
        threads.emplace_back(FormatterWorker, i, 100);
    }
    for (auto& t : threads) t.join();
    std::cout << "\n";
}

// ============================================================
// 통합 실행
// ============================================================

int main() {
    RunPoolDemo();
    RunRNGDemo();
    RunErrorDemo();
    RunFormatterDemo();
    return 0;
}
```

## 사례별 TLS 적합성 요약

| 사례 | TLS 이유 | 락 절감 효과 | 실제 사용처 |
|---|---|---|---|
| 메모리 풀 | 힙 락 제거 | ★★★★★ | jemalloc, tcmalloc |
| 난수 생성기 | 상태 격리 + 재현성 | ★★★★☆ | 시뮬레이션, 게임 |
| 에러 컨텍스트 | `errno` 패턴 확장 | ★★★★☆ | 임베디드, 게임엔진 |
| 포맷 버퍼 | 힙 할당 제로 | ★★★☆☆ | 고빈도 직렬화, 로그 |

TLS의 핵심 원칙은 **"공유하지 않아도 되는 것은 처음부터 공유하지 않는다"** 입니다. 뮤텍스로 보호하는 대신 애초에 경합이 발생하지 않는 구조를 만드는 것이 진짜 고성능 설계입니다.
