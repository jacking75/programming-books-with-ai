# 5. 고급 패턴과 주의사항
Thread-Local Storage는 강력한 도구이지만, 올바르게 사용하지 않으면 예측 불가능한 버그와 성능 문제를 야기할 수 있습니다. 이 섹션에서는 실무에서 마주칠 수 있는 함정들과 그 해결책을 깊이 있게 다루겠습니다.

## 5.1 초기화 순서 문제

### 문제의 본질
thread_local 변수의 초기화는 해당 스레드에서 처음 접근할 때 발생합니다. 이것이 의도하지 않은 시점에 일어나면 데드락이나 예외 상황을 초래할 수 있습니다.

```cpp
// ❌ 위험한 코드
class Logger {
public:
    static thread_local std::ofstream logFile;
    
    static void Init() {
        logFile.open("log.txt", std::ios::app);  // 문제 발생 지점
    }
};

thread_local std::ofstream Logger::logFile;  // 정의

int main() {
    std::thread t1(Logger::Init);  // t1에서만 초기화됨
    std::thread t2(Logger::Init);  // t2에서만 초기화됨
    t1.join();
    t2.join();
}
```

이 코드의 문제점은 각 스레드에서 `Init()`을 호출할 때까지 `logFile`이 초기화되지 않으며, 여러 스레드가 동시에 초기화를 시도할 수 있다는 점입니다.

### 해결책: 안전한 초기화 패턴

```cpp
#include <thread>
#include <mutex>
#include <iostream>

// 방법 1: 스레드마다 명시적 초기화
class SafeLogger {
private:
    static thread_local bool initialized;
    static thread_local int threadId;
    static std::mutex initMutex;
    
public:
    static void EnsureInitialized() {
        if (!initialized) {
            std::lock_guard<std::mutex> lock(initMutex);
            if (!initialized) {
                threadId = std::this_thread::get_id();
                std::cout << "Thread " << threadId << " initialized\n";
                initialized = true;
            }
        }
    }
    
    static int GetThreadId() {
        EnsureInitialized();
        return threadId;
    }
};

thread_local bool SafeLogger::initialized = false;
thread_local int SafeLogger::threadId = -1;
std::mutex SafeLogger::initMutex;


// 방법 2: RAII 패턴을 활용한 자동 초기화
class ThreadContext {
private:
    int contextId;
    std::string contextName;
    
    ThreadContext(int id, const std::string& name) 
        : contextId(id), contextName(name) {
        std::cout << "Context initialized: " << contextName << "\n";
    }
    
public:
    static ThreadContext& Get() {
        thread_local ThreadContext instance(
            std::hash<std::thread::id>{}(std::this_thread::get_id()),
            "ThreadContext_" + std::to_string(
                std::hash<std::thread::id>{}(std::this_thread::get_id())
            )
        );
        return instance;
    }
    
    int GetContextId() const { return contextId; }
    const std::string& GetContextName() const { return contextName; }
};


// 방법 3: 초기화 순서 명시적 제어
class ControlledInit {
private:
    static thread_local struct {
        bool isInitialized;
        std::vector<int> buffer;
        
        void Initialize() {
            if (isInitialized) return;
            buffer.resize(1024);
            for (int& val : buffer) val = 0;
            isInitialized = true;
        }
    } state;
    
public:
    static void Init() {
        state.Initialize();
    }
    
    static std::vector<int>& GetBuffer() {
        state.Initialize();
        return state.buffer;
    }
};
```

## 5.2 동적 라이브러리에서의 TLS

### Win32 API에서의 동작 방식
DLL을 동적으로 로드할 때, 그 DLL 내의 TLS는 독특한 방식으로 작동합니다:

```cpp
// MyLibrary.dll 내부
// ============================================
thread_local int globalCounter = 0;  // 각 스레드마다 별도 저장소

extern "C" __declspec(dllexport) void IncrementCounter() {
    globalCounter++;  // 스레드 로컬
}

extern "C" __declspec(dllexport) int GetCounter() {
    return globalCounter;
}


// 주 프로그램
// ============================================
#include <windows.h>
#include <iostream>
#include <thread>

typedef void (*PFN_IncrementCounter)();
typedef int (*PFN_GetCounter)();

int main() {
    // DLL 동적 로드
    HMODULE hLib = LoadLibraryA("MyLibrary.dll");
    
    auto pIncrement = (PFN_IncrementCounter)GetProcAddress(
        hLib, "IncrementCounter"
    );
    auto pGetCounter = (PFN_GetCounter)GetProcAddress(
        hLib, "GetCounter"
    );
    
    // 메인 스레드에서
    pIncrement();
    std::cout << "Main thread: " << pGetCounter() << "\n";  // 1
    
    // 워커 스레드에서
    std::thread t([pIncrement, pGetCounter]() {
        pIncrement();
        pIncrement();
        std::cout << "Worker thread: " << pGetCounter() << "\n";  // 2 (독립적!)
    });
    
    t.join();
    std::cout << "Main thread again: " << pGetCounter() << "\n";  // 1
    
    FreeLibrary(hLib);
    return 0;
}
```

### DLL 언로드 시 주의사항

```cpp
// ⚠️ 위험한 패턴
class DllResource {
public:
    thread_local static std::unique_ptr<int> resource;
    
    static void Initialize() {
        resource = std::make_unique<int>(42);
    }
};

// DLL이 언로드되면서 thread_local 소멸자 호출
// 만약 다른 스레드가 여전히 이 DLL 함수를 실행 중이면?
// -> 미정의 동작(Undefined Behavior)


// ✓ 안전한 패턴: 명시적 정리
class SafeDllResource {
private:
    thread_local static std::unique_ptr<int> resource;
    thread_local static bool isInitialized;
    
public:
    static void Initialize() {
        if (!isInitialized) {
            resource = std::make_unique<int>(42);
            isInitialized = true;
        }
    }
    
    static void Cleanup() {
        resource.reset();
        isInitialized = false;
    }
    
    static int* Get() {
        if (!isInitialized) return nullptr;
        return resource.get();
    }
};

// DLL 진입점
BOOL APIENTRY DllMain(
    HMODULE hModule,
    DWORD ul_reason_for_call,
    LPVOID lpReserved
) {
    switch (ul_reason_for_call) {
        case DLL_THREAD_ATTACH:
            // 스레드가 DLL을 로드할 때 호출
            SafeDllResource::Initialize();
            break;
            
        case DLL_THREAD_DETACH:
            // 스레드가 종료될 때 호출
            SafeDllResource::Cleanup();
            break;
            
        case DLL_PROCESS_DETACH:
            // DLL이 언로드될 때 호출
            SafeDllResource::Cleanup();
            break;
    }
    return TRUE;
}
```

### Win32 TLS API를 사용한 DLL 호환성
C++ thread_local 대신 Win32 TLS API를 사용하면 더 명시적으로 제어할 수 있습니다:

```cpp
#include <windows.h>

// TLS 인덱스 (전역 변수)
DWORD g_tlsIndex = TLS_OUT_OF_INDEXES;

// DLL 초기화
void InitializeTls() {
    g_tlsIndex = TlsAlloc();
    if (g_tlsIndex == TLS_OUT_OF_INDEXES) {
        // 오류 처리
        return;
    }
}

// TLS 정리
void CleanupTls() {
    if (g_tlsIndex != TLS_OUT_OF_INDEXES) {
        TlsFree(g_tlsIndex);
        g_tlsIndex = TLS_OUT_OF_INDEXES;
    }
}

// 스레드별 데이터 설정
void SetThreadData(int* data) {
    if (g_tlsIndex != TLS_OUT_OF_INDEXES) {
        TlsSetValue(g_tlsIndex, (LPVOID)data);
    }
}

// 스레드별 데이터 조회
int* GetThreadData() {
    if (g_tlsIndex == TLS_OUT_OF_INDEXES) return nullptr;
    return (int*)TlsGetValue(g_tlsIndex);
}

// DllMain에서의 사용
BOOL APIENTRY DllMain(
    HMODULE hModule,
    DWORD ul_reason_for_call,
    LPVOID lpReserved
) {
    switch (ul_reason_for_call) {
        case DLL_PROCESS_ATTACH:
            InitializeTls();
            break;
            
        case DLL_THREAD_ATTACH:
            {
                int* threadData = new int(0);
                SetThreadData(threadData);
            }
            break;
            
        case DLL_THREAD_DETACH:
            {
                int* threadData = GetThreadData();
                delete threadData;
            }
            break;
            
        case DLL_PROCESS_DETACH:
            CleanupTls();
            break;
    }
    return TRUE;
}
```

## 5.3 성능 고려사항

### 접근 비용 분석
thread_local 변수의 접근 비용은 몇 가지 요인에 영향을 받습니다:

```cpp
#include <chrono>
#include <iostream>
#include <vector>
#include <thread>

// 성능 테스트 프레임워크
class PerformanceAnalyzer {
public:
    struct Result {
        const char* name;
        uint64_t iterations;
        double durationMs;
        double opsPerSecond;
        
        void Print() const {
            std::cout << name << ":\n"
                      << "  Duration: " << durationMs << " ms\n"
                      << "  Ops/sec: " << opsPerSecond / 1e6 << "M\n";
        }
    };
    
    template<typename Func>
    static Result Measure(
        const char* name,
        uint64_t iterations,
        Func func
    ) {
        auto start = std::chrono::high_resolution_clock::now();
        
        for (uint64_t i = 0; i < iterations; ++i) {
            func();
        }
        
        auto end = std::chrono::high_resolution_clock::now();
        auto duration = std::chrono::duration_cast<
            std::chrono::nanoseconds
        >(end - start);
        
        double durationMs = duration.count() / 1e6;
        double opsPerSecond = (iterations / (duration.count() / 1e9));
        
        return { name, iterations, durationMs, opsPerSecond };
    }
};


// 테스트 1: TLS 접근 vs 일반 변수
thread_local int tlsCounter = 0;
int globalCounter = 0;  // ❌ 스레드 세이프하지 않음

void TestAccessCost() {
    const uint64_t ITERATIONS = 100'000'000;
    
    // TLS 접근
    auto tlsResult = PerformanceAnalyzer::Measure(
        "TLS access",
        ITERATIONS,
        []() { tlsCounter++; }
    );
    
    // 일반 변수 접근 (스레드 세이프 없음)
    auto globalResult = PerformanceAnalyzer::Measure(
        "Global variable access (unsafe)",
        ITERATIONS,
        []() { globalCounter++; }
    );
    
    tlsResult.Print();
    globalResult.Print();
    
    std::cout << "TLS는 일반 변수보다 약 "
              << tlsResult.durationMs / globalResult.durationMs
              << "배 느림\n";
}


// 테스트 2: 캐싱의 영향
struct CachedAccess {
    thread_local static int tlsValue;
    
    // ❌ 비효율적: 매번 접근
    static void Inefficient() {
        for (int i = 0; i < 100; ++i) {
            tlsValue += i;  // TLS 접근 100회
        }
    }
    
    // ✓ 효율적: 캐싱
    static void Efficient() {
        int cached = tlsValue;  // TLS 접근 1회
        for (int i = 0; i < 100; ++i) {
            cached += i;
        }
        tlsValue = cached;  // TLS 접근 1회
    }
};

thread_local int CachedAccess::tlsValue = 0;

void TestCachingEffect() {
    const uint64_t ITERATIONS = 10'000'000;
    
    auto inefficient = PerformanceAnalyzer::Measure(
        "TLS without caching",
        ITERATIONS,
        []() { CachedAccess::Inefficient(); }
    );
    
    auto efficient = PerformanceAnalyzer::Measure(
        "TLS with caching",
        ITERATIONS,
        []() { CachedAccess::Efficient(); }
    );
    
    inefficient.Print();
    efficient.Print();
    
    std::cout << "캐싱을 통해 "
              << inefficient.durationMs / efficient.durationMs
              << "배 개선\n";
}


// 테스트 3: 메모리 크기의 영향
void TestMemorySizeEffect() {
    const uint64_t ITERATIONS = 50'000'000;
    
    // 작은 TLS
    thread_local int smallTls = 0;
    auto smallResult = PerformanceAnalyzer::Measure(
        "Small TLS (int)",
        ITERATIONS,
        []() { smallTls++; }
    );
    
    // 중간 크기 TLS
    thread_local std::array<int, 64> mediumTls{};
    auto mediumResult = PerformanceAnalyzer::Measure(
        "Medium TLS (64 ints)",
        ITERATIONS,
        [](){ ++mediumTls[0]; }
    );
    
    // 큰 TLS
    thread_local std::vector<int> largeTls(1024);
    auto largeResult = PerformanceAnalyzer::Measure(
        "Large TLS (1024 ints vector)",
        ITERATIONS,
        [](){ ++largeTls[0]; }
    );
    
    smallResult.Print();
    mediumResult.Print();
    largeResult.Print();
}

int main() {
    std::cout << "=== TLS 성능 분석 ===\n\n";
    
    TestAccessCost();
    std::cout << "\n";
    
    TestCachingEffect();
    std::cout << "\n";
    
    TestMemorySizeEffect();
    
    return 0;
}
```

### 캐시 일관성 관점에서의 고려

```cpp
// ⚠️ False Sharing 문제는 TLS에서는 없지만,
// TLS를 포함하는 전역 구조체에서는 발생 가능

struct BadDesign {
    thread_local static int counter;      // 각 스레드마다 별도
    static std::atomic<int> globalValue;  // 모든 스레드 공유
    // counter는 False Sharing 영향 없음
    // globalValue는 반복적인 원자 연산으로 캐시 경합 발생
};

struct GoodDesign {
    thread_local static int counter;      // 각 스레드마다 별도
    
    // 캐시 라인(64바이트) 정렬
    alignas(64) static std::atomic<int> globalValue;
    
    // counter는 TLS이므로 자동으로 격리됨
};
```

## 5.4 메모리 누수 방지

### TLS 메모리 관리의 핵심
thread_local이 포함하는 동적 할당 객체는 각 스레드마다 정리되어야 합니다:

```cpp
#include <memory>
#include <iostream>
#include <thread>
#include <vector>

// 문제 상황 1: 명시적 delete 누락
class BadMemoryManagement {
public:
    thread_local static int* buffer;
    
    static void Initialize() {
        buffer = new int[1024];  // 할당
    }
    
    // Cleanup() 메서드가 없어서 메모리 누수!
};

thread_local int* BadMemoryManagement::buffer = nullptr;


// 해결책 1: std::unique_ptr 사용 (권장)
class GoodMemoryManagement {
private:
    thread_local static std::unique_ptr<int[]> buffer;
    
public:
    static void Initialize() {
        buffer = std::make_unique<int[]>(1024);
        // 스레드 종료 시 자동으로 delete됨
    }
    
    static int* GetBuffer() {
        if (!buffer) Initialize();
        return buffer.get();
    }
};

thread_local std::unique_ptr<int[]> 
    GoodMemoryManagement::buffer = nullptr;


// 해결책 2: RAII 래퍼
class ThreadResourceGuard {
private:
    class Resource {
    public:
        int* buffer;
        std::string name;
        
        Resource() : buffer(nullptr) {
            std::cout << "Resource created for thread\n";
        }
        
        ~Resource() {
            delete[] buffer;
            std::cout << "Resource destroyed for thread\n";
        }
    };
    
    thread_local static Resource resource;
    
public:
    static int* GetBuffer() {
        if (resource.buffer == nullptr) {
            resource.buffer = new int[1024];
        }
        return resource.buffer;
    }
    
    // 스레드 종료 시 ~Resource() 자동 호출
};

thread_local ThreadResourceGuard::Resource 
    ThreadResourceGuard::resource;


// 해결책 3: 명시적 정리 콜백
class ExplicitCleanup {
private:
    thread_local static struct {
        std::vector<void*> allocations;
        
        ~Allocator() {
            std::cout << "Cleaning up " << allocations.size() 
                      << " allocations\n";
            for (void* p : allocations) {
                delete[] (char*)p;
            }
            allocations.clear();
        }
    } allocator;
    
public:
    static void* Allocate(size_t size) {
        void* p = operator new(size);
        allocator.allocations.push_back(p);
        return p;
    }
    
    static void Reset() {
        allocator.~Allocator();
        new (&allocator) decltype(allocator)();
    }
};

thread_local ExplicitCleanup::decltype(ExplicitCleanup::allocator) 
    ExplicitCleanup::allocator;


// 해결책 4: 스마트 포인터 컨테이너
class SmartPointerContainer {
private:
    thread_local static std::vector<
        std::unique_ptr<int[]>
    > buffers;
    
public:
    static int* AllocateBuffer(size_t size) {
        auto buffer = std::make_unique<int[]>(size);
        int* ptr = buffer.get();
        buffers.push_back(std::move(buffer));
        return ptr;
    }
    
    static size_t GetBufferCount() {
        return buffers.size();
    }
    
    // ~buffers() 소멸자가 모든 할당 정리
};

thread_local std::vector<std::unique_ptr<int[]>>
```

### Win32 API를 사용한 명시적 정리

```cpp
#include <windows.h>

// Win32 TLS API를 사용할 때의 정리
DWORD g_tlsIndex = TLS_OUT_OF_INDEXES;

struct ThreadData {
    char* buffer;
    int size;
    
    ThreadData() : buffer(nullptr), size(0) {}
    ~ThreadData() {
        if (buffer) {
            delete[] buffer;
            buffer = nullptr;
        }
    }
};

// 스레드 로컬 저장소에 구조체 저장
void SetupThreadData() {
    ThreadData* pData = new ThreadData();
    pData->buffer = new char[4096];
    pData->size = 4096;
    
    TlsSetValue(g_tlsIndex, (LPVOID)pData);
}

// 정리 함수
void CleanupThreadData() {
    ThreadData* pData = (ThreadData*)TlsGetValue(g_tlsIndex);
    if (pData) {
        delete pData;  // 소멸자가 buffer도 정리
        TlsSetValue(g_tlsIndex, nullptr);
    }
}

BOOL APIENTRY DllMain(
    HMODULE hModule,
    DWORD ul_reason_for_call,
    LPVOID lpReserved
) {
    switch (ul_reason_for_call) {
        case DLL_PROCESS_ATTACH:
            g_tlsIndex = TlsAlloc();
            break;
            
        case DLL_THREAD_ATTACH:
            SetupThreadData();
            break;
            
        case DLL_THREAD_DETACH:
            CleanupThreadData();  // 중요!
            break;
            
        case DLL_PROCESS_DETACH:
            TlsFree(g_tlsIndex);
            break;
    }
    return TRUE;
}
```

### 메모리 누수 검출

```cpp
#include <map>
#include <thread>

class MemoryLeakDetector {
private:
    static std::map<std::thread::id, size_t> allocations;
    static std::mutex mtx;
    
public:
    static void* AllocateTracked(size_t size) {
        {
            std::lock_guard<std::mutex> lock(mtx);
            allocations[std::this_thread::get_id()] += size;
        }
        return operator new(size);
    }
    
    static void DeallocateTracked(void* ptr, size_t size) {
        {
            std::lock_guard<std::mutex> lock(mtx);
            auto it = allocations.find(std::this_thread::get_id());
            if (it != allocations.end()) {
                it->second -= std::min(it->second, size);
            }
        }
        operator delete(ptr);
    }
    
    static void ReportLeaks() {
        std::lock_guard<std::mutex> lock(mtx);
        std::cout << "Memory allocation report:\n";
        for (const auto& [threadId, size] : allocations) {
            if (size > 0) {
                std::cout << "  Thread " << threadId 
                          << ": " << size << " bytes leaked\n";
            }
        }
    }
};

std::map<std::thread::id, size_t> 
    MemoryLeakDetector::allocations;
std::mutex MemoryLeakDetector::mtx;
```
