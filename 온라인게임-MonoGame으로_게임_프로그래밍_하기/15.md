# 15장. 파티클 시스템

## 15.1 파티클의 개념
파티클 시스템은 게임에서 불, 연기, 폭발, 먼지 같은 효과를 표현하는 기술이다. 수많은 작은 객체들(파티클)이 각각의 물리 속성을 가지고 움직이면서 동적이고 자연스러운 이펙트를 만들어낸다.

파티클의 기본 개념을 이해하려면 다음 특성들을 알아야 한다. 각 파티클은 위치, 속도, 가속도, 생명 시간(lifetime), 크기, 색상, 회전 각도 같은 속성을 가진다. 게임이 실행될 때마다 파티클은 자신의 속성에 따라 이동하고 변화하며, 생명 시간이 끝나면 사라진다.

파티클 시스템을 사용하면 다음과 같은 이점이 있다. 첫째, 복잡한 시각 효과를 적은 코드로 표현할 수 있다. 둘째, 수백 개의 작은 객체를 효율적으로 관리할 수 있다. 셋째, 게임의 느낌과 분위기를 크게 향상시킨다. 이전 장에서 배운 오브젝트 풀링과 함께 사용하면 성능도 최적화할 수 있다.

다음은 파티클 시스템의 기본 동작 흐름을 보여준다.

```
┌─────────────────────────────────────────┐
│   파티클 시스템의 작동 흐름                 │
├─────────────────────────────────────────┤
│                                         │
│  1. 파티클 생성 (위치, 속도 설정)           │
│     ↓                                   │
│  2. 매 프레임마다 파티클 업데이트           │
│     ├─ 위치 계산 (속도 × 시간)             │
│     ├─ 속도 변화 (중력, 바람 등)           │
│     ├─ 생명 시간 감소                     │
│     └─ 알파값 감소 (페이드 아웃)           │
│     ↓                                   │
│  3. 파티클 그리기                         │
│     ↓                                   │
│  4. 생명 시간이 0이 되면 삭제              │
│     ↓                                   │
│  5. 모든 파티클이 사라질 때까지 반복         │
│                                         │
└─────────────────────────────────────────┘
```

## 15.2 간단한 파티클 이펙트
이번 섹션에서는 기본적인 파티클 시스템을 구현한다. 먼저 개별 파티클을 나타내는 `Particle` 클래스를 만들겠다.

```csharp
using Microsoft.Xna.Framework;
using Microsoft.Xna.Framework.Graphics;

namespace ParticleSystemDemo
{
    public class Particle
    {
        // 파티클의 물리 속성
        public Vector2 Position { get; set; }
        public Vector2 Velocity { get; set; }
        public Vector2 Acceleration { get; set; }
        
        // 파티클의 시각적 속성
        public Color Color { get; set; }
        public float Rotation { get; set; }
        public float Scale { get; set; }
        
        // 생명 주기 관련
        public float Lifetime { get; set; }
        public float MaxLifetime { get; set; }
        public bool IsActive { get; set; }
        
        // 렌더링에 필요한 텍스처
        private Texture2D texture;
        private Vector2 origin;
        
        public Particle(Texture2D particleTexture, Vector2 startPosition, 
                        Vector2 startVelocity, float lifetime)
        {
            texture = particleTexture;
            Position = startPosition;
            Velocity = startVelocity;
            Acceleration = Vector2.Zero;
            
            Lifetime = lifetime;
            MaxLifetime = lifetime;
            IsActive = true;
            
            Color = Color.White;
            Rotation = 0f;
            Scale = 1f;
            
            // 텍스처의 중심점을 기준으로 그리기 위해 원점 설정
            origin = new Vector2(texture.Width / 2f, texture.Height / 2f);
        }
        
        public void Update(GameTime gameTime)
        {
            if (!IsActive)
                return;
            
            float deltaTime = (float)gameTime.ElapsedGameTime.TotalSeconds;
            
            // 속도에 가속도 적용
            Velocity += Acceleration * deltaTime;
            
            // 위치 업데이트 (선형 운동)
            Position += Velocity * deltaTime;
            
            // 생명 시간 감소
            Lifetime -= deltaTime;
            
            // 생명 시간이 끝나면 비활성화
            if (Lifetime <= 0f)
            {
                IsActive = false;
                Lifetime = 0f;
            }
        }
        
        public void Draw(SpriteBatch spriteBatch)
        {
            if (!IsActive)
                return;
            
            // 생명 시간에 따라 알파값 계산 (페이드 아웃)
            float alpha = Lifetime / MaxLifetime;
            Color colorWithAlpha = Color * alpha;
            
            spriteBatch.Draw(
                texture,
                Position,
                null,
                colorWithAlpha,
                Rotation,
                origin,
                Scale,
                SpriteEffects.None,
                0f
            );
        }
    }
}
```

위 코드는 개별 파티클을 표현하는 클래스이다. 각 파티클은 위치, 속도, 가속도를 가진 물리 객체로 동작한다. `Update` 메서드에서 매 프레임마다 위치를 계산하고 생명 시간을 감소시킨다. 생명 시간이 0이 되면 `IsActive` 플래그를 false로 설정하여 더 이상 업데이트되지 않도록 한다. `Draw` 메서드에서는 남은 생명 시간의 비율로 알파값을 계산하여 시간이 지나면서 점점 투명해지는 효과를 만든다.

이제 파티클 여러 개를 관리하는 `ParticleEmitter` 클래스를 만들겠다.

```csharp
using Microsoft.Xna.Framework;
using Microsoft.Xna.Framework.Graphics;
using System;
using System.Collections.Generic;

namespace ParticleSystemDemo
{
    public class ParticleEmitter
    {
        private List<Particle> particles;
        private Texture2D particleTexture;
        
        // 파티클 생성 설정
        private Vector2 emitterPosition;
        private float emissionRate;  // 초당 생성할 파티클 수
        private float timeSinceLastEmission;
        
        // 파티클 초기값 범위
        private float minSpeed;
        private float maxSpeed;
        private float minAngle;
        private float maxAngle;
        private float minLifetime;
        private float maxLifetime;
        private Vector2 gravityVector;
        
        private Random random;
        private bool isActive;
        
        public ParticleEmitter(Texture2D texture, Vector2 position, int maxParticles = 1000)
        {
            particleTexture = texture;
            emitterPosition = position;
            particles = new List<Particle>(maxParticles);
            
            // 기본값 설정
            emissionRate = 100f;  // 초당 100개
            minSpeed = 50f;
            maxSpeed = 200f;
            minAngle = -MathHelper.PiOver2;
            maxAngle = MathHelper.PiOver2;
            minLifetime = 0.5f;
            maxLifetime = 2f;
            gravityVector = new Vector2(0f, 200f);
            
            random = new Random();
            isActive = true;
            timeSinceLastEmission = 0f;
        }
        
        public void SetPosition(Vector2 newPosition)
        {
            emitterPosition = newPosition;
        }
        
        public void SetEmissionRate(float rate)
        {
            emissionRate = rate;
        }
        
        public void SetSpeedRange(float min, float max)
        {
            minSpeed = min;
            maxSpeed = max;
        }
        
        public void SetAngleRange(float min, float max)
        {
            minAngle = min;
            maxAngle = max;
        }
        
        public void SetLifetimeRange(float min, float max)
        {
            minLifetime = min;
            maxLifetime = max;
        }
        
        public void SetGravity(Vector2 gravity)
        {
            gravityVector = gravity;
        }
        
        public void Stop()
        {
            isActive = false;
        }
        
        public void Resume()
        {
            isActive = true;
        }
        
        public bool IsEmitting()
        {
            return isActive && particles.Count > 0;
        }
        
        public void Update(GameTime gameTime)
        {
            float deltaTime = (float)gameTime.ElapsedGameTime.TotalSeconds;
            
            // 파티클 생성
            if (isActive)
            {
                timeSinceLastEmission += deltaTime;
                float particleInterval = 1f / emissionRate;
                
                while (timeSinceLastEmission >= particleInterval)
                {
                    EmitParticle();
                    timeSinceLastEmission -= particleInterval;
                }
            }
            
            // 모든 파티클 업데이트
            for (int i = particles.Count - 1; i >= 0; i--)
            {
                particles[i].Acceleration = gravityVector;
                particles[i].Update(gameTime);
                
                // 비활성 파티클 제거
                if (!particles[i].IsActive)
                {
                    particles.RemoveAt(i);
                }
            }
        }
        
        public void Draw(SpriteBatch spriteBatch)
        {
            foreach (var particle in particles)
            {
                particle.Draw(spriteBatch);
            }
        }
        
        private void EmitParticle()
        {
            // 각도와 속도를 무작위로 설정하여 다양한 방향으로 파티클 방출
            float angle = minAngle + (float)random.NextDouble() * (maxAngle - minAngle);
            float speed = minSpeed + (float)random.NextDouble() * (maxSpeed - minSpeed);
            
            // 속도 벡터 계산
            Vector2 velocity = new Vector2(
                (float)Math.Cos(angle) * speed,
                (float)Math.Sin(angle) * speed
            );
            
            // 생명 시간 무작위 설정
            float lifetime = minLifetime + (float)random.NextDouble() * (maxLifetime - minLifetime);
            
            // 새 파티클 생성 및 리스트에 추가
            var particle = new Particle(particleTexture, emitterPosition, velocity, lifetime);
            particles.Add(particle);
        }
    }
}
```

`ParticleEmitter` 클래스는 파티클을 생성하고 관리한다. `Update` 메서드에서 `emissionRate`에 따라 주기적으로 새 파티클을 생성한다. 모든 파티클의 업데이트를 수행하고 비활성 파티클을 자동으로 제거한다. 각 파티클의 각도와 속도는 무작위로 설정되어 폭발이나 분사 효과를 자연스럽게 표현한다.

이제 파티클 이펙트를 실제로 사용하는 게임 클래스를 만들어보겠다.

```csharp
using Microsoft.Xna.Framework;
using Microsoft.Xna.Framework.Graphics;
using Microsoft.Xna.Framework.Input;

namespace ParticleSystemDemo
{
    public class Game1 : Game
    {
        private GraphicsDeviceManager graphics;
        private SpriteBatch spriteBatch;
        private Texture2D particleTexture;
        private ParticleEmitter emitter;
        
        public Game1()
        {
            graphics = new GraphicsDeviceManager(this);
            Content.RootDirectory = "Content";
        }
        
        protected override void Initialize()
        {
            graphics.PreferredBackBufferWidth = 800;
            graphics.PreferredBackBufferHeight = 600;
            graphics.ApplyChanges();
            
            base.Initialize();
        }
        
        protected override void LoadContent()
        {
            spriteBatch = new SpriteBatch(GraphicsDevice);
            
            // 작은 흰색 원형 텍스처 생성 (파티클로 사용)
            particleTexture = new Texture2D(GraphicsDevice, 8, 8);
            Color[] colorData = new Color[8 * 8];
            
            // 원형 모양으로 채우기
            for (int i = 0; i < 8; i++)
            {
                for (int j = 0; j < 8; j++)
                {
                    float dx = i - 3.5f;
                    float dy = j - 3.5f;
                    float distance = (float)System.Math.Sqrt(dx * dx + dy * dy);
                    
                    if (distance <= 3.5f)
                        colorData[j * 8 + i] = Color.White;
                    else
                        colorData[j * 8 + i] = Color.Transparent;
                }
            }
            
            particleTexture.SetData(colorData);
            
            // 파티클 이미터 생성
            emitter = new ParticleEmitter(particleTexture, new Vector2(400, 300), 1000);
            emitter.SetEmissionRate(150f);
            emitter.SetSpeedRange(50f, 300f);
            emitter.SetAngleRange(-MathHelper.Pi, MathHelper.Pi);
            emitter.SetLifetimeRange(0.8f, 2.5f);
            emitter.SetGravity(new Vector2(0f, 150f));
        }
        
        protected override void Update(GameTime gameTime)
        {
            KeyboardState keyboard = Keyboard.GetState();
            
            if (keyboard.IsKeyDown(Keys.Escape))
                Exit();
            
            // 마우스 위치에 이미터 이동
            MouseState mouse = Mouse.GetState();
            emitter.SetPosition(new Vector2(mouse.X, mouse.Y));
            
            emitter.Update(gameTime);
            
            base.Update(gameTime);
        }
        
        protected override void Draw(GameTime gameTime)
        {
            GraphicsDevice.Clear(Color.Black);
            
            spriteBatch.Begin();
            emitter.Draw(spriteBatch);
            spriteBatch.End();
            
            base.Draw(gameTime);
        }
    }
}
```

위 게임 클래스에서는 8x8 크기의 흰색 원형 텍스처를 동적으로 생성한다. 각 픽셀을 중심에서의 거리로 계산하여 원형 모양을 만든다. 마우스가 움직일 때마다 파티클 이미터의 위치가 마우스를 따라가므로, 마우스를 움직이면 폭발하는 파티클 이펙트를 볼 수 있다.

## 15.3 파티클 풀링
파티클을 매번 생성하고 삭제하는 것은 메모리 할당과 가비지 컬렉션으로 인해 성능을 저하시킨다. 파티클 풀링은 미리 정해진 수의 파티클 객체들을 만들어두고, 필요할 때 재사용하는 기법이다.

다음은 파티클 풀을 구현한 개선된 이미터이다.

```csharp
using Microsoft.Xna.Framework;
using Microsoft.Xna.Framework.Graphics;
using System;
using System.Collections.Generic;

namespace ParticleSystemDemo
{
    public class PooledParticleEmitter
    {
        private List<Particle> particlePool;
        private List<Particle> activeParticles;
        private Texture2D particleTexture;
        
        private Vector2 emitterPosition;
        private float emissionRate;
        private float timeSinceLastEmission;
        
        private float minSpeed;
        private float maxSpeed;
        private float minAngle;
        private float maxAngle;
        private float minLifetime;
        private float maxLifetime;
        private Vector2 gravityVector;
        
        private Random random;
        private bool isActive;
        
        public PooledParticleEmitter(Texture2D texture, Vector2 position, int poolSize = 1000)
        {
            particleTexture = texture;
            emitterPosition = position;
            
            // 파티클 풀 초기화
            particlePool = new List<Particle>(poolSize);
            activeParticles = new List<Particle>(poolSize);
            
            // 미리 정해진 수의 파티클 객체 생성
            for (int i = 0; i < poolSize; i++)
            {
                var particle = new Particle(texture, Vector2.Zero, Vector2.Zero, 0f);
                particlePool.Add(particle);
            }
            
            emissionRate = 100f;
            minSpeed = 50f;
            maxSpeed = 200f;
            minAngle = -MathHelper.PiOver2;
            maxAngle = MathHelper.PiOver2;
            minLifetime = 0.5f;
            maxLifetime = 2f;
            gravityVector = new Vector2(0f, 200f);
            
            random = new Random();
            isActive = true;
            timeSinceLastEmission = 0f;
        }
        
        public void SetPosition(Vector2 newPosition)
        {
            emitterPosition = newPosition;
        }
        
        public void SetEmissionRate(float rate)
        {
            emissionRate = rate;
        }
        
        public void SetSpeedRange(float min, float max)
        {
            minSpeed = min;
            maxSpeed = max;
        }
        
        public void SetAngleRange(float min, float max)
        {
            minAngle = min;
            maxAngle = max;
        }
        
        public void SetLifetimeRange(float min, float max)
        {
            minLifetime = min;
            maxLifetime = max;
        }
        
        public void SetGravity(Vector2 gravity)
        {
            gravityVector = gravity;
        }
        
        public void Stop()
        {
            isActive = false;
        }
        
        public void Resume()
        {
            isActive = true;
        }
        
        public bool IsEmitting()
        {
            return activeParticles.Count > 0;
        }
        
        public void Update(GameTime gameTime)
        {
            float deltaTime = (float)gameTime.ElapsedGameTime.TotalSeconds;
            
            // 파티클 생성
            if (isActive)
            {
                timeSinceLastEmission += deltaTime;
                float particleInterval = 1f / emissionRate;
                
                while (timeSinceLastEmission >= particleInterval && particlePool.Count > 0)
                {
                    EmitParticle();
                    timeSinceLastEmission -= particleInterval;
                }
            }
            
            // 활성 파티클 업데이트
            for (int i = activeParticles.Count - 1; i >= 0; i--)
            {
                var particle = activeParticles[i];
                particle.Acceleration = gravityVector;
                particle.Update(gameTime);
                
                // 파티클이 비활성 상태가 되면 풀로 반환
                if (!particle.IsActive)
                {
                    activeParticles.RemoveAt(i);
                    particlePool.Add(particle);
                }
            }
        }
        
        public void Draw(SpriteBatch spriteBatch)
        {
            foreach (var particle in activeParticles)
            {
                particle.Draw(spriteBatch);
            }
        }
        
        private void EmitParticle()
        {
            // 풀에서 파티클 가져오기
            if (particlePool.Count == 0)
                return;
            
            var particle = particlePool[particlePool.Count - 1];
            particlePool.RemoveAt(particlePool.Count - 1);
            
            // 파티클 초기화
            float angle = minAngle + (float)random.NextDouble() * (maxAngle - minAngle);
            float speed = minSpeed + (float)random.NextDouble() * (maxSpeed - minSpeed);
            
            Vector2 velocity = new Vector2(
                (float)Math.Cos(angle) * speed,
                (float)Math.Sin(angle) * speed
            );
            
            float lifetime = minLifetime + (float)random.NextDouble() * (maxLifetime - minLifetime);
            
            particle.Position = emitterPosition;
            particle.Velocity = velocity;
            particle.Lifetime = lifetime;
            particle.MaxLifetime = lifetime;
            particle.IsActive = true;
            particle.Color = Color.White;
            particle.Scale = 1f;
            particle.Rotation = 0f;
            
            activeParticles.Add(particle);
        }
    }
}
```

파티클 풀링의 핵심은 세 가지이다. 첫째, 초기화 시점에 필요한 모든 파티클 객체를 미리 생성한다. 둘째, 파티클이 필요하면 풀에서 가져와 초기화한 후 사용한다. 셋째, 파티클이 생명을 마치면 풀로 반환한다. 이렇게 하면 게임이 실행되는 동안 메모리 할당이 발생하지 않아 가비지 컬렉션으로 인한 성능 저하를 방지할 수 있다.

다음은 여러 종류의 파티클 이펙트를 쉽게 만들 수 있도록 `ParticleEffectFactory` 클래스를 만들었다.

```csharp
using Microsoft.Xna.Framework;
using Microsoft.Xna.Framework.Graphics;

namespace ParticleSystemDemo
{
    public class ParticleEffectFactory
    {
        private Texture2D particleTexture;
        
        public ParticleEffectFactory(Texture2D texture)
        {
            particleTexture = texture;
        }
        
        // 폭발 효과
        public PooledParticleEmitter CreateExplosion(Vector2 position)
        {
            var emitter = new PooledParticleEmitter(particleTexture, position);
            emitter.SetEmissionRate(500f);
            emitter.SetSpeedRange(100f, 500f);
            emitter.SetAngleRange(-MathHelper.Pi, MathHelper.Pi);
            emitter.SetLifetimeRange(0.5f, 1.5f);
            emitter.SetGravity(new Vector2(0f, 300f));
            return emitter;
        }
        
        // 연기 효과
        public PooledParticleEmitter CreateSmoke(Vector2 position)
        {
            var emitter = new PooledParticleEmitter(particleTexture, position);
            emitter.SetEmissionRate(100f);
            emitter.SetSpeedRange(20f, 80f);
            emitter.SetAngleRange(-MathHelper.Pi / 4, MathHelper.Pi / 4);
            emitter.SetLifetimeRange(1f, 3f);
            emitter.SetGravity(new Vector2(0f, -50f));  // 위로 상승
            return emitter;
        }
        
        // 불 효과
        public PooledParticleEmitter CreateFire(Vector2 position)
        {
            var emitter = new PooledParticleEmitter(particleTexture, position);
            emitter.SetEmissionRate(200f);
            emitter.SetSpeedRange(50f, 200f);
            emitter.SetAngleRange(-MathHelper.PiOver2 - 0.3f, -MathHelper.PiOver2 + 0.3f);
            emitter.SetLifetimeRange(0.3f, 1f);
            emitter.SetGravity(new Vector2(0f, -200f));
            return emitter;
        }
        
        // 치유 효과 (위로 떠오르는 별)
        public PooledParticleEmitter CreateHeal(Vector2 position)
        {
            var emitter = new PooledParticleEmitter(particleTexture, position);
            emitter.SetEmissionRate(50f);
            emitter.SetSpeedRange(30f, 100f);
            emitter.SetAngleRange(-MathHelper.PiOver2, -MathHelper.PiOver2);
            emitter.SetLifetimeRange(1f, 2f);
            emitter.SetGravity(Vector2.Zero);  // 중력 없음
            return emitter;
        }
    }
}
```

`ParticleEffectFactory` 클래스는 다양한 게임 상황에 맞는 파티클 이펙트를 쉽게 생성한다. 폭발은 모든 방향으로 빠르게 흩어지고, 연기는 위로 천천히 올라가며, 불은 위쪽으로 분사되는 식으로 각 효과가 다르게 설정된다.

파티클 풀링이 얼마나 효율적인지 보여주기 위해 다음 예제에서는 풀링과 비풀링 방식을 비교한다.

```csharp
using Microsoft.Xna.Framework;
using Microsoft.Xna.Framework.Graphics;
using Microsoft.Xna.Framework.Input;

namespace ParticleSystemDemo
{
    public class ParticlePoolingDemo : Game
    {
        private GraphicsDeviceManager graphics;
        private SpriteBatch spriteBatch;
        private SpriteFont font;
        private Texture2D particleTexture;
        
        private PooledParticleEmitter pooledEmitter;
        private ParticleEmitter nonPooledEmitter;
        private ParticleEffectFactory effectFactory;
        
        private bool usePooling;
        private Vector2 mousePosition;
        
        public ParticlePoolingDemo()
        {
            graphics = new GraphicsDeviceManager(this);
            Content.RootDirectory = "Content";
            usePooling = true;
        }
        
        protected override void Initialize()
        {
            graphics.PreferredBackBufferWidth = 1000;
            graphics.PreferredBackBufferHeight = 700;
            graphics.ApplyChanges();
            
            base.Initialize();
        }
        
        protected override void LoadContent()
        {
            spriteBatch = new SpriteBatch(GraphicsDevice);
            
            // 간단한 폰트 로드 (내장 폰트가 없으면 화면에 텍스트를 출력하지 않아도 됨)
            try
            {
                font = Content.Load<SpriteFont>("Arial");
            }
            catch
            {
                // 폰트 로드 실패 시 무시
            }
            
            // 파티클 텍스처 생성
            particleTexture = CreateParticleTexture();
            
            // 풀링과 비풀링 이미터 생성
            pooledEmitter = new PooledParticleEmitter(particleTexture, Vector2.Zero, 500);
            pooledEmitter.SetEmissionRate(100f);
            
            nonPooledEmitter = new ParticleEmitter(particleTexture, Vector2.Zero, 500);
            nonPooledEmitter.SetEmissionRate(100f);
            
            effectFactory = new ParticleEffectFactory(particleTexture);
        }
        
        private Texture2D CreateParticleTexture()
        {
            Texture2D texture = new Texture2D(GraphicsDevice, 8, 8);
            Color[] colorData = new Color[8 * 8];
            
            for (int i = 0; i < 8; i++)
            {
                for (int j = 0; j < 8; j++)
                {
                    float dx = i - 3.5f;
                    float dy = j - 3.5f;
                    float distance = (float)System.Math.Sqrt(dx * dx + dy * dy);
                    
                    if (distance <= 3.5f)
                        colorData[j * 8 + i] = Color.White;
                    else
                        colorData[j * 8 + i] = Color.Transparent;
                }
            }
            
            texture.SetData(colorData);
            return texture;
        }
        
        protected override void Update(GameTime gameTime)
        {
            KeyboardState keyboard = Keyboard.GetState();
            MouseState mouse = Mouse.GetState();
            
            if (keyboard.IsKeyDown(Keys.Escape))
                Exit();
            
            // 스페이스바로 풀링 모드 전환
            if (keyboard.IsKeyDown(Keys.Space))
                usePooling = true;
            else
                usePooling = false;
            
            mousePosition = new Vector2(mouse.X, mouse.Y);
            
            if (usePooling)
            {
                pooledEmitter.SetPosition(mousePosition);
                pooledEmitter.Update(gameTime);
            }
            else
            {
                nonPooledEmitter.SetPosition(mousePosition);
                nonPooledEmitter.Update(gameTime);
            }
            
            base.Update(gameTime);
        }
        
        protected override void Draw(GameTime gameTime)
        {
            GraphicsDevice.Clear(Color.Black);
            
            spriteBatch.Begin();
            
            if (usePooling)
            {
                pooledEmitter.Draw(spriteBatch);
            }
            else
            {
                nonPooledEmitter.Draw(spriteBatch);
            }
            
            // 현재 모드 표시
            if (font != null)
            {
                string modeText = usePooling ? "풀링 방식 (스페이스 누르는 중)" : "비풀링 방식";
                spriteBatch.DrawString(font, modeText, new Vector2(10, 10), Color.White);
                spriteBatch.DrawString(font, "스페이스바를 눌러 모드 전환", new Vector2(10, 40), Color.Yellow);
            }
            
            spriteBatch.End();
            
            base.Draw(gameTime);
        }
    }
}
```

이 데모에서는 스페이스바를 누르고 있을 때 풀링 방식을 사용하고, 놓으면 비풀링 방식으로 전환된다. 같은 파티클 수를 생성하더라도 풀링 방식이 더 부드럽게 작동함을 느낄 수 있다.

## 과제
파티클 시스템의 개념과 구현을 학습했는지 확인하기 위해 다음 과제를 수행하기 바란다.

### 과제 1: 색상이 변하는 파티클 이펙트
기존 `ParticleEmitter` 클래스를 확장하여 다음 기능을 추가하시오.

1. 파티클의 색상이 시간에 따라 변하도록 구현하시오. 예를 들어, 처음에는 빨간색으로 시작하여 점점 노란색으로 변하고 마지막에 투명해지는 방식이다.
2. `Particle` 클래스에 `StartColor`와 `EndColor` 속성을 추가하고, `Update` 메서드에서 남은 생명 시간의 비율로 현재 색상을 계산하시오.
3. `ParticleEmitter`에서 색상 범위를 설정할 수 있는 `SetColorRange(Color startColor, Color endColor)` 메서드를 추가하시오.

### 과제 2: 스케일이 변하는 파티클
파티클이 시간에 따라 크기가 변하도록 구현하시오.

1. `Particle` 클래스에 `MinScale`과 `MaxScale` 속성을 추가하시오.
2. 파티클이 생성될 때 작은 크기로 시작하여 중간에 최대 크기가 되고 사라질 때 다시 작아지는 방식으로 구현하시오.
3. 이를 활용해 불 이펙트가 더 자연스럽게 보이도록 수정하시오.

### 과제 3: 여러 이미터를 동시에 관리하는 파티클 시스템
다음을 수행하시오.

1. `ParticleSystemManager` 클래스를 만들어 여러 개의 파티클 이미터를 관리하시오.
2. 키보드 입력으로 화면의 다양한 위치에서 다양한 종류의 이펙트를 발생시키시오.
   - E 키: 폭발 이펙트 생성
   - S 키: 연기 이펙트 생성
   - F 키: 불 이펙트 생성
   - H 키: 치유 이펙트 생성
3. `ParticleSystemManager`의 `Update` 메서드에서 모든 이미터를 업데이트하고 `Draw` 메서드에서 모두 그리시오.

### 과제 4: 파티클 풀 성능 분석
기존의 풀링과 비풀링 방식을 직접 비교하시오.

1. 두 방식 모두에서 초당 생성되는 파티클 수와 메모리 사용량을 화면에 표시하시오.
2. 100개부터 1000개까지 파티클 개수를 증가시키면서 성능 차이를 관찰하시오.
3. 풀링 방식이 비풀링 방식보다 얼마나 효율적인지 설명하시오.

### 과제 5: 원형 이미터 만들기
원형으로 파티클을 방출하는 이미터를 만들시오.

1. `CircularParticleEmitter` 클래스를 만들어 원 테두리를 따라 파티클을 방출하시오.
2. 원의 반지름, 원 위의 점 개수, 방사 속도를 설정할 수 있도록 하시오.
3. 이를 활용해 마법 원진이나 행성 궤도 같은 이펙트를 만들어보시오.
