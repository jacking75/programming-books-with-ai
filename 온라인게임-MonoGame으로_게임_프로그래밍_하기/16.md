# 16장. 카메라와 뷰포트

## 16.1 2D 카메라 구현
게임에서 카메라는 플레이어가 게임 월드의 어느 부분을 볼 수 있는지를 결정한다. 카메라가 없다면 화면은 항상 같은 장소만 보여주겠지만, 카메라를 구현하면 플레이어를 따라 이동하거나 전체 맵을 볼 수 있도록 뷰를 변경할 수 있다.

2D 카메라의 핵심 개념은 변환 행렬(transformation matrix)이다. MonoGame의 SpriteBatch는 `Begin` 메서드에서 행렬을 받아서 모든 스프라이트를 그 행렬로 변환한 후 화면에 그린다. 따라서 적절한 행렬을 전달하면 카메라 효과를 만들 수 있다.

다음은 2D 카메라의 기본 동작 원리를 시각화한 것이다.

```
┌──────────────────────────────────────┐
│      게임 월드 (무한히 큼)              │
│  ┌───────────────────────────────┐   │
│  │   카메라가 볼 수 있는 영역       │   │
│  │  (뷰포트와 같은 크기)            │   │
│  │  ┌─────────────────────────┐  │   │
│  │  │  실제 모니터에 그려지는     │ │   │
│  │  │  화면 (800x600)          │  │   │
│  │  └─────────────────────────┘  │   │
│  └───────────────────────────────┘   │
└──────────────────────────────────────┘

카메라가 이동하면 보이는 월드의 부분이 바뀐다.
```

카메라의 위치, 회전, 줌 레벨 등의 정보를 관리하는 `Camera2D` 클래스를 만들겠다.

```csharp
using Microsoft.Xna.Framework;
using Microsoft.Xna.Framework.Graphics;

namespace CameraDemo
{
    public class Camera2D
    {
        // 카메라의 위치
        public Vector2 Position { get; set; }
        
        // 카메라의 회전 각도 (라디안)
        public float Rotation { get; set; }
        
        // 카메라의 줌 레벨 (1.0f = 100%)
        public float Zoom { get; set; }
        
        // 카메라가 바라보는 뷰포트 크기
        private Vector2 viewportSize;
        
        // 카메라의 원점 (회전의 중심)
        public Vector2 Origin { get; set; }
        
        public Camera2D(float viewportWidth, float viewportHeight)
        {
            viewportSize = new Vector2(viewportWidth, viewportHeight);
            Position = Vector2.Zero;
            Rotation = 0f;
            Zoom = 1f;
            Origin = viewportSize / 2f;
        }
        
        public void SetViewportSize(float width, float height)
        {
            viewportSize = new Vector2(width, height);
            Origin = viewportSize / 2f;
        }
        
        // 카메라의 변환 행렬을 계산하는 메서드
        // 이 행렬을 SpriteBatch.Begin()에 전달하면 카메라 효과가 적용된다
        public Matrix GetViewMatrix()
        {
            // 1단계: 카메라 위치로 이동
            // 2단계: 원점 중심으로 줌 적용
            // 3단계: 원점 중심으로 회전
            
            return
                Matrix.CreateTranslation(new Vector3(-Position.X, -Position.Y, 0f)) *
                Matrix.CreateScale(Zoom, Zoom, 1f) *
                Matrix.CreateRotationZ(Rotation) *
                Matrix.CreateTranslation(new Vector3(Origin.X, Origin.Y, 0f));
        }
        
        // 월드 좌표를 화면 좌표로 변환
        public Vector2 WorldToScreen(Vector2 worldPosition)
        {
            Vector3 worldPos3D = new Vector3(worldPosition, 0f);
            Vector3 screenPos3D = Vector3.Transform(worldPos3D, GetViewMatrix());
            return new Vector2(screenPos3D.X, screenPos3D.Y);
        }
        
        // 화면 좌표를 월드 좌표로 변환
        public Vector2 ScreenToWorld(Vector2 screenPosition)
        {
            Vector3 screenPos3D = new Vector3(screenPosition, 0f);
            Vector3 worldPos3D = Vector3.Transform(screenPos3D, 
                Matrix.Invert(GetViewMatrix()));
            return new Vector2(worldPos3D.X, worldPos3D.Y);
        }
        
        // 특정 위치를 카메라의 중심으로 이동
        public void LookAt(Vector2 targetPosition)
        {
            Position = targetPosition - (viewportSize / (2f * Zoom));
        }
        
        // 카메라 움직임을 부드럽게 하기 위한 선형 보간
        public void SmoothLookAt(Vector2 targetPosition, float lerpFactor)
        {
            Vector2 targetCameraPosition = targetPosition - (viewportSize / (2f * Zoom));
            Position = Vector2.Lerp(Position, targetCameraPosition, lerpFactor);
        }
    }
}
```

`Camera2D` 클래스의 핵심은 `GetViewMatrix` 메서드이다. 이 메서드는 세 가지 변환을 순서대로 적용한다. 먼저 카메라 위치만큼 음수로 평행이동하여 카메라가 월드 원점을 바라보도록 한다. 다음으로 줌을 적용하고, 마지막으로 원점 중심으로 회전한 후 뷰포트의 중심으로 평행이동한다. 이 행렬을 SpriteBatch.Begin에 전달하면 모든 스프라이트가 카메라 변환을 받는다.

이제 카메라를 사용하는 간단한 게임을 만들어보겠다.

```csharp
using Microsoft.Xna.Framework;
using Microsoft.Xna.Framework.Graphics;
using Microsoft.Xna.Framework.Input;

namespace CameraDemo
{
    public class CameraGame : Game
    {
        private GraphicsDeviceManager graphics;
        private SpriteBatch spriteBatch;
        private Camera2D camera;
        
        // 플레이어 스프라이트
        private Texture2D playerTexture;
        private Vector2 playerPosition;
        private float playerSpeed = 150f;
        
        // 배경 그리드 및 테스트 객체들
        private Texture2D gridTexture;
        private Texture2D objectTexture;
        private Vector2[] objectPositions;
        
        // 카메라 모드
        private int cameraMode = 0;  // 0: 고정, 1: 플레이어 추적, 2: 매끄러운 추적
        
        public CameraGame()
        {
            graphics = new GraphicsDeviceManager(this);
            Content.RootDirectory = "Content";
        }
        
        protected override void Initialize()
        {
            graphics.PreferredBackBufferWidth = 800;
            graphics.PreferredBackBufferHeight = 600;
            graphics.ApplyChanges();
            
            // 카메라 초기화
            camera = new Camera2D(800, 600);
            
            // 플레이어 초기 위치
            playerPosition = new Vector2(400, 300);
            
            // 테스트 객체들 위치 설정
            objectPositions = new Vector2[]
            {
                new Vector2(100, 100),
                new Vector2(300, 150),
                new Vector2(500, 200),
                new Vector2(700, 250),
                new Vector2(200, 400),
                new Vector2(600, 450),
                new Vector2(1000, 300),
                new Vector2(1200, 500)
            };
            
            base.Initialize();
        }
        
        protected override void LoadContent()
        {
            spriteBatch = new SpriteBatch(GraphicsDevice);
            
            // 플레이어 텍스처 생성 (파란색 사각형)
            playerTexture = new Texture2D(GraphicsDevice, 32, 32);
            Color[] playerData = new Color[32 * 32];
            for (int i = 0; i < playerData.Length; i++)
                playerData[i] = Color.Blue;
            playerTexture.SetData(playerData);
            
            // 객체 텍스처 생성 (빨간색 사각형)
            objectTexture = new Texture2D(GraphicsDevice, 48, 48);
            Color[] objectData = new Color[48 * 48];
            for (int i = 0; i < objectData.Length; i++)
                objectData[i] = Color.Red;
            objectTexture.SetData(objectData);
            
            // 그리드 텍스처 생성 (흰색 선으로 된 격자)
            gridTexture = new Texture2D(GraphicsDevice, 1, 1);
            gridTexture.SetData(new[] { Color.White });
        }
        
        protected override void Update(GameTime gameTime)
        {
            KeyboardState keyboard = Keyboard.GetState();
            
            if (keyboard.IsKeyDown(Keys.Escape))
                Exit();
            
            float deltaTime = (float)gameTime.ElapsedGameTime.TotalSeconds;
            
            // 플레이어 움직임
            if (keyboard.IsKeyDown(Keys.W))
                playerPosition.Y -= playerSpeed * deltaTime;
            if (keyboard.IsKeyDown(Keys.S))
                playerPosition.Y += playerSpeed * deltaTime;
            if (keyboard.IsKeyDown(Keys.A))
                playerPosition.X -= playerSpeed * deltaTime;
            if (keyboard.IsKeyDown(Keys.D))
                playerPosition.X += playerSpeed * deltaTime;
            
            // 카메라 모드 전환
            if (keyboard.IsKeyDown(Keys.D1))
                cameraMode = 0;
            if (keyboard.IsKeyDown(Keys.D2))
                cameraMode = 1;
            if (keyboard.IsKeyDown(Keys.D3))
                cameraMode = 2;
            
            // 카메라 모드에 따른 처리
            switch (cameraMode)
            {
                case 0:  // 고정 카메라
                    camera.Position = Vector2.Zero;
                    break;
                case 1:  // 플레이어를 따라가는 카메라
                    camera.LookAt(playerPosition);
                    break;
                case 2:  // 부드럽게 따라가는 카메라
                    camera.SmoothLookAt(playerPosition, 0.1f);
                    break;
            }
            
            base.Update(gameTime);
        }
        
        protected override void Draw(GameTime gameTime)
        {
            GraphicsDevice.Clear(Color.Black);
            
            // 카메라 변환 행렬을 SpriteBatch에 전달
            spriteBatch.Begin(transformMatrix: camera.GetViewMatrix());
            
            // 그리드 그리기 (월드 공간에서)
            DrawGrid();
            
            // 객체들 그리기
            foreach (var objectPos in objectPositions)
            {
                spriteBatch.Draw(
                    objectTexture,
                    objectPos,
                    Color.Red
                );
            }
            
            // 플레이어 그리기
            spriteBatch.Draw(
                playerTexture,
                playerPosition,
                Color.Blue
            );
            
            spriteBatch.End();
            
            // UI 텍스트 (카메라 변환이 적용되지 않는 화면 좌표)
            spriteBatch.Begin();
            DrawUI();
            spriteBatch.End();
            
            base.Draw(gameTime);
        }
        
        private void DrawGrid()
        {
            // 카메라의 뷰포트 영역을 포함하도록 그리드 그리기
            Vector2 cameraTopLeft = camera.ScreenToWorld(Vector2.Zero);
            Vector2 cameraBottomRight = camera.ScreenToWorld(
                new Vector2(GraphicsDevice.Viewport.Width, 
                           GraphicsDevice.Viewport.Height));
            
            int gridSize = 50;
            
            // 수직선
            for (int x = (int)(cameraTopLeft.X / gridSize) * gridSize; 
                 x < cameraBottomRight.X; x += gridSize)
            {
                spriteBatch.Draw(
                    gridTexture,
                    new Rectangle(x, (int)cameraTopLeft.Y, 1, 
                                 (int)(cameraBottomRight.Y - cameraTopLeft.Y)),
                    Color.DarkGray * 0.5f
                );
            }
            
            // 수평선
            for (int y = (int)(cameraTopLeft.Y / gridSize) * gridSize; 
                 y < cameraBottomRight.Y; y += gridSize)
            {
                spriteBatch.Draw(
                    gridTexture,
                    new Rectangle((int)cameraTopLeft.X, y,
                                 (int)(cameraBottomRight.X - cameraTopLeft.X), 1),
                    Color.DarkGray * 0.5f
                );
            }
        }
        
        private void DrawUI()
        {
            // 실제 게임에서는 로드한 폰트를 사용하지만, 여기서는 텍스트 설명만 제시
            string modeText = cameraMode switch
            {
                0 => "모드: 고정 카메라 (1키 누르기)",
                1 => "모드: 플레이어 추적 (2키 누르기)",
                2 => "모드: 매끄러운 추적 (3키 누르기)",
                _ => ""
            };
            
            // 폰트가 있다면 여기에 DrawString 호출
            // spriteBatch.DrawString(font, modeText, new Vector2(10, 10), Color.White);
        }
    }
}
```

이 코드의 목적은 카메라 시스템이 게임에서 어떻게 동작하는지 보여주는 것이다. `Draw` 메서드에서 중요한 부분은 `spriteBatch.Begin(transformMatrix: camera.GetViewMatrix())`이다. 이렇게 하면 그 이후에 그려지는 모든 스프라이트가 카메라 변환을 받는다. 그리드를 그릴 때 `camera.ScreenToWorld` 메서드를 사용하여 카메라가 볼 수 있는 영역만 그리고 있다. 이렇게 하면 불필요한 부분을 그리지 않아 성능이 향상된다.

## 16.2 줌과 팬
줌은 카메라가 얼마나 크게 또는 작게 월드를 보는지를 조절한다. 줌 값이 2.0이면 모든 것이 2배 크게 보이고, 0.5면 반으로 작게 보인다. 팬(pan)은 부드럽게 카메라를 이동시키는 기술이다.

다음은 줌과 팬 기능을 추가한 확장된 카메라 클래스이다.

```csharp
using Microsoft.Xna.Framework;
using Microsoft.Xna.Framework.Graphics;

namespace CameraDemo
{
    public class AdvancedCamera2D
    {
        public Vector2 Position { get; set; }
        public float Rotation { get; set; }
        public float Zoom { get; private set; }
        
        private Vector2 viewportSize;
        public Vector2 Origin { get; set; }
        
        // 줌 관련 속성
        private float minZoom = 0.5f;
        private float maxZoom = 4f;
        private float targetZoom;
        private float zoomSpeed = 2f;
        
        // 팬 관련 속성
        private Vector2 targetPosition;
        private bool isPanning;
        private float panSpeed = 5f;
        
        public AdvancedCamera2D(float viewportWidth, float viewportHeight)
        {
            viewportSize = new Vector2(viewportWidth, viewportHeight);
            Position = Vector2.Zero;
            Rotation = 0f;
            Zoom = 1f;
            targetZoom = 1f;
            Origin = viewportSize / 2f;
            targetPosition = Position;
            isPanning = false;
        }
        
        public void SetZoomLimits(float min, float max)
        {
            minZoom = min;
            maxZoom = max;
        }
        
        // 줌 아웃 (더 작게 보기)
        public void ZoomOut()
        {
            targetZoom = MathHelper.Clamp(targetZoom * 1.2f, minZoom, maxZoom);
        }
        
        // 줌 인 (더 크게 보기)
        public void ZoomIn()
        {
            targetZoom = MathHelper.Clamp(targetZoom / 1.2f, minZoom, maxZoom);
        }
        
        // 특정 줌 값으로 설정
        public void SetZoom(float newZoom)
        {
            targetZoom = MathHelper.Clamp(newZoom, minZoom, maxZoom);
        }
        
        // 목표 위치로 팬하기
        public void PanTo(Vector2 target, float duration)
        {
            targetPosition = target;
            isPanning = true;
            panSpeed = Vector2.Distance(Position, target) / duration;
        }
        
        // 줌과 팬을 함께 적용
        public void FocusOn(Vector2 target, float zoom, float duration)
        {
            SetZoom(zoom);
            PanTo(target - (viewportSize / (2f * zoom)), duration);
        }
        
        // 매 프레임 호출하여 줌과 팬 업데이트
        public void Update(GameTime gameTime)
        {
            float deltaTime = (float)gameTime.ElapsedGameTime.TotalSeconds;
            
            // 줌 부드럽게 변환
            if (Zoom != targetZoom)
            {
                Zoom = MathHelper.Lerp(Zoom, targetZoom, deltaTime * zoomSpeed);
                
                if (MathHelper.Distance(Zoom, targetZoom) < 0.01f)
                    Zoom = targetZoom;
            }
            
            // 팬 처리
            if (isPanning)
            {
                Vector2 direction = targetPosition - Position;
                float distance = direction.Length();
                
                if (distance > 1f)
                {
                    direction.Normalize();
                    Position += direction * panSpeed * deltaTime;
                }
                else
                {
                    Position = targetPosition;
                    isPanning = false;
                }
            }
        }
        
        public Matrix GetViewMatrix()
        {
            return
                Matrix.CreateTranslation(new Vector3(-Position.X, -Position.Y, 0f)) *
                Matrix.CreateScale(Zoom, Zoom, 1f) *
                Matrix.CreateRotationZ(Rotation) *
                Matrix.CreateTranslation(new Vector3(Origin.X, Origin.Y, 0f));
        }
        
        public Vector2 WorldToScreen(Vector2 worldPosition)
        {
            Vector3 worldPos3D = new Vector3(worldPosition, 0f);
            Vector3 screenPos3D = Vector3.Transform(worldPos3D, GetViewMatrix());
            return new Vector2(screenPos3D.X, screenPos3D.Y);
        }
        
        public Vector2 ScreenToWorld(Vector2 screenPosition)
        {
            Vector3 screenPos3D = new Vector3(screenPosition, 0f);
            Vector3 worldPos3D = Vector3.Transform(screenPos3D, 
                Matrix.Invert(GetViewMatrix()));
            return new Vector2(worldPos3D.X, worldPos3D.Y);
        }
    }
}
```

`AdvancedCamera2D` 클래스에서 중요한 부분은 부드러운 줌 처리이다. `targetZoom`에 목표값을 설정하고 `Update` 메서드에서 현재 줌값을 목표값으로 선형 보간한다. 이렇게 하면 줌이 끊기지 않고 자연스럽게 변한다. 마찬가지로 팬도 직선으로 부드럽게 이동한다.

다음은 줌과 팬 기능을 테스트하는 예제이다.

```csharp
using Microsoft.Xna.Framework;
using Microsoft.Xna.Framework.Graphics;
using Microsoft.Xna.Framework.Input;

namespace CameraDemo
{
    public class ZoomPanGame : Game
    {
        private GraphicsDeviceManager graphics;
        private SpriteBatch spriteBatch;
        private AdvancedCamera2D camera;
        
        private Texture2D tileTexture;
        private Vector2 selectedObjectPosition;
        private bool hasSelectedObject = false;
        
        public ZoomPanGame()
        {
            graphics = new GraphicsDeviceManager(this);
            Content.RootDirectory = "Content";
        }
        
        protected override void Initialize()
        {
            graphics.PreferredBackBufferWidth = 800;
            graphics.PreferredBackBufferHeight = 600;
            graphics.ApplyChanges();
            
            camera = new AdvancedCamera2D(800, 600);
            camera.SetZoomLimits(0.5f, 3f);
            
            base.Initialize();
        }
        
        protected override void LoadContent()
        {
            spriteBatch = new SpriteBatch(GraphicsDevice);
            
            tileTexture = new Texture2D(GraphicsDevice, 64, 64);
            Color[] tileData = new Color[64 * 64];
            for (int i = 0; i < tileData.Length; i++)
                tileData[i] = Color.Green;
            tileTexture.SetData(tileData);
        }
        
        protected override void Update(GameTime gameTime)
        {
            KeyboardState keyboard = Keyboard.GetState();
            MouseState mouse = Mouse.GetState();
            
            if (keyboard.IsKeyDown(Keys.Escape))
                Exit();
            
            // 줌 조절
            if (keyboard.IsKeyDown(Keys.Up))
                camera.ZoomIn();
            if (keyboard.IsKeyDown(Keys.Down))
                camera.ZoomOut();
            
            // 마우스 클릭으로 팬
            if (mouse.LeftButton == ButtonState.Pressed)
            {
                Vector2 mouseWorldPos = camera.ScreenToWorld(
                    new Vector2(mouse.X, mouse.Y));
                camera.PanTo(mouseWorldPos - new Vector2(400, 300), 1f);
                
                hasSelectedObject = true;
                selectedObjectPosition = mouseWorldPos;
            }
            
            camera.Update(gameTime);
            
            base.Update(gameTime);
        }
        
        protected override void Draw(GameTime gameTime)
        {
            GraphicsDevice.Clear(Color.Black);
            
            spriteBatch.Begin(transformMatrix: camera.GetViewMatrix());
            
            // 타일맵 그리기
            for (int x = 0; x < 20; x++)
            {
                for (int y = 0; y < 15; y++)
                {
                    spriteBatch.Draw(
                        tileTexture,
                        new Vector2(x * 64, y * 64),
                        Color.Green
                    );
                }
            }
            
            // 선택된 객체 그리기
            if (hasSelectedObject)
            {
                spriteBatch.Draw(
                    tileTexture,
                    selectedObjectPosition,
                    Color.Yellow
                );
            }
            
            spriteBatch.End();
            
            base.Draw(gameTime);
        }
    }
}
```

## 16.3 스크롤링 배경
스크롤링 배경은 배경이 카메라의 움직임에 따라 다른 속도로 이동하는 패럴랙스 효과를 만든다. 이것은 게임에 깊이감을 더해준다.

다음은 패럴랙스 스크롤링을 구현한 배경 클래스이다.

```csharp
using Microsoft.Xna.Framework;
using Microsoft.Xna.Framework.Graphics;

namespace CameraDemo
{
    public class ParallaxBackground
    {
        private Texture2D[] layers;
        private float[] parallaxFactors;
        private Vector2 cameraPosition;
        private Vector2 originalCameraPosition;
        
        // 배경 크기
        private float layerWidth;
        private float layerHeight;
        
        public ParallaxBackground(int layerCount)
        {
            layers = new Texture2D[layerCount];
            parallaxFactors = new float[layerCount];
            cameraPosition = Vector2.Zero;
            originalCameraPosition = Vector2.Zero;
        }
        
        // 레이어 추가 (인덱스, 텍스처, 패럴랙스 계수)
        public void AddLayer(int index, Texture2D texture, float parallaxFactor)
        {
            if (index >= 0 && index < layers.Length)
            {
                layers[index] = texture;
                parallaxFactors[index] = parallaxFactor;
                
                if (layerWidth == 0)
                {
                    layerWidth = texture.Width;
                    layerHeight = texture.Height;
                }
            }
        }
        
        // 카메라 위치 업데이트
        public void UpdateCameraPosition(Vector2 newCameraPosition)
        {
            cameraPosition = newCameraPosition;
        }
        
        // 배경 그리기
        public void Draw(SpriteBatch spriteBatch)
        {
            for (int layerIndex = 0; layerIndex < layers.Length; layerIndex++)
            {
                if (layers[layerIndex] == null)
                    continue;
                
                // 패럴랙스 효과를 위해 오프셋 계산
                float parallaxOffset = cameraPosition.X * parallaxFactors[layerIndex];
                
                // 배경이 반복되도록 하기 위해 모듈로 연산 사용
                float wrappedOffset = parallaxOffset % layerWidth;
                if (wrappedOffset < 0)
                    wrappedOffset += layerWidth;
                
                // 배경 타일 그리기 (반복되도록 2번 그리기)
                for (int i = -1; i <= 1; i++)
                {
                    spriteBatch.Draw(
                        layers[layerIndex],
                        new Vector2(i * layerWidth - wrappedOffset, 0),
                        Color.White
                    );
                }
            }
        }
    }
}
```

`ParallaxBackground` 클래스는 여러 개의 배경 레이어를 관리한다. 각 레이어는 `parallaxFactor`라는 값을 가지는데, 이것이 배경이 얼마나 느리게 움직일지를 결정한다. 예를 들어, 먼 산은 0.3의 계수를 가질 수 있고, 가까운 나무는 0.8의 계수를 가질 수 있다. 이렇게 하면 게임이 3차원처럼 보인다.

이제 패럴랙스 배경을 사용하는 게임을 만들어보겠다.

```csharp
using Microsoft.Xna.Framework;
using Microsoft.Xna.Framework.Graphics;
using Microsoft.Xna.Framework.Input;

namespace CameraDemo
{
    public class ParallaxGame : Game
    {
        private GraphicsDeviceManager graphics;
        private SpriteBatch spriteBatch;
        private AdvancedCamera2D camera;
        private ParallaxBackground background;
        
        private Texture2D playerTexture;
        private Vector2 playerPosition;
        private float playerSpeed = 200f;
        
        // 배경 레이어를 위한 텍스처들
        private Texture2D farMountainTexture;
        private Texture2D nearMountainTexture;
        private Texture2D treeTexture;
        
        public ParallaxGame()
        {
            graphics = new GraphicsDeviceManager(this);
            Content.RootDirectory = "Content";
        }
        
        protected override void Initialize()
        {
            graphics.PreferredBackBufferWidth = 800;
            graphics.PreferredBackBufferHeight = 600;
            graphics.ApplyChanges();
            
            camera = new AdvancedCamera2D(800, 600);
            background = new ParallaxBackground(3);
            
            playerPosition = new Vector2(400, 300);
            
            base.Initialize();
        }
        
        protected override void LoadContent()
        {
            spriteBatch = new SpriteBatch(GraphicsDevice);
            
            // 플레이어 텍스처
            playerTexture = new Texture2D(GraphicsDevice, 32, 32);
            Color[] playerData = new Color[32 * 32];
            for (int i = 0; i < playerData.Length; i++)
                playerData[i] = Color.Blue;
            playerTexture.SetData(playerData);
            
            // 배경 텍스처들 (간단한 색상 텍스처로 대체)
            farMountainTexture = CreateColoredTexture(800, 600, Color.DarkBlue);
            nearMountainTexture = CreateColoredTexture(800, 600, Color.Blue);
            treeTexture = CreateColoredTexture(800, 600, Color.Green);
            
            // 배경 레이어 추가
            background.AddLayer(0, farMountainTexture, 0.2f);  // 가장 먼 산
            background.AddLayer(1, nearMountainTexture, 0.5f); // 중간 산
            background.AddLayer(2, treeTexture, 0.8f);         // 가까운 나무
        }
        
        private Texture2D CreateColoredTexture(int width, int height, Color color)
        {
            Texture2D texture = new Texture2D(GraphicsDevice, width, height);
            Color[] data = new Color[width * height];
            for (int i = 0; i < data.Length; i++)
                data[i] = color;
            texture.SetData(data);
            return texture;
        }
        
        protected override void Update(GameTime gameTime)
        {
            KeyboardState keyboard = Keyboard.GetState();
            
            if (keyboard.IsKeyDown(Keys.Escape))
                Exit();
            
            float deltaTime = (float)gameTime.ElapsedGameTime.TotalSeconds;
            
            // 플레이어 움직임
            if (keyboard.IsKeyDown(Keys.A))
                playerPosition.X -= playerSpeed * deltaTime;
            if (keyboard.IsKeyDown(Keys.D))
                playerPosition.X += playerSpeed * deltaTime;
            
            // 카메라가 플레이어를 부드럽게 따라가기
            camera.SmoothLookAt(playerPosition, 0.1f);
            
            // 배경 업데이트
            background.UpdateCameraPosition(camera.Position);
            
            camera.Update(gameTime);
            
            base.Update(gameTime);
        }
        
        protected override void Draw(GameTime gameTime)
        {
            GraphicsDevice.Clear(Color.Black);
            
            // 배경 그리기 (카메라 변환 없이)
            spriteBatch.Begin();
            background.Draw(spriteBatch);
            spriteBatch.End();
            
            // 게임 오브젝트 그리기 (카메라 변환 포함)
            spriteBatch.Begin(transformMatrix: camera.GetViewMatrix());
            
            spriteBatch.Draw(
                playerTexture,
                playerPosition,
                Color.Blue
            );
            
            spriteBatch.End();
            
            base.Draw(gameTime);
        }
    }
}
```

이 코드의 핵심은 배경과 게임 오브젝트를 다르게 처리한다는 것이다. 배경은 `spriteBatch.Begin()`으로 화면 좌표에 그려지고, 게임 오브젝트는 `spriteBatch.Begin(transformMatrix: camera.GetViewMatrix())`로 월드 좌표에 그려진다. 배경은 `UpdateCameraPosition` 메서드로 카메라의 움직임을 추적하지만, 패럴랙스 계수에 의해 더 느리게 움직인다.

## 과제
카메라와 뷰포트 시스템을 이해했는지 확인하기 위해 다음 과제들을 수행하기 바란다.

### 과제 1: 경계 제한이 있는 카메라
기존 `Camera2D` 클래스를 수정하여 다음 기능을 추가하시오.

1. 카메라가 게임 월드의 경계를 넘어가지 않도록 제한하는 메서드를 만들어라. `SetBounds(Vector2 minPosition, Vector2 maxPosition)` 메서드를 추가하시오.
2. `Update` 메서드 또는 `LookAt` 메서드를 수정하여 설정된 경계를 벗어나지 않도록 카메라 위치를 조정하시오.
3. 플레이어가 움직일 때 카메라가 맵의 경계를 넘지 않도록 테스트하시오.

### 과제 2: 화면 흔들림 효과
카메라에 진동 효과를 추가하시오.

1. `Camera2D` 클래스에 `Shake(float intensity, float duration)` 메서드를 추가하시오.
2. 흔들림은 카메라 위치에 무작위 오프셋을 더하는 방식으로 구현하시오.
3. 시간이 지나면서 강도가 감소하여 자연스럽게 멈추도록 하시오.
4. 게임에서 특정 키를 누르면 화면이 흔들리도록 테스트하시오.

### 과제 3: 미니맵 구현
카메라 시스템을 활용하여 미니맵을 만들시오.

1. 게임 화면의 우측 상단에 미니맵을 그리시오.
2. 미니맵은 전체 게임 월드를 축소한 버전이어야 한다.
3. 미니맵 위에 플레이어의 현재 위치를 표시하시오.
4. 미니맵 위에 카메라가 현재 보고 있는 영역을 사각형으로 표시하시오.

### 과제 4: 다중 레이어 패럴랙스 배경 개선
현재의 `ParallaxBackground`를 개선하시오.

1. 수직 스크롤도 지원하도록 수정하시오. (현재는 수평만 지원)
2. 여러 배경 레이어가 다른 높이에 있을 수 있도록 하시오.
3. 각 레이어의 수평 및 수직 패럴랙스 계수를 독립적으로 설정할 수 있게 하시오.

### 과제 5: 카메라 포인트 애니메이션
여러 카메라 포인트를 순서대로 방문하는 기능을 구현하시오.

2. 게임이 시작될 때 여러 포인트를 자동으로 방문하는 카메라 투어를 만들시오.
3. 각 포인트 사이의 전환이 부드럽게 이루어지도록 선형 보간을 사용하시오.
4. 튜토리얼이나 게임 오버 화면에서 이 기능을 사용하시오.

### 과제 6: 정사영 모드와 원근 모드 전환

1. 표준 2D 정사영 카메라 (현재 구현)

4. 게임플레이 중에 모드를 바꾸었을 때 부드럽게 전환되도록 하시오.
