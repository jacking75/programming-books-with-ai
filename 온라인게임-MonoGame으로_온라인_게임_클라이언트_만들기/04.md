# 4장: 연결 관리와 메시지 처리

## 4.1 서버 연결 및 연결 해제
온라인 게임 클라이언트에서 가장 중요한 작업 중 하나는 서버와의 안정적인 연결을 관리하는 것이다. 이번 섹션에서는 TCP 소켓을 사용하여 서버에 연결하고, 연결을 종료하는 방법을 배운다.

### TCP 연결의 생명주기
TCP 연결은 다음과 같은 상태를 거친다:

```
[연결 대기] → [연결 시도] → [연결 성공] → [통신 중] → [연결 해제]
```

클라이언트는 먼저 특정 IP 주소와 포트 번호에 연결을 시도한다. 서버가 요청을 받으면 연결이 확립되고, 이후 양방향 통신이 가능해진다. 게임을 종료하거나 서버와의 연결이 끊어질 때는 깔끔하게 연결을 해제해야 한다.

### NetworkClient 클래스 구현
게임 클라이언트에서 네트워크 통신을 관리할 `NetworkClient` 클래스를 만들어 보자. 이 클래스는 서버 연결, 패킷 송수신, 연결 해제 등 모든 네트워크 작업을 담당한다.

```csharp
using System;
using System.Net.Sockets;
using System.Collections.Generic;
using System.Threading;
using System.Threading.Tasks;

/// <summary>
/// 게임 클라이언트의 네트워크 통신을 담당하는 클래스다.
/// 서버와의 연결, 패킷 송수신, 메시지 대기열 관리 등을 수행한다.
/// </summary>
public class NetworkClient
{
    private TcpClient _tcpClient;
    private NetworkStream _stream;
    private bool _isConnected;
    private CancellationTokenSource _cancellationTokenSource;
    
    // 수신한 패킷을 저장할 큐
    private readonly Queue<byte[]> _receivedPackets = new Queue<byte[]>();
    private readonly object _lockObject = new object();
    
    public bool IsConnected => _isConnected;
    
    /// <summary>
    /// 서버에 연결한다.
    /// </summary>
    /// <param name="serverIp">연결할 서버의 IP 주소</param>
    /// <param name="port">연결할 포트 번호</param>
    /// <returns>연결 성공 여부를 나타내는 Task</returns>
    public async Task<bool> ConnectAsync(string serverIp, int port)
    {
        try
        {
            _tcpClient = new TcpClient();
            
            // 5초 타임아웃으로 서버 연결 시도
            var connectTask = _tcpClient.ConnectAsync(serverIp, port);
            var timeoutTask = Task.Delay(5000);
            
            var completedTask = await Task.WhenAny(connectTask, timeoutTask);
            
            // 타임아웃 발생
            if (completedTask == timeoutTask)
            {
                Console.WriteLine("[오류] 서버 연결 타임아웃: 5초 이내 응답이 없습니다");
                _tcpClient?.Close();
                return false;
            }
            
            _stream = _tcpClient.GetStream();
            _isConnected = true;
            
            // 취소 토큰 초기화
            _cancellationTokenSource = new CancellationTokenSource();
            
            Console.WriteLine($"[성공] 서버에 연결되었습니다: {serverIp}:{port}");
            
            return true;
        }
        catch (SocketException ex)
        {
            Console.WriteLine($"[오류] 소켓 연결 실패: {ex.Message}");
            _isConnected = false;
            return false;
        }
        catch (Exception ex)
        {
            Console.WriteLine($"[오류] 연결 중 예외 발생: {ex.Message}");
            _isConnected = false;
            return false;
        }
    }
    
    /// <summary>
    /// 패킷을 서버로 전송한다.
    /// </summary>
    /// <param name="packetData">전송할 패킷 데이터 (바이트 배열)</param>
    /// <returns>전송 성공 여부를 나타내는 Task</returns>
    public async Task<bool> SendPacketAsync(byte[] packetData)
    {
        if (!_isConnected || _stream == null)
        {
            Console.WriteLine("[오류] 서버에 연결되어 있지 않습니다");
            return false;
        }
        
        try
        {
            // 패킷 길이를 4바이트로 먼저 전송 (헤더)
            byte[] lengthHeader = BitConverter.GetBytes(packetData.Length);
            
            // 시스템이 빅 엔디안이면 리틀 엔디안으로 변환
            if (BitConverter.IsLittleEndian)
            {
                Array.Reverse(lengthHeader);
            }
            
            // 헤더 + 패킷 데이터를 함께 전송
            await _stream.WriteAsync(lengthHeader, 0, 4);
            await _stream.WriteAsync(packetData, 0, packetData.Length);
            await _stream.FlushAsync();
            
            return true;
        }
        catch (IOException ex)
        {
            Console.WriteLine($"[오류] 패킷 전송 실패: {ex.Message}");
            _isConnected = false;
            return false;
        }
        catch (Exception ex)
        {
            Console.WriteLine($"[오류] 전송 중 예외 발생: {ex.Message}");
            return false;
        }
    }
    
    /// <summary>
    /// 수신한 패킷이 있는지 확인하고 반환한다.
    /// </summary>
    /// <returns>수신한 패킷 (없으면 null)</returns>
    public byte[] GetReceivedPacket()
    {
        lock (_lockObject)
        {
            if (_receivedPackets.Count > 0)
            {
                return _receivedPackets.Dequeue();
            }
        }
        
        return null;
    }
    
    /// <summary>
    /// 서버와의 연결을 종료한다.
    /// </summary>
    public void Disconnect()
    {
        _isConnected = false;
        
        try
        {
            // 비동기 수신 작업 취소
            _cancellationTokenSource?.Cancel();
            
            // 스트림 정리
            _stream?.Close();
            _stream?.Dispose();
            
            // TCP 클라이언트 정리
            _tcpClient?.Close();
            _tcpClient?.Dispose();
            
            Console.WriteLine("[정보] 서버 연결이 종료되었습니다");
        }
        catch (Exception ex)
        {
            Console.WriteLine($"[오류] 연결 해제 중 예외 발생: {ex.Message}");
        }
    }
}
```

이 코드의 중요한 부분들을 설명한다:

**ConnectAsync 메서드**: 서버에 비동기로 연결한다. 5초의 타임아웃을 설정하여 서버가 응답하지 않을 때 무한 대기하지 않도록 한다. `Task.WhenAny`를 사용하여 연결 작업과 타임아웃 중 먼저 완료되는 것을 기준으로 처리한다.

**SendPacketAsync 메서드**: 패킷을 전송하기 전에 먼저 패킷의 길이를 4바이트 헤더로 보낸다. 이를 통해 수신자는 받아야 할 데이터의 크기를 미리 알 수 있다. 엔디안(endian) 처리를 통해 다양한 시스템에서 호환되도록 한다.

**GetReceivedPacket 메서드**: 스레드 안전성을 위해 `lock` 키워드로 임계 영역을 보호한다. 여러 스레드가 동시에 큐에 접근할 때 데이터 손상을 방지한다.

**Disconnect 메서드**: 모든 리소스를 체계적으로 정리한다. 네트워크 리소스를 제대로 정리하지 않으면 메모리 누수가 발생할 수 있다.

## 4.2 비동기 패킷 수신
서버로부터 데이터를 수신하는 작업은 계속 진행되어야 한다. 게임이 실행되는 동안 끊임없이 서버의 메시지를 받아야 하기 때문이다. 이를 위해 비동기 수신 메커니즘을 구현해야 한다.

### 비동기 수신의 필요성
만약 수신 작업이 동기(synchronous)적으로 진행된다면, 데이터가 올 때까지 게임 스레드가 대기하게 되어 게임이 멈춘다. 따라서 별도의 백그라운드 스레드에서 계속 수신하면서, 수신한 데이터를 큐에 저장하는 방식을 사용한다.

```
[메인 게임 루프]                  [백그라운드 수신 스레드]
  ↓                               ↓
업데이트 진행                    계속 수신 시도
  ↓                               ↓
렌더링                          데이터 수신 → 큐에 저장
  ↓                               ↓
큐에서 패킷 확인  ←─── 안전하게 접근 ←── 큐에서 꺼내기
```

이제 `NetworkClient` 클래스에 백그라운드 수신 기능을 추가한다.

```csharp
/// <summary>
/// 백그라운드에서 계속 패킷을 수신한다.
/// 별도의 비동기 작업으로 동작하며, 게임 루프를 블로킹하지 않는다.
/// </summary>
public async Task StartReceivingAsync()
{
    if (!_isConnected || _stream == null)
    {
        Console.WriteLine("[오류] 연결되어 있지 않아 수신을 시작할 수 없습니다");
        return;
    }
    
    try
    {
        // 5KB 크기의 버퍼 - 패킷 헤더와 데이터를 받을 충분한 크기
        byte[] buffer = new byte[5120];
        
        while (_isConnected && !_cancellationTokenSource.Token.IsCancellationRequested)
        {
            try
            {
                // 먼저 4바이트 헤더 (패킷 길이)를 읽는다
                int bytesRead = await _stream.ReadAsync(buffer, 0, 4, _cancellationTokenSource.Token);
                
                if (bytesRead == 0)
                {
                    // 서버가 연결을 끊음
                    Console.WriteLine("[정보] 서버가 연결을 종료했습니다");
                    _isConnected = false;
                    break;
                }
                
                // 헤더에서 패킷 길이를 읽는다
                int packetLength = BitConverter.ToInt32(buffer, 0);
                
                // 엔디안 처리 (리틀 엔디안으로 변환)
                if (BitConverter.IsLittleEndian)
                {
                    Array.Reverse(buffer, 0, 4);
                }
                
                // 패킷 길이가 유효한 범위인지 확인 (0 ~ 5000바이트)
                if (packetLength < 0 || packetLength > 5000)
                {
                    Console.WriteLine($"[경고] 유효하지 않은 패킷 길이: {packetLength}");
                    continue;
                }
                
                // 패킷 데이터를 읽는다
                int dataRead = 0;
                while (dataRead < packetLength)
                {
                    int remaining = packetLength - dataRead;
                    int read = await _stream.ReadAsync(
                        buffer, 
                        dataRead, 
                        remaining, 
                        _cancellationTokenSource.Token
                    );
                    
                    if (read == 0)
                    {
                        Console.WriteLine("[정보] 패킷 수신 중 연결이 끊어졌습니다");
                        _isConnected = false;
                        break;
                    }
                    
                    dataRead += read;
                }
                
                if (!_isConnected)
                {
                    break;
                }
                
                // 수신한 패킷을 큐에 추가
                byte[] packetData = new byte[packetLength];
                Array.Copy(buffer, 0, packetData, 0, packetLength);
                
                lock (_lockObject)
                {
                    _receivedPackets.Enqueue(packetData);
                }
            }
            catch (OperationCanceledException)
            {
                // 취소 요청으로 인한 정상적인 종료
                Console.WriteLine("[정보] 수신 작업이 취소되었습니다");
                break;
            }
            catch (IOException ex)
            {
                Console.WriteLine($"[오류] 수신 중 IO 오류 발생: {ex.Message}");
                _isConnected = false;
                break;
            }
        }
    }
    catch (Exception ex)
    {
        Console.WriteLine($"[오류] 수신 중 예외 발생: {ex.Message}");
        _isConnected = false;
    }
}
```

이 메서드의 동작 방식을 자세히 설명한다:

**헤더 읽기**: 먼저 4바이트 헤더를 읽어 패킷의 크기를 파악한다. 이렇게 하면 정확히 몇 바이트를 더 읽어야 하는지 알 수 있다.

**패킷 길이 검증**: 수신한 패킷 길이가 합리적인 범위 내인지 확인한다. 만약 범위를 벗어나면 프로토콜 오류이므로 무시하고 계속 진행한다.

**데이터 수신**: 패킷의 전체 데이터를 수신할 때까지 반복해서 읽는다. 네트워크에서는 한 번에 모든 데이터가 오지 않을 수 있기 때문이다.

**취소 토큰 처리**: 게임이 종료될 때 깔끔하게 수신을 중단하기 위해 취소 토큰을 사용한다.

**스레드 안전성**: `lock` 키워드로 큐 접근을 보호하여 여러 스레드가 동시에 접근할 때 발생하는 데이터 손상을 방지한다.

## 4.3 패킷 큐(Queue) 기반 메시지 처리
이제 수신한 패킷을 게임 로직에서 활용하는 방법을 배운다. 게임의 메인 루프에서 매 프레임마다 큐에서 패킷을 꺼내 처리한다.

### 메시지 처리 패턴

```
[수신 스레드]                    [메인 게임 루프]
    ↓                                ↓
  패킷 수신                        Update 호출
    ↓                                ↓
  큐에 저장 ←─────────── 락으로 보호 ←─ GetReceivedPacket()
    ↓                                ↓
  계속 수신                      패킷 처리 (핸들러 호출)
                                     ↓
                                  렌더링
```

먼저 패킷을 파싱하고 처리할 `PacketHandler` 클래스를 만든다:

```csharp
using System;

/// <summary>
/// 수신한 패킷을 파싱하고 처리하는 클래스다.
/// 패킷 타입을 구분하여 적절한 핸들러를 호출한다.
/// </summary>
public class PacketHandler
{
    /// <summary>
    /// 패킷 타입을 정의하는 상수들
    /// </summary>
    public const byte PACKET_TYPE_POSITION = 1;      // 플레이어 위치 업데이트
    public const byte PACKET_TYPE_CHAT = 2;          // 채팅 메시지
    public const byte PACKET_TYPE_LOGIN = 3;         // 로그인 응답
    public const byte PACKET_TYPE_PLAYER_JOIN = 4;   // 새 플레이어 입장
    public const byte PACKET_TYPE_PLAYER_LEAVE = 5;  // 플레이어 퇴장
    
    /// <summary>
    /// 패킷을 파싱하여 적절한 메시지로 변환한다.
    /// </summary>
    /// <param name="packetData">수신한 패킷 데이터</param>
    /// <returns>파싱된 메시지 객체 (또는 null)</returns>
    public static object ParsePacket(byte[] packetData)
    {
        if (packetData == null || packetData.Length < 1)
        {
            return null;
        }
        
        byte packetType = packetData[0];
        
        try
        {
            switch (packetType)
            {
                case PACKET_TYPE_POSITION:
                    return ParsePositionMessage(packetData);
                    
                case PACKET_TYPE_CHAT:
                    return ParseChatMessage(packetData);
                    
                case PACKET_TYPE_LOGIN:
                    return ParseLoginMessage(packetData);
                    
                case PACKET_TYPE_PLAYER_JOIN:
                    return ParsePlayerJoinMessage(packetData);
                    
                case PACKET_TYPE_PLAYER_LEAVE:
                    return ParsePlayerLeaveMessage(packetData);
                    
                default:
                    Console.WriteLine($"[경고] 알 수 없는 패킷 타입: {packetType}");
                    return null;
            }
        }
        catch (Exception ex)
        {
            Console.WriteLine($"[오류] 패킷 파싱 중 예외 발생: {ex.Message}");
            return null;
        }
    }
    
    /// <summary>
    /// 위치 업데이트 메시지를 파싱한다.
    /// 패킷 형식: [타입(1)] [플레이어ID(4)] [X좌표(4)] [Y좌표(4)]
    /// </summary>
    private static PositionMessage ParsePositionMessage(byte[] packetData)
    {
        if (packetData.Length < 9)
        {
            throw new ArgumentException("위치 메시지가 너무 짧습니다");
        }
        
        int playerId = BitConverter.ToInt32(packetData, 1);
        float x = BitConverter.ToSingle(packetData, 5);
        float y = BitConverter.ToSingle(packetData, 9);
        
        return new PositionMessage
        {
            PlayerId = playerId,
            X = x,
            Y = y
        };
    }
    
    /// <summary>
    /// 채팅 메시지를 파싱한다.
    /// 패킷 형식: [타입(1)] [플레이어ID(4)] [메시지길이(2)] [메시지(가변)]
    /// </summary>
    private static ChatMessage ParseChatMessage(byte[] packetData)
    {
        if (packetData.Length < 7)
        {
            throw new ArgumentException("채팅 메시지가 너무 짧습니다");
        }
        
        int playerId = BitConverter.ToInt32(packetData, 1);
        ushort messageLength = BitConverter.ToUInt16(packetData, 5);
        
        if (packetData.Length < 7 + messageLength)
        {
            throw new ArgumentException("채팅 메시지 데이터가 불완전합니다");
        }
        
        string message = System.Text.Encoding.UTF8.GetString(
            packetData, 
            7, 
            messageLength
        );
        
        return new ChatMessage
        {
            PlayerId = playerId,
            Text = message
        };
    }
    
    /// <summary>
    /// 로그인 응답 메시지를 파싱한다.
    /// 패킷 형식: [타입(1)] [플레이어ID(4)] [성공여부(1)]
    /// </summary>
    private static LoginMessage ParseLoginMessage(byte[] packetData)
    {
        if (packetData.Length < 6)
        {
            throw new ArgumentException("로그인 메시지가 너무 짧습니다");
        }
        
        int playerId = BitConverter.ToInt32(packetData, 1);
        bool success = packetData[5] != 0;
        
        return new LoginMessage
        {
            PlayerId = playerId,
            Success = success
        };
    }
    
    /// <summary>
    /// 플레이어 입장 메시지를 파싱한다.
    /// 패킷 형식: [타입(1)] [플레이어ID(4)] [이름길이(2)] [이름(가변)]
    /// </summary>
    private static PlayerJoinMessage ParsePlayerJoinMessage(byte[] packetData)
    {
        if (packetData.Length < 7)
        {
            throw new ArgumentException("플레이어 입장 메시지가 너무 짧습니다");
        }
        
        int playerId = BitConverter.ToInt32(packetData, 1);
        ushort nameLength = BitConverter.ToUInt16(packetData, 5);
        
        if (packetData.Length < 7 + nameLength)
        {
            throw new ArgumentException("플레이어 이름 데이터가 불완전합니다");
        }
        
        string name = System.Text.Encoding.UTF8.GetString(
            packetData, 
            7, 
            nameLength
        );
        
        return new PlayerJoinMessage
        {
            PlayerId = playerId,
            PlayerName = name
        };
    }
    
    /// <summary>
    /// 플레이어 퇴장 메시지를 파싱한다.
    /// 패킷 형식: [타입(1)] [플레이어ID(4)]
    /// </summary>
    private static PlayerLeaveMessage ParsePlayerLeaveMessage(byte[] packetData)
    {
        if (packetData.Length < 5)
        {
            throw new ArgumentException("플레이어 퇴장 메시지가 너무 짧습니다");
        }
        
        int playerId = BitConverter.ToInt32(packetData, 1);
        
        return new PlayerLeaveMessage
        {
            PlayerId = playerId
        };
    }
}
```

`PacketHandler` 클래스는 다음과 같은 역할을 한다:

**패킷 타입 구분**: 패킷의 첫 바이트로 메시지 종류를 판단한다. 이를 통해 어떤 방식으로 파싱할지 결정한다.

**구조체 파싱**: 각 메시지 타입에 맞게 바이트 배열을 해석하여 의미 있는 데이터로 변환한다.

**에러 처리**: 패킷이 손상되었거나 불완전한 경우 예외를 발생시킨다.

이제 게임에서 실제로 메시지를 처리하는 방법을 보자. 게임의 메인 루프에서 매 프레임마다 패킷을 확인한다:

```csharp
using System;
using System.Collections.Generic;

/// <summary>
/// 게임의 네트워크 메시지 처리를 관리하는 클래스다.
/// 수신한 패킷을 파싱하고, 등록된 핸들러를 호출한다.
/// </summary>
public class MessageProcessor
{
    private readonly NetworkClient _networkClient;
    
    // 메시지 타입별 핸들러를 저장할 딕셔너리
    private readonly Dictionary<Type, Action<object>> _handlers 
        = new Dictionary<Type, Action<object>>();
    
    public MessageProcessor(NetworkClient networkClient)
    {
        _networkClient = networkClient;
    }
    
    /// <summary>
    /// 특정 메시지 타입에 대한 핸들러를 등록한다.
    /// </summary>
    /// <param name="messageType">메시지 클래스 타입</param>
    /// <param name="handler">메시지를 처리할 델리게이트</param>
    public void RegisterHandler(Type messageType, Action<object> handler)
    {
        _handlers[messageType] = handler;
    }
    
    /// <summary>
    /// 큐에 있는 모든 패킷을 처리한다.
    /// 게임 루프의 Update 메서드에서 매 프레임 호출해야 한다.
    /// </summary>
    public void ProcessMessages()
    {
        while (true)
        {
            // 큐에서 패킷을 꺼낸다
            byte[] packetData = _networkClient.GetReceivedPacket();
            
            if (packetData == null)
            {
                // 더 이상 받을 패킷이 없음
                break;
            }
            
            // 패킷을 파싱한다
            object message = PacketHandler.ParsePacket(packetData);
            
            if (message == null)
            {
                // 파싱 실패
                continue;
            }
            
            // 메시지 타입을 확인하여 적절한 핸들러를 호출한다
            Type messageType = message.GetType();
            
            if (_handlers.TryGetValue(messageType, out var handler))
            {
                try
                {
                    handler(message);
                }
                catch (Exception ex)
                {
                    Console.WriteLine($"[오류] 메시지 처리 중 예외 발생: {ex.Message}");
                }
            }
            else
            {
                Console.WriteLine($"[경고] {messageType.Name}에 대한 핸들러가 등록되어 있지 않습니다");
            }
        }
    }
}
```

`MessageProcessor` 클래스의 설계 원리를 설명한다:

**핸들러 등록 패턴**: 게임 시작 시 각 메시지 타입별로 처리 함수를 등록한다. 이렇게 하면 새로운 메시지 타입을 추가할 때 기존 코드를 수정할 필요 없다.

**큐 처리**: `ProcessMessages` 메서드에서 큐의 모든 패킷을 처리한다. 한 번에 하나씩 처리하므로 프레임 처리 시간이 너무 길어지지 않는다.

**예외 처리**: 하나의 메시지 처리에서 오류가 발생해도 다음 메시지 처리는 계속된다. 이렇게 하면 한 번의 오류로 인해 전체 네트워크 시스템이 중단되는 것을 방지한다.

### 실제 사용 예시
게임에서 실제로 네트워크 클라이언트를 사용하는 방식을 보자:

```csharp
public class GameManager
{
    private NetworkClient _networkClient;
    private MessageProcessor _messageProcessor;
    
    public async void Initialize()
    {
        // 네트워크 클라이언트 생성
        _networkClient = new NetworkClient();
        
        // 서버에 연결
        bool connected = await _networkClient.ConnectAsync("127.0.0.1", 5000);
        
        if (!connected)
        {
            Console.WriteLine("서버 연결 실패");
            return;
        }
        
        // 백그라운드에서 패킷 수신 시작
        _ = _networkClient.StartReceivingAsync();
        
        // 메시지 프로세서 생성
        _messageProcessor = new MessageProcessor(_networkClient);
        
        // 핸들러 등록
        _messageProcessor.RegisterHandler(
            typeof(PositionMessage),
            msg => HandlePositionMessage((PositionMessage)msg)
        );
        
        _messageProcessor.RegisterHandler(
            typeof(ChatMessage),
            msg => HandleChatMessage((ChatMessage)msg)
        );
        
        _messageProcessor.RegisterHandler(
            typeof(PlayerJoinMessage),
            msg => HandlePlayerJoinMessage((PlayerJoinMessage)msg)
        );
    }
    
    public void Update()
    {
        // 매 프레임마다 메시지 처리
        _messageProcessor.ProcessMessages();
    }
    
    private void HandlePositionMessage(PositionMessage msg)
    {
        Console.WriteLine($"플레이어 {msg.PlayerId}가 ({msg.X}, {msg.Y})로 이동했습니다");
    }
    
    private void HandleChatMessage(ChatMessage msg)
    {
        Console.WriteLine($"플레이어 {msg.PlayerId}: {msg.Text}");
    }
    
    private void HandlePlayerJoinMessage(PlayerJoinMessage msg)
    {
        Console.WriteLine($"플레이어 {msg.PlayerName}(ID: {msg.PlayerId})이 입장했습니다");
    }
    
    public void Shutdown()
    {
        _networkClient.Disconnect();
    }
}
```

이 구조의 장점들을 정리하면:

1. **비동기 수신**: 게임 루프를 블로킹하지 않으면서 지속적으로 서버의 메시지를 수신한다.

2. **스레드 안전**: 락을 사용하여 수신 스레드와 메인 스레드 사이의 데이터 접근을 안전하게 관리한다.

3. **확장성**: 새로운 메시지 타입을 추가하면 핸들러만 등록하면 되므로 기존 코드 수정이 최소화된다.

4. **오류 격리**: 하나의 메시지 처리 오류가 다른 메시지 처리에 영향을 주지 않는다.
  
</br>  
  

## 종합 정리
4장에서 배운 내용을 정리한다. 먼저 게임 클라이언트가 서버와 통신하기 위해서는 연결을 수립하고, 지속적으로 데이터를 수신하며, 수신한 데이터를 처리해야 한다. `NetworkClient` 클래스는 연결 관리와 패킷 송수신을 담당하고, `PacketHandler`는 수신한 바이너리 데이터를 의미 있는 메시지로 변환한다. `MessageProcessor`는 메시지를 게임 로직과 연결하는 중간 역할을 한다.

이런 구조를 사용하면 게임이 백그라운드에서 지속적으로 서버와 통신하면서도, 메인 게임 루프는 자신의 속도대로 진행될 수 있다. 또한 새로운 메시지 타입을 추가하는 것도 쉬워진다.

**문제 8** (심화): 현재 구현에서 `GetReceivedPacket` 메서드와 `StartReceivingAsync` 메서드에서 모두 `_lockObject`로 `_receivedPackets` 큐를 보호한다. 만약 이 락을 사용하지 않았다면 어떤 문제가 발생할 수 있는지 구체적인 시나리오를 들어 설명하시오.

**예시 답안**: 
락이 없다면 다음과 같은 문제가 발생할 수 있다:
1. **데이터 손상**: 수신 스레드가 패킷을 큐에 Enqueue하는 도중에, 메인 스레드가 Dequeue를 시도하면 큐의 내부 구조가 일관성 없는 상태가 될 수 있다.
2. **패킷 손실**: 수신 스레드가 Count를 확인한 직후 메인 스레드가 마지막 패킷을 꺼내면, 수신 스레드는 큐에 아무것도 없는데 패킷을 추가하려고 할 수 있다.
3. **예외 발생**: 큐의 불일치된 상태로 인해 `ArgumentOutOfRangeException` 등의 예외가 발생할 수 있다.

따라서 락을 사용하여 한 번에 하나의 스레드만 큐에 접근하도록 보장해야 한다.
