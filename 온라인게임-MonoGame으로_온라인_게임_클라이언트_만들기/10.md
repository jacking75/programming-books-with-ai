## 10.1 고급 네트워킹 라이브러리
지금까지 우리는 .NET의 기본 라이브러리(`System.Net.Sockets`)만 사용해서 네트워크 통신을 구현했다. 이것은 학습 목적으로는 훌륭하지만, 실제 게임을 만들 때는 더 강력한 기능이 필요할 수 있다. 이번 섹션에서는 실무에서 자주 사용되는 네트워킹 라이브러리들을 소개하겠다.

### Netcode for GameObjects
Netcode for GameObjects는 Unity에서 만들었지만, 개념은 MonoGame에도 적용할 수 있다. 이 라이브러리의 핵심 아이디어는 네트워크에서 동기화되는 객체를 추상화하는 것이다.

```csharp
// Netcode for GameObjects의 개념을 MonoGame에서 구현한 간단한 예제

// 네트워크에서 동기화되는 객체의 기본 인터페이스
public interface INetworkObject
{
    int NetworkId { get; }
    void SerializeState(BinaryWriter writer);
    void DeserializeState(BinaryReader reader);
    void OnOwnershipChanged(int newOwnerId);
}

// 네트워크 게임 객체의 추상 클래스
public abstract class NetworkGameObject : INetworkObject
{
    public int NetworkId { get; protected set; }
    public int OwnerId { get; protected set; }
    public bool IsOwner { get; protected set; }

    protected NetworkGameObject(int networkId, int ownerId)
    {
        NetworkId = networkId;
        OwnerId = ownerId;
        IsOwner = false;
    }

    // 클라이언트가 소유한 객체의 상태를 업데이트하고 네트워크에 전송
    public virtual void UpdateAndSync()
    {
        if (!IsOwner)
            return;

        UpdateState();
        SendStateToServer();
    }

    protected virtual void UpdateState()
    {
        // 게임 로직에 따라 상태 업데이트
    }

    protected virtual void SendStateToServer()
    {
        // 상태를 서버로 전송
    }

    public abstract void SerializeState(BinaryWriter writer);
    public abstract void DeserializeState(BinaryReader reader);
    public virtual void OnOwnershipChanged(int newOwnerId)
    {
        OwnerId = newOwnerId;
        IsOwner = (newOwnerId == ClientManager.Instance.ClientId);
    }
}

// 네트워크 플레이어 객체 구현
public class NetworkPlayer : NetworkGameObject
{
    public float X { get; set; }
    public float Y { get; set; }
    public float VelocityX { get; set; }
    public float VelocityY { get; set; }
    public string PlayerName { get; set; }

    public NetworkPlayer(int networkId, int ownerId) 
        : base(networkId, ownerId)
    {
        PlayerName = "";
    }

    protected override void UpdateState()
    {
        // 이 플레이어가 소유한 객체라면 입력을 처리하고 위치를 업데이트
        // 다른 클라이언트의 소유 객체라면 네트워크로부터 받은 상태로 렌더링만 함
    }

    public override void SerializeState(BinaryWriter writer)
    {
        writer.Write(NetworkId);
        writer.Write(X);
        writer.Write(Y);
        writer.Write(VelocityX);
        writer.Write(VelocityY);
        writer.Write(PlayerName ?? "");
    }

    public override void DeserializeState(BinaryReader reader)
    {
        int networkId = reader.ReadInt32();
        X = reader.ReadSingle();
        Y = reader.ReadSingle();
        VelocityX = reader.ReadSingle();
        VelocityY = reader.ReadSingle();
        PlayerName = reader.ReadString();
    }
}

// 네트워크 객체 매니저: 모든 네트워크 객체를 관리
public class NetworkObjectManager
{
    private Dictionary<int, INetworkObject> networkObjects 
        = new Dictionary<int, INetworkObject>();
    private int nextNetworkId = 1;

    // 새로운 네트워크 객체 생성
    public T SpawnNetworkObject<T>(int ownerId) where T : INetworkObject, new()
    {
        var obj = new T();
        var networkObj = obj as NetworkGameObject;
        networkObj.NetworkId = nextNetworkId++;
        networkObj.OnOwnershipChanged(ownerId);
        
        networkObjects[networkObj.NetworkId] = obj;
        return obj;
    }

    // 네트워크 객체 제거
    public void DestroyNetworkObject(int networkId)
    {
        networkObjects.Remove(networkId);
    }

    // 특정 네트워크 ID로 객체 찾기
    public INetworkObject GetNetworkObject(int networkId)
    {
        networkObjects.TryGetValue(networkId, out var obj);
        return obj;
    }

    // 모든 객체의 상태를 직렬화
    public byte[] SerializeAllObjects()
    {
        using (var ms = new MemoryStream())
        using (var bw = new BinaryWriter(ms))
        {
            bw.Write((ushort)networkObjects.Count);
            foreach (var obj in networkObjects.Values)
            {
                obj.SerializeState(bw);
            }
            return ms.ToArray();
        }
    }

    // 모든 객체의 상태를 역직렬화
    public void DeserializeAllObjects(byte[] data)
    {
        using (var ms = new MemoryStream(data))
        using (var br = new BinaryReader(ms))
        {
            ushort count = br.ReadUInt16();
            for (int i = 0; i < count; i++)
            {
                // 첫 번째로 읽은 NetworkId를 사용하여 객체 찾기
                int networkId = br.PeekChar() > -1 ? 
                    BitConverter.ToInt32(new byte[4], 0) : 0;
                
                if (networkObjects.TryGetValue(networkId, out var obj))
                {
                    obj.DeserializeState(br);
                }
            }
        }
    }
}
```

이 패턴의 장점은 각 게임 객체가 자신의 상태를 어떻게 직렬화하는지 알고 있다는 것이다. 새로운 객체 타입을 추가할 때마다 직렬화/역직렬화 로직을 함께 작성하므로, 프로토콜을 한 곳에 중앙집중식으로 관리하지 않아도 된다. 이는 코드가 커질수록 유지보수 비용을 크게 줄인다.

### Mirror 네트워크 라이브러리
Mirror는 Unity를 위한 고수준 네트워킹 라이브러리지만, 그 설계 철학은 모든 게임 네트워크 프로그래밍에 적용할 수 있다. Mirror의 핵심은 RPC(Remote Procedure Call)이다.

```csharp
// RPC 개념을 MonoGame에 구현
public class NetworkCommand
{
    public int CommandId { get; set; }
    public string MethodName { get; set; }
    public object[] Parameters { get; set; }
    public int SenderId { get; set; }
}

// RPC를 처리하는 기본 클래스
public class NetworkBehaviour
{
    protected int NetworkId { get; set; }
    protected NetworkManager networkManager;

    protected NetworkBehaviour(NetworkManager manager, int networkId)
    {
        networkManager = manager;
        NetworkId = networkId;
    }

    // RPC를 다른 클라이언트에게 호출
    protected void CallRpcOnServer(string methodName, params object[] parameters)
    {
        networkManager.SendRpc(NetworkId, methodName, parameters);
    }

    // RPC를 모든 클라이언트에게 호출 (서버가 중계)
    protected void CallRpcOnAllClients(string methodName, params object[] parameters)
    {
        networkManager.BroadcastRpc(NetworkId, methodName, parameters);
    }
}

// 플레이어 행동을 RPC로 구현
public class PlayerBehaviour : NetworkBehaviour
{
    public string PlayerName { get; set; }
    public float Health { get; set; }

    public PlayerBehaviour(NetworkManager manager, int networkId) 
        : base(manager, networkId)
    {
        Health = 100f;
    }

    // 로컬에서만 실행되는 메서드
    public void TakeDamage(float damage)
    {
        Health -= damage;
        
        // 모든 클라이언트에 알림
        CallRpcOnAllClients("OnTakeDamage", damage);
    }

    // RPC로 호출되는 메서드
    [NetworkRpc]
    public void OnTakeDamage(float damage)
    {
        // 모든 클라이언트에서 실행
        Console.WriteLine($"{PlayerName} took {damage} damage. Health: {Health}");
    }

    // 공격을 다른 플레이어에게 알림
    public void AttackPlayer(int targetNetworkId, float damage)
    {
        CallRpcOnServer("OnAttackedByPlayer", targetNetworkId, damage);
    }

    [NetworkRpc]
    public void OnAttackedByPlayer(int attackerId, float damage)
    {
        // 피해를 입었을 때의 로직
    }
}

// 네트워크 RPC를 처리하는 매니저
public class NetworkRpcManager
{
    private Dictionary<(int, string), Action<byte[]>> rpcHandlers 
        = new Dictionary<(int, string), Action<byte[]>>();

    // RPC 핸들러 등록
    public void RegisterRpcHandler(int networkId, string methodName, 
        Action<byte[]> handler)
    {
        rpcHandlers[(networkId, methodName)] = handler;
    }

    // 수신한 RPC 패킷 처리
    public void ProcessRpcPacket(int networkId, string methodName, byte[] data)
    {
        var key = (networkId, methodName);
        if (rpcHandlers.TryGetValue(key, out var handler))
        {
            handler(data);
        }
    }
}
```

RPC 패턴은 직관적이다. "플레이어가 공격했다"는 사실을 전체 상태로 표현하지 말고, "플레이어가 공격했다"는 이벤트 자체를 네트워크로 전송한다. 이는 네트워크 트래픽을 줄이고 코드를 더 읽기 쉽게 만든다.

### Photon PUN
Photon PUN은 상용 네트워킹 솔루션으로, 게임 개발자들이 자주 사용한다. Photon의 강점은 클라우드 호스팅 서버를 제공한다는 것이다. 자신의 게임 서버를 직접 구축하고 운영하는 대신, Photon의 인프라를 사용할 수 있다.

```csharp
// Photon의 개념을 MonoGame에 적용한 예제

// 포톤 클라우드 서버에 연결
public class PhotonCloudClient
{
    private string appId;
    private string appVersion = "1.0";
    private PhotonConnection connection;

    public PhotonCloudClient(string appId)
    {
        this.appId = appId;
    }

    // 클라우드 서버에 연결
    public async Task ConnectToCloud()
    {
        connection = new PhotonConnection();
        await connection.ConnectAsync("ns.photonengine.com", appId);
    }

    // 방 생성
    public async Task<Room> CreateRoom(string roomName, int maxPlayers)
    {
        var room = new Room
        {
            Name = roomName,
            MaxPlayers = maxPlayers,
            IsOpen = true,
            IsVisible = true
        };

        await connection.SendRequest(new CreateRoomRequest { Room = room });
        return room;
    }

    // 게임 방 목록 조회
    public async Task<List<Room>> GetAvailableRooms()
    {
        var response = await connection.SendRequest(new ListRoomsRequest());
        return response.Rooms;
    }

    // 특정 방에 입장
    public async Task<bool> JoinRoom(string roomName)
    {
        var request = new JoinRoomRequest { RoomName = roomName };
        var response = await connection.SendRequest(request);
        return response.Success;
    }
}

// Photon의 이벤트 기반 메시지 처리
public class PhotonEventBus
{
    private Dictionary<byte, List<Action<object[]>>> eventHandlers 
        = new Dictionary<byte, List<Action<object[]>>>();

    // 이벤트 핸들러 구독
    public void Subscribe(byte eventCode, Action<object[]> handler)
    {
        if (!eventHandlers.ContainsKey(eventCode))
        {
            eventHandlers[eventCode] = new List<Action<object[]>>();
        }
        eventHandlers[eventCode].Add(handler);
    }

    // 이벤트 발행
    public void PublishEvent(byte eventCode, params object[] data)
    {
        if (eventHandlers.TryGetValue(eventCode, out var handlers))
        {
            foreach (var handler in handlers)
            {
                handler(data);
            }
        }
    }
}
```

### 라이브러리 선택 가이드
각 라이브러리는 서로 다른 장단점이 있다. 프로젝트의 특성에 따라 선택하면 된다.

- **자체 구현** (이 책에서 배운 방식): 완전한 제어, 학습 목적으로 최적. 다만 버그를 직접 관리해야 함.
- **Netcode for GameObjects**: 객체 지향적 설계, 중규모 게임에 적합.
- **Mirror**: RPC 기반, 빠른 프로토타이핑. Unity 생태계.
- **Photon PUN**: 클라우드 호스팅, 대규모 멀티플레이. 상용 서비스.

## 10.2 게임 확장 방향
현재 우리가 만든 게임은 기본적인 멀티플레이 기능만 가지고 있다. 실제 게임으로 발전시키려면 다음과 같은 기능들을 추가해야 한다.

### 게임플레이 확장
첫 번째로 고려할 수 있는 확장은 게임플레이의 다양성을 늘리는 것이다.

```csharp
// 아이템 시스템 구현 예제
public class Item
{
    public int ItemId { get; set; }
    public string ItemName { get; set; }
    public ItemType Type { get; set; }
    public int Value { get; set; }
}

public enum ItemType
{
    Weapon,
    Armor,
    Consumable,
    Quest
}

public class Inventory
{
    private List<Item> items = new List<Item>();
    private const int MAX_SLOTS = 20;

    public bool AddItem(Item item)
    {
        if (items.Count >= MAX_SLOTS)
            return false;

        items.Add(item);
        return true;
    }

    public bool RemoveItem(int itemId)
    {
        var item = items.FirstOrDefault(i => i.ItemId == itemId);
        if (item == null)
            return false;

        items.Remove(item);
        return true;
    }

    public List<Item> GetItems() => new List<Item>(items);
}

// 전투 시스템 구현
public class CombatSystem
{
    public class AttackResult
    {
        public bool Hit { get; set; }
        public float Damage { get; set; }
        public int CriticalChance { get; set; }
    }

    // 공격 계산
    public static AttackResult CalculateAttack(
        float attackerPower, 
        float targetDefense,
        float range,
        float distanceToTarget)
    {
        // 거리가 범위를 벗어나면 미스
        if (distanceToTarget > range)
        {
            return new AttackResult { Hit = false, Damage = 0 };
        }

        // 기본 데미지 계산
        float baseDamage = attackerPower - (targetDefense * 0.5f);
        baseDamage = Math.Max(1, baseDamage); // 최소 1 데미지

        // 크리티컬 계산 (20% 확률)
        var random = new Random();
        bool isCritical = random.Next(100) < 20;
        float finalDamage = isCritical ? baseDamage * 1.5f : baseDamage;

        return new AttackResult
        {
            Hit = true,
            Damage = finalDamage,
            CriticalChance = isCritical ? 1 : 0
        };
    }
}

// 퀘스트 시스템 구현
public class Quest
{
    public int QuestId { get; set; }
    public string QuestName { get; set; }
    public string Description { get; set; }
    public QuestObjective[] Objectives { get; set; }
    public int RewardGold { get; set; }
    public Item[] RewardItems { get; set; }
    public QuestState State { get; set; }
}

public class QuestObjective
{
    public int ObjectiveId { get; set; }
    public string Description { get; set; }
    public int TargetCount { get; set; }
    public int CurrentCount { get; set; }

    public bool IsComplete() => CurrentCount >= TargetCount;
}

public enum QuestState
{
    NotStarted,
    InProgress,
    Completed,
    Abandoned
}

public class QuestManager
{
    private Dictionary<int, Quest> activeQuests = new Dictionary<int, Quest>();

    public void AcceptQuest(Quest quest)
    {
        quest.State = QuestState.InProgress;
        activeQuests[quest.QuestId] = quest;
    }

    public bool UpdateQuestProgress(int questId, int objectiveId, int count)
    {
        if (!activeQuests.TryGetValue(questId, out var quest))
            return false;

        var objective = quest.Objectives.FirstOrDefault(o => o.ObjectiveId == objectiveId);
        if (objective == null)
            return false;

        objective.CurrentCount += count;

        // 모든 목표가 완료되면 퀘스트 완료
        if (quest.Objectives.All(o => o.IsComplete()))
        {
            quest.State = QuestState.Completed;
            return true;
        }

        return false;
    }

    public List<Quest> GetActiveQuests()
    {
        return activeQuests.Values.Where(q => q.State == QuestState.InProgress).ToList();
    }
}
```

이러한 시스템들은 게임을 단순한 이동 게임에서 실제 RPG로 변환한다. 각 시스템은 독립적으로 개발할 수 있으므로, 점진적으로 게임을 확장할 수 있다.

### 게임 맵 확장
큰 맵을 효율적으로 관리하기 위해서는 공간 분할(spatial partitioning)이 필요하다.

```csharp
// 쿼드트리를 사용한 공간 분할
public class QuadTree
{
    private const int MAX_OBJECTS = 4;
    private const int MAX_LEVELS = 5;

    public class Node
    {
        public Rectangle Bounds { get; set; }
        public List<NetworkPlayer> Objects { get; set; }
        public Node[] Children { get; set; }
        public int Level { get; set; }

        public Node(Rectangle bounds, int level)
        {
            Bounds = bounds;
            Level = level;
            Objects = new List<NetworkPlayer>();
            Children = null;
        }
    }

    private Node root;

    public QuadTree(int width, int height)
    {
        root = new Node(new Rectangle(0, 0, width, height), 0);
    }

    // 플레이어를 쿼드트리에 삽입
    public void Insert(NetworkPlayer player)
    {
        InsertNode(root, player);
    }

    private void InsertNode(Node node, NetworkPlayer player)
    {
        // 현재 노드의 범위를 벗어나면 삽입 불가
        if (!node.Bounds.Contains((int)player.X, (int)player.Y))
            return;

        // 리프 노드인 경우
        if (node.Children == null)
        {
            node.Objects.Add(player);

            // 오브젝트가 많아지면 분할
            if (node.Objects.Count > MAX_OBJECTS && node.Level < MAX_LEVELS)
            {
                Subdivide(node);
            }
            return;
        }

        // 자식 노드에 재귀적으로 삽입
        foreach (var child in node.Children)
        {
            InsertNode(child, player);
        }
    }

    // 노드를 4개의 자식 노드로 분할
    private void Subdivide(Node node)
    {
        int x = node.Bounds.X;
        int y = node.Bounds.Y;
        int width = node.Bounds.Width / 2;
        int height = node.Bounds.Height / 2;

        node.Children = new Node[4];
        node.Children[0] = new Node(new Rectangle(x, y, width, height), node.Level + 1);
        node.Children[1] = new Node(new Rectangle(x + width, y, width, height), node.Level + 1);
        node.Children[2] = new Node(new Rectangle(x, y + height, width, height), node.Level + 1);
        node.Children[3] = new Node(new Rectangle(x + width, y + height, width, height), node.Level + 1);

        // 기존 오브젝트를 자식 노드로 이동
        foreach (var obj in node.Objects)
        {
            foreach (var child in node.Children)
            {
                InsertNode(child, obj);
            }
        }
        node.Objects.Clear();
    }

    // 특정 영역 내의 플레이어 조회
    public List<NetworkPlayer> Query(Rectangle area)
    {
        var result = new List<NetworkPlayer>();
        QueryNode(root, area, result);
        return result;
    }

    private void QueryNode(Node node, Rectangle area, List<NetworkPlayer> result)
    {
        if (!node.Bounds.Intersects(area))
            return;

        // 리프 노드인 경우
        if (node.Children == null)
        {
            foreach (var obj in node.Objects)
            {
                if (area.Contains((int)obj.X, (int)obj.Y))
                {
                    result.Add(obj);
                }
            }
            return;
        }

        // 자식 노드 순회
        foreach (var child in node.Children)
        {
            QueryNode(child, area, result);
        }
    }
}
```

이 쿼드트리 구조를 사용하면, 특정 플레이어 주변의 다른 플레이어들만 찾을 수 있다. 예를 들어, 플레이어가 시야 범위(500x500 픽셀) 내의 다른 플레이어들의 정보만 받으면 된다. 이렇게 하면 게임에 수천 명의 플레이어가 있더라도, 각 플레이어는 시야 범위 내의 몇십 명의 정보만 처리하면 된다.

```csharp
public class PlayerSaveData
{

// 게임 상태 저장 시스템
public class GameSaveManager
{
    private string savePath = "saves";

    public GameSaveManager()
    {
        if (!Directory.Exists(savePath))
        {
            Directory.CreateDirectory(savePath);
        }
    }

    // 플레이어 데이터를 파일로 저장
    public bool SavePlayer(PlayerSaveData saveData)
    {
        try
        {
            string fileName = Path.Combine(savePath, $"{saveData.PlayerId}.sav");
            
            using (var fs = new FileStream(fileName, FileMode.Create))
            using (var bw = new BinaryWriter(fs))
            {
                // 플레이어 기본 정보 저장
                bw.Write(saveData.PlayerId);
                bw.Write(saveData.PlayerName);
                bw.Write(saveData.Level);
                bw.Write(saveData.Experience);
                bw.Write(saveData.X);
                bw.Write(saveData.Y);
                bw.Write(saveData.Health);
                bw.Write(saveData.MaxHealth);

                // 인벤토리 저장
                bw.Write((ushort)saveData.Inventory.Count);
                foreach (var item in saveData.Inventory)
                {
                    bw.Write(item.ItemId);
                    bw.Write(item.ItemName);
                    bw.Write((byte)item.Type);
                    bw.Write(item.Value);
                }

                // 완료한 퀘스트 저장
                bw.Write((ushort)saveData.CompletedQuests.Count);
                foreach (var quest in saveData.CompletedQuests)
                {
                    bw.Write(quest.QuestId);
                    bw.Write(quest.QuestName);
                }

                bw.Write(saveData.LastSavedTime.Ticks);
            }

            return true;
        }
        catch (Exception ex)
        {
            Console.WriteLine($"Save failed: {ex.Message}");
            return false;
        }
    }

    // 파일에서 플레이어 데이터를 불러오기
    public PlayerSaveData LoadPlayer(int playerId)
    {
        try
        {
            string fileName = Path.Combine(savePath, $"{playerId}.sav");
            
            if (!File.Exists(fileName))
                return null;

            using (var fs = new FileStream(fileName, FileMode.Open))
            using (var br = new BinaryReader(fs))
            {
                var saveData = new PlayerSaveData
                {
                    PlayerId = br.ReadInt32(),
                    PlayerName = br.ReadString(),
                    Level = br.ReadInt32(),
                    Experience = br.ReadSingle(),
                    X = br.ReadSingle(),
                    Y = br.ReadSingle(),
                    Health = br.ReadSingle(),
                    MaxHealth = br.ReadSingle(),
                    Inventory = new List<Item>(),
                    CompletedQuests = new List<Quest>()
                };

                // 인벤토리 불러오기
                ushort itemCount = br.ReadUInt16();
                for (int i = 0; i < itemCount; i++)
                {
                    saveData.Inventory.Add(new Item
                    {
                        ItemId = br.ReadInt32(),
                        ItemName = br.ReadString(),
                        Type = (ItemType)br.ReadByte(),
                        Value = br.ReadInt32()
                    });
                }

                // 완료한 퀘스트 불러오기
                ushort questCount = br.ReadUInt16();
                for (int i = 0; i < questCount; i++)
                {
                    saveData.CompletedQuests.Add(new Quest
                    {
                        QuestId = br.ReadInt32(),
                        QuestName = br.ReadString()
                    });
                }

                saveData.LastSavedTime = new DateTime(br.ReadInt64());

                return saveData;
            }
        }
        catch (Exception ex)
        {
            Console.WriteLine($"Load failed: {ex.Message}");
            return null;
        }
    }

    // 모든 세이브 파일 목록 조회
    public List<PlayerSaveData> GetAllSaves()
    {
        var result = new List<PlayerSaveData>();
        
        var files = Directory.GetFiles(savePath, "*.sav");
        foreach (var file in files)
        {
            // 파일명에서 플레이어 ID 추출
            int playerId = int.Parse(Path.GetFileNameWithoutExtension(file));
            var saveData = LoadPlayer(playerId);
            if (saveData != null)
            {
                result.Add(saveData);
            }
        }

        return result;
    }

    // 세이브 파일 삭제
    public bool DeleteSave(int playerId)
    {
        try
        {
            string fileName = Path.Combine(savePath, $"{playerId}.sav");
            if (File.Exists(fileName))
            {
                File.Delete(fileName);
                return true;
            }
            return false;
        }
        catch (Exception ex)
        {
            Console.WriteLine($"Delete failed: {ex.Message}");
            return false;
        }
    }
}
```

```csharp
public class AnimationFrame

public class Animation
{
    public string Name { get; set; }

    public Animation()
    {
        Frames = new List<AnimationFrame>();

public class Animator
{
    private Animation currentAnimation;
    private int currentFrameIndex = 0;
    private float elapsedTime = 0;
    private bool isPlaying = false;

    public void PlayAnimation(Animation animation)
    {
        currentAnimation = animation;
        currentFrameIndex = 0;
        elapsedTime = 0;
        isPlaying = true;
    }

    public void Update(float deltaTime)
    {
        if (!isPlaying || currentAnimation == null || currentAnimation.Frames.Count == 0)
            return;

        elapsedTime += deltaTime;

        // 현재 프레임의 지속 시간이 지났으면 다음 프레임으로
        float frameDuration = currentAnimation.Frames[currentFrameIndex].Duration;
        if (elapsedTime >= frameDuration)
        {
            elapsedTime -= frameDuration;
            currentFrameIndex++;

            // 애니메이션 끝에 도달
            if (currentFrameIndex >= currentAnimation.Frames.Count)
            {
                if (currentAnimation.IsLooping)
                {
                    currentFrameIndex = 0;
                }
                else
                {
                    isPlaying = false;
                    currentFrameIndex = currentAnimation.Frames.Count - 1;
                }
            }
        }
    }

    public Texture2D GetCurrentFrame()
    {
        if (currentAnimation == null || currentAnimation.Frames.Count == 0)
            return null;

        return currentAnimation.Frames[currentFrameIndex].Texture;
    }

    public bool IsAnimationFinished => !isPlaying;
}

public class Particle
{
    public Vector2 Position { get; set; }
    public Vector2 Velocity { get; set; }
    public float Lifetime { get; set; }
    public float MaxLifetime { get; set; }
    public Color Color { get; set; }
    public float Size { get; set; }
}

public class ParticleEmitter
{
    private List<Particle> particles = new List<Particle>();
    private Texture2D particleTexture;
    private Random random = new Random();

    public ParticleEmitter(Texture2D texture)
    {
        particleTexture = texture;
    }

    // 파티클 방출
    public void Emit(Vector2 position, int count, Vector2 velocityRange)
    {
        for (int i = 0; i < count; i++)
        {
            var particle = new Particle
            {
                Position = position,
                Velocity = new Vector2(
                    (float)(random.NextDouble() - 0.5) * velocityRange.X,
                    (float)(random.NextDouble() - 0.5) * velocityRange.Y
                ),
                Lifetime = 1f,
                MaxLifetime = 1f,
                Color = Color.White,
                Size = 2f
            };

            particles.Add(particle);
        }
    }

    public void Update(float deltaTime)
    {
        for (int i = particles.Count - 1; i >= 0; i--)
        {
            var particle = particles[i];
            particle.Lifetime -= deltaTime;

            if (particle.Lifetime <= 0)
            {
                particles.RemoveAt(i);
                continue;
            }

            // 위치 업데이트
            particle.Position += particle.Velocity * deltaTime;

            // 알파값 감소 (페이드 아웃)
            float alphaRatio = particle.Lifetime / particle.MaxLifetime;
            particle.Color = Color.White * alphaRatio;
        }
    }

    public void Draw(SpriteBatch spriteBatch)
    {
        foreach (var particle in particles)
        {
            spriteBatch.Draw(
                particleTexture,
                particle.Position,
                null,
                particle.Color,
                0,
                Vector2.Zero,
                particle.Size / particleTexture.Width,
                SpriteEffects.None,
                0
            );
        }
    }
}
```

- **AI**: 적 AI 구현
- **물리 엔진**: 충돌, 중력 등
- **그래픽 프로그래밍**: 셰이더, 이펙트
- **사운드 프로그래밍**: 3D 오디오, 음성 채팅
- **플랫폼별 최적화**: PC, 모바일, 콘솔

## 마치며
이 책을 통해 여러분은 MonoGame으로 온라인 게임 클라이언트를 만드는 기초를 배웠다. 처음에는 두려워 보이던 네트워크 프로그래밍도, 차근차근 배우면 이해할 수 있는 주제임을 깨달았을 것이다.

게임 개발은 프로그래밍, 그래픽, 사운드, 게임 디자인 등 여러 분야가 모여있는 종합 예술이다. 이 책에서는 프로그래밍 측면에만 집중했지만, 진정한 게임을 만들려면 다른 분야들도 함께 배워야 한다.

가장 중요한 것은 계속 만들어보는 것이다. 작은 게임부터 시작해서, 점차 복잡한 게임으로 나아가면서 실력을 쌓아가길 바란다. 실수와 실패는 좋은 선생님이다. 문제가 발생했을 때 그 원인을 찾고 해결하는 과정에서 가장 많이 배운다.

그리고 기억하자. 게임 개발은 혼자 하는 것보다 팀으로 하는 것이 훨씬 더 재미있고 효율적이다. 온라인 게임은 특히 그렇다. 여러분의 게임이 많은 사람들에게 즐거움을 주기를 바란다.

행운을 빈다.

## 학습 확인 과제
이번 장에서 배운 내용을 바탕으로 다음 과제를 수행해보자.

### 과제 1: 게임 엔진 선택 및 평가
현재 배운 내용을 바탕으로 다음 네트워킹 라이브러리들을 조사하고, 각각의 장단점을 정리해보자:

1. **자체 구현** (이 책의 방식)
2. **Netcode for GameObjects** 개념
3. **Mirror** 라이브러리
4. **Photon PUN** 서비스
5. **Grimoire** 라이브러리

각 라이브러리에 대해 다음 항목을 비교해보자:
- 학습 난이도
- 개발 속도
- 성능
- 비용
- 커뮤니티 지원
- 확장성

조사 결과를 보고서로 작성하여, 어떤 라이브러리가 어떤 상황에서 최적인지 논의해보자.

### 과제 2: 게임 확장 기능 구현
지금까지 만든 게임에 다음 기능 중 하나를 선택해서 구현해보자:

#### 옵션 A: 아이템 시스템
- 아이템 획득/사용 시스템
- 인벤토리 UI
- 아이템 드롭 시각화
- 아이템 효과 적용 (데미지 증가, 체력 회복 등)

#### 옵션 B: 전투 시스템
- 기본 공격 메커니즘
- 스킬 시스템 (여러 스킬, 쿨타임)
- 크리티컬 데미지
- 상태 이상 (스턴, 중독 등)

#### 옵션 C: 퀘스트 시스템
- 퀘스트 NPC
- 퀘스트 진행도 추적
- 퀘스트 완료 및 보상
- 퀘스트 UI

각 옵션에 대해 다음을 수행해야 한다:
- 필요한 데이터 구조 설계
- 게임 로직 구현
- 네트워크 동기화 (필요한 경우)
- 사용자 인터페이스 구현
- 기능 테스트

### 과제 3: 성능 보고서 작성

게임을 1시간 동안 플레이하면서 다음 데이터를 수집해보자:

1. **프레임 레이트**
   - 최소 FPS
   - 최대 FPS
   - 평균 FPS
   - 표준편차

2. **네트워크 성능**
   - 초당 전송/수신 패킷 수
   - 초당 전송/수신 바이트 수
   - 평균 지연 시간 (핑)
   - 패킷 손실률

3. **메모리 사용**
   - 초기 메모리 사용량
   - 최대 메모리 사용량
   - 메모리 누수 여부

4. **시스템 리소스**
   - CPU 사용률
   - 그래픽카드 사용률

수집한 데이터를 그래프로 시각화하고, 성능 병목 지점을 분석해보자.

### 과제 4: 저장/로딩 시스템 구현
게임 상태를 저장하고 불러올 수 있는 기능을 구현해보자:

#### 저장할 데이터
- 플레이어 위치
- 플레이어 통계 (레벨, 경험치, 보유 골드)
- 인벤토리
- 완료한 퀘스트
- 게임 진행도

#### 기능 요구사항
- 게임을 종료할 때 자동 저장
- 게임을 시작할 때 자동 로드
- 수동 저장 슬롯 (3개 이상)
- 저장 파일 정보 표시 (시간, 플레이 시간 등)
- 저장 파일 삭제 기능

#### 추가 요구사항
- 저장 파일 암호화
- 저장 파일 무결성 검증
- 여러 버전의 호환성 유지

### 과제 5: UI/UX 개선
게임의 사용자 인터페이스를 개선해보자:

#### 개선할 항목
- **메인 메뉴**: 새 게임, 계속하기, 설정, 종료
- **게임 HUD**: 체력, 마나, 레벨, 경험치바
- **인벤토리 창**: 아이템 목록, 아이템 정보, 사용/버림
- **설정 창**: 음량, 그래픽 설정, 키 바인딩
- **플레이어 목록**: 온라인 플레이어, 길드원 등

#### 기술 요구사항
- 반응형 UI (다양한 해상도 지원)
- 부드러운 애니메이션
- 마우스/키보드 네비게이션
- 키보드 단축키 지원

### 과제 6: 팀 프로젝트 제안
만약 동료 학생들과 함께 게임을 개발한다면, 다음과 같은 역할 분담을 고려해보자:

- **네트워크 엔지니어**: 게임 서버 구축 및 네트워크 코드
- **게임 프로그래머**: 게임 로직 및 게임플레이
- **UI/UX 개발자**: 사용자 인터페이스
- **그래픽스 프로그래머**: 렌더링 및 애니메이션
- **QA/테스터**: 버그 찾기 및 성능 테스트
- **게임 디자이너**: 게임 밸런싱 및 콘텐츠 설계

각 역할의 책임을 정의하고, 팀 프로젝트 계획을 수립해보자. 특히 다음을 고려해야 한다:
- 개발 일정
- 커뮤니케이션 방식
- 코드 관리 (Git 등)
- 정기적인 평가와 피드백

이 과제들을 통해 게임 개발이 단순한 프로그래밍을 넘어 복잡한 시스템과 팀 협력을 요구하는 분야임을 깨달을 것이다.
