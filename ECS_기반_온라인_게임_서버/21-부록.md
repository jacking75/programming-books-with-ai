## 부록 1: ECS 패턴 모음

### ECS 기본 구조
ECS(Entity-Component-System)는 게임 개발에서 자주 사용되는 아키텍처 패턴이다. 기존의 객체지향 프로그래밍과 다르게 데이터와 로직을 분리해 성능과 확장성을 향상시킨다.

```csharp
// Entity: 게임 객체의 고유 식별자
public readonly struct Entity
{
    public readonly int Id;
    
    public Entity(int id)
    {
        Id = id;
    }
}

// Component: 순수 데이터 구조
public struct PositionComponent
{
    public float X;
    public float Y;
    public float Z;
}

// System: 비즈니스 로직 담당
public class MovementSystem
{
    private readonly ComponentManager<PositionComponent> _positionComponents;
    
    public MovementSystem(ComponentManager<PositionComponent> positionComponents)
    {
        _positionComponents = positionComponents;
    }
    
    public void Update(float deltaTime)
    {
        foreach (var (entity, position) in _positionComponents.GetAll())
        {
            // 위치 업데이트 로직
            position.X += 1.0f * deltaTime;
        }
    }
}
```

2. **Object Pooling**

```csharp
public class ObjectPool<T> where T : class, new()
{
    private readonly Stack<T> _pool = new();
    private readonly Action<T> _resetAction;
    
    public ObjectPool(int initialCapacity, Action<T> resetAction = null)
    {
        _resetAction = resetAction;
        
        for (int i = 0; i < initialCapacity; i++)
        {
            _pool.Push(new T());
        }
    }
    
    public T Get()
    {
        return _pool.Count > 0 ? _pool.Pop() : new T();
    }
    
    public void Return(T obj)
    {
        _resetAction?.Invoke(obj);
        _pool.Push(obj);
    }
}
```

## 부록 4: 자주 발생하는 문제와 해결책

### 동시성 문제

**문제**: 다수의 클라이언트가 동일 엔티티를 수정할 때 발생하는 경쟁 상태

**해결책**: 낙관적 동시성 제어

```csharp
public class OptimisticConcurrencySystem
{
    private readonly Dictionary<int, int> _entityVersions = new();
    
    public bool TryUpdate(int entityId, int clientVersion, Action updateAction)
    {
        lock (_entityVersions)
        {
            if (!_entityVersions.TryGetValue(entityId, out int currentVersion))
            {
                currentVersion = 0;
                _entityVersions[entityId] = 0;
            }
            
            if (clientVersion != currentVersion)
                return false;
                
            updateAction();
            _entityVersions[entityId] = currentVersion + 1;
            return true;
        }
    }
}
```

### 메모리 누수 패턴

**문제**: 패킷 처리 중 예외가 발생해 리소스가 정리되지 않는 상황

**해결책**: 안전한 리소스 해제 패턴

```csharp
public class SafeResourceHandler : IDisposable
{
    private readonly List<IDisposable> _resources = new();
    private bool _disposed = false;
    
    public T AddResource<T>(T resource) where T : IDisposable
    {
        _resources.Add(resource);
        return resource;
    }
    
    public void Dispose()
    {
        if (_disposed)
            return;
            
        foreach (var resource in _resources)
        {
            try
            {
                resource.Dispose();
            }
            catch (Exception ex)
            {
                // 로그 기록
                Console.WriteLine($"리소스 해제 중 오류: {ex.Message}");
            }
        }
        
        _resources.Clear();
        _disposed = true;
    }
}
```
  
  
### ECS 통합 서버 예제

다음은 위 모든 개념이 통합된 간단한 ECS 게임 서버와 클라이언트 예제이다:

```csharp
// 서버 측 코드
public class GameServer
{
    private readonly World _world = new();
    private readonly INetworkService _networkService;
    private readonly ObjectPool<Player> _playerPool;
    
    public GameServer(string host, int port)
    {
        _networkService = new NetworkService();
        _networkService.Initialize(host, port);
        
        _playerPool = new ObjectPool<Player>(100, player => player.Reset());
        
        // 패킷 핸들러 등록
        _networkService.RegisterPacketHandler<PlayerMovePacket>(HandlePlayerMove);
        
        // 시스템 등록
        _world.RegisterSystem(new MovementSystem());
        _world.RegisterSystem(new CombatSystem());
    }
    
    public void Start()
    {
        // 게임 루프 시작
        var gameLoop = new GameLoop();
        gameLoop.FixedUpdateAction = Update;
        gameLoop.Run();
    }
    
    private void Update(float deltaTime)
    {
        // 네트워크 메시지 처리
        _networkService.Update();
        
        // ECS 시스템 업데이트
        _world.Update(deltaTime);
    }
    
    private void HandlePlayerMove(PlayerMovePacket packet)
    {
        // 플레이어 이동 처리
        if (_world.TryGetEntity(packet.PlayerId, out var entity))
        {
            var positionComponent = _world.GetComponent<PositionComponent>(entity);
            positionComponent.X = packet.PosX;
            positionComponent.Y = packet.PosY;
            positionComponent.Z = packet.PosZ;
            
            // 변경사항 브로드캐스트
            _networkService.Broadcast(packet);
        }
    }
}

// 클라이언트 측 코드 (콘솔)
public class GameClient
{
    private readonly INetworkService _networkService;
    private readonly int _playerId;
    
    public GameClient(string host, int port, int playerId)
    {
        _networkService = new NetworkService();
        _networkService.Initialize(host, port);
        _playerId = playerId;
        
        // 패킷 핸들러 등록
        _networkService.RegisterPacketHandler<PlayerMovePacket>(HandlePlayerMove);
    }
    
    public void Connect()
    {
        _networkService.Connect();
        Console.WriteLine("서버에 연결됨");
    }
    
    public void Run()
    {
        while (true)
        {
            // 네트워크 메시지 처리
            _networkService.Update();
            
            // 사용자 입력 처리
            if (Console.KeyAvailable)
            {
                var key = Console.ReadKey(true).Key;
                HandleInput(key);
            }
            
            Thread.Sleep(10);
        }
    }
    
    private void HandleInput(ConsoleKey key)
    {
        float x = 0, y = 0, z = 0;
        
        switch (key)
        {
            case ConsoleKey.W: z += 1; break;
            case ConsoleKey.S: z -= 1; break;
            case ConsoleKey.A: x -= 1; break;
            case ConsoleKey.D: x += 1; break;
            default: return;
        }
        
        // 이동 패킷 전송
        var movePacket = new PlayerMovePacket(_playerId, x, y, z);
        _networkService.Send(movePacket);
    }
    
    private void HandlePlayerMove(PlayerMovePacket packet)
    {
        // 다른 플레이어 이동 처리
        if (packet.PlayerId != _playerId)
        {
            Console.WriteLine($"플레이어 {packet.PlayerId}가 ({packet.PosX}, {packet.PosY}, {packet.PosZ})로 이동");
        }
    }
}
```
