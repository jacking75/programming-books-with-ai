# Chapter 4: C#과 SQLKata로 기본 CRUD 마스터하기
이전 장에서 Hypertable의 마법을 경험했다. 이제 본격적으로 게임 서버에서 발생하는 다양한 이벤트 데이터를 다루는 방법을 배울 차례다. 이번 장에서는 SQLKata의 Fluent API를 활용하여 데이터를 삽입(INSERT), 조회(SELECT), 수정(UPDATE), 삭제(DELETE)하는 모든 기본 작업을 마스터한다. 단순히 문법을 배우는 것이 아니라, 실제 게임 서버에서 발생하는 플레이어 행동 로그, 아이템 사용 기록, 전투 데이터 등을 처리하면서 자연스럽게 익힐 것이다.

## 4.1 SQLKata 기본 문법 익히기
SQLKata는 C#에서 SQL 쿼리를 타입 안전하게 작성할 수 있게 해주는 쿼리 빌더 라이브러리다. 문자열로 SQL을 작성하는 대신, C# 메서드 체이닝을 통해 직관적으로 쿼리를 구성할 수 있다.

**SQLKata의 핵심 장점은 다음과 같다:**

타입 안전성이 보장된다. 컴파일 타임에 오류를 잡을 수 있어 런타임 에러가 줄어든다. 문자열 SQL에서 흔히 발생하는 오타나 잘못된 컬럼명 문제를 사전에 방지한다.

가독성이 뛰어나다. 복잡한 WHERE 조건이나 JOIN을 메서드 체이닝으로 표현하면 코드가 자연스럽게 읽힌다. SQL 문자열을 여러 줄로 나누는 것보다 훨씬 깔끔하다.

데이터베이스 독립적이다. PostgreSQL, MySQL, SQL Server 등 다양한 데이터베이스를 지원한다. Compiler만 바꾸면 같은 코드로 여러 데이터베이스를 사용할 수 있다.

SQL 인젝션 공격을 자동으로 방어한다. 모든 값이 파라미터화되므로 보안 걱정이 줄어든다.

**기본 구조를 살펴보자:**

```csharp
using SqlKata;
using SqlKata.Execution;
using SqlKata.Compilers;
using Npgsql;
using System;
using System.Threading.Tasks;

namespace TimescaleGameServer
{
    public class SQLKataBasics
    {
        private readonly string _connectionString;

        public SQLKataBasics(string connectionString)
        {
            _connectionString = connectionString;
        }

        // QueryFactory 생성 - SQLKata의 핵심 객체
        private QueryFactory CreateQueryFactory()
        {
            var connection = new NpgsqlConnection(_connectionString);
            var compiler = new PostgresCompiler();
            return new QueryFactory(connection, compiler);
        }

        // 가장 기본적인 SELECT 쿼리
        public async Task BasicSelectExample()
        {
            using var db = CreateQueryFactory();

            // SELECT * FROM player_login_logs LIMIT 10
            var results = await db.Query("player_login_logs")
                .Limit(10)
                .GetAsync();

            foreach (var row in results)
            {
                Console.WriteLine($"플레이어 ID: {row.player_id}, 시각: {row.login_time}");
            }
        }

        // WHERE 조건 추가
        public async Task SelectWithWhereExample()
        {
            using var db = CreateQueryFactory();

            // SELECT * FROM player_login_logs 
            // WHERE login_result = 'SUCCESS' AND player_id = 1001
            var results = await db.Query("player_login_logs")
                .Where("login_result", "SUCCESS")
                .Where("player_id", 1001)
                .GetAsync();

            Console.WriteLine($"플레이어 1001의 성공한 로그인: {results.Count()}건");
        }

        // 특정 컬럼만 선택
        public async Task SelectSpecificColumnsExample()
        {
            using var db = CreateQueryFactory();

            // SELECT player_id, login_time, device_type 
            // FROM player_login_logs 
            // ORDER BY login_time DESC 
            // LIMIT 5
            var results = await db.Query("player_login_logs")
                .Select("player_id", "login_time", "device_type")
                .OrderByDesc("login_time")
                .Limit(5)
                .GetAsync();

            foreach (var row in results)
            {
                Console.WriteLine($"{row.player_id} - {row.device_type} - {row.login_time}");
            }
        }
    }
}
```

**SQLKata의 메서드 체이닝 패턴을 이해하자.** 각 메서드는 Query 객체를 반환하므로 계속 이어서 조건을 추가할 수 있다. 이것이 Fluent API의 핵심이다.

```
Query 객체 생성
    ↓
.Where() → Query 객체 반환
    ↓
.OrderBy() → Query 객체 반환
    ↓
.Limit() → Query 객체 반환
    ↓
.GetAsync() → 실제 쿼리 실행 및 결과 반환
```

## 4.2 INSERT: 플레이어 이벤트 로깅하기
게임 서버에서 가장 빈번하게 발생하는 작업은 이벤트 로깅이다. 플레이어가 로그인하고, 아이템을 사용하고, 전투를 벌이는 모든 순간을 기록해야 한다. SQLKata로 이런 데이터를 효율적으로 삽입하는 방법을 배워보자.

**단일 레코드 삽입의 기본:**

```csharp
public class GameEventLogger
{
    private readonly string _connectionString;

    public GameEventLogger(string connectionString)
    {
        _connectionString = connectionString;
    }

    private QueryFactory CreateQueryFactory()
    {
        var connection = new NpgsqlConnection(_connectionString);
        var compiler = new PostgresCompiler();
        return new QueryFactory(connection, compiler);
    }

    // 플레이어 로그인 이벤트 저장
    public async Task<int> LogPlayerLogin(long playerId, string ipAddress, string deviceType)
    {
        using var db = CreateQueryFactory();

        var result = await db.Query("player_login_logs").InsertAsync(new
        {
            login_time = DateTime.UtcNow,
            player_id = playerId,
            ip_address = ipAddress,
            device_type = deviceType,
            login_result = "SUCCESS",
            session_id = Guid.NewGuid(),
            server_id = 1
        });

        return result;
    }

    // 아이템 사용 로그 테이블 생성 및 Hypertable 변환 (처음 한 번만)
    public async Task CreateItemUsageTable()
    {
        using var connection = new NpgsqlConnection(_connectionString);
        await connection.OpenAsync();

        var createTableSql = @"
            CREATE TABLE IF NOT EXISTS item_usage_logs (
                usage_time TIMESTAMPTZ NOT NULL,
                player_id BIGINT NOT NULL,
                item_id INT NOT NULL,
                item_name VARCHAR(100),
                quantity INT,
                usage_type VARCHAR(20),
                game_context JSONB
            );
        ";

        await using (var cmd = new NpgsqlCommand(createTableSql, connection))
        {
            await cmd.ExecuteNonQueryAsync();
        }

        var createHypertableSql = @"
            SELECT create_hypertable('item_usage_logs', 'usage_time',
                if_not_exists => TRUE
            );
        ";

        await using (var cmd = new NpgsqlCommand(createHypertableSql, connection))
        {
            await cmd.ExecuteNonQueryAsync();
        }

        Console.WriteLine("아이템 사용 로그 테이블 준비 완료");
    }

    // 아이템 사용 로그 저장
    public async Task<int> LogItemUsage(long playerId, int itemId, string itemName, 
        int quantity, string usageType)
    {
        using var db = CreateQueryFactory();

        var result = await db.Query("item_usage_logs").InsertAsync(new
        {
            usage_time = DateTime.UtcNow,
            player_id = playerId,
            item_id = itemId,
            item_name = itemName,
            quantity = quantity,
            usage_type = usageType,
            game_context = new 
            { 
                level = 45, 
                zone = "던전_3층", 
                party_size = 4 
            }
        });

        return result;
    }
}
```

**배치 삽입으로 성능 극대화하기.** 실제 게임 서버에서는 초당 수백, 수천 건의 이벤트가 발생한다. 각 이벤트마다 개별 INSERT를 수행하면 네트워크와 데이터베이스 오버헤드가 크다. 배치 삽입을 사용하면 성능이 10배 이상 향상된다.

```csharp
public class BatchEventLogger
{
    private readonly string _connectionString;

    public BatchEventLogger(string connectionString)
    {
        _connectionString = connectionString;
    }

    private QueryFactory CreateQueryFactory()
    {
        var connection = new NpgsqlConnection(_connectionString);
        var compiler = new PostgresCompiler();
        return new QueryFactory(connection, compiler);
    }

    // 여러 로그인 이벤트를 한 번에 저장
    public async Task<int> LogPlayerLoginsBatch(List<PlayerLoginLog> loginEvents)
    {
        using var db = CreateQueryFactory();

        var data = loginEvents.Select(log => new
        {
            login_time = log.login_time,
            player_id = log.player_id,
            ip_address = log.ip_address,
            device_type = log.device_type,
            login_result = log.login_result,
            session_id = log.session_id,
            server_id = log.server_id
        }).ToArray();

        var result = await db.Query("player_login_logs").InsertAsync(data);
        return result;
    }

    // 실전 시나리오: 점심시간 피크타임 로그인 급증 처리
    public async Task SimulatePeakTimeLogins()
    {
        var loginEvents = new List<PlayerLoginLog>();
        var random = new Random();
        var baseTime = DateTime.UtcNow;

        // 1분간 500명의 로그인 시뮬레이션
        for (int i = 0; i < 500; i++)
        {
            loginEvents.Add(new PlayerLoginLog
            {
                login_time = baseTime.AddSeconds(random.Next(60)),
                player_id = 10000 + i,
                ip_address = $"192.168.{random.Next(1, 255)}.{random.Next(1, 255)}",
                device_type = random.Next(3) switch
                {
                    0 => "Android",
                    1 => "iOS",
                    _ => "PC"
                },
                login_result = random.Next(100) < 98 ? "SUCCESS" : "FAILED",
                session_id = Guid.NewGuid(),
                server_id = random.Next(1, 6)
            });
        }

        var stopwatch = System.Diagnostics.Stopwatch.StartNew();
        await LogPlayerLoginsBatch(loginEvents);
        stopwatch.Stop();

        Console.WriteLine($"500건의 로그인 이벤트를 {stopwatch.ElapsedMilliseconds}ms에 저장 완료");
    }
}
```

```
  - 총 소요 시간: 약 2,500ms
  - 초당 처리량: 200건

배치 INSERT (500건):
  - 한 번의 네트워크 왕복
  - 총 소요 시간: 약 150ms
  - 초당 처리량: 3,333건
  
약 16배 빠르다!
```

## 4.3 SELECT: 시간 범위로 데이터 조회하기
TimescaleDB의 진가는 시간 범위 조회에서 드러난다. 게임 운영팀에서 "지난주 월요일 오후 3시부터 5시 사이의 로그인 데이터를 보여달라"고 요청했다고 가정해보자.

**시간 범위 조회의 기본 패턴:**

```csharp
public class TimeRangeQueryService
{
    private readonly string _connectionString;

    public TimeRangeQueryService(string connectionString)
    {
        _connectionString = connectionString;
    }

    private QueryFactory CreateQueryFactory()
    {
        var connection = new NpgsqlConnection(_connectionString);
        var compiler = new PostgresCompiler();
        return new QueryFactory(connection, compiler);
    }

    // 특정 날짜 범위의 로그인 조회
    public async Task<IEnumerable<PlayerLoginLog>> GetLoginsBetween(
        DateTime startTime, DateTime endTime)
    {
        using var db = CreateQueryFactory();

        var results = await db.Query("player_login_logs")
            .WhereBetween("login_time", startTime, endTime)
            .OrderBy("login_time")
            .GetAsync<PlayerLoginLog>();

        return results;
    }

    // 최근 N시간 데이터 조회
    public async Task<IEnumerable<PlayerLoginLog>> GetLoginsLastHours(int hours)
    {
        using var db = CreateQueryFactory();

        var cutoffTime = DateTime.UtcNow.AddHours(-hours);

        var results = await db.Query("player_login_logs")
            .Where("login_time", ">=", cutoffTime)
            .OrderByDesc("login_time")
            .GetAsync<PlayerLoginLog>();

        return results;
    }

    // 오늘 하루 데이터 조회
    public async Task<IEnumerable<PlayerLoginLog>> GetTodayLogins()
    {
        using var db = CreateQueryFactory();

        var today = DateTime.UtcNow.Date;
        var tomorrow = today.AddDays(1);

        var results = await db.Query("player_login_logs")
            .Where("login_time", ">=", today)
            .Where("login_time", "<", tomorrow)
            .OrderBy("login_time")
            .GetAsync<PlayerLoginLog>();

        return results;
    }

    // 특정 주의 데이터 조회 (월요일부터 일요일까지)
    public async Task<IEnumerable<PlayerLoginLog>> GetWeekLogins(DateTime weekStart)
    {
        using var db = CreateQueryFactory();

        // weekStart는 월요일이어야 함
        var weekEnd = weekStart.AddDays(7);

        var results = await db.Query("player_login_logs")
            .WhereBetween("login_time", weekStart, weekEnd)
            .OrderBy("login_time")
            .GetAsync<PlayerLoginLog>();

        Console.WriteLine($"{weekStart:yyyy-MM-dd} 주의 로그인 {results.Count()}건");
        return results;
    }

    // 페이지네이션 지원 조회
    public async Task<IEnumerable<PlayerLoginLog>> GetLoginsWithPagination(
        DateTime startTime, DateTime endTime, int page, int pageSize)
    {
        using var db = CreateQueryFactory();

        var offset = (page - 1) * pageSize;

        var results = await db.Query("player_login_logs")
            .WhereBetween("login_time", startTime, endTime)
            .OrderByDesc("login_time")
            .Offset(offset)
            .Limit(pageSize)
            .GetAsync<PlayerLoginLog>();

        return results;
    }

    // 전체 개수도 함께 조회 (페이지네이션용)
    public async Task<(IEnumerable<PlayerLoginLog> data, int totalCount)> 
        GetLoginsWithCount(DateTime startTime, DateTime endTime, int page, int pageSize)
    {
        using var db = CreateQueryFactory();

        var offset = (page - 1) * pageSize;

        // 데이터 조회
        var data = await db.Query("player_login_logs")
            .WhereBetween("login_time", startTime, endTime)
            .OrderByDesc("login_time")
            .Offset(offset)
            .Limit(pageSize)
            .GetAsync<PlayerLoginLog>();

        // 전체 개수 조회
        var totalCount = await db.Query("player_login_logs")
            .WhereBetween("login_time", startTime, endTime)
            .CountAsync<int>();

        return (data, totalCount);
    }
}
```

## 4.4 WHERE 조건: 특정 플레이어 행동 추적
게임 운영에서 특정 플레이어의 행동을 추적해야 하는 경우가 많다. 고객 문의나 어뷰징 의심 사례를 조사할 때 필수적이다.

**다양한 WHERE 조건 활용:**

```csharp
public class PlayerTrackingService
{
    private readonly string _connectionString;

    public PlayerTrackingService(string connectionString)
    {
        _connectionString = connectionString;
    }

    private QueryFactory CreateQueryFactory()
    {
        var connection = new NpgsqlConnection(_connectionString);
        var compiler = new PostgresCompiler();
        return new QueryFactory(connection, compiler);
    }

    // 특정 플레이어의 최근 로그인 이력
    public async Task<IEnumerable<PlayerLoginLog>> GetPlayerLoginHistory(
        long playerId, int days = 30)
    {
        using var db = CreateQueryFactory();

        var cutoffDate = DateTime.UtcNow.AddDays(-days);

        var results = await db.Query("player_login_logs")
            .Where("player_id", playerId)
            .Where("login_time", ">=", cutoffDate)
            .OrderByDesc("login_time")
            .GetAsync<PlayerLoginLog>();

        return results;
    }

    // 특정 IP에서의 로그인 시도 조회 (다중 계정 의심)
    public async Task<IEnumerable<dynamic>> GetLoginsByIpAddress(string ipAddress)
    {
        using var db = CreateQueryFactory();

        var results = await db.Query("player_login_logs")
            .Where("ip_address", ipAddress)
            .Where("login_time", ">=", DateTime.UtcNow.AddDays(-7))
            .Select("player_id", "login_time", "device_type", "login_result")
            .OrderByDesc("login_time")
            .GetAsync();

        return results;
    }

    // 로그인 실패 케이스만 조회
    public async Task<IEnumerable<PlayerLoginLog>> GetFailedLogins(
        DateTime startTime, DateTime endTime)
    {
        using var db = CreateQueryFactory();

        var results = await db.Query("player_login_logs")
            .WhereBetween("login_time", startTime, endTime)
            .Where("login_result", "FAILED")
            .OrderByDesc("login_time")
            .GetAsync<PlayerLoginLog>();

        return results;
    }

    // 여러 조건 조합: 특정 기기에서 실패한 로그인
    public async Task<IEnumerable<PlayerLoginLog>> GetFailedMobileLogins(int hours = 24)
    {
        using var db = CreateQueryFactory();

        var cutoffTime = DateTime.UtcNow.AddHours(-hours);

        var results = await db.Query("player_login_logs")
            .Where("login_time", ">=", cutoffTime)
            .Where("login_result", "FAILED")
            .WhereIn("device_type", new[] { "Android", "iOS" })
            .OrderByDesc("login_time")
            .GetAsync<PlayerLoginLog>();

        return results;
    }

    // OR 조건 사용
    public async Task<IEnumerable<PlayerLoginLog>> GetProblematicLogins()
```

```csharp
public class JoinQueryService
{
    private readonly string _connectionString;

    public JoinQueryService(string connectionString)
    {
        _connectionString = connectionString;
    }

    private QueryFactory CreateQueryFactory()
    {
        var connection = new NpgsqlConnection(_connectionString);
        var compiler = new PostgresCompiler();
        return new QueryFactory(connection, compiler);
    }

    // INNER JOIN: 로그인 로그에 플레이어 정보 결합
    public async Task<IEnumerable<dynamic>> GetLoginsWithPlayerInfo(DateTime startTime)
    {
        using var db = CreateQueryFactory();

        var results = await db.Query("player_login_logs as pll")
            .Join("player_info as pi", "pll.player_id", "pi.player_id")
            .Where("pll.login_time", ">=", startTime)
            .Select(
                "pll.login_time",
                "pll.player_id",
                "pi.player_name",
                "pi.level",
                "pi.guild_name",
                "pll.device_type",
                "pll.login_result"
            )
            .OrderByDesc("pll.login_time")
            .GetAsync();

        return results;
    }

    // LEFT JOIN: 모든 플레이어와 최근 로그인 정보
    public async Task<IEnumerable<dynamic>> GetAllPlayersWithLastLogin()
    {
        using var db = CreateQueryFactory();

        // 각 플레이어의 최근 로그인 시각
        var results = await db.Query("player_info as pi")
            .LeftJoin("player_login_logs as pll", j => j
                .On("pi.player_id", "pll.player_id")
            )
            .Select(
                "pi.player_id",
                "pi.player_name",
                "pi.level",
                "pi.guild_name"
            )
            .SelectRaw("MAX(pll.login_time) as last_login_time")
            .GroupBy("pi.player_id", "pi.player_name", "pi.level", "pi.guild_name")
            .OrderBy("pi.player_name")
            .GetAsync();

        return results;
    }

    // 길드별 로그인 통계
    public async Task<IEnumerable<dynamic>> GetGuildLoginStats(DateTime startTime)
    {
        using var db = CreateQueryFactory();

        var results = await db.Query("player_info as pi")
            .Join("player_login_logs as pll", "pi.player_id", "pll.player_id")
            .Where("pll.login_time", ">=", startTime)
            .Where("pll.login_result", "SUCCESS")
            .Select("pi.guild_name")
            .SelectRaw("COUNT(*) as login_count")
            .SelectRaw("COUNT(DISTINCT pi.player_id) as unique_players")
            .GroupBy("pi.guild_name")
            .OrderByDesc("login_count")
            .GetAsync();

        return results;
    }

    // 레벨 구간별 로그인 패턴 분석
    public async Task<IEnumerable<dynamic>> GetLoginPatternByLevelRange()
    {
        using var db = CreateQueryFactory();

        var results = await db.Query("player_login_logs as pll")
            .Join("player_info as pi", "pll.player_id", "pi.player_id")
            .Where("pll.login_time", ">=", DateTime.UtcNow.AddDays(-7))
            .SelectRaw(@"
                CASE 
                    WHEN pi.level < 30 THEN '초보(1-29)'
                    WHEN pi.level < 50 THEN '중급(30-49)'
                    WHEN pi.level < 70 THEN '고급(50-69)'
                    ELSE '최상급(70+)'
                END as level_range
            ")
            .SelectRaw("COUNT(*) as login_count")
            .SelectRaw("COUNT(DISTINCT pll.player_id) as unique_players")
            .SelectRaw("AVG(pi.level) as avg_level")
            .GroupByRaw(@"
                CASE 
                    WHEN pi.level < 30 THEN '초보(1-29)'
                    WHEN pi.level < 50 THEN '중급(30-49)'
                    WHEN pi.level < 70 THEN '고급(50-69)'
                    ELSE '최상급(70+)'
                END
            ")
            .OrderBy("level_range")
            .GetAsync();

        return results;
    }
}
```

## 4.6 집계 함수: COUNT, SUM, AVG 활용
게임 운영에서 통계는 필수다. 일일 활성 사용자(DAU), 평균 플레이 시간, 총 매출 등을 계산해야 한다. SQLKata의 집계 함수를 마스터하자.

**기본 집계 함수들:**

```csharp
public class AggregationService
{
    private readonly string _connectionString;

    public AggregationService(string connectionString)
    {
        _connectionString = connectionString;
    }

    private QueryFactory CreateQueryFactory()
    {
        var connection = new NpgsqlConnection(_connectionString);
        var compiler = new PostgresCompiler();
        return new QueryFactory(connection, compiler);
    }

    // 총 로그인 횟수
    public async Task<int> GetTotalLoginCount(DateTime startTime, DateTime endTime)
    {
        using var db = CreateQueryFactory();

        var count = await db.Query("player_login_logs")
            .WhereBetween("login_time", startTime, endTime)
            .CountAsync<int>();

        return count;
    }

    // 고유 플레이어 수 (DAU 계산)
    public async Task<int> GetDailyActiveUsers(DateTime date)
    {
        using var db = CreateQueryFactory();

        var startOfDay = date.Date;
        var endOfDay = startOfDay.AddDays(1);

        var dau = await db.Query("player_login_logs")
            .WhereBetween("login_time", startOfDay, endOfDay)
            .Where("login_result", "SUCCESS")
            .CountAsync<int>("DISTINCT player_id");

        return dau;
    }

    // 시간대별 통계
    public async Task<IEnumerable<dynamic>> GetHourlyStats(DateTime date)
    {
        using var db = CreateQueryFactory();

        var startOfDay = date.Date;
        var endOfDay = startOfDay.AddDays(1);

        var results = await db.Query("player_login_logs")
            .WhereBetween("login_time", startOfDay, endOfDay)
            .SelectRaw("DATE_TRUNC('hour', login_time) as hour")
            .SelectRaw("COUNT(*) as total_logins")
            .SelectRaw("COUNT(DISTINCT player_id) as unique_players")
            .SelectRaw("SUM(CASE WHEN login_result = 'SUCCESS' THEN 1 ELSE 0 END) as successful_logins")
            .SelectRaw("SUM(CASE WHEN login_result = 'FAILED' THEN 1 ELSE 0 END) as failed_logins")
            .SelectRaw("ROUND(100.0 * SUM(CASE WHEN login_result = 'SUCCESS' THEN 1 ELSE 0 END) / COUNT(*), 2) as success_rate")
            .GroupByRaw("DATE_TRUNC('hour', login_time)")
            .OrderByRaw("hour")
            .GetAsync();

        return results;
    }

    // 서버별 부하 분석
    public async Task<IEnumerable<dynamic>> GetServerLoadStats(DateTime startTime)
    {
        using var db = CreateQueryFactory();

        var results = await db.Query("player_login_logs")
            .Where("login_time", ">=", startTime)
            .Select("server_id")
            .SelectRaw("COUNT(*) as login_attempts")
            .SelectRaw("COUNT(DISTINCT player_id) as unique_players")
            .SelectRaw("AVG(CASE WHEN login_result = 'SUCCESS' THEN 1.0 ELSE 0.0 END) as success_rate")
            .GroupBy("server_id")
            .OrderBy("server_id")
            .GetAsync();

        return results;
    }

    // 디바이스 타입별 분포
    public async Task<IEnumerable<dynamic>> GetDeviceDistribution(DateTime startTime)
    {
        using var db = CreateQueryFactory();

        var results = await db.Query("player_login_logs")
            .Where("login_time", ">=", startTime)
            .Select("device_type")
            .SelectRaw("COUNT(*) as login_count")
            .SelectRaw("COUNT(DISTINCT player_id) as unique_players")
            .SelectRaw("ROUND(100.0 * COUNT(*) / SUM(COUNT(*)) OVER (), 2) as percentage")
            .GroupBy("device_type")
            .OrderByDesc("login_count")
            .GetAsync();

        return results;
    }

    // 일주일 추세 분석
    public async Task<IEnumerable<dynamic>> GetWeeklyTrend()
    {
        using var db = CreateQueryFactory();

        var weekAgo = DateTime.UtcNow.Date.AddDays(-7);

        var results = await db.Query("player_login_logs")
            .Where("login_time", ">=", weekAgo)
            .SelectRaw("DATE_TRUNC('day', login_time) as day")
            .SelectRaw("COUNT(*) as total_logins")
            .SelectRaw("COUNT(DISTINCT player_id) as dau")
            .SelectRaw("COUNT(DISTINCT CASE WHEN device_type = 'Android' THEN player_id END) as android_users")
            .SelectRaw("COUNT(DISTINCT CASE WHEN device_type = 'iOS' THEN player_id END) as ios_users")
            .SelectRaw("COUNT(DISTINCT CASE WHEN device_type = 'PC' THEN player_id END) as pc_users")
            .GroupByRaw("DATE_TRUNC('day', login_time)")
            .OrderByRaw("day")
            .GetAsync();

        return results;
    }

    // 실전: 일일 리포트 생성
    public async Task GenerateDailyReport(DateTime date)
    {
        using var db = CreateQueryFactory();

        var startOfDay = date.Date;
        var endOfDay = startOfDay.AddDays(1);

        // 전체 통계
        var totalLogins = await GetTotalLoginCount(startOfDay, endOfDay);
        var dau = await GetDailyActiveUsers(date);
        
        // 시간대별 통계
        var hourlyStats = await GetHourlyStats(date);
        
        // 디바이스 분포
        var deviceDist = await GetDeviceDistribution(startOfDay);

        // 리포트 출력
        Console.WriteLine($"\n========== {date:yyyy-MM-dd} 일일 리포트 ==========");
        Console.WriteLine($"총 로그인 시도: {totalLogins:N0}건");
        Console.WriteLine($"일일 활성 사용자(DAU): {dau:N0}명");
        Console.WriteLine($"1인당 평균 로그인: {(double)totalLogins / dau:F2}회\n");

        Console.WriteLine("--- 시간대별 접속 현황 ---");

            Console.WriteLine($"{stat.hour:HH:00} - 접속 {stat.total_logins}건, " +
                            $"고유 유저 {stat.unique_players}명, 성공률 {stat.success_rate}%");
        }

        Console.WriteLine("\n--- 디바이스 분포 ---");
        foreach (var device in deviceDist)
        {
            Console.WriteLine($"{device.device_type}: {device.login_count}건 ({device.percentage}%)");
        }
    }
}
```

## 4.7 비동기 처리: async/await 패턴
게임 서버는 동시에 수많은 요청을 처리해야 한다. 비동기 프로그래밍은 필수다. SQLKata는 모든 데이터베이스 작업에 비동기 메서드를 제공한다.  
  
**비동기 처리의 핵심 개념:**

```csharp
public class AsyncPatternService
{
    private readonly string _connectionString;

    public AsyncPatternService(string connectionString)
    {
        _connectionString = connectionString;
    }

    private QueryFactory CreateQueryFactory()
    {
        var connection = new NpgsqlConnection(_connectionString);
        var compiler = new PostgresCompiler();
        return new QueryFactory(connection, compiler);
    }

    // 기본 비동기 조회
    public async Task<IEnumerable<PlayerLoginLog>> GetRecentLoginsAsync(int count)
    {
        using var db = CreateQueryFactory();

        // GetAsync는 비동기로 실행되며, 스레드를 블록하지 않는다
        var results = await db.Query("player_login_logs")
            .OrderByDesc("login_time")
            .Limit(count)
            .GetAsync<PlayerLoginLog>();

        return results;
    }

    // 여러 쿼리를 병렬로 실행
    public async Task<(int todayLogins, int yesterdayLogins, int weekLogins)> 
        GetMultiPeriodStatsAsync()
    {
        var today = DateTime.UtcNow.Date;
        var yesterday = today.AddDays(-1);
        var weekAgo = today.AddDays(-7);

        using var db = CreateQueryFactory();

        // 세 개의 쿼리를 동시에 실행
        var todayTask = db.Query("player_login_logs")
            .Where("login_time", ">=", today)
            .CountAsync<int>();

        var yesterdayTask = db.Query("player_login_logs")
            .WhereBetween("login_time", yesterday, today)
            .CountAsync<int>();

        var weekTask = db.Query("player_login_logs")
            .Where("login_time", ">=", weekAgo)
            .CountAsync<int>();

        // 모든 작업이 완료될 때까지 대기
        await Task.WhenAll(todayTask, yesterdayTask, weekTask);

        return (todayTask.Result, yesterdayTask.Result, weekTask.Result);
    }

    // 배치 삽입도 비동기로
    public async Task<int> LogMultipleEventsAsync(List<PlayerLoginLog> events)
    {
        using var db = CreateQueryFactory();

        var data = events.Select(e => new
        {
            login_time = e.login_time,
            player_id = e.player_id,
            ip_address = e.ip_address,
            device_type = e.device_type,
            login_result = e.login_result,
            session_id = e.session_id,
            server_id = e.server_id
        }).ToArray();

        // InsertAsync는 논블로킹 방식으로 실행
        var result = await db.Query("player_login_logs").InsertAsync(data);
        return result;
    }

    // 실전: 백그라운드에서 주기적으로 통계 수집
    public async Task RunPeriodicStatsCollection(CancellationToken cancellationToken)
    {
        while (!cancellationToken.IsCancellationRequested)
        {
            try
            {
                using var db = CreateQueryFactory();

                // 현재 시각 기준 통계
                var stats = await db.Query("player_login_logs")
                    .Where("login_time", ">=", DateTime.UtcNow.AddMinutes(-5))
                    .SelectRaw("COUNT(*) as total")
                    .SelectRaw("COUNT(DISTINCT player_id) as unique_players")
                    .FirstOrDefaultAsync<dynamic>();

                Console.WriteLine($"[{DateTime.Now:HH:mm:ss}] 최근 5분: " +
                                $"로그인 {stats.total}건, 고유 유저 {stats.unique_players}명");

                // 5분마다 실행
                await Task.Delay(TimeSpan.FromMinutes(5), cancellationToken);
            }
            catch (OperationCanceledException)
            {
                Console.WriteLine("통계 수집이 중단되었다");
                break;
            }
            catch (Exception ex)
            {
                Console.WriteLine($"통계 수집 오류: {ex.Message}");
                await Task.Delay(TimeSpan.FromSeconds(30), cancellationToken);
            }
        }
    }

    // 트랜잭션 처리도 비동기로
    public async Task<bool> TransferItemWithTransactionAsync(
        long fromPlayerId, long toPlayerId, int itemId)
    {
        using var connection = new NpgsqlConnection(_connectionString);
        await connection.OpenAsync();

        using var transaction = await connection.BeginTransactionAsync();
        var compiler = new PostgresCompiler();
        var db = new QueryFactory(connection, compiler);

        try
        {
            // 아이템 차감
            await db.Query("item_usage_logs").InsertAsync(new
            {
                usage_time = DateTime.UtcNow,
                player_id = fromPlayerId,
                item_id = itemId,
                quantity = -1,
                usage_type = "TRANSFER_OUT"
            });

            // 아이템 증가
            await db.Query("item_usage_logs").InsertAsync(new
            {
                usage_time = DateTime.UtcNow,
                player_id = toPlayerId,
                item_id = itemId,
                quantity = 1,
                usage_type = "TRANSFER_IN"
            });

            await transaction.CommitAsync();
            return true;
        }
        catch (Exception ex)
        {
            await transaction.RollbackAsync();
            Console.WriteLine($"아이템 이전 실패: {ex.Message}");
            return false;
        }
    }
}
```

**비동기 처리의 주요 이점:**

응답성이 향상된다. UI 스레드나 요청 처리 스레드가 블록되지 않아 애플리케이션이 더 빠르게 응답한다.

확장성이 개선된다. 스레드 풀의 스레드를 효율적으로 사용하여 더 많은 동시 요청을 처리할 수 있다.

리소스 사용이 최적화된다. I/O 작업 중에 스레드가 다른 작업을 처리할 수 있어 CPU 활용률이 높아진다.

**주의사항:**

`async void`는 사용하지 않는다. 이벤트 핸들러가 아닌 이상 항상 `async Task`를 사용한다.

`await` 없이 `Task`를 반환하지 않는다. 예외가 제대로 전파되지 않을 수 있다.

`Task.Wait()`나 `.Result`로 동기적으로 기다리지 않는다. 데드락이 발생할 수 있다.

**실전 게임 서버 이벤트 로거 완성본:**

```csharp
public class GameServerEventLogger
{
    private readonly string _connectionString;
    private readonly SemaphoreSlim _batchSemaphore = new SemaphoreSlim(1, 1);
    private readonly List<PlayerLoginLog> _batchBuffer = new List<PlayerLoginLog>();
    private const int BATCH_SIZE = 100;

    public GameServerEventLogger(string connectionString)
    {
        _connectionString = connectionString;
    }

    private QueryFactory CreateQueryFactory()
    {
        var connection = new NpgsqlConnection(_connectionString);
        var compiler = new PostgresCompiler();
        return new QueryFactory(connection, compiler);
    }

    // 로그인 이벤트 버퍼링 (메모리에 모아두었다가 배치 처리)
    public async Task QueueLoginEvent(PlayerLoginLog loginEvent)
    {
        await _batchSemaphore.WaitAsync();
        try
        {
            _batchBuffer.Add(loginEvent);

            if (_batchBuffer.Count >= BATCH_SIZE)
            {
                await FlushBatchAsync();
            }
        }
        finally
        {
            _batchSemaphore.Release();
        }
    }

    // 버퍼의 내용을 데이터베이스에 저장
    private async Task FlushBatchAsync()
    {
        if (_batchBuffer.Count == 0) return;

        using var db = CreateQueryFactory();

        var data = _batchBuffer.Select(e => new
        {
            login_time = e.login_time,
            player_id = e.player_id,
            ip_address = e.ip_address,
            device_type = e.device_type,
            login_result = e.login_result,
            session_id = e.session_id,
            server_id = e.server_id
        }).ToArray();

        await db.Query("player_login_logs").InsertAsync(data);
        Console.WriteLine($"[{DateTime.Now:HH:mm:ss}] {data.Length}건의 로그를 저장했다");

        _batchBuffer.Clear();
    }

    // 주기적으로 버퍼 플러시 (타이머 기반)
    public async Task StartPeriodicFlush(CancellationToken cancellationToken)
    {
        while (!cancellationToken.IsCancellationRequested)
        {
            await Task.Delay(TimeSpan.FromSeconds(10), cancellationToken);

            await _batchSemaphore.WaitAsync(cancellationToken);
            try
            {
                await FlushBatchAsync();
            }
            finally
            {
                _batchSemaphore.Release();
            }
        }
    }
}
```
