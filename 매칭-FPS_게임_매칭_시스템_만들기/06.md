# 6. 실전 구현 예제

## 6.1 기본 매칭 시스템 구현
FPS 게임을 위한 기본 매칭 시스템은 플레이어의 매칭 요청을 받아 적절한 상대를 찾아 게임 세션을 생성하는 핵심 컴포넌트다. 여기서는 10만 이상의 동시 유저를 처리할 수 있는 확장 가능한 구조를 구현한다.

먼저 플레이어와 매칭 요청을 표현하는 핵심 모델을 정의한다.

```csharp
public class Player
{
    public string PlayerId { get; set; }
    public double MMR { get; set; }
    public int Ping { get; set; }
    public string Region { get; set; }
    public DateTime JoinedQueueAt { get; set; }
    public PlayerPreferences Preferences { get; set; }
}

public class PlayerPreferences
{
    public int MaxPingThreshold { get; set; } = 100; // ms
    public int MaxWaitTime { get; set; } = 120; // seconds
    public GameMode PreferredGameMode { get; set; }
}

public class MatchRequest
{
    public string RequestId { get; set; }
    public Player Player { get; set; }
    public DateTime RequestTime { get; set; }
    public double ExpandedMMRRange { get; set; }

    public MatchRequest(Player player)
    {
        RequestId = Guid.NewGuid().ToString();
        Player = player;
        RequestTime = DateTime.UtcNow;
        ExpandedMMRRange = 100; // 초기 MMR 허용 범위
    }
}

public class Match
{
    public string MatchId { get; set; }
    public List<Player> Team1 { get; set; }
    public List<Player> Team2 { get; set; }
    public double AverageMMR { get; set; }
    public int AveragePing { get; set; }
    public DateTime CreatedAt { get; set; }

    public Match()
    {
        MatchId = Guid.NewGuid().ToString();
        Team1 = new List<Player>();
        Team2 = new List<Player>();
        CreatedAt = DateTime.UtcNow;
    }
}

public enum GameMode
{
    TeamDeathmatch,
    CaptureTheFlag,
    Domination,
    SearchAndDestroy
}
```

### 멀티스레드 환경에서의 중복 매칭 문제
멀티스레드로 매칭 워커를 여러 개 실행하면 동일한 플레이어가 두 워커에 의해 동시에 후보로 선택되어 **중복 매칭**이 발생할 수 있다. 예를 들어 워커 A와 워커 B가 동시에 같은 큐를 처리할 때, 둘 다 같은 플레이어를 후보로 추출하고 각각 별도의 매칭을 성사시키는 상황이 생긴다. 이 문제를 방지하기 위해 아래 세 가지 메커니즘을 조합하여 사용한다.

**첫째, `ConcurrentQueue`의 원자적 Dequeue를 활용한다.** `ConcurrentQueue<T>.TryDequeue()`는 스레드 안전한 원자적 연산으로, 여러 워커가 동시에 호출하더라도 하나의 요청은 반드시 하나의 워커만 꺼낼 수 있다. 큐에서 꺼낸 순간 해당 요소는 큐에서 사라지므로, 다른 워커가 동일한 요소를 꺼낼 수 없다.

**둘째, `_activeRequests`에서 `TryRemove`로 소유권을 원자적으로 획득한다.** `TryRemove` 단일 연산으로 대체하면 확인과 제거가 원자적으로 처리되어 중복이 완전히 방지된다.

**셋째, 매칭 성공 시 즉시 `_activeRequests`에서 해당 플레이어를 제거한다.** 매칭이 확정된 플레이어는 즉시 활성 요청 목록에서 제거되므로, 이후 어떤 워커도 해당 플레이어를 유효한 후보로 인식하지 않는다.

이 세 가지를 조합하면 단일 서버 내 멀티스레드 환경에서의 중복 매칭을 효과적으로 방지할 수 있다. 단, 분산 환경(다중 서버)에서는 6.3절에서 설명하는 Redis 분산 락이 추가로 필요하다.

```csharp
public class MatchmakingService
{
    private readonly ConcurrentDictionary<GameMode, ConcurrentQueue<MatchRequest>> _matchQueues;
    private readonly ConcurrentDictionary<string, MatchRequest> _activeRequests;
    private readonly ConcurrentDictionary<string, byte> _restoringRequests;
    private readonly ILogger<MatchmakingService> _logger;
    private readonly MatchmakingConfig _config;
    private CancellationTokenSource _cancellationTokenSource;

    public MatchmakingService(ILogger<MatchmakingService> logger, MatchmakingConfig config)
    {
        _matchQueues = new ConcurrentDictionary<GameMode, ConcurrentQueue<MatchRequest>>();
        _activeRequests = new ConcurrentDictionary<string, MatchRequest>();
        _restoringRequests = new ConcurrentDictionary<string, byte>();
        _logger = logger;
        _config = config;

        // 각 게임 모드별 큐 초기화
        foreach (GameMode mode in Enum.GetValues(typeof(GameMode)))
        {
            _matchQueues[mode] = new ConcurrentQueue<MatchRequest>();
        }
    }

    public async Task StartMatchmakingAsync()
    {
        _cancellationTokenSource = new CancellationTokenSource();
        var token = _cancellationTokenSource.Token;

        // 여러 매칭 워커를 병렬로 실행하여 처리량 증가
        var workers = new List<Task>();
        for (int i = 0; i < _config.MatchmakingWorkerCount; i++)
        {
            workers.Add(Task.Run(() => MatchmakingWorkerAsync(token), token));
        }

        // 대기 시간에 따른 범위 확장 워커
        workers.Add(Task.Run(() => RangeExpansionWorkerAsync(token), token));

        await Task.WhenAll(workers);
    }

    public async Task<string> EnqueuePlayerAsync(Player player, GameMode gameMode)
    {
        var request = new MatchRequest(player)
        {
            Player = player
        };

        player.Preferences.PreferredGameMode = gameMode;

        if (!_matchQueues.TryGetValue(gameMode, out var queue))
        {
            throw new InvalidOperationException($"Invalid game mode: {gameMode}");
        }

        _activeRequests[request.RequestId] = request;
        queue.Enqueue(request);

        _logger.LogInformation(
            "Player {PlayerId} enqueued for {GameMode}. MMR: {MMR}, Queue size: {QueueSize}",
            player.PlayerId, gameMode, player.MMR, queue.Count);

        return request.RequestId;
    }

    public async Task<bool> DequeuePlayerAsync(string requestId)
    {
        if (_activeRequests.TryRemove(requestId, out var request))
        {
            _logger.LogInformation("Player request {RequestId} removed from queue", requestId);
            return true;
        }
        return false;
    }

    private async Task MatchmakingWorkerAsync(CancellationToken cancellationToken)
    {
        while (!cancellationToken.IsCancellationRequested)
        {
            try
            {
                foreach (var gameMode in _matchQueues.Keys)
                {
                    await ProcessMatchQueueAsync(gameMode, cancellationToken);
                }

                // CPU 과부하 방지를 위한 짧은 대기
                await Task.Delay(_config.WorkerTickInterval, cancellationToken);
            }
            catch (OperationCanceledException)
            {
                // 정상 종료 시 발생하는 예외이므로 별도 처리 없이 루프 종료
                break;
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "Error in matchmaking worker");
            }
        }
    }

    private async Task ProcessMatchQueueAsync(GameMode gameMode, CancellationToken cancellationToken)
    {
        if (!_matchQueues.TryGetValue(gameMode, out var queue) || queue.IsEmpty)
            return;

        var candidateRequests = new List<MatchRequest>();
        var playersNeeded = _config.PlayersPerMatch; // 예: 10명 (5v5)

        // 큐에서 후보 추출
        // ConcurrentQueue.TryDequeue()는 원자적 연산이므로 동일 요청을
        // 두 워커가 동시에 꺼내는 것은 불가능하다.
        // 하지만 꺼낸 후 activeRequests 확인이 필요하다:
        // 이미 타임아웃 처리되어 activeRequests에서 제거된 요청이
        // 큐에 남아 있을 수 있기 때문이다.
        while (candidateRequests.Count < playersNeeded * 3 && queue.TryDequeue(out var request))
        {
            // TryRemove는 확인과 제거를 단일 원자적 연산으로 처리하므로
            // 한 번 TryRemove에 성공한 워커만 해당 플레이어를 후보로 사용한다.
            if (_activeRequests.TryRemove(request.RequestId, out var activeRequest))
            {
                candidateRequests.Add(activeRequest);
            }
            // TryRemove 실패 = 이미 다른 워커가 처리했거나 타임아웃으로 제거됨 → 건너뜀
        }

        if (candidateRequests.Count < playersNeeded)
        {
            // 후보 부족 - 대기 시간이 긴 플레이어가 다음 틱에서 먼저 처리될 수 있도록
            // RequestTime 오름차순(오래된 순)으로 정렬하여 복원한다.
            // ConcurrentQueue는 Enqueue 순서대로 Dequeue되므로
            // 오래 기다린 플레이어를 먼저 넣으면 다음 틱에서 먼저 추출된다.
            var orderedByWait = candidateRequests
                .OrderBy(r => r.RequestTime)
                .ToList();

            RestoreCandidates(orderedByWait, queue);

            _logger.LogDebug(
                "Not enough candidates for {GameMode}. Have {Count}, need {Need}. Restored to queue.",
                gameMode, candidateRequests.Count, playersNeeded);
            return;
        }

        // 매칭 시도
        var match = TryCreateMatch(candidateRequests, playersNeeded);

        if (match != null)
        {
            // 매칭 성공 - 매칭에 사용된 플레이어는 이미 TryRemove로 activeRequests에서
            // 제거되었으므로 추가 제거 작업이 필요 없다.
            // 매칭에 포함되지 않은 나머지 후보는 대기 시간 오름차순으로 정렬하여 복원한다.
            var matchedPlayerIds = match.Team1.Concat(match.Team2)
                .Select(p => p.PlayerId)
                .ToHashSet();

            var unmatched = candidateRequests
                .Where(r => !matchedPlayerIds.Contains(r.Player.PlayerId))
                .OrderBy(r => r.RequestTime) // 오래된 순 정렬 후 복원
                .ToList();

            RestoreCandidates(unmatched, queue);

            await OnMatchCreatedAsync(match);

            _logger.LogInformation(
                "Match {MatchId} created with {PlayerCount} players. Avg MMR: {AvgMMR}",
                match.MatchId, match.Team1.Count + match.Team2.Count, match.AverageMMR);
        }
        else
        {
            // 매칭 조건 불만족 - 동일 조합의 반복 실패를 줄이기 위해
            // 오래된 요청을 앞에 배치하여 다음 틱에서 RangeExpansionWorker가
            // 확장한 범위가 적용될 가능성을 높인다.
            var orderedByWait = candidateRequests
                .OrderBy(r => r.RequestTime)
                .ToList();

            RestoreCandidates(orderedByWait, queue);

            _logger.LogDebug(
                "Match creation failed for {GameMode} with {Count} candidates. Restored to queue.",
                gameMode, candidateRequests.Count);
        }
    }

    /// <summary>
    /// 후보 요청을 activeRequests와 큐에 안전하게 복원한다.
    /// _restoringRequests를 이용해 동일 요청이 중복으로 큐에 삽입되는 것을 방지한다.
    /// </summary>
    private void RestoreCandidates(
        List<MatchRequest> candidates,
        ConcurrentQueue<MatchRequest> queue)
    {
        foreach (var req in candidates)
        {
            // TryAdd로 복원 중 여부를 원자적으로 등록한다.
            // 다른 워커가 이미 같은 요청을 복원 중이면 TryAdd가 false를 반환하므로
            // 중복 삽입이 발생하지 않는다.
            if (!_restoringRequests.TryAdd(req.RequestId, 0))
            {
                // 이미 다른 워커가 이 요청을 복원 중 → 건너뜀
                continue;
            }

            try
            {
                _activeRequests[req.RequestId] = req;
                queue.Enqueue(req);
            }
            finally
            {
                // 복원 완료 후 반드시 복원 추적 제거
                // finally 블록으로 예외 발생 시에도 항상 정리되도록 보장한다.
                _restoringRequests.TryRemove(req.RequestId, out _);
            }
        }
    }

    private Match TryCreateMatch(List<MatchRequest> candidates, int playersNeeded)
    {
        // MMR 기준으로 정렬
        var sortedCandidates = candidates
            .OrderBy(r => r.Player.MMR)
            .ToList();

        // 가능한 조합 중 최적의 매칭 찾기
        for (int i = 0; i <= sortedCandidates.Count - playersNeeded; i++)
        {
            var selectedPlayers = sortedCandidates
                .Skip(i)
                .Take(playersNeeded)
                .ToList();

            // MMR 범위 체크
            var mmrRange = selectedPlayers.Max(r => r.Player.MMR) -
                          selectedPlayers.Min(r => r.Player.MMR);

            var maxAllowedRange = selectedPlayers.Min(r => r.ExpandedMMRRange);

            if (mmrRange > maxAllowedRange)
                continue;

            // 핑 체크
            var avgPing = selectedPlayers.Average(r => r.Player.Ping);
            if (avgPing > _config.MaxAveragePing)
                continue;

            // 팀 밸런싱
            var match = CreateBalancedTeams(selectedPlayers.Select(r => r.Player).ToList());

            if (IsMatchBalanced(match))
            {
                return match;
            }
        }

        return null;
    }

    private Match CreateBalancedTeams(List<Player> players)
    {
        var match = new Match();

        // MMR 기준 내림차순 정렬
        var sortedPlayers = players.OrderByDescending(p => p.MMR).ToList();

        // 지그재그 방식으로 팀 배정 (1,4,5,8,9 vs 2,3,6,7,10)
        for (int i = 0; i < sortedPlayers.Count; i++)
        {
            if (i % 4 < 2)
                match.Team1.Add(sortedPlayers[i]);
            else
                match.Team2.Add(sortedPlayers[i]);
        }

        match.AverageMMR = players.Average(p => p.MMR);
        match.AveragePing = (int)players.Average(p => p.Ping);

        return match;
    }

    private bool IsMatchBalanced(Match match)
    {
        var team1Avg = match.Team1.Average(p => p.MMR);
        var team2Avg = match.Team2.Average(p => p.MMR);
        var difference = Math.Abs(team1Avg - team2Avg);

        return difference <= _config.MaxTeamMMRDifference;
    }

    private async Task RangeExpansionWorkerAsync(CancellationToken cancellationToken)
    {
        while (!cancellationToken.IsCancellationRequested)
        {
            try
            {
                foreach (var request in _activeRequests.Values)
                {
                    var waitTime = (DateTime.UtcNow - request.RequestTime).TotalSeconds;

                    // 대기 시간에 따라 MMR 범위 확장
                    if (waitTime > 30)
                    {
                        request.ExpandedMMRRange = 100 + (waitTime - 30) * 5;
                    }

                    // 최대 대기 시간 초과 시 처리
                    if (waitTime > request.Player.Preferences.MaxWaitTime)
                    {
                        await HandleTimeoutAsync(request);
                    }
                }

                await Task.Delay(5000, cancellationToken); // 5초마다 체크
            }
            catch (OperationCanceledException)
            {
                // 정상 종료 시 발생하는 예외이므로 별도 처리 없이 루프 종료
                break;
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "Error in range expansion worker");
            }
        }
    }

    private async Task HandleTimeoutAsync(MatchRequest request)
    {
        _logger.LogWarning(
            "Matchmaking timeout for player {PlayerId} after {WaitTime} seconds",
            request.Player.PlayerId,
            (DateTime.UtcNow - request.RequestTime).TotalSeconds);

        await DequeuePlayerAsync(request.RequestId);
        await OnMatchmakingTimeoutAsync(request.Player);
    }

    protected virtual async Task OnMatchCreatedAsync(Match match)
    {
        // 매칭 완료 이벤트 처리
        // 게임 서버에 세션 생성 요청 등
        await Task.CompletedTask;
    }

    protected virtual async Task OnMatchmakingTimeoutAsync(Player player)
    {
        // 타임아웃 이벤트 처리
        await Task.CompletedTask;
    }

    public void Stop()
    {
        _cancellationTokenSource?.Cancel();
    }
}


public class MatchmakingConfig
{
    public int MatchmakingWorkerCount { get; set; } = 10;
    public int WorkerTickInterval { get; set; } = 100; // ms
    public int PlayersPerMatch { get; set; } = 10; // 5v5
    public int MaxAveragePing { get; set; } = 150; // ms
    public double MaxTeamMMRDifference { get; set; } = 50;
}
```

### 멀티스레드 중복 매칭 방지 메커니즘 요약
단일 서버 내 멀티스레드 환경에서 중복 매칭을 방지하는 세 가지 핵심 원칙은 다음과 같다.

**첫째, `ConcurrentQueue.TryDequeue()`로 큐 단계에서 1차 분리한다.** 원자적 연산이므로 동일 요소를 두 워커가 동시에 가져가는 것 자체가 불가능하다.

**둘째, `ConcurrentDictionary.TryRemove()`로 소유권을 원자적으로 획득한다.** 큐에서 꺼낸 요청이 아직 유효한지 확인하는 동시에 소유권을 단일 워커에게 귀속시킨다. 이 연산에 성공한 워커만 해당 플레이어를 매칭에 사용할 수 있다.

**셋째, 미사용 후보를 반드시 복원한다.** `TryRemove`로 `_activeRequests`에서 꺼낸 요청이 최종 매칭에 포함되지 않으면, 반드시 `_activeRequests`에 재삽입하고 큐에도 다시 넣어 다음 매칭 기회를 보장해야 한다.

## 6.2 MMR 계산 로직 구현
MMR은 플레이어의 실력을 수치화하는 핵심 요소다. FPS 게임에서는 단순한 승패뿐만 아니라 개인 기여도도 반영하는 복합적인 계산이 필요하다.

```csharp
public class MMRCalculator
{
    private readonly MMRConfig _config;
    private readonly ILogger<MMRCalculator> _logger;

    public MMRCalculator(MMRConfig config, ILogger<MMRCalculator> logger)
    {
        _config = config;
        _logger = logger;
    }

    // ELO 기반 기본 MMR 계산
    public double CalculateNewMMR(
        double currentMMR,
        double opponentAverageMMR,
        bool won,
        PlayerPerformance performance)
    {
        // 기대 승률 계산
        double expectedScore = CalculateExpectedScore(currentMMR, opponentAverageMMR);

        // 실제 결과 (1.0 = 승리, 0.0 = 패배)
        double actualScore = won ? 1.0 : 0.0;

        // K-Factor 계산 (경기 수에 따라 조정)
        double kFactor = CalculateKFactor(currentMMR, performance.MatchesPlayed);

        // 기본 MMR 변화량
        double baseChange = kFactor * (actualScore - expectedScore);

        // 개인 성과 보정
        double performanceMultiplier = CalculatePerformanceMultiplier(performance);

        // 최종 MMR 변화량
        double mmrChange = baseChange * performanceMultiplier;

        // 변화량 제한
        mmrChange = Math.Clamp(mmrChange, -_config.MaxMMRChange, _config.MaxMMRChange);

        double newMMR = currentMMR + mmrChange;

        _logger.LogInformation(
            "MMR Update - Current: {Current}, Change: {Change:F2}, New: {New:F2}, Win: {Win}",
            currentMMR, mmrChange, newMMR, won);

        return newMMR;
    }

    private double CalculateExpectedScore(double playerMMR, double opponentMMR)
    {
        // ELO 공식
        return 1.0 / (1.0 + Math.Pow(10, (opponentMMR - playerMMR) / 400.0));
    }

    private double CalculateKFactor(double currentMMR, int matchesPlayed)
    {
        // 초보자는 높은 K-Factor (빠른 MMR 변화)
        if (matchesPlayed < _config.NewPlayerMatchThreshold)
        {
            return _config.NewPlayerKFactor;
        }

        // 고랭크는 낮은 K-Factor (안정적인 MMR)
        if (currentMMR >= _config.HighRankMMRThreshold)
        {
            return _config.HighRankKFactor;
        }

        return _config.DefaultKFactor;
    }

    private double CalculatePerformanceMultiplier(PlayerPerformance performance)
    {
        double multiplier = 1.0;

        // K/D 비율 보정
        double kdRatio = performance.Kills / Math.Max(performance.Deaths, 1.0);
        if (kdRatio > _config.ExcellentKDThreshold)
        {
            multiplier += _config.KDRBonusMultiplier;
        }
        else if (kdRatio < _config.PoorKDThreshold)
        {
            multiplier -= _config.KDRPenaltyMultiplier;
        }

        // 목표 기여도 보정
        double objectiveScore = performance.ObjectivePoints /
            Math.Max(performance.TeamTotalObjectivePoints, 1.0);

        if (objectiveScore > 0.3) // 팀 목표의 30% 이상 기여
        {
            multiplier += _config.ObjectiveBonusMultiplier;
        }

        // 팀워크 보정 (어시스트)
        if (performance.Assists > performance.Kills * 0.5)
        {
            multiplier += _config.TeamworkBonusMultiplier;
        }

        // 최종 배율 제한
        return Math.Clamp(multiplier, _config.MinPerformanceMultiplier,
            _config.MaxPerformanceMultiplier);
    }

    // Glicko-2 기반 신뢰도를 포함한 MMR 계산
    public (double NewRating, double NewDeviation, double NewVolatility)
        CalculateGlicko2(
            double rating,
            double ratingDeviation,
            double volatility,
            List<GameResult> results)
    {
        // Glicko-2 스케일로 변환
        double mu = (rating - 1500) / 173.7178;
        double phi = ratingDeviation / 173.7178;

        // v 계산 (예상 분산의 역수)
        double v = 0;
        foreach (var result in results)
        {
            double opponentMu = (result.OpponentRating - 1500) / 173.7178;
            double opponentPhi = result.OpponentDeviation / 173.7178;

            double g = GFunction(opponentPhi);
            double e = EFunction(mu, opponentMu, opponentPhi);

            v += Math.Pow(g, 2) * e * (1 - e);
        }
        v = 1.0 / v;

        // Delta 계산
        double delta = 0;
        foreach (var result in results)
        {
            double opponentMu = (result.OpponentRating - 1500) / 173.7178;
            double opponentPhi = result.OpponentDeviation / 173.7178;

            double g = GFunction(opponentPhi);
            double e = EFunction(mu, opponentMu, opponentPhi);
            double s = result.Won ? 1.0 : 0.0;

            delta += g * (s - e);
        }
        delta *= v;

        // 새로운 volatility 계산
        double newVolatility = CalculateNewVolatility(phi, v, delta, volatility);

        // 새로운 rating deviation 계산
        double phiStar = Math.Sqrt(Math.Pow(phi, 2) + Math.Pow(newVolatility, 2));
        double newPhi = 1.0 / Math.Sqrt(1.0 / Math.Pow(phiStar, 2) + 1.0 / v);

        // 새로운 rating 계산
        double newMu = mu;
        foreach (var result in results)
        {
            double opponentMu = (result.OpponentRating - 1500) / 173.7178;
            double opponentPhi = result.OpponentDeviation / 173.7178;

            double g = GFunction(opponentPhi);
            double e = EFunction(mu, opponentMu, opponentPhi);
            double s = result.Won ? 1.0 : 0.0;

            newMu += Math.Pow(newPhi, 2) * g * (s - e);
        }

        // 원래 스케일로 변환
        double newRating = newMu * 173.7178 + 1500;
        double newDeviation = newPhi * 173.7178;

        return (newRating, newDeviation, newVolatility);
    }

    private double GFunction(double phi)
    {
        return 1.0 / Math.Sqrt(1.0 + 3.0 * Math.Pow(phi, 2) / Math.Pow(Math.PI, 2));
    }

    private double EFunction(double mu, double opponentMu, double opponentPhi)
    {
        double g = GFunction(opponentPhi);
        return 1.0 / (1.0 + Math.Exp(-g * (mu - opponentMu)));
    }

    private double CalculateNewVolatility(double phi, double v, double delta,
        double volatility)
    {
        double tau = _config.VolatilityChangeConstraint;
        double a = Math.Log(Math.Pow(volatility, 2));
        double epsilon = 0.000001;

        double A = a;
        double B;

        if (Math.Pow(delta, 2) > Math.Pow(phi, 2) + v)
        {
            B = Math.Log(Math.Pow(delta, 2) - Math.Pow(phi, 2) - v);
        }
        else
        {
            int k = 1;
            while (F(a - k * tau, delta, phi, v, a) < 0)
            {
                k++;
            }
            B = a - k * tau;
        }

        double fA = F(A, delta, phi, v, a);
        double fB = F(B, delta, phi, v, a);

        while (Math.Abs(B - A) > epsilon)
        {
            double C = A + (A - B) * fA / (fB - fA);
            double fC = F(C, delta, phi, v, a);

            if (fC * fB < 0)
            {
                A = B;
                fA = fB;
            }
            else
            {
                fA = fA / 2.0;
            }

            B = C;
            fB = fC;
        }

        return Math.Exp(A / 2.0);
    }

    private double F(double x, double delta, double phi, double v, double a)
    {
        double ex = Math.Exp(x);
        double num1 = ex * (Math.Pow(delta, 2) - Math.Pow(phi, 2) - v - ex);
        double denom1 = 2.0 * Math.Pow(Math.Pow(phi, 2) + v + ex, 2);

        double tau = _config.VolatilityChangeConstraint;
        double num2 = x - a;
        double denom2 = Math.Pow(tau, 2);

        return num1 / denom1 - num2 / denom2;
    }
}
```

## 6.3 앞 구현의 문제와 개선

### 현재 구현의 근본적 한계
현재 방식은 **큐를 순서대로 훑어서 묶는 단순 배치(Batch) 방식**이다. 이 방식의 문제를 구체적으로 살펴보면 다음과 같다.

```
큐 상태 (MMR 순서가 뒤섞인 실제 상황):
[A:1500] [B:2800] [C:1520] [D:2750] [E:1510] [F:2900] ...

3배수(30명)를 앞에서 순서대로 꺼내면:
→ MMR 1500~2900이 섞인 후보군 형성
→ TryCreateMatch에서 슬라이딩 윈도우로 찾지만
→ 1500대끼리 10명, 2800대끼리 10명이 각각 큐에 있어도
→ 이번 배치에서는 절대 만날 수 없음
→ MMR 범위 초과로 매칭 실패 → 전원 복원 → 반복
```
  
**아래는 실제 서비스에서 사용하는 접근 방식이다.**  
  
### 방식 1: MMR 버킷 분리 큐 (가장 보편적)
Riot Games(리그 오브 레전드), Valve(CS2, Dota 2) 등 대부분의 대형 게임이 사용하는 방식이다. 큐 자체를 MMR 구간별로 처음부터 분리하여 관리한다.

```
큐를 MMR 구간별로 분리:
Bronze  큐: [A:1100] [B:1050] [C:1200] ...
Silver  큐: [D:1500] [E:1480] [F:1550] ...
Gold    큐: [G:1800] [H:1820] [I:1790] ...
Platinum큐: [J:2100] [K:2050] ...

각 구간 큐에서 독립적으로 매칭 시도
→ 같은 구간끼리만 묶이므로 처음부터 MMR 조건 충족 가능성 극대화
→ 인원 부족 시 인접 구간 큐와 병합
```

### 방식 2: 플레이어 중심 능동 탐색 (Active Search)
순서대로 꺼내는 것이 아니라 **각 플레이어가 자신의 MMR 범위에 맞는 상대를 능동적으로 탐색**하는 방식이다. Blizzard의 Overwatch, Epic의 Fortnite가 이 변형 방식을 사용한다.

```
A(MMR:1500, 허용범위:±100)가 큐에 있을 때:
→ 큐 전체를 탐색하여 MMR 1400~1600인 플레이어 목록 수집
→ 그 중 핑 조건도 만족하는 플레이어만 후보군 형성
→ 후보군이 충분하면 매칭 확정

시간이 지날수록 A의 허용범위 확장:
30초 후: ±150, 60초 후: ±200, 90초 후: ±300 ...
```

### 방식 3: 점수 기반 최적화 매칭 (Score-based Optimization)
Microsoft Azure PlayFab, Amazon GameLift 같은 매칭 플랫폼이 제공하는 방식으로, 모든 대기 플레이어 조합에 **매칭 품질 점수**를 계산하여 가장 점수가 높은 조합을 선택한다.

```
대기 중인 플레이어 전원을 대상으로:
Score(A,B) = f(MMR차이, 핑차이, 대기시간, 팀밸런스)

가장 높은 Score 조합 선택
→ 전역 최적해에 가까운 매칭 가능
→ 단, 플레이어 수가 많아지면 계산량이 급격히 증가
→ 실제로는 후보군을 제한하거나 근사 알고리즘 적용
```

### 개선된 구현: MMR 버킷 분리 + 능동 탐색 조합
실제 서비스에 가장 가까운 방식으로 현재 코드를 재설계한다.

```csharp
public class MatchmakingService
{
    // 게임 모드 → MMR 버킷 → 해당 버킷의 요청 목록
    // 큐를 처음부터 MMR 구간별로 분리하여 관리한다.
    // SortedList를 사용하여 버킷 내에서 대기 시간 순 정렬을 유지한다.
    private readonly ConcurrentDictionary<GameMode,
        ConcurrentDictionary<int, SortedList<DateTime, MatchRequest>>> _mmrBucketQueues;

    private readonly ConcurrentDictionary<string, MatchRequest> _activeRequests;
    private readonly ConcurrentDictionary<string, byte> _restoringRequests;
    private readonly ILogger<MatchmakingService> _logger;
    private readonly MatchmakingConfig _config;
    private CancellationTokenSource _cancellationTokenSource;

    // MMR 버킷 크기: 200 단위로 구간 분리
    // 예: 0~199, 200~399, 400~599 ... 2800~2999, 3000+
    private const int BucketSize = 200;

    public MatchmakingService(ILogger<MatchmakingService> logger, MatchmakingConfig config)
    {
        _mmrBucketQueues = new ConcurrentDictionary<GameMode,
            ConcurrentDictionary<int, SortedList<DateTime, MatchRequest>>>();
        _activeRequests = new ConcurrentDictionary<string, MatchRequest>();
        _restoringRequests = new ConcurrentDictionary<string, byte>();
        _logger = logger;
        _config = config;

        foreach (GameMode mode in Enum.GetValues(typeof(GameMode)))
        {
            _mmrBucketQueues[mode] =
                new ConcurrentDictionary<int, SortedList<DateTime, MatchRequest>>();
        }
    }

    /// <summary>
    /// 플레이어의 MMR에 해당하는 버킷 인덱스를 반환한다.
    /// 예: MMR 1523 → 버킷 7 (1400~1599 구간)
    /// </summary>
    private int GetBucketIndex(double mmr)
    {
        return (int)(mmr / BucketSize);
    }

    public async Task<string> EnqueuePlayerAsync(Player player, GameMode gameMode)
    {
        var request = new MatchRequest(player);
        player.Preferences.PreferredGameMode = gameMode;

        if (!_mmrBucketQueues.TryGetValue(gameMode, out var buckets))
            throw new InvalidOperationException($"Invalid game mode: {gameMode}");

        int bucketIndex = GetBucketIndex(player.MMR);

        // 해당 버킷이 없으면 새로 생성
        var bucket = buckets.GetOrAdd(
            bucketIndex,
            _ => new SortedList<DateTime, MatchRequest>());

        // activeRequests 등록 후 버킷에 삽입
        // SortedList의 키를 RequestTime으로 사용하여 대기 시간 순 정렬을 자동 유지한다.
        _activeRequests[request.RequestId] = request;

        lock (bucket)
        {
            // 극히 드물게 같은 DateTime이 충돌할 수 있으므로 Ticks 단위 미세 조정
            var key = request.RequestTime;
            while (bucket.ContainsKey(key))
                key = key.AddTicks(1);

            bucket.Add(key, request);
        }

        _logger.LogInformation(
            "Player {PlayerId} enqueued. MMR: {MMR}, Bucket: {Bucket}, GameMode: {GameMode}",
            player.PlayerId, player.MMR, bucketIndex, gameMode);

        return request.RequestId;
    }

    public async Task<bool> DequeuePlayerAsync(string requestId)
    {
        if (!_activeRequests.TryRemove(requestId, out var request))
            return false;

        // 버킷에서도 제거
        RemoveFromBucket(request);

        _logger.LogInformation(
            "Player {PlayerId} dequeued from matchmaking.", request.Player.PlayerId);
        return true;
    }

    private void RemoveFromBucket(MatchRequest request)
    {
        var gameMode = request.Player.Preferences.PreferredGameMode;

        if (!_mmrBucketQueues.TryGetValue(gameMode, out var buckets))
            return;

        int bucketIndex = GetBucketIndex(request.Player.MMR);

        if (!buckets.TryGetValue(bucketIndex, out var bucket))
            return;

        lock (bucket)
        {
            // RequestId로 해당 항목을 찾아 제거
            var keyToRemove = bucket
                .FirstOrDefault(kv => kv.Value.RequestId == request.RequestId).Key;

            if (keyToRemove != default)
                bucket.Remove(keyToRemove);
        }
    }

    public async Task StartMatchmakingAsync()
    {
        _cancellationTokenSource = new CancellationTokenSource();
        var token = _cancellationTokenSource.Token;

        var workers = new List<Task>();
        for (int i = 0; i < _config.MatchmakingWorkerCount; i++)
        {
            workers.Add(Task.Run(() => MatchmakingWorkerAsync(token), token));
        }

        workers.Add(Task.Run(() => RangeExpansionWorkerAsync(token), token));

        await Task.WhenAll(workers);
    }

    private async Task MatchmakingWorkerAsync(CancellationToken cancellationToken)
    {
        while (!cancellationToken.IsCancellationRequested)
        {
            try
            {
                foreach (var gameMode in _mmrBucketQueues.Keys)
                {
                    await ProcessAllBucketsAsync(gameMode, cancellationToken);
                }

                await Task.Delay(_config.WorkerTickInterval, cancellationToken);
            }
            catch (OperationCanceledException)
            {
                break;
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "Error in matchmaking worker");
            }
        }
    }

    private async Task ProcessAllBucketsAsync(
        GameMode gameMode,
        CancellationToken cancellationToken)
    {
        if (!_mmrBucketQueues.TryGetValue(gameMode, out var buckets))
            return;

        // 각 버킷을 독립적으로 처리한다.
        // 버킷 내 플레이어는 이미 MMR 구간이 유사하므로
        // 처음부터 조건 충족 가능성이 높다.
        foreach (var (bucketIndex, bucket) in buckets)
        {
            if (cancellationToken.IsCancellationRequested)
                break;

            await ProcessBucketAsync(gameMode, bucketIndex, bucket, buckets);
        }
    }

    private async Task ProcessBucketAsync(
        GameMode gameMode,
        int bucketIndex,
        SortedList<DateTime, MatchRequest> bucket,
        ConcurrentDictionary<int, SortedList<DateTime, MatchRequest>> allBuckets)
    {
        List<MatchRequest> candidates;

        lock (bucket)
        {
            if (bucket.Count == 0)
                return;

            // 버킷 내 후보를 대기 시간 순(오래된 순)으로 수집한다.
            // SortedList가 RequestTime 기준으로 정렬되어 있으므로
            // Values를 순서대로 읽으면 자동으로 대기 시간 순이 된다.
            candidates = bucket.Values
                .Where(r => _activeRequests.ContainsKey(r.RequestId))
                .ToList();
        }

        if (candidates.Count < _config.PlayersPerMatch)
        {
            // 현재 버킷 인원 부족 시 인접 버킷 병합 탐색
            // 대기 시간이 길어진 플레이어는 ExpandedMMRRange가 넓어져 있으므로
            // 인접 버킷과의 병합 매칭이 성공할 가능성이 높아진다.
            candidates = CollectCandidatesFromAdjacentBuckets(
                gameMode, bucketIndex, allBuckets, _config.PlayersPerMatch * 2);

            if (candidates.Count < _config.PlayersPerMatch)
            {
                _logger.LogDebug(
                    "Bucket {Bucket} for {GameMode}: only {Count} candidates. Skipping.",
                    bucketIndex, gameMode, candidates.Count);
                return;
            }
        }

        // 후보군에서 최적 매칭 탐색
        // 현재 구현(슬라이딩 윈도우)과 달리 모든 후보를 대상으로
        // MMR 조건을 능동적으로 평가하여 최적 조합을 선택한다.
        var match = FindBestMatch(candidates, _config.PlayersPerMatch);

        if (match != null)
        {
            // 매칭 확정: 선택된 플레이어를 activeRequests와 버킷에서 제거
            foreach (var player in match.Team1.Concat(match.Team2))
            {
                if (_activeRequests.TryRemove(
                    candidates.First(r => r.Player.PlayerId == player.PlayerId).RequestId,
                    out var removed))
                {
                    RemoveFromBucket(removed);
                }
            }

            await OnMatchCreatedAsync(match);

            _logger.LogInformation(
                "Match {MatchId} created from bucket {Bucket}. " +
                "Players: {Count}, Avg MMR: {MMR:F0}, Avg Ping: {Ping}ms",
                match.MatchId, bucketIndex,
                match.Team1.Count + match.Team2.Count,
                match.AverageMMR, match.AveragePing);
        }
        else
        {
            _logger.LogDebug(
                "No valid match found in bucket {Bucket} for {GameMode} " +
                "with {Count} candidates.",
                bucketIndex, gameMode, candidates.Count);
        }
    }

    /// <summary>
    /// 인접 버킷을 탐색하여 후보를 수집한다.
    /// 중심 버킷에서 바깥쪽으로 한 단계씩 확장하며 수집하므로
    /// MMR 차이가 작은 플레이어를 우선적으로 포함한다.
    /// </summary>
    private List<MatchRequest> CollectCandidatesFromAdjacentBuckets(
        GameMode gameMode,
        int centerBucketIndex,
        ConcurrentDictionary<int, SortedList<DateTime, MatchRequest>> allBuckets,
        int maxCandidates)
    {
        var result = new List<MatchRequest>();

        // 중심 버킷부터 시작하여 양쪽으로 한 단계씩 확장
        // offset 0 → 중심, offset 1 → 중심±1, offset 2 → 중심±2 ...
        for (int offset = 0; offset <= _config.MaxBucketSearchRange; offset++)
        {
            var offsetsToCheck = offset == 0
                ? new[] { 0 }
                : new[] { offset, -offset };

            foreach (var delta in offsetsToCheck)
            {
                int targetBucket = centerBucketIndex + delta;

                if (!allBuckets.TryGetValue(targetBucket, out var bucket))
                    continue;

                List<MatchRequest> bucketCandidates;
                lock (bucket)
                {
                    bucketCandidates = bucket.Values
                        .Where(r => _activeRequests.ContainsKey(r.RequestId))
                        .ToList();
                }

                result.AddRange(bucketCandidates);

                if (result.Count >= maxCandidates)
                    return result.Take(maxCandidates).ToList();
            }
        }

        return result;
    }

    /// <summary>
    /// 후보군에서 최적 매칭을 탐색한다.
    /// 단순 슬라이딩 윈도우가 아닌 점수 기반 평가로
    /// MMR, 핑, 팀 밸런스를 종합적으로 고려한 최적 조합을 선택한다.
    /// </summary>
    private Match FindBestMatch(List<MatchRequest> candidates, int playersNeeded)
    {
        // MMR 기준 정렬 (버킷 분리로 이미 유사한 MMR끼리 모여 있음)
        var sorted = candidates
            .OrderBy(r => r.Player.MMR)
            .ToList();

        Match bestMatch = null;
        double bestScore = double.MinValue;

        // 슬라이딩 윈도우로 연속된 MMR 구간의 후보 조합을 평가한다.
        // 버킷으로 사전 분류되어 있으므로 윈도우 내 MMR 차이가 이미 작아
        // 대부분의 윈도우가 조건을 충족할 가능성이 높다.
        for (int i = 0; i <= sorted.Count - playersNeeded; i++)
        {
            var window = sorted.Skip(i).Take(playersNeeded).ToList();

            // MMR 범위 체크
            double mmrRange = window.Max(r => r.Player.MMR) - window.Min(r => r.Player.MMR);
            double maxAllowedRange = window.Min(r => r.ExpandedMMRRange);

            if (mmrRange > maxAllowedRange)
                continue;

            // 핑 체크
            double avgPing = window.Average(r => r.Player.Ping);
            if (avgPing > _config.MaxAveragePing)
                continue;

            // 팀 밸런싱
            var match = CreateBalancedTeams(window.Select(r => r.Player).ToList());
            if (!IsMatchBalanced(match))
                continue;

            // 매칭 품질 점수 계산
            // 더 좋은 조합이 있으면 교체한다.
            double score = CalculateMatchScore(window, match);
            if (score > bestScore)
            {
                bestScore = score;
                bestMatch = match;
            }
        }

        return bestMatch;
    }

    /// <summary>
    /// 매칭 품질 점수를 계산한다. 점수가 높을수록 더 좋은 매칭이다.
    /// MMR 균형, 핑 품질, 대기 시간 공정성을 종합적으로 반영한다.
    /// </summary>
    private double CalculateMatchScore(List<MatchRequest> window, Match match)
    {
        double score = 1000.0;

        // 1. MMR 편차 페널티: 편차가 작을수록 좋은 매칭
        double mmrRange = window.Max(r => r.Player.MMR) - window.Min(r => r.Player.MMR);
        score -= mmrRange * _config.MMRRangePenaltyWeight;

        // 2. 팀 간 MMR 차이 페널티
        double team1Avg = match.Team1.Average(p => p.MMR);
        double team2Avg = match.Team2.Average(p => p.MMR);
        score -= Math.Abs(team1Avg - team2Avg) * _config.TeamMMRDiffPenaltyWeight;

        // 3. 핑 페널티: 평균 핑이 낮을수록 좋은 매칭
        double avgPing = window.Average(r => r.Player.Ping);
        score -= avgPing * _config.PingPenaltyWeight;

        // 4. 대기 시간 보너스: 오래 기다린 플레이어가 포함될수록 가산점
        // 이 보너스가 없으면 조건이 완벽한 신규 플레이어끼리만 계속 매칭되고
        // 오래 기다린 플레이어는 계속 후순위로 밀릴 수 있다.
        double avgWaitSeconds = window
            .Average(r => (DateTime.UtcNow - r.RequestTime).TotalSeconds);
        score += avgWaitSeconds * _config.WaitTimeBonusWeight;

        return score;
    }

    private Match CreateBalancedTeams(List<Player> players)
    {
        var match = new Match();
        var sortedPlayers = players.OrderByDescending(p => p.MMR).ToList();

        // 지그재그 방식으로 팀 배정 (1,4,5,8,9 vs 2,3,6,7,10)
        for (int i = 0; i < sortedPlayers.Count; i++)
        {
            if (i % 4 < 2)
                match.Team1.Add(sortedPlayers[i]);
            else
                match.Team2.Add(sortedPlayers[i]);
        }

        match.AverageMMR = players.Average(p => p.MMR);
        match.AveragePing = (int)players.Average(p => p.Ping);

        return match;
    }

    private bool IsMatchBalanced(Match match)
    {
        var team1Avg = match.Team1.Average(p => p.MMR);
        var team2Avg = match.Team2.Average(p => p.MMR);
        return Math.Abs(team1Avg - team2Avg) <= _config.MaxTeamMMRDifference;
    }

    private async Task RangeExpansionWorkerAsync(CancellationToken cancellationToken)
    {
        while (!cancellationToken.IsCancellationRequested)
        {
            try
            {
                foreach (var request in _activeRequests.Values)
                {
                    var waitTime = (DateTime.UtcNow - request.RequestTime).TotalSeconds;

                    if (waitTime > 30)
                    {
                        request.ExpandedMMRRange = 100 + (waitTime - 30) * 5;
                    }

                    if (waitTime > request.Player.Preferences.MaxWaitTime)
                    {
                        await HandleTimeoutAsync(request);
                    }
                }

                await Task.Delay(5000, cancellationToken);
            }
            catch (OperationCanceledException)
            {
                break;
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "Error in range expansion worker");
            }
        }
    }

    private async Task HandleTimeoutAsync(MatchRequest request)
    {
        _logger.LogWarning(
            "Matchmaking timeout for player {PlayerId} after {WaitTime:F0} seconds",
            request.Player.PlayerId,
            (DateTime.UtcNow - request.RequestTime).TotalSeconds);

        await DequeuePlayerAsync(request.RequestId);
        await OnMatchmakingTimeoutAsync(request.Player);
    }

    protected virtual async Task OnMatchCreatedAsync(Match match)
    {
        await Task.CompletedTask;
    }

    protected virtual async Task OnMatchmakingTimeoutAsync(Player player)
    {
        await Task.CompletedTask;
    }

    public void Stop()
    {
        _cancellationTokenSource?.Cancel();
    }
}
```

## 6.4 분산 매칭 큐 구현
10만 이상의 동시 유저를 처리하기 위해서는 단일 서버로는 한계가 있다. Redis를 활용한 분산 매칭 큐를 구현하여 수평 확장이 가능한 구조를 만든다.

---

### 분산 환경에서의 중복 매칭 문제
6.1절의 인메모리 방식은 단일 서버 내 스레드 간 동기화에는 효과적이지만, **서버가 여러 대로 늘어나는 분산 환경에서는 적용할 수 없다**. 서버 A와 서버 B가 각자의 메모리를 가지고 있으므로, 서버 A의 `ConcurrentDictionary`가 서버 B의 처리 상태를 알 수 없기 때문이다.

분산 환경에서 중복 매칭을 방지하기 위해 두 가지 메커니즘을 조합한다.

**첫째, Redis Sorted Set의 member 유일성을 활용한다.** Redis Sorted Set은 동일한 member(여기서는 `playerId`)를 중복으로 저장하지 않는다. 같은 플레이어가 여러 버킷에 동시에 등록되더라도 각 버킷 내에서는 단 한 번만 존재한다. 매칭 후 해당 플레이어를 모든 버킷에서 일괄 제거함으로써 잔류 데이터로 인한 중복 매칭을 방지한다.

**둘째, Redis `SET NX`(Set if Not eXists) 명령으로 플레이어 단위 분산 락을 획득한다.** 매칭 생성 직전에 참여 플레이어 전원에 대해 락을 시도하며, 모든 락 획득에 성공한 서버만 매칭을 생성할 수 있다. `SET NX`는 Redis 서버에서 단일 명령어로 원자적으로 실행되므로 두 서버가 동시에 같은 플레이어에 락을 시도하면 반드시 하나만 성공한다. 또한 TTL을 설정하여 락을 보유한 서버가 장애로 죽더라도 일정 시간 후 자동으로 락이 해제되어 데드락을 방지한다.

이 두 메커니즘을 조합하면 단일 서버와 분산 환경 모두에서 중복 매칭을 안전하게 방지할 수 있다.

---

```csharp
public class DistributedMatchmakingQueue
{
    private readonly IConnectionMultiplexer _redis;
    private readonly IDatabase _redisDb;
    private readonly ILogger<DistributedMatchmakingQueue> _logger;
    private readonly string _queueKeyPrefix = "matchmaking:queue:";
    private readonly string _playerDataKeyPrefix = "matchmaking:player:";
    private readonly string _lockKeyPrefix = "matchmaking:lock:";

    public DistributedMatchmakingQueue(
        IConnectionMultiplexer redis,
        ILogger<DistributedMatchmakingQueue> logger)
    {
        _redis = redis;
        _redisDb = redis.GetDatabase();
        _logger = logger;
    }

    // MMR 범위별 버킷에 플레이어 추가
    public async Task<bool> EnqueueAsync(Player player, GameMode gameMode)
    {
        try
        {
            // MMR을 기준으로 버킷 결정 (100 단위)
            int bucket = (int)(player.MMR / 100) * 100;
            string playerKey = $"{_playerDataKeyPrefix}{player.PlayerId}";

            // 플레이어 데이터에 소속 버킷 목록을 함께 저장한다.
            // 매칭 후 제거 시 모든 버킷(0~5000)을 순회하는 대신
            // 실제로 등록된 버킷만 조회하여 불필요한 Redis 호출을 줄인다.
            var adjacentBuckets = GetAdjacentBuckets(bucket, 2);
            var playerData = new PlayerQueueData
            {
                PlayerId = player.PlayerId,
                MMR = player.MMR,
                Ping = player.Ping,
                Region = player.Region,
                JoinedAt = DateTime.UtcNow.Ticks,
                GameMode = gameMode.ToString(),
                RegisteredBuckets = adjacentBuckets // [수정] 버킷 목록 저장
            };

            await _redisDb.StringSetAsync(
                playerKey,
                JsonSerializer.Serialize(playerData),
                TimeSpan.FromMinutes(10)); // TTL 설정

            // Sorted Set에 추가 (score는 대기 시작 시간)
            // Redis Sorted Set은 동일 member(playerId)의 중복 저장을 허용하지 않으므로
            // 같은 플레이어가 여러 버킷에 등록되어도 버킷 내 중복은 발생하지 않는다.
            double score = DateTime.UtcNow.Ticks;
            string queueKey = $"{_queueKeyPrefix}{gameMode}:{bucket}";
            await _redisDb.SortedSetAddAsync(queueKey, player.PlayerId, score);

            // 인접 버킷에도 추가하여 검색 범위 확장
            foreach (var adjBucket in adjacentBuckets)
            {
                string adjQueueKey = $"{_queueKeyPrefix}{gameMode}:{adjBucket}";
                await _redisDb.SortedSetAddAsync(adjQueueKey, player.PlayerId, score);
            }

            _logger.LogInformation(
                "Player {PlayerId} enqueued in bucket {Bucket} for {GameMode}",
                player.PlayerId, bucket, gameMode);

            return true;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Failed to enqueue player {PlayerId}", player.PlayerId);
            return false;
        }
    }

    // 매칭 가능한 플레이어 그룹 검색
    public async Task<List<Player>> FindMatchCandidatesAsync(
        GameMode gameMode,
        int count,
        double targetMMR,
        int maxMMRRange)
    {
        var candidates = new List<Player>();

        // 타겟 MMR 주변 버킷들 검색
        int targetBucket = (int)(targetMMR / 100) * 100;
        int bucketRange = (maxMMRRange / 100) + 1;

        var bucketsToSearch = GetAdjacentBuckets(targetBucket, bucketRange);

        foreach (var bucket in bucketsToSearch)
        {
            string queueKey = $"{_queueKeyPrefix}{gameMode}:{bucket}";

            // 대기 시간 순으로 플레이어 가져오기
            var playerIds = await _redisDb.SortedSetRangeByScoreAsync(
                queueKey,
                take: count * 2); // 여유있게 가져오기

            foreach (var playerId in playerIds)
            {
                string playerKey = $"{_playerDataKeyPrefix}{playerId}";
                var playerDataJson = await _redisDb.StringGetAsync(playerKey);

                if (!playerDataJson.HasValue)
                    continue;

                var playerQueueData = JsonSerializer.Deserialize<PlayerQueueData>(playerDataJson);

                // MMR 범위 체크
                if (Math.Abs(playerQueueData.MMR - targetMMR) <= maxMMRRange)
                {
                    candidates.Add(new Player
                    {
                        PlayerId = playerQueueData.PlayerId,
                        MMR = playerQueueData.MMR,
                        Ping = playerQueueData.Ping,
                        Region = playerQueueData.Region
                    });

                    if (candidates.Count >= count)
                        return candidates;
                }
            }
        }

        return candidates;
    }

    // 분산 락을 이용한 안전한 매칭 생성
    public async Task<Match> TryCreateMatchWithLockAsync(
        GameMode gameMode,
        List<string> playerIds,
        TimeSpan lockTimeout)
    {
        // 플레이어마다 고유한 락 키를 생성한다.
        // lockValue를 동일한 Guid로 설정하여 이 호출이 설정한 락임을 식별할 수 있게 한다.
        // (락 해제 시 자신이 설정한 락만 해제하는 안전장치로 활용 가능)
        string lockValue = Guid.NewGuid().ToString();
        var playerLocks = new List<string>();
        bool allLocksAcquired = true;

        try
        {
            foreach (var playerId in playerIds)
            {
                string playerLockKey = $"{_lockKeyPrefix}player:{playerId}";

                // [핵심] Redis SET NX (Set if Not eXists) + TTL 원자적 실행
                // - NX: 키가 없을 때만 설정 성공 → 두 서버가 동시에 시도해도 하나만 성공
                // - TTL: 락 보유 서버가 장애로 죽어도 자동 만료 → 데드락 방지
                // 이 두 속성의 조합이 분산 환경에서 중복 매칭을 방지하는 핵심이다.
                bool lockAcquired = await _redisDb.StringSetAsync(
                    playerLockKey,
                    lockValue,
                    lockTimeout,
                    When.NotExists); // ← Redis SET NX

                if (lockAcquired)
                {
                    playerLocks.Add(playerLockKey);
                }
                else
                {
                    // 하나라도 실패하면 전체 취소
                    // 이미 획득한 락을 즉시 해제하여 다른 서버가 재시도할 수 있게 한다
                    allLocksAcquired = false;
                    break;
                }
            }

            if (!allLocksAcquired)
            {
                _logger.LogWarning("Failed to acquire all player locks for match creation");
                return null;
            }

            // 락 획득 성공 후 플레이어 데이터 재검증
            // 락 획득 직전에 다른 서버가 이미 매칭을 생성했을 경우를 대비하여
            // 플레이어 데이터가 여전히 존재하는지 확인한다.
            var players = new List<Player>();
            foreach (var playerId in playerIds)
            {
                string playerKey = $"{_playerDataKeyPrefix}{playerId}";
                var playerDataJson = await _redisDb.StringGetAsync(playerKey);

                if (!playerDataJson.HasValue)
                {
                    _logger.LogWarning(
                        "Player {PlayerId} data not found during match creation", playerId);
                    return null; // 이미 다른 매칭에서 처리됨
                }

                var playerQueueData = JsonSerializer.Deserialize<PlayerQueueData>(playerDataJson);
                players.Add(new Player
                {
                    PlayerId = playerQueueData.PlayerId,
                    MMR = playerQueueData.MMR,
                    Ping = playerQueueData.Ping,
                    Region = playerQueueData.Region
                });
            }

            // 매칭 생성
            var match = CreateBalancedMatch(players);

            // 큐에서 플레이어 제거
            await RemovePlayersFromQueueAsync(playerIds, gameMode);

            return match;
        }
        finally
        {
            // 성공·실패 여부와 관계없이 반드시 모든 락을 해제한다.
            // finally 블록을 사용하여 예외 발생 시에도 락이 반드시 해제되도록 보장한다.
            foreach (var playerLock in playerLocks)
            {
                await _redisDb.KeyDeleteAsync(playerLock);
            }
        }
    }

    private async Task RemovePlayersFromQueueAsync(List<string> playerIds, GameMode gameMode)
    {
        foreach (var playerId in playerIds)
        {
            string playerKey = $"{_playerDataKeyPrefix}{playerId}";
            var playerDataJson = await _redisDb.StringGetAsync(playerKey);

            if (playerDataJson.HasValue)
            {
                var playerQueueData = JsonSerializer.Deserialize<PlayerQueueData>(playerDataJson);

                // [수정] 저장된 버킷 목록만 조회하여 제거한다.
                // 원본 코드는 0~5000 전체 버킷을 순회(51회 호출)했으나,
                // 플레이어가 실제로 등록된 버킷 목록을 저장해두면
                // 필요한 버킷(통상 5개 내외)만 대상으로 제거할 수 있다.
                foreach (var bucket in playerQueueData.RegisteredBuckets)
                {
                    string queueKey = $"{_queueKeyPrefix}{gameMode}:{bucket}";
                    await _redisDb.SortedSetRemoveAsync(queueKey, playerId);
                }
            }

            // 플레이어 원본 데이터 삭제
            await _redisDb.KeyDeleteAsync(playerKey);
        }
    }

    private List<int> GetAdjacentBuckets(int centerBucket, int range)
    {
        var buckets = new List<int>();

        for (int i = -range; i <= range; i++)
        {
            int bucket = centerBucket + (i * 100);
            if (bucket >= 0 && bucket <= 5000)
            {
                buckets.Add(bucket);
            }
        }

        return buckets;
    }

    private Match CreateBalancedMatch(List<Player> players)
    {
        var match = new Match();
        var sorted = players.OrderByDescending(p => p.MMR).ToList();

        for (int i = 0; i < sorted.Count; i++)
        {
            if (i % 4 < 2)
                match.Team1.Add(sorted[i]);
            else
                match.Team2.Add(sorted[i]);
        }

        match.AverageMMR = players.Average(p => p.MMR);
        match.AveragePing = (int)players.Average(p => p.Ping);

        return match;
    }

    // 통계 정보 조회
    public async Task<QueueStatistics> GetQueueStatisticsAsync(GameMode gameMode)
    {
        var stats = new QueueStatistics { GameMode = gameMode };

        for (int bucket = 0; bucket <= 5000; bucket += 100)
        {
            string queueKey = $"{_queueKeyPrefix}{gameMode}:{bucket}";
            long count = await _redisDb.SortedSetLengthAsync(queueKey);

            stats.TotalPlayers += count;
            stats.PlayersByBucket[bucket] = count;
        }

        return stats;
    }
}
```

### 분산 환경 중복 매칭 방지 메커니즘 요약
분산 환경에서 중복 매칭을 방지하는 세 가지 핵심 원칙은 다음과 같다.

**첫째, Redis Sorted Set의 member 유일성으로 큐 단계에서 중복을 차단한다.** 동일 `playerId`는 각 버킷 내에 단 하나만 존재하며, 매칭 후 `RegisteredBuckets` 목록을 기반으로 정확히 해당 버킷들에서만 제거한다.

**둘째, `SET NX` + TTL 조합으로 매칭 생성 직전 원자적 소유권을 확보한다.** 모든 플레이어에 대해 락 획득에 성공한 서버만 매칭을 생성할 수 있으며, TTL로 장애 시 자동 복구된다.

**셋째, 락 획득 후 플레이어 데이터를 재검증한다.** 락 획득 직전에 다른 서버가 이미 해당 플레이어를 처리했을 경우를 대비하여 플레이어 데이터의 존재 여부를 다시 확인한다.

## 6.5 핑 기반 필터링 구현
전세계에서 단일 위치의 게임 서버를 운영할 때 핑은 매우 중요한 요소다. 효과적인 핑 기반 필터링으로 플레이어 경험을 향상시킨다.

```csharp
public class PingBasedMatchmaker
{
    private readonly ILogger<PingBasedMatchmaker> _logger;
    private readonly PingMatchmakingConfig _config;
    private readonly IGeoLocationService _geoLocationService;

    public PingBasedMatchmaker(
        ILogger<PingBasedMatchmaker> logger,
        PingMatchmakingConfig config,
        IGeoLocationService geoLocationService)
    {
        _logger = logger;
        _config = config;
        _geoLocationService = geoLocationService;
    }

    // 핑 기반 후보 필터링
    public List<Player> FilterByPing(
        List<Player> candidates,
        int maxPing,
        PingFilterStrategy strategy = PingFilterStrategy.Strict)
    {
        switch (strategy)
        {
            case PingFilterStrategy.Strict:
                return FilterStrictPing(candidates, maxPing);

            case PingFilterStrategy.Adaptive:
                return FilterAdaptivePing(candidates, maxPing);

            case PingFilterStrategy.Regional:
                return FilterRegionalPing(candidates, maxPing);

            default:
                return candidates;
        }
    }

    private List<Player> FilterStrictPing(List<Player> candidates, int maxPing)
    {
        return candidates.Where(p => p.Ping <= maxPing).ToList();
    }

    private List<Player> FilterAdaptivePing(List<Player> candidates, int maxPing)
    {
        // 후보가 충분하면 엄격하게, 부족하면 완화
        if (candidates.Count >= _config.MinPlayersForMatch * 2)
        {
            var filtered = candidates.Where(p => p.Ping <= maxPing).ToList();

            if (filtered.Count >= _config.MinPlayersForMatch)
                return filtered;
        }

        // 완화된 기준 적용
        int relaxedPing = (int)(maxPing * _config.PingRelaxationMultiplier);
        return candidates.Where(p => p.Ping <= relaxedPing).ToList();
    }

    private List<Player> FilterRegionalPing(List<Player> candidates, int maxPing)
    {
        // 지역별로 그룹화
        var regionalGroups = candidates
            .GroupBy(p => p.Region)
            .OrderByDescending(g => g.Count())
            .ToList();

        // 가장 많은 플레이어가 있는 지역 우선
        foreach (var group in regionalGroups)
        {
            var filtered = group.Where(p => p.Ping <= maxPing).ToList();

            if (filtered.Count >= _config.MinPlayersForMatch)
                return filtered;
        }

        // 적합한 지역이 없으면 전체에서 필터링
        return FilterAdaptivePing(candidates, maxPing);
    }

    // 핑 기반 매칭 품질 점수 계산
    public double CalculateMatchQualityScore(Match match)
    {
        double score = 100.0;

        // 평균 핑 페널티
        double avgPing = match.AveragePing;
        if (avgPing > _config.OptimalPing)
        {
            double pingPenalty = (avgPing - _config.OptimalPing) /
                _config.OptimalPing * _config.PingPenaltyWeight;
            score -= Math.Min(pingPenalty, 30.0);
        }

        // 핑 편차 페널티 (플레이어 간 핑 차이)
        var allPlayers = match.Team1.Concat(match.Team2).ToList();
        double pingStdDev = CalculateStandardDeviation(
            allPlayers.Select(p => (double)p.Ping).ToList());

        if (pingStdDev > _config.MaxPingStandardDeviation)
        {
            double deviationPenalty = (pingStdDev - _config.MaxPingStandardDeviation) /
                _config.MaxPingStandardDeviation * _config.DeviationPenaltyWeight;
            score -= Math.Min(deviationPenalty, 20.0);
        }

        // MMR 밸런스 보너스
        double team1Avg = match.Team1.Average(p => p.MMR);
        double team2Avg = match.Team2.Average(p => p.MMR);
        double mmrDifference = Math.Abs(team1Avg - team2Avg);

        if (mmrDifference < _config.ExcellentMMRDifference)
        {
            score += 10.0;
        }

        return Math.Max(score, 0);
    }

    // 동적 핑 임계값 조정
    public int CalculateDynamicPingThreshold(
        Player player,
        double waitTimeSeconds,
        int currentQueueSize)
    {
        int basePing = player.Preferences.MaxPingThreshold;

        // 대기 시간에 따른 완화
        if (waitTimeSeconds > _config.WaitTimeThresholdForRelaxation)
        {
            double waitTimeFactor = (waitTimeSeconds - _config.WaitTimeThresholdForRelaxation) /
                _config.WaitTimeThresholdForRelaxation;

            int waitTimeIncrease = (int)(basePing * waitTimeFactor *
                _config.PingIncreasePerWaitTime);

            basePing += Math.Min(waitTimeIncrease, _config.MaxPingIncrease);
        }

        // 큐 크기에 따른 조정
        if (currentQueueSize < _config.LowQueueSizeThreshold)
        {
            basePing = (int)(basePing * _config.LowQueuePingMultiplier);
        }

        return Math.Min(basePing, _config.AbsoluteMaxPing);
    }

    // 지역 기반 핑 예측
    public async Task<int> EstimatePingAsync(string playerRegion, string serverLocation)
    {
        // 지역 간 평균 핑 캐시 조회
        var cachedPing = await _geoLocationService.GetCachedPingAsync(
            playerRegion, serverLocation);

        if (cachedPing.HasValue)
            return cachedPing.Value;

        // 거리 기반 추정
        double distance = await _geoLocationService.CalculateDistanceAsync(
            playerRegion, serverLocation);

        // 거리당 핑 계산 (대략 1ms per 100km + 기본 레이턴시)
        int estimatedPing = (int)(distance / 100.0) + _config.BaseLatency;

        // 캐시에 저장
        await _geoLocationService.CachePingAsync(
            playerRegion, serverLocation, estimatedPing);

        return estimatedPing;
    }

    // 핑 최적화를 위한 플레이어 재배치
    public Match OptimizeMatchForPing(Match match)
    {
        var allPlayers = match.Team1.Concat(match.Team2)
            .OrderBy(p => p.Ping)
            .ToList();

        // 핑이 비슷한 플레이어끼리 팀 구성
        var optimizedMatch = new Match
        {
            MatchId = match.MatchId,
            CreatedAt = match.CreatedAt
        };

        // 교대로 배치하여 양 팀의 평균 핑이 비슷하도록
        for (int i = 0; i < allPlayers.Count; i++)
        {
            if (i % 2 == 0)
                optimizedMatch.Team1.Add(allPlayers[i]);
            else
                optimizedMatch.Team2.Add(allPlayers[i]);
        }

        // MMR도 고려하여 미세 조정
        optimizedMatch = BalanceTeamsByMMRAndPing(optimizedMatch);

        optimizedMatch.AverageMMR = allPlayers.Average(p => p.MMR);
        optimizedMatch.AveragePing = (int)allPlayers.Average(p => p.Ping);

        return optimizedMatch;
    }

    private Match BalanceTeamsByMMRAndPing(Match match)
    {
        // 스왑을 통한 최적화 (간단한 휴리스틱)
        for (int i = 0; i < 10; i++) // 최대 10번 시도
        {
            double team1MMR = match.Team1.Average(p => p.MMR);
            double team2MMR = match.Team2.Average(p => p.MMR);
            double team1Ping = match.Team1.Average(p => p.Ping);
            double team2Ping = match.Team2.Average(p => p.Ping);

            double mmrDiff = Math.Abs(team1MMR - team2MMR);
            double pingDiff = Math.Abs(team1Ping - team2Ping);

            if (mmrDiff < 50 && pingDiff < 20)
                break; // 충분히 밸런스가 맞음

            // 가장 효과적인 스왑 찾기
            (int team1Idx, int team2Idx, double improvement) bestSwap =
                FindBestSwap(match, team1MMR, team2MMR, team1Ping, team2Ping);

            if (bestSwap.improvement <= 0)
                break;

            // 스왑 실행
            var temp = match.Team1[bestSwap.team1Idx];
            match.Team1[bestSwap.team1Idx] = match.Team2[bestSwap.team2Idx];
            match.Team2[bestSwap.team2Idx] = temp;
        }

        return match;
    }

    private (int team1Idx, int team2Idx, double improvement) FindBestSwap(
        Match match,
        double team1MMR,
        double team2MMR,
        double team1Ping,
        double team2Ping)
    {
        double currentScore = CalculateImbalanceScore(
            team1MMR, team2MMR, team1Ping, team2Ping);

        int bestT1Idx = -1;
        int bestT2Idx = -1;
        double bestImprovement = 0;

        for (int i = 0; i < match.Team1.Count; i++)
        {
            for (int j = 0; j < match.Team2.Count; j++)
            {
                // 스왑 시뮬레이션
                double newTeam1MMR = team1MMR - match.Team1[i].MMR + match.Team2[j].MMR;
                double newTeam2MMR = team2MMR - match.Team2[j].MMR + match.Team1[i].MMR;
                double newTeam1Ping = team1Ping - match.Team1[i].Ping + match.Team2[j].Ping;
                double newTeam2Ping = team2Ping - match.Team2[j].Ping + match.Team1[i].Ping;

                double newScore = CalculateImbalanceScore(
                    newTeam1MMR, newTeam2MMR, newTeam1Ping, newTeam2Ping);

                double improvement = currentScore - newScore;

                if (improvement > bestImprovement)
                {
                    bestImprovement = improvement;
                    bestT1Idx = i;
                    bestT2Idx = j;
                }
            }
        }

        return (bestT1Idx, bestT2Idx, bestImprovement);
    }

    private double CalculateImbalanceScore(
        double team1MMR,
        double team2MMR,
        double team1Ping,
        double team2Ping)
    {
        double mmrImbalance = Math.Abs(team1MMR - team2MMR);
        double pingImbalance = Math.Abs(team1Ping - team2Ping);

        return mmrImbalance * _config.MMRWeight + pingImbalance * _config.PingWeight;
    }

    private double CalculateStandardDeviation(List<double> values)
    {
        double avg = values.Average();
        double sumOfSquares = values.Sum(v => Math.Pow(v - avg, 2));
        return Math.Sqrt(sumOfSquares / values.Count);
    }
}

public enum PingFilterStrategy
{
    Strict,      // 엄격한 핑 제한
    Adaptive,    // 상황에 따라 조정
    Regional     // 지역 우선
}

public class PingMatchmakingConfig
{

    public int MaxPingStandardDeviation { get; set; } = 30; // ms
    public double PingPenaltyWeight { get; set; } = 25.0;
    public double DeviationPenaltyWeight { get; set; } = 15.0;
    public double ExcellentMMRDifference { get; set; } = 30.0;

    public int MinPlayersForMatch { get; set; } = 10;
    public double PingRelaxationMultiplier { get; set; } = 1.3;

    public int WaitTimeThresholdForRelaxation { get; set; } = 60; // seconds
    public double PingIncreasePerWaitTime { get; set; } = 0.1;
    public int MaxPingIncrease { get; set; } = 100; // ms
    public int AbsoluteMaxPing { get; set; } = 300; // ms

    public int LowQueueSizeThreshold { get; set; } = 50;
    public double LowQueuePingMultiplier { get; set; } = 1.5;

    public int BaseLatency { get; set; } = 10; // ms

    public double MMRWeight { get; set; } = 1.0;
    public double PingWeight { get; set; } = 0.5;
}
```

이러한 구현을 통해 대규모 FPS 게임의 매칭 시스템을 효과적으로 구축할 수 있다. 각 컴포넌트는 독립적으로 테스트하고 최적화할 수 있으며, Redis를 활용한 분산 아키텍처로 수평 확장이 가능하다. 핑 기반 필터링은 전세계 단일 서버 환경에서 플레이어 경험을 크게 향상시킨다.

### 중복 매칭 방지 전체 레이어 요약
단일 서버와 분산 환경 모두를 포괄하는 방어 레이어를 정리하면 다음과 같다.

| 레이어 | 메커니즘 | 적용 환경 | 원자성 | 역할 |
|--------|---------|-----------|--------|------|
| L1 | `ConcurrentQueue.TryDequeue()` | 단일 서버 | ✅ 완전 | 큐 단계 1차 분리 |
| L2 | `ConcurrentDictionary.TryRemove()` | 단일 서버 | ✅ 완전 | 소유권 원자적 획득 |
| L3 | Redis Sorted Set member 유일성 | 분산 | ✅ 완전 | 버킷 내 중복 차단 |
| L4 | Redis `SET NX` + TTL | 분산 | ✅ 완전 | 매칭 생성 독점 보장 |
| L5 | 락 획득 후 데이터 재검증 | 분산 | ✅ 완전 | 최종 일관성 확인 |
