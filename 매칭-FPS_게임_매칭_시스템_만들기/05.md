## 5.1 핵심 클래스 설계

### Player, MatchRequest, Match 모델
FPS 게임의 매칭 시스템에서 가장 기본이 되는 도메인 모델은 Player, MatchRequest, Match다. 이들은 매칭 시스템의 모든 동작의 중심이 된다.

Player 클래스는 게임 플레이어의 핵심 정보를 담는다. MMR뿐만 아니라 통계, 선호도, 제재 상태 등을 포함한다.

```csharp
public class Player
{
    public string Id { get; set; }
    public string Username { get; set; }
    public int MMR { get; set; }
    public double Uncertainty { get; set; }  // Glicko-2의 RD 값
    public string Region { get; set; }
    public PlayerStats Stats { get; set; }
    public PlayerPreferences Preferences { get; set; }
    public DateTime LastMatchTime { get; set; }
    public bool IsBanned { get; set; }
    public DateTime? BanExpiresAt { get; set; }
    
    public bool CanJoinMatch()
    {
        if (IsBanned && BanExpiresAt.HasValue && 
            DateTime.UtcNow < BanExpiresAt.Value)
        {
            return false;
        }
        return true;
    }
    
    public int GetDisplayRank()
    {
        // MMR을 티어로 변환 (Bronze, Silver, Gold, etc.)
        return MMR / 200;
    }
}

public class PlayerStats
{
    public int TotalMatches { get; set; }
    public int Wins { get; set; }
    public int Losses { get; set; }
    public double KDRatio { get; set; }
    public double HeadshotPercentage { get; set; }
    public double AverageDamagePerRound { get; set; }
    
    public double WinRate => TotalMatches > 0 
        ? (double)Wins / TotalMatches 
        : 0.0;
}

public class PlayerPreferences
{
    public List<string> PreferredMaps { get; set; } = new();
    public List<string> PreferredModes { get; set; } = new();
    public int MaxAcceptablePing { get; set; } = 100;
    public bool AllowCrossPlatform { get; set; } = true;
}
```

MatchRequest는 플레이어의 매칭 요청을 나타낸다. 요청 시간, 핑 정보, 파티 정보 등을 포함한다.

```csharp
public class MatchRequest
{
    public string RequestId { get; set; }
    public Player Player { get; set; }
    public int PingMs { get; set; }
    public DateTime SubmitTime { get; set; }
    public DateTime? ExpiresAt { get; set; }
    public MatchRequestStatus Status { get; set; }
    public List<string> PartyMemberIds { get; set; } = new();
    public string PreferredGameMode { get; set; }
    
    public TimeSpan WaitTime => DateTime.UtcNow - SubmitTime;
    
    public bool IsExpired()
    {
        return ExpiresAt.HasValue && DateTime.UtcNow > ExpiresAt.Value;
    }
    
    public bool IsPartyLeader()
    {
        return PartyMemberIds.Count > 0 && PartyMemberIds[0] == Player.Id;
    }
    
    public int GetDynamicMMRRange()
    {
        // 대기 시간에 따라 MMR 허용 범위 확장
        const int INITIAL_RANGE = 100;
        const int MAX_RANGE = 500;
        const int RANGE_INCREASE_PER_30_SEC = 50;
        
        int additionalRange = (int)(WaitTime.TotalSeconds / 30) * RANGE_INCREASE_PER_30_SEC;
        return Math.Min(INITIAL_RANGE + additionalRange, MAX_RANGE);
    }
    
    public int GetDynamicPingRange()
    {
        const int INITIAL_PING_RANGE = 30;
        const int MAX_PING_RANGE = 100;
        const int PING_INCREASE_PER_60_SEC = 20;
        
        int additionalRange = (int)(WaitTime.TotalSeconds / 60) * PING_INCREASE_PER_60_SEC;
        return Math.Min(INITIAL_PING_RANGE + additionalRange, MAX_PING_RANGE);
    }
}

public enum MatchRequestStatus
{
    Pending,
    Matched,
    Cancelled,
    Expired
}
```

Match 클래스는 완성된 매치를 나타낸다. 두 팀의 플레이어들과 게임 서버 정보를 담는다.

```csharp
public class Match
{
    public string Id { get; set; }
    public List<Player> Team1 { get; set; } = new();
    public List<Player> Team2 { get; set; } = new();
    public string GameMode { get; set; }
    public string MapName { get; set; }
    public DateTime CreatedAt { get; set; }
    public GameServerInfo GameServer { get; set; }
    public MatchQuality Quality { get; set; }
    
    public List<Player> AllPlayers => Team1.Concat(Team2).ToList();
    
    public double GetMMRBalance()
    {
        double team1Avg = Team1.Average(p => p.MMR);
        double team2Avg = Team2.Average(p => p.MMR);
        return Math.Abs(team1Avg - team2Avg);
    }
    
    public bool IsBalanced(double maxMMRDifference = 50)
    {
        return GetMMRBalance() <= maxMMRDifference;
    }
}

public class GameServerInfo
{
    public string ServerId { get; set; }
    public string Endpoint { get; set; }
    public int Port { get; set; }
    public string Region { get; set; }
    public string Token { get; set; }  // 인증 토큰
}

public class MatchQuality
{
    public double MMRBalanceScore { get; set; }  // 0-1, 1이 최상
    public double PingQualityScore { get; set; }   // 0-1, 1이 최상
    public double WaitTimeScore { get; set; }      // 0-1, 1이 최상
    public double OverallScore { get; set; }
    
    public static MatchQuality Calculate(Match match, 
        Dictionary<Player, int> pingMap, 
        Dictionary<Player, TimeSpan> waitTimes)
    {
        var quality = new MatchQuality();
        
        // MMR 균형 점수
        double mmrDiff = match.GetMMRBalance();
        quality.MMRBalanceScore = Math.Max(0, 1.0 - (mmrDiff / 100.0));
        
        // 핑 품질 점수
        var pings = match.AllPlayers.Select(p => pingMap[p]).ToList();
        double avgPing = pings.Average();
        double pingSpread = pings.Max() - pings.Min();
        quality.PingQualityScore = Math.Max(0, 
            (1.0 - avgPing / 150.0) * 0.6 + 
            (1.0 - pingSpread / 80.0) * 0.4);
        
        // 대기 시간 점수
        double avgWaitSeconds = waitTimes.Values.Average(wt => wt.TotalSeconds);
        quality.WaitTimeScore = Math.Max(0, 1.0 - (avgWaitSeconds / 120.0));
        
        // 종합 점수
        quality.OverallScore = 
            quality.MMRBalanceScore * 0.5 +
            quality.PingQualityScore * 0.3 +
            quality.WaitTimeScore * 0.2;
        
        return quality;
    }
}
```

### MMR Calculator
MMR 계산은 매칭 시스템의 핵심이다. ELO, Glicko-2, TrueSkill 등 다양한 알고리즘을 지원하도록 인터페이스를 정의한다.

```csharp
public interface IMMRCalculator
{
    int CalculateNewMMR(Player player, bool won, List<Player> teammates, 
        List<Player> opponents);
    double CalculateWinProbability(Player player, List<Player> opponents);
    void UpdateAfterMatch(Match match, bool team1Won);
}

public class EloMMRCalculator : IMMRCalculator
{
    private const int K_FACTOR = 32;
    private const int K_FACTOR_NEW_PLAYER = 40;  // 신규 플레이어는 더 큰 변동
    private const int NEW_PLAYER_THRESHOLD = 20;  // 20경기 이하는 신규
    
    public int CalculateNewMMR(Player player, bool won, 
        List<Player> teammates, List<Player> opponents)
    {
        // 팀 평균 MMR 계산
        double teamMMR = (teammates.Sum(p => p.MMR) + player.MMR) / (teammates.Count + 1);
        double opponentMMR = opponents.Average(p => p.MMR);
        
        // 승률 예측
        double expectedScore = CalculateExpectedScore(teamMMR, opponentMMR);
        double actualScore = won ? 1.0 : 0.0;
        
        // K 팩터 결정
        int kFactor = player.Stats.TotalMatches < NEW_PLAYER_THRESHOLD 
            ? K_FACTOR_NEW_PLAYER 
            : K_FACTOR;
        
        // 새 MMR 계산
        int mmrChange = (int)Math.Round(kFactor * (actualScore - expectedScore));
        int newMMR = player.MMR + mmrChange;
        
        // MMR 최소/최대 제한
        return Math.Clamp(newMMR, 0, 5000);
    }
    
    public double CalculateWinProbability(Player player, List<Player> opponents)
    {
        double opponentAvgMMR = opponents.Average(p => p.MMR);
        return CalculateExpectedScore(player.MMR, opponentAvgMMR);
    }
    
    private double CalculateExpectedScore(double playerMMR, double opponentMMR)
    {
        return 1.0 / (1.0 + Math.Pow(10, (opponentMMR - playerMMR) / 400.0));
    }
    
    public void UpdateAfterMatch(Match match, bool team1Won)
    {
        var winners = team1Won ? match.Team1 : match.Team2;
        var losers = team1Won ? match.Team2 : match.Team1;
        
        foreach (var player in winners)
        {
            int newMMR = CalculateNewMMR(player, true, 
                winners.Where(p => p.Id != player.Id).ToList(), 
                losers);
            player.MMR = newMMR;
        }
        
        foreach (var player in losers)
        {
            int newMMR = CalculateNewMMR(player, false, 
                losers.Where(p => p.Id != player.Id).ToList(), 
                winners);
            player.MMR = newMMR;
        }
    }
}
```

Glicko-2는 불확실성(RD)을 고려하여 더 정교한 레이팅을 제공한다.

```csharp
public class Glicko2MMRCalculator : IMMRCalculator
{
    private const double TAU = 0.5;  // 시스템 상수
    private const double INITIAL_RD = 350.0;  // 초기 RD (불확실성)
    private const double INITIAL_VOLATILITY = 0.06;
    
    public int CalculateNewMMR(Player player, bool won, 
        List<Player> teammates, List<Player> opponents)
    {
        // Glicko-2 스케일로 변환 (MMR을 μ로)
        double mu = (player.MMR - 1500) / 173.7178;
        double phi = player.Uncertainty / 173.7178;
        
        // 상대 팀 평균
        double opponentAvgMMR = opponents.Average(p => p.MMR);
        double opponentMu = (opponentAvgMMR - 1500) / 173.7178;
        double opponentPhi = INITIAL_RD / 173.7178;
        
        // g 함수
        double g = 1.0 / Math.Sqrt(1.0 + 3.0 * opponentPhi * opponentPhi / (Math.PI * Math.PI));
        
        // E 함수 (예상 점수)
        double e = 1.0 / (1.0 + Math.Exp(-g * (mu - opponentMu)));
        
        // v (분산)
        double v = 1.0 / (g * g * e * (1.0 - e));
        
        // Δ (성능 차이)
        double score = won ? 1.0 : 0.0;
        double delta = v * g * (score - e);
        
        // 새로운 μ와 φ 계산
        double phiStar = Math.Sqrt(phi * phi + 0.06 * 0.06);
        double phiPrime = 1.0 / Math.Sqrt(1.0 / (phiStar * phiStar) + 1.0 / v);
        double muPrime = mu + phiPrime * phiPrime * g * (score - e);
        
        // MMR 스케일로 복원
        int newMMR = (int)Math.Round(muPrime * 173.7178 + 1500);
        player.Uncertainty = phiPrime * 173.7178;
        
        return Math.Clamp(newMMR, 0, 5000);
    }
    
    public double CalculateWinProbability(Player player, List<Player> opponents)
    {
        double mu = (player.MMR - 1500) / 173.7178;
        double opponentAvgMMR = opponents.Average(p => p.MMR);
        double opponentMu = (opponentAvgMMR - 1500) / 173.7178;
        double opponentPhi = INITIAL_RD / 173.7178;
        
        double g = 1.0 / Math.Sqrt(1.0 + 3.0 * opponentPhi * opponentPhi / (Math.PI * Math.PI));
        return 1.0 / (1.0 + Math.Exp(-g * (mu - opponentMu)));
    }
    
    public void UpdateAfterMatch(Match match, bool team1Won)
    {
        var winners = team1Won ? match.Team1 : match.Team2;
        var losers = team1Won ? match.Team2 : match.Team1;
        
        foreach (var player in winners)
        {
            int newMMR = CalculateNewMMR(player, true,
                winners.Where(p => p.Id != player.Id).ToList(),
                losers);
            player.MMR = newMMR;
        }
        
        foreach (var player in losers)
        {
            int newMMR = CalculateNewMMR(player, false,
                losers.Where(p => p.Id != player.Id).ToList(),
                winners);
            player.MMR = newMMR;
        }
    }
}
```

### MatchmakingService
MatchmakingService는 모든 매칭 로직을 조율하는 핵심 서비스다. 요청 접수, 후보 검색, 팀 구성, 매치 생성을 담당한다.

```csharp
public class MatchmakingService
{
    private readonly IMatchingQueue _queue;
    private readonly IMMRCalculator _mmrCalculator;
    private readonly ITeamBalancer _teamBalancer;
    private readonly IGameServerAllocator _serverAllocator;
    private readonly CancellationTokenSource _cancellationTokenSource;
    
    public MatchmakingService(
        IMatchingQueue queue,
        IMMRCalculator mmrCalculator,
        ITeamBalancer teamBalancer,
        IGameServerAllocator serverAllocator)
    {
        _queue = queue;
        _mmrCalculator = mmrCalculator;
        _teamBalancer = teamBalancer;
        _serverAllocator = serverAllocator;
        _cancellationTokenSource = new CancellationTokenSource();
    }
    
    public async Task<string> SubmitMatchRequest(MatchRequest request)
    {
        // 플레이어 검증
        if (!request.Player.CanJoinMatch())
        {
            throw new InvalidOperationException("Player is banned");
        }
        
        // 요청 ID 생성
        request.RequestId = Guid.NewGuid().ToString();
        request.SubmitTime = DateTime.UtcNow;
        request.ExpiresAt = DateTime.UtcNow.AddMinutes(5);
        request.Status = MatchRequestStatus.Pending;
        
        // 큐에 추가
        await _queue.Enqueue(request);
        
        return request.RequestId;
    }
    
    public async Task CancelMatchRequest(string requestId)
    {
        await _queue.Remove(requestId);
    }
    
    public void StartMatchmaking()
    {
        Task.Run(async () => await MatchmakingLoop(_cancellationTokenSource.Token));
    }
    
    public void StopMatchmaking()
    {
        _cancellationTokenSource.Cancel();
    }
    
    private async Task MatchmakingLoop(CancellationToken cancellationToken)
    {
        while (!cancellationToken.IsCancellationRequested)
        {
            try
            {
                await ProcessMatchingCycle();
                await Task.Delay(100, cancellationToken); // 100ms 주기
            }
            catch (OperationCanceledException)
            {
                break;
            }
            catch (Exception ex)
            {
                // 로깅
                Console.WriteLine($"Matching cycle error: {ex.Message}");
                await Task.Delay(1000, cancellationToken);
            }
        }
    }
    
    private async Task ProcessMatchingCycle()
    {
        // 만료된 요청 제거
        await _queue.RemoveExpiredRequests();
        
        // 각 지역별로 매칭 시도
        var regions = await _queue.GetActiveRegions();
        
        var matchTasks = regions.Select(region => 
            TryCreateMatchesForRegion(region));
        
        await Task.WhenAll(matchTasks);
    }
    
    private async Task TryCreateMatchesForRegion(string region)
    {
        const int PLAYERS_PER_MATCH = 10;
        const int MAX_MATCHES_PER_CYCLE = 50;
        
        int matchesCreated = 0;
        
        while (matchesCreated < MAX_MATCHES_PER_CYCLE)
        {
            // 가장 오래 기다린 플레이어를 시드로 선택
            var seedRequest = await _queue.GetOldestRequest(region);
            if (seedRequest == null)
                break;
            
            // 후보 플레이어 검색
            var candidates = await FindCandidates(seedRequest, PLAYERS_PER_MATCH - 1);
            
            if (candidates.Count < PLAYERS_PER_MATCH - 1)
                break;  // 충분한 플레이어가 없음
            
            // 매치 생성 시도
            var match = await TryCreateMatch(seedRequest, candidates);
            
            if (match != null)
            {
                matchesCreated++;
                await FinalizeMatch(match);
            }
            else
            {
                // 매치 생성 실패, 다음 사이클에서 재시도
                break;
            }
        }
    }
    
    private async Task<List<MatchRequest>> FindCandidates(
        MatchRequest seedRequest, int count)
    {
        int mmrRange = seedRequest.GetDynamicMMRRange();
        int pingRange = seedRequest.GetDynamicPingRange();
        
        var candidates = await _queue.GetCandidates(
            seedRequest.Player.Region,
            seedRequest.Player.MMR,
            mmrRange,
            seedRequest.PingMs,
            pingRange,
            count * 2);  // 여유있게 가져옴
        
        // 점수 기반 정렬
        var scored = candidates
            .Select(c => new
            {
                Request = c,
                Score = CalculateMatchScore(seedRequest, c)
            })
            .OrderBy(x => x.Score)
            .Take(count)
            .Select(x => x.Request)
            .ToList();
        
        return scored;
    }
    
    private double CalculateMatchScore(MatchRequest seed, MatchRequest candidate)
    {
        double mmrDiff = Math.Abs(seed.Player.MMR - candidate.Player.MMR);
        double mmrScore = mmrDiff / 500.0;
        
        double pingDiff = Math.Abs(seed.PingMs - candidate.PingMs);
        double pingScore = pingDiff / 100.0;
        
        double waitTimeDiff = Math.Abs(
            (seed.WaitTime - candidate.WaitTime).TotalSeconds);
        double waitScore = waitTimeDiff / 60.0;
        
        return mmrScore * 0.5 + pingScore * 0.3 + waitScore * 0.2;
    }
    
    private async Task<Match> TryCreateMatch(
        MatchRequest seedRequest, List<MatchRequest> candidates)
    {
        var allRequests = new List<MatchRequest> { seedRequest };
        allRequests.AddRange(candidates);
        
        // 원자적으로 모든 요청을 큐에서 제거
        var removed = await _queue.TryRemoveAll(
            allRequests.Select(r => r.RequestId).ToList());
        
        if (removed.Count != allRequests.Count)
        {
            // 일부 플레이어가 이미 다른 매치에 할당됨
            // 제거된 플레이어들을 다시 큐에 넣음
            foreach (var req in removed)
            {
                await _queue.Enqueue(req);
            }
            return null;
        }
        
        // 팀 밸런싱
        var players = allRequests.Select(r => r.Player).ToList();
        var (team1, team2) = _teamBalancer.BalanceTeams(players);
        
        var match = new Match
        {
            Id = Guid.NewGuid().ToString(),
            Team1 = team1,
            Team2 = team2,
            CreatedAt = DateTime.UtcNow,
            GameMode = seedRequest.PreferredGameMode ?? "Default"
        };
        
        // 매치 품질 계산
        var pingMap = allRequests.ToDictionary(r => r.Player, r => r.PingMs);
        var waitTimeMap = allRequests.ToDictionary(r => r.Player, r => r.WaitTime);
        match.Quality = MatchQuality.Calculate(match, pingMap, waitTimeMap);
        
        // 품질이 너무 낮으면 거부
        if (match.Quality.OverallScore < 0.3)
        {
            foreach (var req in allRequests)
            {
                await _queue.Enqueue(req);
            }
            return null;
        }
        
        return match;
    }
    
    private async Task FinalizeMatch(Match match)
    {
        // 게임 서버 할당
        match.GameServer = await _serverAllocator.AllocateServer(match);
        
        // 플레이어들에게 알림
        await NotifyPlayers(match);
        
        // 매치 데이터 저장
        await SaveMatch(match);
    }
    
    private async Task NotifyPlayers(Match match)
    {
        // 실제 구현에서는 SignalR, WebSocket 등을 사용
        foreach (var player in match.AllPlayers)
        {
            await SendNotification(player.Id, new
            {
                MatchId = match.Id,
                GameServer = match.GameServer.Endpoint,
                Team = match.Team1.Contains(player) ? 1 : 2,
                Teammates = match.Team1.Contains(player) 
                    ? match.Team1.Where(p => p.Id != player.Id).Select(p => p.Username)
                    : match.Team2.Where(p => p.Id != player.Id).Select(p => p.Username),
                Opponents = match.Team1.Contains(player)
                    ? match.Team2.Select(p => p.Username)
                    : match.Team1.Select(p => p.Username)
            });
        }
    }
    
    private async Task SendNotification(string playerId, object data)
    {
        // SignalR Hub를 통한 실시간 알림
        // await _hubContext.Clients.User(playerId).SendAsync("MatchFound", data);
        await Task.CompletedTask;
    }
    
    private async Task SaveMatch(Match match)
    {
        // 데이터베이스에 매치 정보 저장
        await Task.CompletedTask;
    }
}
```

## 5.2 동시성 처리

### async/await 패턴
10만 명의 동시 유저를 처리하려면 스레드를 효율적으로 사용해야 한다. async/await 패턴은 I/O 작업 중에 스레드를 블로킹하지 않고 다른 작업에 활용할 수 있게 한다.

```csharp
public class AsyncMatchingQueue : IMatchingQueue
{
    private readonly IDatabase _redis;
    private readonly SemaphoreSlim _enqueueSemaphore = new SemaphoreSlim(100, 100);
    
    public async Task Enqueue(MatchRequest request)
    {
        await _enqueueSemaphore.WaitAsync();
        try
        {
            // Redis 작업은 I/O 바운드이므로 async로 처리
            await _redis.SortedSetAddAsync(
                GetQueueKey(request.Player.Region),
                request.RequestId,
                request.Player.MMR);
            
            await _redis.HashSetAsync(
                GetRequestKey(request.RequestId),
                ToHashEntries(request));
            
            await _redis.KeyExpireAsync(
                GetRequestKey(request.RequestId),
                TimeSpan.FromMinutes(5));
        }
        finally
        {
            _enqueueSemaphore.Release();
        }
    }
    
    public async Task<List<MatchRequest>> GetCandidates(
        string region, int targetMMR, int mmrRange,
        int targetPing, int pingRange, int count)
    {
        // 비동기 병렬 처리
        var tasks = new List<Task<List<MatchRequest>>>();
        
        // 여러 MMR 범위를 병렬로 쿼리
        int bucketSize = 100;
        for (int mmr = targetMMR - mmrRange; 
             mmr <= targetMMR + mmrRange; 
             mmr += bucketSize)
        {
            tasks.Add(GetCandidatesInRange(region, mmr, mmr + bucketSize));
        }
        
        var results = await Task.WhenAll(tasks);
        
        // 결과 병합 및 필터링
        var allCandidates = results.SelectMany(r => r)
            .Where(r => Math.Abs(r.Player.MMR - targetMMR) <= mmrRange &&
                       Math.Abs(r.PingMs - targetPing) <= pingRange)
            .Take(count)
            .ToList();
        
        return allCandidates;
    }
    
    private async Task<List<MatchRequest>> GetCandidatesInRange(
        string region, int minMMR, int maxMMR)
    {
        var requestIds = await _redis.SortedSetRangeByScoreAsync(
            GetQueueKey(region),
            minMMR,
            maxMMR);
        
        var requests = new List<MatchRequest>();
        
        // 각 요청의 상세 정보를 병렬로 가져옴
        var tasks = requestIds.Select(id => 
            GetRequestDetails(id.ToString()));
        
        var results = await Task.WhenAll(tasks);
        
        return results.Where(r => r != null).ToList();
    }
    
    private string GetQueueKey(string region) => $"queue:{region}";
    private string GetRequestKey(string requestId) => $"request:{requestId}";
    
    private HashEntry[] ToHashEntries(MatchRequest request)
    {
        return new[]
        {
            new HashEntry("playerId", request.Player.Id),
            new HashEntry("mmr", request.Player.MMR),
            new HashEntry("ping", request.PingMs),
            new HashEntry("submitTime", request.SubmitTime.Ticks)
        };
    }
    
    private async Task<MatchRequest> GetRequestDetails(string requestId)
    {
        var hash = await _redis.HashGetAllAsync(GetRequestKey(requestId));
        if (hash.Length == 0)
            return null;
        
        return FromHashEntries(requestId, hash);
    }
    
    private MatchRequest FromHashEntries(string requestId, HashEntry[] hash)
    {
        return new MatchRequest
        {
            RequestId = requestId,
            Player = new Player
            {
                Id = hash.First(h => h.Name == "playerId").Value,
                MMR = (int)hash.First(h => h.Name == "mmr").Value
            },
            PingMs = (int)hash.First(h => h.Name == "ping").Value,
            SubmitTime = new DateTime((long)hash.First(h => h.Name == "submitTime").Value)
        };
    }
}
```

Task.WhenAll을 사용하면 여러 비동기 작업을 병렬로 실행할 수 있다. 하지만 너무 많은 작업을 동시에 실행하면 리소스 고갈이 발생할 수 있으므로 제한이 필요하다.

```csharp
public class ThrottledAsyncProcessor
{
    private readonly SemaphoreSlim _throttler;
    
    public ThrottledAsyncProcessor(int maxConcurrency)
    {
        _throttler = new SemaphoreSlim(maxConcurrency, maxConcurrency);
    }
    
    public async Task<List<T>> ProcessInParallel<T>(
        List<Func<Task<T>>> tasks, int maxConcurrency = 50)
    {
        var results = new List<T>();
        var activeTasks = new List<Task<T>>();
        
        foreach (var taskFactory in tasks)
        {
            await _throttler.WaitAsync();
            
            var task = Task.Run(async () =>
            {
                try
                {
                    return await taskFactory();
                }
                finally
                {
                    _throttler.Release();
                }
            });
            
            activeTasks.Add(task);
            
            // 최대 동시 실행 수에 도달하면 일부 완료 대기
            if (activeTasks.Count >= maxConcurrency)
            {
                var completed = await Task.WhenAny(activeTasks);
                results.Add(await completed);
                activeTasks.Remove(completed);
            }
        }
        
        // 남은 작업 완료 대기
        results.AddRange(await Task.WhenAll(activeTasks));
        
        return results;
    }
}
```

### ConcurrentQueue와 Thread-safe 컬렉션
.NET의 Concurrent 컬렉션은 락 없이 동시 액세스를 지원한다. 내부적으로 Compare-And-Swap(CAS) 같은 원자적 연산을 사용한다.

```csharp
public class ConcurrentMatchingPool
{
    private readonly ConcurrentDictionary<string, ConcurrentQueue<MatchRequest>> _regionQueues;
    private readonly ConcurrentDictionary<string, MatchRequest> _activeRequests;
    private readonly ConcurrentBag<Match> _completedMatches;
    
    public ConcurrentMatchingPool()
    {
        _regionQueues = new ConcurrentDictionary<string, ConcurrentQueue<MatchRequest>>();
        _activeRequests = new ConcurrentDictionary<string, MatchRequest>();
        _completedMatches = new ConcurrentBag<Match>();
    }
    
    public void AddRequest(MatchRequest request)
    {
        // GetOrAdd는 원자적 연산
        var queue = _regionQueues.GetOrAdd(
            request.Player.Region,
            _ => new ConcurrentQueue<MatchRequest>());
        
        queue.Enqueue(request);
        _activeRequests.TryAdd(request.RequestId, request);
    }
    
    public bool TryRemoveRequest(string requestId, out MatchRequest request)
    {
        // TryRemove는 원자적 연산
        return _activeRequests.TryRemove(requestId, out request);
    }
    
    public List<MatchRequest> GetSnapshot(string region)
    {
        if (!_regionQueues.TryGetValue(region, out var queue))
            return new List<MatchRequest>();
        
        // ToArray는 스냅샷을 생성 (이후 변경에 영향받지 않음)
        return queue.ToArray().ToList();
    }
    
    public void AddCompletedMatch(Match match)
    {
        _completedMatches.Add(match);
    }
}
```

### 락프리(Lock-free) 자료구조
극한의 성능을 위해 완전히 락프리인 자료구조를 구현할 수 있다. Interlocked 클래스는 원자적 연산을 제공한다.

```csharp
public class LockFreeCounter
{
    private long _count;
    
    public long Increment()
    {
        return Interlocked.Increment(ref _count);
    }
    
    public long Decrement()
    {
        return Interlocked.Decrement(ref _count);
    }
    
    public long Add(long value)
    {
        return Interlocked.Add(ref _count, value);
    }
    
    public long Value => Interlocked.Read(ref _count);
}

public class LockFreeMatchingMetrics
{
    private long _totalRequestsReceived;
    private long _totalMatchesCreated;
    private long _totalPlayersMatched;
    private long _currentQueueSize;
    
    public void RecordRequest()
    {
        Interlocked.Increment(ref _totalRequestsReceived);
        Interlocked.Increment(ref _currentQueueSize);
    }
    
    public void RecordMatch(int playerCount)
    {
        Interlocked.Increment(ref _totalMatchesCreated);
        Interlocked.Add(ref _totalPlayersMatched, playerCount);

    }
    
    public MatchingMetricsSnapshot GetSnapshot()
    {
        return new MatchingMetricsSnapshot
        {
            TotalRequestsReceived = Interlocked.Read(ref _totalRequestsReceived),
            TotalMatchesCreated = Interlocked.Read(ref _totalMatchesCreated),
            TotalPlayersMatched = Interlocked.Read(ref _totalPlayersMatched),
            CurrentQueueSize = Interlocked.Read(ref _currentQueueSize)
        };
    }
}
```

더 복잡한 락프리 자료구조는 Compare-Exchange를 사용한다.

```csharp
public class LockFreeStack<T> where T : class
{
    private class Node
    {
        public T Value;
        public Node Next;
    }
    
    private Node _head;
    
    public void Push(T value)
    {
        var newNode = new Node { Value = value };
        
        while (true)
        {
            Node currentHead = _head;
            newNode.Next = currentHead;
            
            // CompareExchange: _head가 currentHead와 같으면 newNode로 교체
            if (Interlocked.CompareExchange(ref _head, newNode, currentHead) == currentHead)
            {
                return;  // 성공
            }
            // 실패하면 재시도 (다른 스레드가 먼저 수정함)
        }
    }
    
    public bool TryPop(out T value)
    {
        while (true)
        {
            Node currentHead = _head;
            
            if (currentHead == null)
            {
                value = null;
                return false;
            }
            
            // CompareExchange로 head를 다음 노드로 변경
            if (Interlocked.CompareExchange(ref _head, currentHead.Next, currentHead) == currentHead)
            {
                value = currentHead.Value;
                return true;
            }
            // 실패하면 재시도
        }
    }
}
```

### 객체 풀링
매 매칭마다 새로운 객체를 생성하면 가비지 컬렉션(GC) 압력이 증가한다. 객체 풀링을 사용하면 객체를 재사용하여 메모리 할당을 줄일 수 있다.

```csharp
public class ObjectPool<T> where T : class, new()
{
    private readonly ConcurrentBag<T> _objects;
    private readonly Func<T> _objectGenerator;
    private readonly Action<T> _resetAction;
    
    public ObjectPool(Func<T> objectGenerator, Action<T> resetAction = null)
    {
        _objects = new ConcurrentBag<T>();
        _objectGenerator = objectGenerator ?? (() => new T());
        _resetAction = resetAction;
    }
    
    public T Rent()
    {
        if (_objects.TryTake(out T item))
        {
            return item;
        }
        
        return _objectGenerator();
    }
    
    public void Return(T item)
    {
        _resetAction?.Invoke(item);
        _objects.Add(item);
    }
}

public class MatchRequestPool
{
    private static readonly ObjectPool<MatchRequest> _pool = new ObjectPool<MatchRequest>(
        () => new MatchRequest(),
        request =>
        {
            // 재사용 전에 상태 초기화
            request.RequestId = null;
            request.Player = null;
            request.PingMs = 0;
            request.Status = MatchRequestStatus.Pending;
            request.PartyMemberIds.Clear();
        });
    
    public static MatchRequest Create(Player player, int pingMs)
    {
        var request = _pool.Rent();
        request.RequestId = Guid.NewGuid().ToString();
        request.Player = player;
        request.PingMs = pingMs;
        request.SubmitTime = DateTime.UtcNow;
        return request;
    }
    
    public static void Recycle(MatchRequest request)
    {
        _pool.Return(request);
    }
}
```

### 프로파일링 포인트
성능 문제를 발견하고 해결하려면 주요 지점에 프로파일링 코드를 추가해야 한다.

```csharp
public class PerformanceMonitor
{
    private readonly ConcurrentDictionary<string, PerformanceMetric> _metrics;
    
    public PerformanceMonitor()
    {
        _metrics = new ConcurrentDictionary<string, PerformanceMetric>();
    }
    
    public IDisposable Measure(string operationName)
    {
        return new PerformanceMeasurement(this, operationName);
    }
    
    internal void RecordMeasurement(string operationName, long elapsedMs)
    {
        _metrics.AddOrUpdate(
            operationName,
            new PerformanceMetric { Count = 1, TotalMs = elapsedMs, MinMs = elapsedMs, MaxMs = elapsedMs },
            (key, existing) =>
            {
                existing.Count++;
                existing.TotalMs += elapsedMs;
                existing.MinMs = Math.Min(existing.MinMs, elapsedMs);
                existing.MaxMs = Math.Max(existing.MaxMs, elapsedMs);
                return existing;
            });
    }
    
    public Dictionary<string, PerformanceMetric> GetMetrics()
    {
        return new Dictionary<string, PerformanceMetric>(_metrics);
    }
    
    private class PerformanceMeasurement : IDisposable
    {
        private readonly PerformanceMonitor _monitor;
        private readonly string _operationName;
        private readonly Stopwatch _stopwatch;
        
        public PerformanceMeasurement(PerformanceMonitor monitor, string operationName)
        {
            _monitor = monitor;
            _operationName = operationName;
            _stopwatch = Stopwatch.StartNew();
        }
        
        public void Dispose()
        {
            _stopwatch.Stop();
            _monitor.RecordMeasurement(_operationName, _stopwatch.ElapsedMilliseconds);
        }
    }
}

{
    public long Count { get; set; }
    public long TotalMs { get; set; }
    public long MinMs { get; set; }
    public long MaxMs { get; set; }
    
    public double AverageMs => Count > 0 ? (double)TotalMs / Count : 0;
}
```
