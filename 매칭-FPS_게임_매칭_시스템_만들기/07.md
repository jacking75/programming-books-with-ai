## 7.1 주요 메트릭
매칭 시스템의 성능과 품질을 지속적으로 관리하기 위해서는 적절한 메트릭 수집과 분석이 필수다. 10만 이상의 동시 유저를 처리하는 환경에서는 실시간 모니터링과 빠른 대응이 시스템 안정성의 핵심이다.

### 평균 매칭 시간
매칭 시간은 플레이어 경험에 가장 직접적인 영향을 미치는 지표다. 이를 효과적으로 측정하고 분석하는 시스템을 구현한다.

```csharp
public class MatchmakingMetricsCollector
{
    private readonly ILogger<MatchmakingMetricsCollector> _logger;
    private readonly IMetricsPublisher _metricsPublisher;
    private readonly ConcurrentDictionary<string, MatchmakingMetrics> _activeMetrics;
    private readonly MetricsConfig _config;
    
    public MatchmakingMetricsCollector(
        ILogger<MatchmakingMetricsCollector> logger,
        IMetricsPublisher metricsPublisher,
        MetricsConfig config)
    {
        _logger = logger;
        _metricsPublisher = metricsPublisher;
        _activeMetrics = new ConcurrentDictionary<string, MatchmakingMetrics>();
        _config = config;
    }
    
    // 매칭 시작 기록
    public void RecordMatchmakingStart(string requestId, Player player, GameMode gameMode)
    {
        var metrics = new MatchmakingMetrics
        {
            RequestId = requestId,
            PlayerId = player.PlayerId,
            GameMode = gameMode,
            PlayerMMR = player.MMR,
            PlayerPing = player.Ping,
            StartTime = DateTime.UtcNow,
            Region = player.Region
        };
        
        _activeMetrics[requestId] = metrics;
    }
    
    // 매칭 완료 기록
    public async Task RecordMatchmakingCompleteAsync(
        string requestId, 
        Match match,
        MatchQualityMetrics qualityMetrics)
    {
        if (!_activeMetrics.TryRemove(requestId, out var metrics))
        {
            _logger.LogWarning("Metrics not found for request {RequestId}", requestId);
            return;
        }
        
        metrics.EndTime = DateTime.UtcNow;
        metrics.WaitTimeSeconds = (metrics.EndTime - metrics.StartTime).TotalSeconds;
        metrics.MatchId = match.MatchId;
        metrics.QualityMetrics = qualityMetrics;
        
        // 메트릭 발행
        await PublishMatchmakingMetricsAsync(metrics);
        
        // 집계 통계 업데이트
        await UpdateAggregateStatisticsAsync(metrics);
        
        _logger.LogInformation(
            "Matchmaking completed - RequestId: {RequestId}, WaitTime: {WaitTime:F2}s, Quality: {Quality:F2}",
            requestId, metrics.WaitTimeSeconds, qualityMetrics.OverallScore);
    }
    
    // 매칭 취소 기록
    public async Task RecordMatchmakingCancelAsync(string requestId, string reason)
    {
        if (!_activeMetrics.TryRemove(requestId, out var metrics))
            return;
        
        metrics.EndTime = DateTime.UtcNow;
        metrics.WaitTimeSeconds = (metrics.EndTime - metrics.StartTime).TotalSeconds;
        metrics.CancellationReason = reason;
        
        await PublishCancellationMetricsAsync(metrics);
    }
    
    private async Task PublishMatchmakingMetricsAsync(MatchmakingMetrics metrics)
    {
        // 실시간 메트릭 발행
        await _metricsPublisher.PublishAsync(new
        {
            MetricName = "MatchmakingWaitTime",
            Value = metrics.WaitTimeSeconds,
            Dimensions = new
            {
                GameMode = metrics.GameMode.ToString(),
                MMRBucket = GetMMRBucket(metrics.PlayerMMR),
                Region = metrics.Region,
                PingRange = GetPingRange(metrics.PlayerPing)
            },
            Timestamp = metrics.EndTime
        });
        
        await _metricsPublisher.PublishAsync(new
        {
            MetricName = "MatchQuality",
            Value = metrics.QualityMetrics.OverallScore,
            Dimensions = new
            {
                GameMode = metrics.GameMode.ToString(),
                MMRBucket = GetMMRBucket(metrics.PlayerMMR)
            },
            Timestamp = metrics.EndTime
        });
    }
    
    private async Task UpdateAggregateStatisticsAsync(MatchmakingMetrics metrics)
    {
        var key = $"{metrics.GameMode}:{GetMMRBucket(metrics.PlayerMMR)}";
        
        // Redis를 사용한 슬라이딩 윈도우 통계
        var redisKey = $"matchmaking:stats:{key}:{DateTime.UtcNow:yyyyMMddHH}";
        
        await _metricsPublisher.IncrementCounterAsync(redisKey, "total_matches");
        await _metricsPublisher.AddToListAsync(redisKey, "wait_times", metrics.WaitTimeSeconds);
        await _metricsPublisher.AddToListAsync(redisKey, "quality_scores", 
            metrics.QualityMetrics.OverallScore);
        
        // TTL 설정 (24시간)
        await _metricsPublisher.SetExpiryAsync(redisKey, TimeSpan.FromHours(24));
    }
    
    // 실시간 통계 조회
    public async Task<MatchmakingStatistics> GetCurrentStatisticsAsync(
        GameMode gameMode, 
        TimeSpan timeWindow)
    {
        var statistics = new MatchmakingStatistics
        {
            GameMode = gameMode,
            TimeWindow = timeWindow,
            CollectedAt = DateTime.UtcNow
        };
        
        var endTime = DateTime.UtcNow;
        var startTime = endTime.Subtract(timeWindow);
        
        // 시간 범위 내의 모든 시간대 키 생성
        var keys = new List<string>();
        for (var time = startTime; time <= endTime; time = time.AddHours(1))
        {
            foreach (var mmrBucket in _config.MMRBuckets)
            {
                var key = $"matchmaking:stats:{gameMode}:{mmrBucket}:{time:yyyyMMddHH}";
                keys.Add(key);
            }
        }
        
        // 모든 키에서 데이터 수집
        var allWaitTimes = new List<double>();
        var allQualityScores = new List<double>();
        long totalMatches = 0;
        
        foreach (var key in keys)
        {
            var waitTimes = await _metricsPublisher.GetListAsync<double>(key, "wait_times");
            var qualityScores = await _metricsPublisher.GetListAsync<double>(key, "quality_scores");
            var matches = await _metricsPublisher.GetCounterAsync(key, "total_matches");
            
            allWaitTimes.AddRange(waitTimes);
            allQualityScores.AddRange(qualityScores);
            totalMatches += matches;
        }
        
        if (allWaitTimes.Any())
        {
            statistics.AverageWaitTime = allWaitTimes.Average();
            statistics.MedianWaitTime = CalculatePercentile(allWaitTimes, 50);
            statistics.P95WaitTime = CalculatePercentile(allWaitTimes, 95);
            statistics.P99WaitTime = CalculatePercentile(allWaitTimes, 99);
            statistics.MinWaitTime = allWaitTimes.Min();
            statistics.MaxWaitTime = allWaitTimes.Max();
        }
        
        if (allQualityScores.Any())
        {
            statistics.AverageQualityScore = allQualityScores.Average();
            statistics.MedianQualityScore = CalculatePercentile(allQualityScores, 50);
        }
        
        statistics.TotalMatches = totalMatches;
        statistics.MatchesPerMinute = totalMatches / timeWindow.TotalMinutes;
        
        return statistics;
    }
    
    private double CalculatePercentile(List<double> values, int percentile)
    {
        var sorted = values.OrderBy(v => v).ToList();
        int index = (int)Math.Ceiling(percentile / 100.0 * sorted.Count) - 1;
        return sorted[Math.Max(0, Math.Min(index, sorted.Count - 1))];
    }
    
    private string GetMMRBucket(double mmr)
    {
        int bucket = (int)(mmr / 100) * 100;
        return $"{bucket}-{bucket + 99}";
    }
    
    private string GetPingRange(int ping)
    {
        if (ping <= 50) return "0-50";
        if (ping <= 100) return "51-100";
        if (ping <= 150) return "101-150";
        return "151+";
    }
}
```

### 매칭 품질 지표

매칭 품질은 게임 경험의 핵심이며, 다양한 요소를 종합적으로 평가해야 한다.

```csharp
public class MatchQualityAnalyzer
{
    private readonly ILogger<MatchQualityAnalyzer> _logger;
    private readonly QualityConfig _config;
    
    public MatchQualityAnalyzer(
        ILogger<MatchQualityAnalyzer> logger,
        QualityConfig config)
    {
        _logger = logger;
        _config = config;
    }
    
    // 종합 품질 점수 계산
    public MatchQualityMetrics AnalyzeMatchQuality(Match match, List<Player> originalCandidates)
    {
        var metrics = new MatchQualityMetrics
        {
            MatchId = match.MatchId,
            AnalyzedAt = DateTime.UtcNow
        };
        
        // 1. MMR 밸런스 점수
        metrics.MMRBalanceScore = CalculateMMRBalanceScore(match);
        
        // 2. 팀 내부 MMR 분산 점수
        metrics.TeamMMRVarianceScore = CalculateTeamVarianceScore(match);
        
        // 3. 핑 품질 점수
        metrics.PingQualityScore = CalculatePingQualityScore(match);
        
        // 4. 대기 시간 효율성 점수
        metrics.WaitTimeEfficiencyScore = CalculateWaitTimeEfficiencyScore(
            match, originalCandidates);
        
        // 5. 지역 분포 점수
        metrics.RegionDistributionScore = CalculateRegionDistributionScore(match);
        
        // 가중치를 적용한 종합 점수
        metrics.OverallScore = 
            metrics.MMRBalanceScore * _config.MMRBalanceWeight +
            metrics.TeamMMRVarianceScore * _config.TeamVarianceWeight +
            metrics.PingQualityScore * _config.PingQualityWeight +
            metrics.WaitTimeEfficiencyScore * _config.WaitTimeWeight +
            metrics.RegionDistributionScore * _config.RegionDistributionWeight;
        
        // 품질 등급 부여
        metrics.QualityGrade = DetermineQualityGrade(metrics.OverallScore);
        
        return metrics;
    }
    
    private double CalculateMMRBalanceScore(Match match)
    {
        var team1Avg = match.Team1.Average(p => p.MMR);
        var team2Avg = match.Team2.Average(p => p.MMR);
        var difference = Math.Abs(team1Avg - team2Avg);
        
        // 차이가 작을수록 높은 점수
        double score = 100.0;
        
        if (difference <= _config.ExcellentMMRDifference)
        {
            score = 100.0;
        }
        else if (difference <= _config.GoodMMRDifference)
        {
            score = 100.0 - (difference - _config.ExcellentMMRDifference) / 
                (_config.GoodMMRDifference - _config.ExcellentMMRDifference) * 20.0;
        }
        else if (difference <= _config.AcceptableMMRDifference)
        {
            score = 80.0 - (difference - _config.GoodMMRDifference) / 
                (_config.AcceptableMMRDifference - _config.GoodMMRDifference) * 30.0;
        }
        else
        {
            score = Math.Max(0, 50.0 - (difference - _config.AcceptableMMRDifference) * 0.5);
        }
        
        return score;
    }
    
    private double CalculateTeamVarianceScore(Match match)
    {
        var team1Variance = CalculateVariance(match.Team1.Select(p => p.MMR).ToList());
        var team2Variance = CalculateVariance(match.Team2.Select(p => p.MMR).ToList());
        var avgVariance = (team1Variance + team2Variance) / 2.0;
        
        // 분산이 작을수록 높은 점수 (팀 내부가 균등함)
        double score = 100.0;
        
        if (avgVariance <= _config.ExcellentVariance)
        {
            score = 100.0;
        }
        else if (avgVariance <= _config.AcceptableVariance)
        {
            score = 100.0 - (avgVariance - _config.ExcellentVariance) / 
                (_config.AcceptableVariance - _config.ExcellentVariance) * 40.0;
        }
        else
        {
            score = Math.Max(0, 60.0 - (avgVariance - _config.AcceptableVariance) * 0.3);
        }
        
        return score;
    }
    
    private double CalculatePingQualityScore(Match match)
    {
        var allPlayers = match.Team1.Concat(match.Team2).ToList();
        var avgPing = allPlayers.Average(p => p.Ping);
        var maxPing = allPlayers.Max(p => p.Ping);
        var pingVariance = CalculateVariance(allPlayers.Select(p => (double)p.Ping).ToList());
        
        double score = 100.0;
        
        // 평균 핑 페널티
        if (avgPing > _config.OptimalPing)
        {
            score -= (avgPing - _config.OptimalPing) / _config.OptimalPing * 20.0;
        }
        
        // 최대 핑 페널티
        if (maxPing > _config.MaxAcceptablePing)
        {
            score -= 15.0;
        }
        
        // 핑 편차 페널티
        if (pingVariance > _config.MaxPingVariance)
        {
            score -= 10.0;
        }
        
        return Math.Max(0, score);
    }
    
    private double CalculateWaitTimeEfficiencyScore(Match match, List<Player> candidates)
    {
        var matchedPlayers = match.Team1.Concat(match.Team2).ToList();
        var avgWaitTime = matchedPlayers.Average(p => 
            (DateTime.UtcNow - p.JoinedQueueAt).TotalSeconds);
        
        // 대기 시간이 짧을수록 높은 점수
        double score = 100.0;
        
        if (avgWaitTime <= _config.TargetWaitTime)
        {
            score = 100.0;
        }
        else if (avgWaitTime <= _config.AcceptableWaitTime)
        {
            score = 100.0 - (avgWaitTime - _config.TargetWaitTime) / 
                (_config.AcceptableWaitTime - _config.TargetWaitTime) * 30.0;
        }
        else
        {
            score = Math.Max(0, 70.0 - (avgWaitTime - _config.AcceptableWaitTime) * 0.5);
        }
        
        return score;
    }
    
    private double CalculateRegionDistributionScore(Match match)
    {
        var allPlayers = match.Team1.Concat(match.Team2).ToList();
        var regionGroups = allPlayers.GroupBy(p => p.Region).ToList();
        
        // 지역이 너무 분산되어 있으면 낮은 점수
        double score = 100.0;
        
        if (regionGroups.Count == 1)
        {
            score = 100.0; // 모두 같은 지역
        }
        else if (regionGroups.Count == 2)
        {
            score = 90.0; // 2개 지역
        }
        else if (regionGroups.Count <= 4)
        {
            score = 70.0; // 3-4개 지역
        }
        else
        {
            score = 50.0; // 5개 이상 지역
        }
        
        return score;
    }
    
    private double CalculateVariance(List<double> values)
    {
        if (!values.Any()) return 0;
        
        var avg = values.Average();
        var sumOfSquares = values.Sum(v => Math.Pow(v - avg, 2));
        return sumOfSquares / values.Count;
    }
    
    private QualityGrade DetermineQualityGrade(double score)
    {
        if (score >= 90) return QualityGrade.Excellent;
        if (score >= 75) return QualityGrade.Good;
        if (score >= 60) return QualityGrade.Fair;
        if (score >= 40) return QualityGrade.Poor;
        return QualityGrade.VeryPoor;
    }
    
    // 게임 결과 기반 사후 품질 분석
    public async Task<PostMatchQualityAnalysis> AnalyzePostMatchQualityAsync(
        Match match,
        GameResult gameResult)
    {
        var analysis = new PostMatchQualityAnalysis
        {
            MatchId = match.MatchId,
            AnalyzedAt = DateTime.UtcNow
        };
        
        // 1. 예측 정확도 분석
        analysis.PredictionAccuracy = AnalyzePredictionAccuracy(match, gameResult);
        
        // 2. 게임 경쟁도 분석 (점수 차이)
        analysis.CompetitivenessScore = AnalyzeCompetitiveness(gameResult);
        
        // 3. 플레이어 이탈률
        analysis.PlayerRetentionRate = CalculatePlayerRetention(gameResult);
        
        // 4. 플레이어 만족도 (보고된 경우)
        if (gameResult.PlayerFeedback != null)
        {
            analysis.AverageSatisfactionScore = 
                gameResult.PlayerFeedback.Average(f => f.SatisfactionScore);
        }
        
        return analysis;
    }
    
    private double AnalyzePredictionAccuracy(Match match, GameResult gameResult)
    {
        var team1MMR = match.Team1.Average(p => p.MMR);
        var team2MMR = match.Team2.Average(p => p.MMR);
        
        // 예상 승률
        double expectedTeam1WinProb = 1.0 / (1.0 + Math.Pow(10, (team2MMR - team1MMR) / 400.0));
        
        // 실제 결과
        bool team1Won = gameResult.WinningTeam == 1;
        double actualResult = team1Won ? 1.0 : 0.0;
        
        // 브라이어 스코어 (낮을수록 정확한 예측)
        double brierScore = Math.Pow(expectedTeam1WinProb - actualResult, 2);
        
        // 0-100 점수로 변환
        return (1.0 - brierScore) * 100.0;
    }
    
    private double AnalyzeCompetitiveness(GameResult gameResult)
    {
        var scoreDiff = Math.Abs(gameResult.Team1Score - gameResult.Team2Score);
        var totalScore = gameResult.Team1Score + gameResult.Team2Score;
        
        if (totalScore == 0) return 50.0;
        
        var scoreRatio = (double)scoreDiff / totalScore;
        
        // 점수 차이가 작을수록 경쟁적
        if (scoreRatio <= 0.1) return 100.0; // 10% 이내 차이
        if (scoreRatio <= 0.2) return 90.0;
        if (scoreRatio <= 0.3) return 75.0;
        if (scoreRatio <= 0.5) return 60.0;
        return 40.0;
    }
    
    private double CalculatePlayerRetention(GameResult gameResult)
    {
        var completedPlayers = gameResult.PlayerStats.Count(p => p.CompletedMatch);
        var totalPlayers = gameResult.PlayerStats.Count;
        
        return (double)completedPlayers / totalPlayers * 100.0;
    }
}
```

### 시스템 처리량

대규모 동시 접속 환경에서 시스템 처리량은 안정성의 핵심 지표다.

```csharp
public class ThroughputMonitor
{
    private readonly ILogger<ThroughputMonitor> _logger;
    private readonly IMetricsPublisher _metricsPublisher;
    private readonly ConcurrentDictionary<string, ThroughputCounter> _counters;
    private Timer _reportingTimer;
    
    public ThroughputMonitor(
        ILogger<ThroughputMonitor> logger,
        IMetricsPublisher metricsPublisher)
    {
        _logger = logger;
        _metricsPublisher = metricsPublisher;
        _counters = new ConcurrentDictionary<string, ThroughputCounter>();
        
        // 1분마다 보고
        _reportingTimer = new Timer(ReportMetrics, null, TimeSpan.FromMinutes(1), 
            TimeSpan.FromMinutes(1));
    }
    
    public void RecordEnqueueOperation(GameMode gameMode)
    {
        var key = $"enqueue:{gameMode}";
        GetOrCreateCounter(key).Increment();
    }
    
    public void RecordMatchCreation(GameMode gameMode)
    {
        var key = $"match_created:{gameMode}";
        GetOrCreateCounter(key).Increment();
    }
    
    public void RecordDequeueOperation(GameMode gameMode)
    {
        var key = $"dequeue:{gameMode}";
        GetOrCreateCounter(key).Increment();
    }
    
    public void RecordMatchmakingIteration()
    {
        GetOrCreateCounter("matchmaking_iterations").Increment();
    }
    
    private ThroughputCounter GetOrCreateCounter(string key)
    {
        return _counters.GetOrAdd(key, k => new ThroughputCounter());
    }
    
    private async void ReportMetrics(object state)
    {
        try
        {
            var snapshot = new Dictionary<string, long>();
            
            foreach (var kvp in _counters)
            {
                var count = kvp.Value.GetAndReset();
                snapshot[kvp.Key] = count;
                
                // 메트릭 발행
                await _metricsPublisher.PublishAsync(new
                {
                    MetricName = $"Throughput.{kvp.Key}",
                    Value = count,
                    Unit = "Count/Minute",
                    Timestamp = DateTime.UtcNow
                });
            }
            
            // 처리량 요약 로그
            _logger.LogInformation(
                "Throughput Summary - Enqueues: {Enqueues}, Matches: {Matches}, Dequeues: {Dequeues}, Iterations: {Iterations}",
                snapshot.Where(s => s.Key.StartsWith("enqueue:")).Sum(s => s.Value),
                snapshot.Where(s => s.Key.StartsWith("match_created:")).Sum(s => s.Value),
                snapshot.Where(s => s.Key.StartsWith("dequeue:")).Sum(s => s.Value),
                snapshot.GetValueOrDefault("matchmaking_iterations"));
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Error reporting throughput metrics");
        }
    }
    
    public async Task<SystemThroughputMetrics> GetCurrentThroughputAsync()

        var startTime = DateTime.UtcNow;
        var startCpuUsage = process.TotalProcessorTime;
        
        Thread.Sleep(100);
        
        var endTime = DateTime.UtcNow;
        var endCpuUsage = process.TotalProcessorTime;
        
        var cpuUsedMs = (endCpuUsage - startCpuUsage).TotalMilliseconds;
        var totalMs = (endTime - startTime).TotalMilliseconds;
        var cpuUsageTotal = cpuUsedMs / (Environment.ProcessorCount * totalMs);
        
        return cpuUsageTotal * 100;
    }
```

## 7.2 A/B 테스트 전략

매칭 시스템의 지속적인 개선을 위해서는 체계적인 A/B 테스트가 필수다. 다양한 매칭 알고리즘과 파라미터를 실험하고 데이터 기반으로 의사결정을 내릴 수 있다.

```csharp
    public async Task<ExperimentVariant> AssignVariantAsync(
        string playerId, 
        string experimentId)
    {
        // 이미 할당된 그룹이 있는지 확인
        var cacheKey = $"experiment:{experimentId}:player:{playerId}";
        var cachedVariant = await _redisDb.StringGetAsync(cacheKey);

        {
            return JsonSerializer.Deserialize<ExperimentVariant>(cachedVariant);
        }

        var experiment = await GetExperimentConfigAsync(experimentId);
        
        if (experiment == null || !experiment.IsActive)
        {
            return ExperimentVariant.Control;
        }
        
        // 플레이어를 실험 그룹에 할당
        var variant = DetermineVariant(playerId, experiment);
        
        // 캐시에 저장
        await _redisDb.StringSetAsync(
            cacheKey,
            JsonSerializer.Serialize(variant),
            TimeSpan.FromDays(30));
        
        // 할당 기록
        await RecordVariantAssignmentAsync(experimentId, playerId, variant);
        
        return variant;

    private ExperimentVariant DetermineVariant(string playerId, ExperimentConfig experiment)
    {
        // 플레이어 ID 기반 일관된 해싱으로 그룹 결정
        var hash = GetConsistentHash(playerId + experiment.Id);
        var normalizedHash = hash % 100;
        
        double cumulativePercentage = 0;
        
        foreach (var variant in experiment.Variants)
        {
            cumulativePercentage += variant.TrafficPercentage;
            
            if (normalizedHash < cumulativePercentage)
            {
                return variant.Variant;
            }
        }
        
        return ExperimentVariant.Control;
    }
    
    private int GetConsistentHash(string input)
    {
        using (var md5 = MD5.Create())
        {
            var hash = md5.ComputeHash(Encoding.UTF8.GetBytes(input));
            return Math.Abs(BitConverter.ToInt32(hash, 0));
        }
    }

            return {count, totalWait / count, totalQuality / count}

    // 실험 결과 분석
    public async Task<ExperimentAnalysis> AnalyzeExperimentAsync(string experimentId)
    {
        var analysis = new ExperimentAnalysis
        {
            ExperimentId = experimentId,
            AnalyzedAt = DateTime.UtcNow
        };
        
        var experiment = await GetExperimentConfigAsync(experimentId);
        
        foreach (var variantConfig in experiment.Variants)
        {
            var stats = await GetVariantStatisticsAsync(experimentId, variantConfig.Variant);
            analysis.VariantStats[variantConfig.Variant] = stats;
        }
        
        // 통계적 유의성 검정
        var controlStats = analysis.VariantStats[ExperimentVariant.Control];
        
        foreach (var variant in analysis.VariantStats.Keys.Where(v => v != ExperimentVariant.Control))
        {
            var testStats = analysis.VariantStats[variant];
            
            // 대기 시간 비교
            var waitTimeTest = PerformTTest(
                controlStats.AverageWaitTime,
                controlStats.WaitTimeStdDev,
                controlStats.SampleCount,
                testStats.AverageWaitTime,
                testStats.WaitTimeStdDev,
                testStats.SampleCount);
            
            // 품질 점수 비교
            var qualityTest = PerformTTest(
                controlStats.AverageQuality,
                controlStats.QualityStdDev,
                controlStats.SampleCount,
                testStats.AverageQuality,
                testStats.QualityStdDev,
                testStats.SampleCount);
            
            analysis.StatisticalTests[variant] = new StatisticalTestResult
            {
                WaitTimeSignificance = waitTimeTest,
                QualitySignificance = qualityTest,
                IsSignificant = waitTimeTest.PValue < 0.05 || qualityTest.PValue < 0.05
            };
        }
        
        // 권장사항 생성
        analysis.Recommendation = GenerateRecommendation(analysis);
        
        return analysis;
    }

    private TTestResult PerformTTest(
        double mean1, double stdDev1, long n1,
        double mean2, double stdDev2, long n2)
    {
        // Welch의 t-검정 (등분산 가정 불필요)
        var variance1 = Math.Pow(stdDev1, 2);
        var variance2 = Math.Pow(stdDev2, 2);
        
        var tStatistic = (mean1 - mean2) / 
            Math.Sqrt(variance1 / n1 + variance2 / n2);
        
        // 자유도 계산
        var df = Math.Pow(variance1 / n1 + variance2 / n2, 2) /
            (Math.Pow(variance1 / n1, 2) / (n1 - 1) + 
             Math.Pow(variance2 / n2, 2) / (n2 - 1));
        
        // p-value 근사 (간단한 구현)
        var pValue = CalculatePValue(Math.Abs(tStatistic), df);
        
        return new TTestResult
        {
            TStatistic = tStatistic,
            DegreesOfFreedom = df,
            PValue = pValue,
            MeanDifference = mean2 - mean1,
            PercentImprovement = ((mean2 - mean1) / mean1) * 100
        };
    }
    
    private double CalculatePValue(double tStat, double df)
    {
        // 간단한 p-value 근사
        // 실제로는 통계 라이브러리 사용 권장
        if (tStat < 1.96) return 0.05; // 95% 신뢰수준
        if (tStat < 2.58) return 0.01; // 99% 신뢰수준
        return 0.001; // 99.9% 신뢰수준
    }
    
    private double CalculateStandardDeviation(List<double> values)
    {
        if (!values.Any()) return 0;
        
        var avg = values.Average();
        var sumOfSquares = values.Sum(v => Math.Pow(v - avg, 2));
        return Math.Sqrt(sumOfSquares / (values.Count - 1));
    }
    
    private string GenerateRecommendation(ExperimentAnalysis analysis)
    {
        var bestVariant = analysis.VariantStats
            .OrderByDescending(kvp => kvp.Value.AverageQuality - kvp.Value.AverageWaitTime * 0.1)
            .First();
        
        if (bestVariant.Key == ExperimentVariant.Control)
        {
            return "현재 설정(Control)을 유지하는 것이 권장된다.";
        }
        
        var testResult = analysis.StatisticalTests[bestVariant.Key];
        
        if (!testResult.IsSignificant)
        {
            return $"{bestVariant.Key} 변형이 약간 더 나은 성능을 보이지만 통계적으로 유의미하지 않다. 더 많은 데이터가 필요하다.";
        }
        
        return $"{bestVariant.Key} 변형으로 전환을 권장한다. " +
               $"품질 점수 {testResult.QualitySignificance.PercentImprovement:F2}% 개선, " +
               $"대기 시간 {testResult.WaitTimeSignificance.PercentImprovement:F2}% 변화.";
    }
```

## 7.3 문제 해결 및 디버깅

대규모 매칭 시스템에서는 다양한 문제가 발생할 수 있다. 효과적인 디버깅과 문제 해결 도구를 구축하여 빠른 대응이 가능하도록 한다.

```csharp
    private async Task<QueueHealthCheck> CheckQueueHealthAsync()

    private async Task<List<string>> FindDataInconsistenciesAsync()
    {
        var issues = new List<string>();
        
        // 큐에는 있지만 플레이어 데이터가 없는 경우
        foreach (GameMode mode in Enum.GetValues(typeof(GameMode)))
        {
            for (int bucket = 0; bucket <= 5000; bucket += 100)
            {
                var queueKey = $"matchmaking:queue:{mode}:{bucket}";
                var playerIds = await _redisDb.SortedSetRangeByScoreAsync(queueKey, take: 100);
                
                foreach (var playerId in playerIds)
                {
                    var playerKey = $"matchmaking:player:{playerId}";
                    var exists = await _redisDb.KeyExistsAsync(playerKey);
                    
                    if (!exists)

                        // 자동 정리
                        await _redisDb.SortedSetRemoveAsync(queueKey, playerId);
                    }
                }
            }
        }
        
        return issues;

    // 특정 플레이어의 매칭 상태 진단
    public async Task<PlayerDiagnostics> DiagnosePlayerAsync(string playerId)
    {
        var diagnostics = new PlayerDiagnostics
        {
            PlayerId = playerId,
            DiagnosedAt = DateTime.UtcNow
        };
        
        // 플레이어 데이터 조회
        var playerKey = $"matchmaking:player:{playerId}";
        var playerDataJson = await _redisDb.StringGetAsync(playerKey);
        
        if (!playerDataJson.HasValue)
        {
            diagnostics.Status = "플레이어 데이터 없음";
            return diagnostics;
        }
        
        var player = JsonSerializer.Deserialize<Player>(playerDataJson);
        diagnostics.PlayerData = player;
        
        // 큐 내 위치 확인
        var queuePositions = await FindPlayerInQueuesAsync(playerId, player);
        diagnostics.QueuePositions = queuePositions;
        
        // 대기 시간 계산
        if (player.JoinedQueueAt != default)
        {
            diagnostics.WaitTimeSeconds = (DateTime.UtcNow - player.JoinedQueueAt).TotalSeconds;
        }
        
        // 예상 매칭 시간 계산
        diagnostics.EstimatedMatchTime = await EstimateMatchTimeAsync(player);
        
        // 매칭 가능한 후보 수 확인
        diagnostics.AvailableCandidatesCount = await CountAvailableCandidatesAsync(player);
        
        // 문제 진단
        diagnostics.Issues = DiagnosePlayerIssues(diagnostics);
        diagnostics.Status = diagnostics.Issues.Any() ? "문제 발견" : "정상";
        
        return diagnostics;
    }
    
    private async Task<Dictionary<string, int>> FindPlayerInQueuesAsync(
        string playerId,
        Player player)
    {
        var positions = new Dictionary<string, int>();
        
        int bucket = (int)(player.MMR / 100) * 100;
        var adjacentBuckets = GetAdjacentBuckets(bucket, 2);
        
        foreach (var b in adjacentBuckets)
        {
            var queueKey = $"matchmaking:queue:{player.Preferences.PreferredGameMode}:{b}";
            var rank = await _redisDb.SortedSetRankAsync(queueKey, playerId);
            
            if (rank.HasValue)
            {
                positions[queueKey] = (int)rank.Value;
            }
        }
        
        return positions;
    }
    
    private async Task<double> EstimateMatchTimeAsync(Player player)
    {
        // 비슷한 MMR 범위의 최근 매칭 시간 기반 추정
        var recentMatches = await GetRecentMatchesInMMRRangeAsync(
            player.MMR - 100, 
            player.MMR + 100);
        
        if (!recentMatches.Any())
            return 60; // 기본값
        
        return recentMatches.Average(m => m.AverageWaitTime);
    }
    
    private async Task<int> CountAvailableCandidatesAsync(Player player)
    {
        int count = 0;
        int bucket = (int)(player.MMR / 100) * 100;
        var adjacentBuckets = GetAdjacentBuckets(bucket, 3);
        
        foreach (var b in adjacentBuckets)
        {
            var queueKey = $"matchmaking:queue:{player.Preferences.PreferredGameMode}:{b}";
            count += (int)await _redisDb.SortedSetLengthAsync(queueKey);
        }
        
        return count;
    }
    
    private List<string> DiagnosePlayerIssues(PlayerDiagnostics diagnostics)
    {
        var issues = new List<string>();
        
        // 대기 시간이 너무 긴 경우
        if (diagnostics.WaitTimeSeconds > 300)
        {
            issues.Add($"비정상적으로 긴 대기 시간: {diagnostics.WaitTimeSeconds:F0}초");
        }
        
        // 후보가 충분하지 않은 경우
        if (diagnostics.AvailableCandidatesCount < 10)
        {
            issues.Add($"매칭 가능한 후보 부족: {diagnostics.AvailableCandidatesCount}명");
        }
        
        // 큐에 없는 경우
        if (!diagnostics.QueuePositions.Any())
        {
            issues.Add("어떤 큐에도 플레이어가 없음 - 데이터 불일치 가능성");
        }
        
        // 핑이 너무 높은 경우
        if (diagnostics.PlayerData.Ping > 200)
        {
            issues.Add($"높은 핑으로 인한 매칭 어려움: {diagnostics.PlayerData.Ping}ms");
        }
        
        // MMR이 극단적인 경우
        if (diagnostics.PlayerData.MMR > 3000 || diagnostics.PlayerData.MMR < 500)
        {
            issues.Add($"극단적인 MMR로 인한 매칭 어려움: {diagnostics.PlayerData.MMR:F0}");
        }
        
        return issues;
    }
    
    // 매칭 실패 원인 분석
    public async Task<MatchFailureAnalysis> AnalyzeMatchFailureAsync(
        List<string> candidatePlayerIds,
        string reason)
    {
        var analysis = new MatchFailureAnalysis
        {
            AnalyzedAt = DateTime.UtcNow,
            Reason = reason,
            CandidateCount = candidatePlayerIds.Count
        };
        
        // 후보 플레이어 데이터 수집
        var players = new List<Player>();
        foreach (var playerId in candidatePlayerIds)
        {
            var playerKey = $"matchmaking:player:{playerId}";
            var playerDataJson = await _redisDb.StringGetAsync(playerKey);
            
            if (playerDataJson.HasValue)
            {
                players.Add(JsonSerializer.Deserialize<Player>(playerDataJson));
            }
        }
        
        // MMR 분포 분석
        if (players.Any())
        {
            analysis.MMRRange = players.Max(p => p.MMR) - players.Min(p => p.MMR);
            analysis.MMRStdDev = CalculateStandardDeviation(players.Select(p => p.MMR).ToList());
        }
        
        // 핑 분포 분석
        if (players.Any())
        {
            analysis.AvgPing = players.Average(p => p.Ping);
            analysis.MaxPing = players.Max(p => p.Ping);
            analysis.PingStdDev = CalculateStandardDeviation(
                players.Select(p => (double)p.Ping).ToList());
        }
        
        // 지역 분포
        analysis.RegionDistribution = players
            .GroupBy(p => p.Region)
            .ToDictionary(g => g.Key, g => g.Count());
        
        // 실패 원인 추론
        analysis.InferredCauses = InferFailureCauses(analysis, players);
        
        return analysis;
    }
    
    private List<string> InferFailureCauses(MatchFailureAnalysis analysis, List<Player> players)
    {
        var causes = new List<string>();
        
        if (analysis.MMRRange > 500)
        {
            causes.Add($"MMR 범위가 너무 넓음: {analysis.MMRRange:F0}");
        }
        
        if (analysis.MaxPing > 200)
        {
            causes.Add($"일부 플레이어의 핑이 너무 높음: {analysis.MaxPing}ms");
        }
        
        if (analysis.PingStdDev > 50)
        {
            causes.Add($"플레이어 간 핑 편차가 큼: {analysis.PingStdDev:F1}ms");
        }
        
        if (analysis.RegionDistribution.Count > 5)
        {
            causes.Add($"지역이 너무 분산됨: {analysis.RegionDistribution.Count}개 지역");
        }
        
        if (players.Count < 10)
        {
            causes.Add($"후보 플레이어 수 부족: {players.Count}명");
        }
        
        return causes;
    }

    private double GetCpuUsage()
    {
        var process = Process.GetCurrentProcess();
        var startTime = DateTime.UtcNow;
        var startCpuUsage = process.TotalProcessorTime;
        
        Thread.Sleep(100);
        
        var endTime = DateTime.UtcNow;
        var endCpuUsage = process.TotalProcessorTime;
        
        var cpuUsedMs = (endCpuUsage - startCpuUsage).TotalMilliseconds;
        var totalMs = (endTime - startTime).TotalMilliseconds;
        var cpuUsageTotal = cpuUsedMs / (Environment.ProcessorCount * totalMs);
        
        return cpuUsageTotal * 100;
    }

    private double CalculateStandardDeviation(List<double> values)
    {
        if (!values.Any()) return 0;
        
        var avg = values.Average();
        var sumOfSquares = values.Sum(v => Math.Pow(v - avg, 2));
        return Math.Sqrt(sumOfSquares / values.Count);
    }
    
    private List<int> GetAdjacentBuckets(int centerBucket, int range)
    {
        var buckets = new List<int>();
        
        for (int i = -range; i <= range; i++)
        {
            int bucket = centerBucket + (i * 100);
            if (bucket >= 0 && bucket <= 5000)
            {
                buckets.Add(bucket);
            }
        }
        
        return buckets;
    }
```
