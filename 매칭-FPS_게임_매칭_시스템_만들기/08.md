## 8.1 주요 FPS 게임들의 매칭 시스템
성공적인 FPS 게임들의 매칭 시스템을 분석하면 실제 운영 환경에서 검증된 설계 패턴과 전략을 배울 수 있다. 각 게임은 고유한 특성에 맞춰 매칭 시스템을 최적화했으며, 이들의 경험은 새로운 시스템을 구축하는 데 귀중한 통찰을 제공한다.

### Counter-Strike: Global Offensive (CS:GO)
CS:GO는 가장 정교한 매칭 시스템 중 하나를 운영한다. Valve는 Glicko-2 레이팅 시스템을 기반으로 자체 수정을 가한 시스템을 사용한다.

```csharp
// CS:GO 스타일 매칭 시스템 구현
public class CSGOStyleMatchmaking
{
    private readonly ILogger<CSGOStyleMatchmaking> _logger;
    private readonly IDatabase _redisDb;
    private readonly Glicko2Calculator _glicko2;
    
    // CS:GO는 신뢰도(RD)를 매우 중요하게 취급한다
    private const double INITIAL_RATING = 1500.0;
    private const double INITIAL_RD = 350.0;  // 높은 불확실성으로 시작
    private const double MIN_RD = 50.0;
    
    public CSGOStyleMatchmaking(
        ILogger<CSGOStyleMatchmaking> logger,
        IConnectionMultiplexer redis,
        Glicko2Calculator glicko2)
    {
        _logger = logger;
        _redisDb = redis.GetDatabase();
        _glicko2 = glicko2;
    }
    
    // CS:GO의 Trust Factor와 유사한 시스템
    public async Task<TrustFactor> CalculateTrustFactorAsync(Player player)
    {
        var trustFactor = new TrustFactor
        {
            PlayerId = player.PlayerId,
            CalculatedAt = DateTime.UtcNow
        };
        
        // 1. 계정 나이 (오래된 계정일수록 신뢰도 높음)
        var accountAge = (DateTime.UtcNow - player.AccountCreatedAt).TotalDays;
        trustFactor.AccountAgeScore = Math.Min(100, accountAge / 365.0 * 100);
        
        // 2. 플레이 시간
        trustFactor.PlaytimeScore = Math.Min(100, player.TotalPlaytimeHours / 500.0 * 100);
        
        // 3. 신고 이력
        var reportsAgainst = await GetReportsAgainstPlayerAsync(player.PlayerId);
        trustFactor.ReportScore = Math.Max(0, 100 - reportsAgainst.Count * 5);
        
        // 4. 팀킬 및 이탈 이력
        var toxicityScore = await CalculateToxicityScoreAsync(player.PlayerId);
        trustFactor.BehaviorScore = 100 - toxicityScore;
        
        // 5. VAC 밴 이력 체크
        var vacBans = await CheckVACBansAsync(player.PlayerId);
        trustFactor.VACScore = vacBans > 0 ? 0 : 100;
        
        // 6. Prime 상태 (유료 기능)
        trustFactor.PrimeScore = player.IsPrime ? 100 : 50;
        
        // 가중 평균 계산
        trustFactor.OverallScore = 
            trustFactor.AccountAgeScore * 0.15 +
            trustFactor.PlaytimeScore * 0.15 +
            trustFactor.ReportScore * 0.25 +
            trustFactor.BehaviorScore * 0.25 +
            trustFactor.VACScore * 0.15 +
            trustFactor.PrimeScore * 0.05;
        
        return trustFactor;
    }
    
    // CS:GO 스타일의 매칭 풀 생성
    public async Task<List<Player>> CreateMatchingPoolAsync(
        Player initiatingPlayer,
        CompetitiveRank targetRank)
    {
        var pool = new List<Player>();
        
        // 1. 랭크 범위 설정 (CS:GO는 보통 ±2 랭크)
        var minRank = (CompetitiveRank)Math.Max(0, (int)targetRank - 2);
        var maxRank = (CompetitiveRank)Math.Min(18, (int)targetRank + 2);
        
        // 2. Trust Factor가 유사한 플레이어 우선
        var initiatingTrustFactor = await CalculateTrustFactorAsync(initiatingPlayer);
        
        for (var rank = minRank; rank <= maxRank; rank++)
        {
            var candidates = await GetPlayersInRankAsync(rank);
            
            foreach (var candidate in candidates)
            {
                // Prime 매칭 (Prime 플레이어끼리 우선 매칭)
                if (initiatingPlayer.IsPrime && !candidate.IsPrime)
                    continue;
                
                // Trust Factor 필터링
                var candidateTrustFactor = await CalculateTrustFactorAsync(candidate);
                var trustDifference = Math.Abs(
                    initiatingTrustFactor.OverallScore - candidateTrustFactor.OverallScore);
                
                // Trust Factor 차이가 30 이상이면 제외
                if (trustDifference > 30)
                    continue;
                
                // 핑 체크
                if (candidate.Ping > 150)
                    continue;
                
                pool.Add(candidate);
            }
        }
        
        return pool;
    }
    
    // CS:GO 스타일의 팀 밸런싱
    public Match BalanceTeamsCSGOStyle(List<Player> pool)
    {
        // CS:GO는 매우 엄격한 팀 밸런싱을 사용
        // 각 팀의 평균 MMR 차이를 최소화
        
        if (pool.Count < 10)
            return null;
        
        var players = pool.OrderByDescending(p => p.MMR).Take(10).ToList();
        
        // 가능한 모든 팀 조합 평가 (5:5)
        var bestMatch = new Match
        {
            MatchId = Guid.NewGuid().ToString(),
            Team1 = new List<Player>(),
            Team2 = new List<Player>()
        };
        
        double bestBalance = double.MaxValue;
        
        // 조합 생성 및 평가 (최적화된 알고리즘)
        var combinations = GenerateBalancedCombinations(players, 5);
        
        foreach (var combination in combinations)
        {
            var team1 = combination;
            var team2 = players.Except(combination).ToList();
            
            var team1Avg = team1.Average(p => p.MMR);
            var team2Avg = team2.Average(p => p.MMR);
            var balance = Math.Abs(team1Avg - team2Avg);
            
            // 팀 내부 분산도 고려
            var team1Variance = CalculateVariance(team1.Select(p => p.MMR).ToList());
            var team2Variance = CalculateVariance(team2.Select(p => p.MMR).ToList());
            var totalVariance = team1Variance + team2Variance;
            
            // 종합 점수 (밸런스와 분산 모두 고려)
            var score = balance * 2.0 + totalVariance * 0.1;
            
            if (score < bestBalance)
            {
                bestBalance = score;
                bestMatch.Team1 = team1;
                bestMatch.Team2 = team2;
            }
        }
        
        return bestMatch;
    }
    
    private IEnumerable<List<Player>> GenerateBalancedCombinations(
        List<Player> players, 
        int teamSize)
    {
        // 최적화된 조합 생성 (모든 조합 대신 상위 1000개만)
        var combinations = new List<List<Player>>();
        var random = new Random();
        
        for (int i = 0; i < 1000; i++)
        {
            var shuffled = players.OrderBy(x => random.Next()).ToList();
            combinations.Add(shuffled.Take(teamSize).ToList());
        }
        
        return combinations;
    }
    
    private double CalculateVariance(List<double> values)
    {
        if (!values.Any()) return 0;
        var avg = values.Average();
        return values.Sum(v => Math.Pow(v - avg, 2)) / values.Count;
    }
    
    private async Task<List<PlayerReport>> GetReportsAgainstPlayerAsync(string playerId)
    {
        var key = $"reports:player:{playerId}";
        var reportsJson = await _redisDb.StringGetAsync(key);
        
        if (!reportsJson.HasValue)
            return new List<PlayerReport>();
        
        return JsonSerializer.Deserialize<List<PlayerReport>>(reportsJson);
    }
    
    private async Task<double> CalculateToxicityScoreAsync(string playerId)
    {
        var statsKey = $"toxicity:player:{playerId}";
        var teamKills = (int)await _redisDb.HashGetAsync(statsKey, "team_kills");
        var abandons = (int)await _redisDb.HashGetAsync(statsKey, "abandons");
        
        // 팀킬 1회당 5점, 이탈 1회당 10점 페널티
        return Math.Min(100, teamKills * 5 + abandons * 10);
    }
    
    private async Task<int> CheckVACBansAsync(string playerId)
    {
        var key = $"vac:bans:{playerId}";
        var bans = await _redisDb.StringGetAsync(key);
        return bans.HasValue ? (int)bans : 0;
    }
    
    private async Task<List<Player>> GetPlayersInRankAsync(CompetitiveRank rank)
    {
        var key = $"matchmaking:rank:{rank}";
        var playerIds = await _redisDb.SetMembersAsync(key);
        
        var players = new List<Player>();
        foreach (var playerId in playerIds)
        {
            var playerKey = $"player:{playerId}";
            var playerJson = await _redisDb.StringGetAsync(playerKey);
            
            if (playerJson.HasValue)
            {
                players.Add(JsonSerializer.Deserialize<Player>(playerJson));
            }
        }
        
        return players;
    }
}
```

### Valorant

Riot Games의 Valorant는 매우 투명하고 공정한 매칭 시스템으로 유명하다. 숨겨진 MMR과 표시되는 랭크를 분리하여 운영한다.

```csharp
public class ValorantStyleMatchmaking
{
    private readonly ILogger<ValorantStyleMatchmaking> _logger;
    private readonly IDatabase _redisDb;
    
    // Valorant는 숨겨진 MMR과 표시 랭크를 분리
    private const double MMR_PER_RANK_TIER = 100.0;
    
    public ValorantStyleMatchmaking(
        ILogger<ValorantStyleMatchmaking> logger,
        IConnectionMultiplexer redis)
    {
        _logger = logger;
        _redisDb = redis.GetDatabase();
    }
    
    // Valorant 스타일의 이중 레이팅 시스템
    public class ValorantRating
    {
        // 숨겨진 MMR (실제 매칭에 사용)
        public double HiddenMMR { get; set; }
        public double MMRConfidence { get; set; }  // 신뢰도
        
        // 표시되는 랭크
        public ValorantRank VisibleRank { get; set; }
        public int RankRating { get; set; }  // 0-100 (랭크 내 진행도)
        
        // 성과 기반 MMR 조정
        public double PerformanceModifier { get; set; }
    }
    
    // Valorant의 성과 기반 MMR 계산
    public async Task<MMRChange> CalculateMMRChangeAsync(
        Player player,
        GameResult gameResult,
        PerformanceStats stats)
    {
        var change = new MMRChange
        {
            PlayerId = player.PlayerId,
            MatchId = gameResult.MatchId
        };
        
        // 1. 기본 MMR 변화 (승패 기반)
        var expectedWinRate = CalculateExpectedWinRate(
            player.HiddenMMR, 
            gameResult.OpponentAverageMMR);
        
        var actualResult = gameResult.IsWin ? 1.0 : 0.0;
        var kFactor = CalculateKFactor(player.MMRConfidence, player.GamesPlayed);
        
        var baseMMRChange = kFactor * (actualResult - expectedWinRate);
        
        // 2. 성과 기반 조정 (Valorant의 핵심 특징)
        var performanceModifier = CalculatePerformanceModifier(
            stats, 
            gameResult,
            player.HiddenMMR);
        
        // 3. 최종 MMR 변화
        change.BaseMMRChange = baseMMRChange * MMR_PER_RANK_TIER;
        change.PerformanceAdjustment = performanceModifier;
        change.TotalMMRChange = change.BaseMMRChange + change.PerformanceAdjustment;
        
        // 4. 신뢰도 업데이트
        change.NewConfidence = UpdateConfidence(player.MMRConfidence);
        
        // 5. 표시 랭크 변화 계산
        var rankChange = CalculateRankChange(
            player.VisibleRank,
            player.RankRating,
            change.TotalMMRChange);
        
        change.RankChange = rankChange;
        
        return change;
    }
    
    private double CalculatePerformanceModifier(
        PerformanceStats stats,
        GameResult gameResult,
        double playerMMR)
    {
        double modifier = 0.0;
        
        // ACS (Average Combat Score) 기반 평가
        var expectedACS = GetExpectedACS(playerMMR);
        var acsRatio = stats.AverageCombatScore / expectedACS;
        
        // ACS가 예상보다 높으면 보너스, 낮으면 페널티
        modifier += (acsRatio - 1.0) * 15.0;
        
        // K/D 비율 고려
        var expectedKD = GetExpectedKD(playerMMR);
        var kdRatio = stats.KillDeathRatio / expectedKD;
        modifier += (kdRatio - 1.0) * 10.0;
        
        // 팀 기여도 (어시스트, 유틸리티 사용 등)
        var teamContribution = CalculateTeamContribution(stats);
        modifier += teamContribution * 0.1;
        
        // MVP 성과
        if (stats.IsMVP)
        {
            modifier += 5.0;
        }
        
        // 점수 차이에 따른 조정 (대패/대승 시 조정)
        var scoreDifference = Math.Abs(gameResult.TeamScore - gameResult.OpponentScore);
        var blowoutModifier = scoreDifference > 8 ? -2.0 : 0.0;
        modifier += blowoutModifier;
        
        // 최대/최소 제한
        return Math.Clamp(modifier, -15.0, 15.0);
    }
    
    private double CalculateTeamContribution(PerformanceStats stats)
    {
        double contribution = 0.0;
        
        // 어시스트 기여도
        contribution += stats.Assists * 0.3;
        
        // 첫 킬 기여도
        contribution += stats.FirstBloods * 0.5;
        
        // 클러치 성공
        contribution += stats.ClutchesWon * 1.0;
        
        // 유틸리티 사용 효율
        contribution += stats.UtilityDamage / 100.0;
        
        return contribution;
    }
    
    // Valorant 스타일의 매칭 범위 확장
    public async Task<MatchingRangeConfig> GetMatchingRangeAsync(
        Player player,
        TimeSpan waitTime)
    {
        var config = new MatchingRangeConfig
        {
            PlayerId = player.PlayerId,
            WaitTime = waitTime
        };
        
        // 기본 MMR 범위
        double baseRange = 100.0;
        
        // 대기 시간에 따른 범위 확장 (Valorant는 점진적 확장)
        if (waitTime.TotalSeconds <= 30)
        {
            config.MMRRange = baseRange;
        }
        else if (waitTime.TotalSeconds <= 60)
        {
            config.MMRRange = baseRange + 50;
        }
        else if (waitTime.TotalSeconds <= 120)
        {
            config.MMRRange = baseRange + 100;
        }
        else if (waitTime.TotalSeconds <= 300)
        {
            config.MMRRange = baseRange + 200;
        }
        else
        {
            config.MMRRange = baseRange + 300;
        }
        
        // 높은 MMR 플레이어는 더 넓은 범위 허용
        if (player.HiddenMMR > 2500)  // Immortal/Radiant
        {
            config.MMRRange *= 1.5;
        }
        
        // 낮은 플레이어 풀에서는 범위 확장
        var playersInRange = await GetPlayersInMMRRangeAsync(
            player.HiddenMMR - config.MMRRange,
            player.HiddenMMR + config.MMRRange);
        
        if (playersInRange < 20)
        {
            config.MMRRange *= 1.3;
            config.IsExpanded = true;
        }
        
        return config;
    }
    
    // Valorant의 파티 매칭 처리
    public async Task<Match> CreatePartyMatchAsync(Party party)
    {
        // Valorant는 파티의 가장 높은 MMR 플레이어 기준으로 매칭
        var highestMMR = party.Members.Max(m => m.HiddenMMR);
        var averageMMR = party.Members.Average(m => m.HiddenMMR);
        
        // MMR 격차가 큰 파티는 페널티
        var mmrSpread = party.Members.Max(m => m.HiddenMMR) - 
                       party.Members.Min(m => m.HiddenMMR);
        
        double effectiveMMR;
        
        if (mmrSpread > 300)
        {
            // 큰 격차: 최고 MMR에 가까운 매칭
            effectiveMMR = highestMMR * 0.9 + averageMMR * 0.1;
            
            _logger.LogInformation(
                "파티 MMR 격차 큼 - Spread: {Spread}, EffectiveMMR: {MMR}",
                mmrSpread, effectiveMMR);
        }
        else if (mmrSpread > 150)
        {
            // 중간 격차: 가중 평균
            effectiveMMR = highestMMR * 0.7 + averageMMR * 0.3;
        }
        else
        {
            // 작은 격차: 평균 사용
            effectiveMMR = averageMMR;
        }
        
        // 상대 팀 찾기
        var opponents = await FindOpponentPartyAsync(
            party.Size,
            effectiveMMR,
            party.PreferredGameMode);
        
        if (opponents != null)
        {
            return CreateMatchFromParties(party, opponents);
        }
        
        // 상대 파티를 찾지 못한 경우 솔로 플레이어로 팀 구성
        var soloPlayers = await FindSoloPlayersAsync(
            5 - party.Size,
            effectiveMMR);
        
        if (soloPlayers.Count >= 5 - party.Size)
        {
            return CreateMixedMatch(party, soloPlayers);
        }
        
        return null;
    }
    
    private double CalculateExpectedWinRate(double playerMMR, double opponentMMR)
    {
        return 1.0 / (1.0 + Math.Pow(10, (opponentMMR - playerMMR) / 400.0));
    }
    
    private double CalculateKFactor(double confidence, int gamesPlayed)
    {
        // 신뢰도가 낮거나 게임 수가 적으면 K-Factor 증가
        if (gamesPlayed < 10)
            return 0.8;  // 배치 게임
        else if (gamesPlayed < 50)
            return 0.5;  // 초기 게임
        else if (confidence < 0.6)
            return 0.4;  // 낮은 신뢰도
        else
            return 0.3;  // 일반
    }
    
    private double UpdateConfidence(double currentConfidence)
    {
        // 게임을 할수록 신뢰도 증가 (최대 1.0)
        return Math.Min(1.0, currentConfidence + 0.01);
    }
    
    private RankChange CalculateRankChange(
        ValorantRank currentRank,
        int currentRating,
        double mmrChange)
    {
        var change = new RankChange();
        
        // MMR 변화를 랭크 레이팅 변화로 변환
        var ratingChange = (int)(mmrChange / 2.0);  // 대략 50% 비율
        
        var newRating = currentRating + ratingChange;
        var newRank = currentRank;
        
        // 랭크업
        while (newRating >= 100)
        {
            newRating -= 100;
            newRank = (ValorantRank)Math.Min((int)ValorantRank.Radiant, (int)newRank + 1);
        }
        
        // 랭크다운
        while (newRating < 0)
        {
            newRating += 100;
            newRank = (ValorantRank)Math.Max((int)ValorantRank.Iron1, (int)newRank - 1);
        }
        
        change.OldRank = currentRank;
        change.NewRank = newRank;
        change.OldRating = currentRating;
        change.NewRating = newRating;
        change.RatingChange = ratingChange;
        
        return change;
    }
    
    private double GetExpectedACS(double mmr)
    {
        // MMR에 따른 예상 ACS (간단한 선형 모델)
        return 150 + (mmr - 1000) * 0.05;
    }
    
    private double GetExpectedKD(double mmr)
    {
        // MMR에 따른 예상 K/D
        return 0.8 + (mmr - 1000) * 0.0002;
    }
    
    private async Task<int> GetPlayersInMMRRangeAsync(double minMMR, double maxMMR)
    {
        // Redis sorted set으로 빠른 범위 쿼리
        var key = "matchmaking:mmr_index";
        return (int)await _redisDb.SortedSetLengthAsync(
            key,
            minMMR,
            maxMMR);
    }
    
    private async Task<Party> FindOpponentPartyAsync(
        int size,
        double targetMMR,
        GameMode gameMode)
    {
        var key = $"matchmaking:parties:{gameMode}:{size}";
        var partyIds = await _redisDb.SortedSetRangeByScoreAsync(
            key,
            targetMMR - 100,
            targetMMR + 100);
        
        foreach (var partyId in partyIds)
        {
            var partyKey = $"party:{partyId}";
            var partyJson = await _redisDb.StringGetAsync(partyKey);
            
            if (partyJson.HasValue)
            {
                return JsonSerializer.Deserialize<Party>(partyJson);
            }
        }
        
        return null;
    }
    
    private async Task<List<Player>> FindSoloPlayersAsync(int count, double targetMMR)
    {
        var key = "matchmaking:solo";
        var playerIds = await _redisDb.SortedSetRangeByScoreAsync(
            key,
            targetMMR - 150,
            targetMMR + 150,
            take: count * 2);
        
        var players = new List<Player>();
        
        foreach (var playerId in playerIds.Take(count))
        {
            var playerKey = $"player:{playerId}";
            var playerJson = await _redisDb.StringGetAsync(playerKey);
            
            if (playerJson.HasValue)
            {
                players.Add(JsonSerializer.Deserialize<Player>(playerJson));
            }
        }
        
        return players;
    }
    
    private Match CreateMatchFromParties(Party party1, Party party2)
    {
        return new Match
        {
            MatchId = Guid.NewGuid().ToString(),
            Team1 = party1.Members,
```

```csharp
    public async Task<SRChange> CalculateSRChangeAsync(
        Player player,
        GameResult gameResult,
        PerformanceStats stats)
    {
        var change = new SRChange
        {
            PlayerId = player.PlayerId,
            Role = player.SelectedRole
        };
        
        var currentSR = GetPlayerRoleSR(player, player.SelectedRole);
        
        // 1. 기본 SR 변화 (승패)
        var baseSRChange = gameResult.IsWin ? 25 : -25;
        
        // 2. 예상 결과와의 차이
        var expectedWinRate = CalculateExpectedWinRate(
            gameResult.TeamAverageSR,
            gameResult.OpponentAverageSR);
        
        var surprise = gameResult.IsWin ? (1.0 - expectedWinRate) : expectedWinRate;
        var surpriseBonus = (int)(surprise * 10);
        
        // 3. 성과 기반 조정 (역할별로 다른 기준)
        var performanceModifier = CalculateRolePerformanceModifier(
            player.SelectedRole,
            stats,
            currentSR);
        
        // 4. 언더독 보너스/페널티
        var underdogModifier = 0;
        if (Math.Abs(gameResult.TeamAverageSR - gameResult.OpponentAverageSR) > 200)
        {
            if (gameResult.IsWin && gameResult.TeamAverageSR < gameResult.OpponentAverageSR)
            {
                underdogModifier = 10;  // 약팀이 이기면 보너스
            }
            else if (!gameResult.IsWin && gameResult.TeamAverageSR > gameResult.OpponentAverageSR)
            {
                underdogModifier = -5;  // 강팀이 지면 추가 페널티
            }
        }
        
        // 5. 연승/연패 보너스
        var streakBonus = await CalculateStreakBonusAsync(player.PlayerId, gameResult.IsWin);
        
        // 최종 SR 변화
        change.BaseSRChange = baseSRChange;
        change.SurpriseBonus = surpriseBonus;
        change.PerformanceModifier = performanceModifier;
        change.UnderdogModifier = underdogModifier;
        change.StreakBonus = streakBonus;
        change.TotalSRChange = baseSRChange + surpriseBonus + 
                              performanceModifier + underdogModifier + streakBonus;
        
        // SR 범위 제한 (0-5000)
        var newSR = Math.Clamp(currentSR + change.TotalSRChange, 0, 5000);
        change.NewSR = newSR;
        change.NewTier = DetermineTier(newSR);
        
        return change;
    }
    
    private int CalculateRolePerformanceModifier(
        PlayerRole role,
        PerformanceStats stats,
        int currentSR)
    {
        int modifier = 0;
        
        switch (role)
        {
            case PlayerRole.Tank:
                // 탱크: 데미지 흡수, 공간 창출
                modifier += CalculateTankPerformance(stats, currentSR);
                break;
                
            case PlayerRole.DPS:
                // 딜러: 킬, 데미지
                modifier += CalculateDPSPerformance(stats, currentSR);
                break;
                
            case PlayerRole.Support:
                // 서포터: 힐링, 어시스트
                modifier += CalculateSupportPerformance(stats, currentSR);
                break;
        }
        
        return Math.Clamp(modifier, -10, 10);
    }
    
    private int CalculateTankPerformance(PerformanceStats stats, int currentSR)
    {
        int modifier = 0;
        
        // 데미지 차단
        var expectedDamageBlocked = GetExpectedDamageBlocked(currentSR);
        if (stats.DamageBlocked > expectedDamageBlocked * 1.2)
            modifier += 5;
        else if (stats.DamageBlocked < expectedDamageBlocked * 0.8)
            modifier -= 3;
        
        // 생존 시간
        if (stats.AverageLifetime > 60)
            modifier += 2;
        
        // 공격적인 탱킹 (엘리미네이션 기여)
        if (stats.Eliminations > 15)
            modifier += 3;
        
        return modifier;
    }
    
    private int CalculateDPSPerformance(PerformanceStats stats, int currentSR)
    {
        int modifier = 0;
        
        // 킬/데스 비율
        if (stats.KillDeathRatio > 2.5)
            modifier += 5;
        else if (stats.KillDeathRatio < 1.0)
            modifier -= 4;
        
        // 데미지 기여
        var expectedDamage = GetExpectedDamage(currentSR);
        if (stats.HeroDamage > expectedDamage * 1.3)
            modifier += 5;
        
        // 결정적 킬 (첫킬, 파이널블로우)
        if (stats.FinalBlows > 20)
            modifier += 3;
        
        return modifier;
    }
    
    private int CalculateSupportPerformance(PerformanceStats stats, int currentSR)
    {
        int modifier = 0;
        
        // 힐링 기여
        var expectedHealing = GetExpectedHealing(currentSR);
        if (stats.Healing > expectedHealing * 1.3)
            modifier += 6;
        else if (stats.Healing < expectedHealing * 0.7)
            modifier -= 4;
        
        // 생존율 (서포터는 살아있어야 팀 지원 가능)
        if (stats.Deaths < 5)
            modifier += 4;
        else if (stats.Deaths > 10)
            modifier -= 3;
        
        // 공격 어시스트
        if (stats.OffensiveAssists > 10)
            modifier += 2;
        
        return modifier;
    }
    
    private async Task<int> CalculateStreakBonusAsync(string playerId, bool isWin)
    {
        var streakKey = $"streak:{playerId}";
        var currentStreak = (int)await _redisDb.StringGetAsync(streakKey);
        
        int newStreak;
        if (isWin)
        {
            newStreak = currentStreak > 0 ? currentStreak + 1 : 1;
        }
        else
        {
            newStreak = currentStreak < 0 ? currentStreak - 1 : -1;
        }
        
        await _redisDb.StringSetAsync(streakKey, newStreak);
        
        // 연승 보너스 (최대 5연승까지)
        if (newStreak >= 3)
        {
            return Math.Min(newStreak - 2, 3);
        }
        
        // 연패 완화 (계속 지면 손실 감소)
        if (newStreak <= -3)
        {
            return Math.Max(newStreak + 2, -3);
        }
        
        return 0;
    }
    
    private double CalculateExpectedWinRate(double teamSR, double opponentSR)
    {
        return 1.0 / (1.0 + Math.Pow(10, (opponentSR - teamSR) / 400.0));
    }
```

```csharp
            // 2. 더 나은 매칭 가능성 확인
            var betterMatchPossible = await CheckBetterMatchPossibilityAsync(
                proposedMatch,
                allCandidates);
            
            if (betterMatchPossible)
            {
                _logger.LogInformation(
                    "더 나은 매칭 가능성으로 대기 - MatchId: {MatchId}",
                    proposedMatch.MatchId);
                return true;
            }
            
            // 3. 플레이어 대기 시간 고려
            var maxWaitTime = proposedMatch.Team1
                .Concat(proposedMatch.Team2)
                .Max(p => (DateTime.UtcNow - p.JoinedQueueAt).TotalSeconds);
            
            // 2분 이상 대기했으면 품질이 낮아도 매칭 진행
            if (maxWaitTime > 120)
            {
                return false;
            }
            
            return false;
        }
        
        private async Task<bool> CheckBetterMatchPossibilityAsync(
            Match currentMatch,
            List<Player> candidates)
        {
            // 대기 중인 다른 플레이어로 더 나은 매칭 가능한지 확인
            var currentPlayers = currentMatch.Team1.Concat(currentMatch.Team2).ToList();
            var otherPlayers = candidates.Except(currentPlayers).ToList();
            
            if (otherPlayers.Count < 2)
                return false;  // 대체 가능한 플레이어 부족
            
            // 간단한 휴리스틱: 비슷한 MMR의 대기 플레이어가 있는지
            var currentAvgMMR = currentPlayers.Average(p => p.MMR);
            var betterCandidates = otherPlayers.Where(p => 
                Math.Abs(p.MMR - currentAvgMMR) < 50).ToList();
            
            return betterCandidates.Count >= 2;
        }
        
        private double CalculateMatchQuality(Match match)
        {
            var team1Avg = match.Team1.Average(p => p.MMR);
            var team2Avg = match.Team2.Average(p => p.MMR);
            var balance = Math.Abs(team1Avg - team2Avg);
            
            // 100점 만점 (차이가 작을수록 높은 점수)
            return Math.Max(0, 100 - balance / 5.0);
        }
    }

    public class PartyBalancing
    {
        public static bool IsPartyMatchupFair(
            List<Party> team1Parties,
            List<Party> team2Parties)
        {
            // 파티 크기 분포가 유사한지 확인
            var team1Sizes = team1Parties.Select(p => p.Size).OrderBy(s => s).ToList();
            var team2Sizes = team2Parties.Select(p => p.Size).OrderBy(s => s).ToList();
            
            // 파티 구성이 유사하면 공정
            // 예: [2, 3] vs [2, 3]는 OK, [5] vs [1, 1, 1, 1, 1]는 불공정
            
            if (team1Sizes.Count != team2Sizes.Count)
                return false;
            
            for (int i = 0; i < team1Sizes.Count; i++)
            {
                if (Math.Abs(team1Sizes[i] - team2Sizes[i]) > 1)
                    return false;
            }
            
            return true;
        }
        
        public static double CalculatePartyCommunicationAdvantage(Party party)
        {
            // 파티는 의사소통 이점이 있음
            // 파티 크기에 따른 MMR 보정값
            return party.Size switch
            {
                2 => 20,   // 듀오: +20 MMR
                3 => 40,   // 트리오: +40 MMR
                4 => 70,   // 4인: +70 MMR
                5 => 100,  // 풀파티: +100 MMR
                _ => 0
            };
        }
    }

            // 비피크 시간
            return new TimeBasedMatchingStrategy
            {
                MaxWaitTime = TimeSpan.FromSeconds(120),  // 긴 대기 허용
                StrictQuality = false,  // 완화된 품질
```
