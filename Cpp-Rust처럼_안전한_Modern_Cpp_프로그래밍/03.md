# 3장: 타입 안전성과 컴파일 타임 검증
타입 시스템은 프로그램의 정확성을 보장하는 첫 번째 방어선이다. 강력한 타입 시스템을 활용하면 많은 버그를 컴파일 타임에 잡아낼 수 있다. 이 장에서는 C++의 타입 시스템을 최대한 활용하여 런타임 오류를 사전에 방지하는 방법을 배운다.

## 3.1 강타입 시스템 구축하기
C++은 정적 타입 언어지만, 암묵적 형변환과 약한 typedef로 인해 타입 안전성이 떨어질 수 있다. 강타입 시스템을 구축하면 이러한 문제를 해결할 수 있다.

### 암묵적 형변환의 위험성

```cpp
// 안전하지 않은 코드: 단위 혼동
void unsafe_unit_confusion() {
    double distance_meters = 100.0;
    double distance_feet = 328.0;
    
    // 컴파일러는 이 실수를 잡아내지 못함
    double total = distance_meters + distance_feet;  // 단위가 다른 값을 더함!
    
    std::cout << "총 거리: " << total << std::endl;  // 의미 없는 값
}

// 안전하지 않은 코드: ID 혼동
void unsafe_id_confusion() {
    using UserId = int;
    using ProductId = int;
    
    UserId user = 123;
    ProductId product = 456;
    
    // 컴파일러는 이 실수를 잡아내지 못함
    if (user == product) {  // 다른 종류의 ID를 비교!
        std::cout << "일치합니다" << std::endl;
    }
}
```

위 코드들은 컴파일은 되지만 논리적으로 잘못되었다. typedef는 타입의 별칭일 뿐 새로운 타입을 만들지 않기 때문에 이러한 실수를 막을 수 없다.

### 강타입 래퍼 구현

```cpp
// 안전한 코드: 강타입 래퍼
template<typename T, typename Tag>
class StrongType {
    T value;
    
public:
    explicit constexpr StrongType(T v) : value(v) {}
    
    constexpr T get() const { return value; }
    
    // 명시적 변환만 허용
    explicit constexpr operator T() const { return value; }
    
    // 같은 타입끼리만 비교 가능
    constexpr bool operator==(const StrongType& other) const {
        return value == other.value;
    }
    
    constexpr auto operator<=>(const StrongType& other) const {
        return value <=> other.value;
    }
};

// 태그 타입으로 서로 다른 강타입 생성
struct MeterTag {};
struct FeetTag {};

using Meters = StrongType<double, MeterTag>;
using Feet = StrongType<double, FeetTag>;

void safe_unit_usage() {
    Meters distance_m{100.0};
    Feet distance_ft{328.0};
    
    // 컴파일 에러: 다른 타입끼리 연산 불가
    // auto total = distance_m.get() + distance_ft.get();
    
    // 명시적 변환 필요
    auto total_meters = distance_m.get() + (distance_ft.get() * 0.3048);
    
    std::cout << "총 거리: " << total_meters << "미터" << std::endl;
}
```

강타입 래퍼는 암묵적 형변환을 방지하여 타입 안전성을 크게 향상시킨다.

### ID 타입 안전성

```cpp
// 안전한 코드: 강타입 ID
struct UserIdTag {};
struct ProductIdTag {};

using UserId = StrongType<int, UserIdTag>;
using ProductId = StrongType<int, ProductIdTag>;

class UserDatabase {
    std::map<UserId, std::string> users;
    
public:
    void add_user(UserId id, const std::string& name) {
        users[id] = name;
    }
    
    std::optional<std::string> get_user(UserId id) const {
        auto it = users.find(id);
        if (it != users.end()) {
            return it->second;
        }
        return std::nullopt;
    }
};

class ProductDatabase {
    std::map<ProductId, std::string> products;
    
public:
    void add_product(ProductId id, const std::string& name) {
        products[id] = name;
    }
    
    std::optional<std::string> get_product(ProductId id) const {
        auto it = products.find(id);
        if (it != products.end()) {
            return it->second;
        }
        return std::nullopt;
    }
};

void safe_id_usage() {
    UserDatabase user_db;
    ProductDatabase product_db;
    
    UserId user{123};
    ProductId product{456};
    
    user_db.add_user(user, "김철수");
    product_db.add_product(product, "노트북");
    
    // 컴파일 에러: 잘못된 ID 타입
    // auto name = user_db.get_user(product);
    
    // 올바른 사용
    if (auto name = user_db.get_user(user)) {
        std::cout << "사용자: " << *name << std::endl;
    }
}
```

### 연산 제한이 있는 강타입
특정 연산만 허용하는 강타입을 만들 수 있다.

```cpp
// 더하기만 가능한 타입
template<typename T, typename Tag>
class AdditiveType {
    T value;
    
public:
    explicit constexpr AdditiveType(T v) : value(v) {}
    
    constexpr T get() const { return value; }
    
    // 더하기만 허용
    constexpr AdditiveType operator+(const AdditiveType& other) const {
        return AdditiveType{value + other.value};
    }
    
    constexpr AdditiveType& operator+=(const AdditiveType& other) {
        value += other.value;
        return *this;
    }
    
    // 곱하기는 스칼라만 가능
    constexpr AdditiveType operator*(T scalar) const {
        return AdditiveType{value * scalar};
    }
};

struct MoneyTag {};
using Money = AdditiveType<double, MoneyTag>;

void money_operations() {
    Money price1{100.0};
    Money price2{200.0};
    
    Money total = price1 + price2;  // OK
    Money discounted = total * 0.9;  // OK
    
    // 컴파일 에러: Money끼리 곱하기 불가
    // Money invalid = price1 * price2;
    
    std::cout << "총액: " << total.get() << "원" << std::endl;
    std::cout << "할인가: " << discounted.get() << "원" << std::endl;
}
```

### 단위 변환 시스템

```cpp
// 안전한 단위 변환 시스템
template<typename T, typename Tag>
class Unit {
    T value;
    
public:
    explicit constexpr Unit(T v) : value(v) {}
    
    constexpr T get() const { return value; }
    
    constexpr Unit operator+(const Unit& other) const {
        return Unit{value + other.value};
    }
    
    constexpr Unit operator-(const Unit& other) const {
        return Unit{value - other.value};
    }
    
    template<typename U>
    constexpr Unit operator*(U scalar) const {
        return Unit{value * scalar};
    }
    
    template<typename U>
    constexpr Unit operator/(U scalar) const {
        return Unit{value / scalar};
    }
};

struct KilometerTag {};
struct MileTag {};

using Kilometers = Unit<double, KilometerTag>;
using Miles = Unit<double, MileTag>;

// 타입 안전한 변환 함수
constexpr Kilometers to_kilometers(Miles miles) {
    return Kilometers{miles.get() * 1.60934};
}

constexpr Miles to_miles(Kilometers km) {
    return Miles{km.get() / 1.60934};
}

void unit_conversion_example() {
    Kilometers distance_km{100.0};
    Miles distance_mi{62.1371};
    
    // 직접 연산 불가
    // auto total = distance_km + distance_mi;  // 컴파일 에러
    
    // 명시적 변환 후 연산
    auto total_km = distance_km + to_kilometers(distance_mi);
    auto total_mi = to_miles(distance_km) + distance_mi;
    
    std::cout << "총 거리: " << total_km.get() << "km" << std::endl;
    std::cout << "총 거리: " << total_mi.get() << "마일" << std::endl;
}
```

## 3.2 C++26 Contract 기능 활용
C++26은 Contract 프로그래밍을 공식적으로 지원한다. Contract는 함수의 사전조건, 사후조건, 불변식을 명시적으로 선언하여 프로그램의 정확성을 검증한다.

### 사전조건(Precondition)과 사후조건(Postcondition)
사전조건은 함수 호출 전에 만족해야 하는 조건이고, 사후조건은 함수 실행 후에 보장되는 조건이다.

```cpp
// 안전하지 않은 코드: 암묵적 가정
int unsafe_divide(int a, int b) {
    // b가 0이 아니라고 가정하지만 명시되지 않음
    return a / b;  // b가 0이면 미정의 동작!
}

void unsafe_usage() {
    int result = unsafe_divide(10, 0);  // 런타임에 크래시
}
```

Contract를 사용하면 가정을 명시적으로 표현할 수 있다.

```cpp
// 안전한 코드: Contract 사용
int safe_divide(int a, int b)
    pre(b != 0)  // 사전조건: b는 0이 아니어야 함
{
    return a / b;
}

void safe_usage() {
    // 컴파일러나 런타임이 사전조건 위반을 감지
    int result = safe_divide(10, 0);  // 계약 위반 감지!
}
```

사후조건은 함수의 반환값이나 상태를 검증한다.

```cpp
// 정렬 함수의 계약
void sort_array(std::vector<int>& arr)
    post(std::is_sorted(arr.begin(), arr.end()))  // 사후조건: 정렬됨
{
    std::sort(arr.begin(), arr.end());
}

// 제곱근 함수의 계약
double square_root(double x)
    pre(x >= 0)  // 사전조건: 음수가 아님
    post(r >= 0 && std::abs(r * r - x) < 0.0001)  // 사후조건: 결과 검증
{
    return std::sqrt(x);
}
```

### 복잡한 사전조건 표현

```cpp
// 범위 검사를 포함한 계약
template<typename T>
T get_element(const std::vector<T>& vec, size_t index)
    pre(index < vec.size())  // 범위 검사
{
    return vec[index];
}

// 여러 조건을 포함한 계약
void transfer_money(Account& from, Account& to, double amount)
    pre(amount > 0)  // 양수 금액
    pre(from.get_balance() >= amount)  // 잔액 충분
    post(from.get_balance() == old(from.get_balance()) - amount)  // 출금 확인
    post(to.get_balance() == old(to.get_balance()) + amount)  // 입금 확인
{
    from.withdraw(amount);
    to.deposit(amount);
}
```

`old()` 키워드는 함수 시작 시점의 값을 참조한다.

### 불변식(Invariant) 검증
불변식은 클래스의 생명주기 동안 항상 유지되어야 하는 조건이다.

```cpp
// 안전하지 않은 코드: 암묵적 불변식
class UnsafeBankAccount {
    double balance;
    
public:
    UnsafeBankAccount(double initial) : balance(initial) {}
    
    void withdraw(double amount) {
        balance -= amount;  // 음수 잔액 가능!
    }
    
    double get_balance() const { return balance; }
};
```

Contract를 사용하여 불변식을 명시적으로 검증한다.

```cpp
// 안전한 코드: 명시적 불변식
class SafeBankAccount {
    double balance;
    
    // 클래스 불변식
    invariant(balance >= 0);  // 잔액은 항상 0 이상
    
public:
    SafeBankAccount(double initial)
        pre(initial >= 0)
        : balance(initial) {}
    
    void withdraw(double amount)
        pre(amount > 0)
        pre(balance >= amount)  // 인출 가능 여부 확인
    {
        balance -= amount;
    }
    
    void deposit(double amount)
        pre(amount > 0)
    {
        balance += amount;
    }
    
    double get_balance() const { return balance; }
};
```

불변식은 모든 public 멤버 함수의 시작과 끝에서 자동으로 검증된다.

### 실전 예제: 스택 자료구조

```cpp
template<typename T, size_t Capacity>
class BoundedStack {
    std::array<T, Capacity> data;
    size_t top_index;
    
    // 불변식: top_index는 항상 유효한 범위
    invariant(top_index <= Capacity);
    
public:
    BoundedStack() : top_index(0) {}
    
    void push(const T& value)
        pre(!is_full())  // 스택이 가득 차지 않았어야 함
        post(size() == old(size()) + 1)  // 크기 증가 확인
        post(top() == value)  // 새 값이 최상위에 있음
    {
        data[top_index++] = value;
    }
    
    T pop()
        pre(!is_empty())  // 스택이 비어있지 않아야 함
        post(size() == old(size()) - 1)  // 크기 감소 확인
    {
        return data[--top_index];
    }
    
    const T& top() const
        pre(!is_empty())
    {
        return data[top_index - 1];
    }
    
    bool is_empty() const {
        return top_index == 0;
    }
    
    bool is_full() const {
        return top_index == Capacity;
    }
    
    size_t size() const {
        return top_index;
    }
};

void stack_usage() {
    BoundedStack<int, 10> stack;
    
    stack.push(1);
    stack.push(2);
    stack.push(3);
    
    // 정상 동작
    std::cout << stack.top() << std::endl;  // 3
    
    // 계약 위반 시도
    BoundedStack<int, 2> small_stack;
    small_stack.push(1);
    small_stack.push(2);
    // small_stack.push(3);  // 사전조건 위반: 스택이 가득 찼음
}
```

## 3.3 컴파일 타임 오류 검출
컴파일 타임에 오류를 잡으면 런타임 비용이 전혀 들지 않으면서도 안전성을 보장할 수 있다.

### `static_assert`와 사용자 정의 메시지

```cpp
// 안전하지 않은 코드: 런타임 검사
template<typename T>
void unsafe_runtime_check(T value) {
    if (sizeof(T) != 4) {
        throw std::runtime_error("T는 4바이트여야 합니다");
    }
    // 처리...
}
```

런타임 검사는 실행 시점에 오버헤드가 발생하고, 테스트되지 않은 경로에서는 감지되지 않을 수 있다.

```cpp
// 안전한 코드: 컴파일 타임 검사
template<typename T>
void safe_compile_time_check(T value) {
    static_assert(sizeof(T) == 4, "T는 4바이트여야 합니다");
    // 처리...
}

void compile_time_check_example() {
    safe_compile_time_check(int32_t{42});  // OK
    // safe_compile_time_check(int64_t{42});  // 컴파일 에러
}
```

C++26에서는 더욱 향상된 메시지를 제공할 수 있다.

```cpp
// C++26: 개선된 static_assert 메시지
template<typename T>
concept Arithmetic = std::is_arithmetic_v<T>;

template<typename T>
void process_number(T value) {
    static_assert(Arithmetic<T>,
        "T는 산술 타입이어야 합니다. "
        "정수 또는 부동소수점 타입을 사용하세요.");
    
    static_assert(sizeof(T) >= 4,
        "T는 최소 4바이트여야 합니다. "
        "현재 크기: " + std::to_string(sizeof(T)) + "바이트");
    
    // 처리...
}
```

Concept을 사용하면 명확한 컴파일 에러 메시지를 얻을 수 있다.

concept Addable = requires(T a, T b) {
    { a + b } -> std::convertible_to<T>;
};

template<Addable T>
T safe_add(T a, T b) {
    return a + b;
}

template<typename T>
concept Numeric = std::is_arithmetic_v<T>;

template<Numeric T>
T multiply(T a, T b) {
    return a * b;
}

void safe_template_usage() {
    auto result1 = safe_add(10, 20);  // OK
    auto result2 = multiply(3.14, 2.0);  // OK
    
    // 명확한 컴파일 에러: std::string은 Addable이 아님
    // auto result3 = safe_add(std::string{"hello"}, std::string{"world"});

### 복잡한 Concept 조합

```cpp
// 여러 조건을 조합한 Concept
template<typename T>
concept Container = requires(T t) {
    typename T::value_type;
    typename T::iterator;
    { t.begin() } -> std::same_as<typename T::iterator>;
    { t.end() } -> std::same_as<typename T::iterator>;
    { t.size() } -> std::convertible_to<size_t>;
};

template<typename T>
concept SortableContainer = Container<T> && requires(T t) {
    requires std::is_same_v<
        decltype(t.begin()),
        typename std::vector<typename T::value_type>::iterator
    >;
};

template<Container C>
void print_container(const C& container) {
    for (const auto& item : container) {
        std::cout << item << " ";
    }
    std::cout << std::endl;
}

template<SortableContainer C>
void sort_and_print(C& container) {
    std::sort(container.begin(), container.end());
    print_container(container);
}
```

constexpr bool is_power_of_two(unsigned int n) {

// 컴파일 타임 문자열 유효성 검사
constexpr bool is_valid_identifier(const char* str) {
    if (!str || !*str) return false;
    
    // 첫 문자는 알파벳 또는 _
    if (!((*str >= 'a' && *str <= 'z') || 
          (*str >= 'A' && *str <= 'Z') || 
          *str == '_')) {
        return false;
    }
    
    // 나머지는 알파벳, 숫자, _
    for (++str; *str; ++str) {
        if (!((*str >= 'a' && *str <= 'z') || 
              (*str >= 'A' && *str <= 'Z') || 
              (*str >= '0' && *str <= '9') || 
              *str == '_')) {
            return false;
        }
    }
    
    return true;
}

template<const char* Name>
class Variable {
    static_assert(is_valid_identifier(Name), 
                  "변수 이름은 유효한 식별자여야 합니다");
    
    int value;
    
public:
    Variable() : value(0) {}
    static constexpr const char* name() { return Name; }
};

constexpr char valid_name[] = "myVariable";
constexpr char invalid_name[] = "123invalid";

void identifier_validation() {
    Variable<valid_name> v1;  // OK
    // Variable<invalid_name> v2;  // 컴파일 에러
}

// 컴파일 타임 범위 검증 배열
template<typename T, size_t N>
class SafeArray {
    T data[N];
    
public:
    template<size_t Index>
    constexpr T& get() {
        static_assert(Index < N, "인덱스가 범위를 벗어났습니다");
        return data[Index];
    }
    
    template<size_t Index>
    constexpr const T& get() const {
        static_assert(Index < N, "인덱스가 범위를 벗어났습니다");
        return data[Index];
    }
    
    // 런타임 인덱스는 예외 발생
    constexpr T& at(size_t index) {
        if (index >= N) {
            throw std::out_of_range("인덱스가 범위를 벗어났습니다");
        }
        return data[index];
    }
};

void safe_array_example() {
    SafeArray<int, 5> arr;
    
    // 컴파일 타임 검증
    arr.get<0>() = 10;  // OK
    arr.get<4>() = 20;  // OK
    // arr.get<5>() = 30;  // 컴파일 에러: 인덱스 초과
    
    // 런타임 검증
    try {
        arr.at(3) = 15;  // OK
        arr.at(5) = 25;  // 예외 발생
    } catch (const std::out_of_range& e) {
        std::cout << "에러: " << e.what() << std::endl;
    }
}

template<typename T>
void process_trivially_copyable(const T& value) {
    static_assert(std::is_trivially_copyable_v<T>,
                  "T는 trivially copyable 타입이어야 합니다");
    
    // memcpy 등을 안전하게 사용 가능
    T copy;
    std::memcpy(&copy, &value, sizeof(T));
}

template<typename T>
class ThreadSafeValue {
    static_assert(std::is_standard_layout_v<T>,
                  "T는 standard layout 타입이어야 합니다");
    static_assert(std::is_trivially_destructible_v<T>,
                  "T는 trivially destructible 타입이어야 합니다");
    
    std::atomic<T> value;
    
public:
    ThreadSafeValue(T initial) : value(initial) {}
    
    T load() const {
        return value.load(std::memory_order_acquire);
    }
    
    void store(T new_value) {
        value.store(new_value, std::memory_order_release);
    }
};

### 실전 예제: 타입 안전한 설정 시스템

```cpp
// 타입 안전한 설정 관리
template<typename T>
concept ConfigValue = 
    std::is_arithmetic_v<T> || 
    std::is_same_v<T, std::string> ||
    std::is_same_v<T, bool>;

template<ConfigValue T>
class TypedConfig {
    T value;
    T min_value;
    T max_value;
    
public:
    constexpr TypedConfig(T val, T min_val, T max_val)
        requires std::is_arithmetic_v<T>
        : value(val), min_value(min_val), max_value(max_val) {
        
        if (value < min_value || value > max_value) {
            throw std::out_of_range("설정 값이 범위를 벗어났습니다");
        }
    }
    
    constexpr TypedConfig(T val)
        requires (!std::is_arithmetic_v<T>)
        : value(val), min_value{}, max_value{} {}
    
    constexpr void set(T new_value)
        requires std::is_arithmetic_v<T>
    {
        if (new_value < min_value || new_value > max_value) {
            throw std::out_of_range("설정 값이 범위를 벗어났습니다");
        }
        value = new_value;
    }
    
    constexpr void set(T new_value)
        requires (!std::is_arithmetic_v<T>)
    {
        value = new_value;
    }
    
    constexpr T get() const { return value; }
};

class ApplicationConfig {
    TypedConfig<int> max_connections{100, 1, 1000};
    TypedConfig<double> timeout_seconds{30.0, 1.0, 300.0};
    TypedConfig<std::string> server_address{"localhost"};
    TypedConfig<bool> enable_logging{true};
    
public:
    void set_max_connections(int value) {
        max_connections.set(value);
    }
    
    int get_max_connections() const {
        return max_connections.get();
    }
    
    void set_timeout(double value) {
        timeout_seconds.set(value);
    }
    
    double get_timeout() const {
        return timeout_seconds.get();
    }
};

void config_usage() {
    ApplicationConfig config;
    
    config.set_max_connections(200);  // OK
    
    try {
        config.set_max_connections(2000);  // 예외: 범위 초과
    } catch (const std::out_of_range& e) {
        std::cout << "설정 오류: " << e.what() << std::endl;
    }
    
    config.set_timeout(60.0);  // OK
    
    std::cout << "최대 연결: " << config.get_max_connections() << std::endl;
    std::cout << "타임아웃: " << config.get_timeout() << "초" << std::endl;
}
```

이 장에서 배운 타입 안전성 기법들을 정리하면:

1. 강타입 래퍼로 암묵적 형변환을 방지하여 논리적 오류를 컴파일 타임에 잡는다
2. Contract를 사용하여 함수의 사전조건, 사후조건, 불변식을 명시적으로 검증한다
3. `static_assert`와 Concept으로 타입 제약을 표현하여 명확한 컴파일 에러를 얻는다
4. constexpr 함수로 컴파일 타임 검증을 수행하여 런타임 오버헤드를 제거한다
