# 5장: 초기화되지 않은 메모리 접근 방지

## 5.1 C++의 미정의 동작(UB) 문제
초기화되지 않은 변수의 값을 읽는 것은 C++에서 가장 흔하면서도 위험한 미정의 동작(Undefined Behavior, UB) 중 하나다. 이러한 버그는 디버그 빌드에서는 정상적으로 작동하는 것처럼 보이다가 릴리스 빌드나 다른 환경에서 예측할 수 없는 결과를 초래한다.

### 미정의 동작이 일으키는 문제들

```cpp
// 안전하지 않은 예시: 초기화되지 않은 변수
int dangerous_function() {
    int value;  // 초기화되지 않음
    
    // 조건에 따라 초기화될 수도 있음
    if (some_condition()) {
        value = 42;
    }
    
    // value가 초기화되지 않았을 수 있음 - UB!
    return value * 2;
}

// 실제 사례: 보안 취약점
class UserSession {
    bool is_authenticated;  // 초기화 누락
    int user_id;           // 초기화 누락
    
public:
    void login(int id) {
        user_id = id;
        is_authenticated = true;
    }
    
    bool check_access() {
        // is_authenticated가 우연히 0이 아닌 값이면?
        // 인증 없이 접근이 허용될 수 있음!
        return is_authenticated;
    }
};

// 메모리 재사용으로 인한 정보 유출
void memory_leak_risk() {
    {
        std::array<char, 256> password;
        std::cin >> password.data();
        // password 사용
    }  // password 범위 종료
    
    {
        std::array<char, 256> buffer;  // 초기화 안 함
        // buffer에 이전 password 데이터가 남아있을 수 있음
        log_debug(buffer.data());  // 민감 정보 유출!
    }
}
```

이러한 문제들은 다음과 같은 심각한 결과를 초래할 수 있다:

1. **예측 불가능한 동작**: 컴파일러, 최적화 레벨, 플랫폼에 따라 다른 결과
2. **보안 취약점**: 초기화되지 않은 메모리에 민감한 정보가 남아있을 수 있음
3. **재현 불가능한 버그**: 스택 상태에 따라 간헐적으로 발생
4. **최적화 문제**: 컴파일러가 UB를 가정하고 예상치 못한 최적화를 수행

### 실제 사례 분석

```cpp
// 케이스 1: 조건부 초기화 실패
int calculate_score(bool bonus_enabled) {
    int score;

        bonus = get_bonus();
    }
    
    // bonus_enabled가 false이면 bonus는 쓰레기 값

void process_data() {

    // 전체 배열 처리 시도

        process(buffer[i]);  // i >= 50일 때 UB!
    }

    bool verbose;

    Config cfg(30);
    
    // verbose는 초기화되지 않음 - UB!

    }
```

## 5.2 C++26 Erroneous Behavior 도입
C++26은 초기화되지 않은 메모리 접근을 더 명확하게 다루기 위해 "Erroneous Behavior" 개념을 도입했다. 이는 미정의 동작(UB)보다 덜 심각하지만 여전히 오류인 동작을 정의한다.

### `[[indeterminate]]` 속성의 이해
>>> `[[indeterminate]]` 속성은 현재 GCC 16 버전에서 구현 되었다.   
  
```cpp
// C++26: indeterminate 속성으로 의도 명시
void safe_conditional_init() {
    [[indeterminate]] int value;
    
    if (should_initialize()) {
        value = 42;
    }
    
    // 컴파일러 경고 또는 에러:
    // "value가 모든 경로에서 초기화되지 않았습니다"
    return value;  // 컴파일 시점에 탐지 가능
}

// 적절한 사용: 성능을 위해 지연 초기화
[[indeterminate]] std::array<int, 1000> buffer;

void initialize_buffer(int start_value) {
    // 명시적으로 모든 요소 초기화
    for (size_t i = 0; i < buffer.size(); ++i) {
        buffer[i] = start_value + i;
    }
}

// 이후 buffer 사용은 안전함을 컴파일러에 알림
```

### 초기화되지 않은 읽기 탐지
현재 사용 가능한 도구들로 초기화되지 않은 변수를 탐지할 수 있다:

```cpp
// 컴파일러 경고 활성화
// GCC/Clang: -Wuninitialized -Wmaybe-uninitialized
// MSVC: /W4

// 안전하지 않은 예시
int unsafe_read() {
    int x;
    int y = x + 1;  // 경고: 'x'가 초기화되지 않고 사용됨
    return y;
}

// 정적 분석으로 탐지 가능
void complex_path(bool condition1, bool condition2) {
    int value;
    
    if (condition1) {
        if (condition2) {
            value = 10;
        }
    } else {
        value = 20;
    }
    
    // condition1=true, condition2=false인 경우 value 미초기화
    std::cout << value << '\n';  // 정적 분석 도구가 탐지
}

// AddressSanitizer로 런타임 탐지
void runtime_detection() {
    int* ptr = new int;  // 초기화 안 함
    
    // ASan 활성화 시 탐지:
    // use-of-uninitialized-value
    std::cout << *ptr << '\n';
    
    delete ptr;
}
```

### 안전하지 않은 방법과 안전한 방법의 비교

```cpp
// === 안전하지 않은 방법 ===
class UnsafeBuffer {
    char data[1024];  // 초기화되지 않음
    size_t size;      // 초기화되지 않음
    
public:
    UnsafeBuffer() {
        // 멤버 초기화 누락 - UB의 온상
    }
    
    void write(const char* src, size_t len) {
        // size가 초기화되지 않았을 수 있음
        if (len <= size) {  // UB!
            std::memcpy(data, src, len);
        }
    }
};

// === 안전한 방법 ===
class SafeBuffer {
    std::array<char, 1024> data{};  // 값 초기화
    size_t size = 0;                 // 명시적 초기화
    
public:
    SafeBuffer() = default;  // 멤버 초기화자로 충분
    
    void write(const char* src, size_t len) {
        // size는 항상 유효한 값
        if (len <= size) {
            std::ranges::copy_n(src, len, data.begin());
        }
    }
};

// === 조건부 초기화 비교 ===
// 안전하지 않은 방법
int unsafe_conditional() {
    int result;
    
    if (auto opt = try_get_value()) {
        result = *opt;
    }
    
    return result;  // opt가 없으면 UB
}

// 안전한 방법 1: 기본값 제공
int safe_with_default() {
    int result = 0;  // 기본값으로 초기화
    
    if (auto opt = try_get_value()) {
        result = *opt;
    }
    
    return result;  // 항상 안전
}

// 안전한 방법 2: std::optional 활용
std::optional<int> safe_with_optional() {
    std::optional<int> result;  // 명시적으로 "값 없음" 상태
    
    if (auto opt = try_get_value()) {
        result = *opt;
    }
    
    return result;  // 호출자가 값 존재 여부 확인 가능
}
```

## 5.3 안전한 변수 초기화 패턴
C++은 여러 종류의 초기화 방식을 제공하며, 각각은 다른 안전성 특성을 가진다. 올바른 초기화 패턴을 선택하는 것이 중요하다.  

### 기본 초기화 vs 값 초기화

```cpp
// 기본 초기화 (Default Initialization)
// - 기본 타입: 초기화되지 않음 (UB 위험)
// - 클래스 타입: 기본 생성자 호출
int a;              // 초기화되지 않음 - 위험!
std::string s;      // 빈 문자열로 초기화 - 안전

// 값 초기화 (Value Initialization)
// - 기본 타입: 0으로 초기화
// - 클래스 타입: 기본 생성자 호출
int b{};            // 0으로 초기화 - 안전
int c = {};         // 0으로 초기화 - 안전
std::string t{};    // 빈 문자열로 초기화 - 안전

// 배열 초기화
int arr1[10];           // 초기화되지 않음 - 위험!
int arr2[10] = {};      // 모두 0으로 초기화 - 안전
int arr3[10]{};         // 모두 0으로 초기화 - 안전

// 구조체 초기화
struct Point {
    int x;
    int y;
};

Point p1;           // x, y 초기화되지 않음 - 위험!
Point p2{};         // x=0, y=0 - 안전
Point p3{10};       // x=10, y=0 - 안전
Point p4{10, 20};   // x=10, y=20 - 안전

// 클래스 멤버 초기화
class SafeClass {
    int value = 0;              // 멤버 초기화자 사용 - 권장
    std::string name;           // 기본 생성자로 초기화 - 안전
    std::vector<int> data{};    // 빈 벡터로 초기화 - 안전
    
public:
    SafeClass() = default;      // 모든 멤버가 안전하게 초기화됨
    
    explicit SafeClass(int v) : value(v) {
        // name과 data는 멤버 초기화자로 이미 초기화됨
    }
};
```

### 실전 응용: 안전한 파서 구현

```cpp
// JSON 파서 예시
class JsonParser {
public:
    struct Token {
        enum class Type {
            String, Number, Boolean, Null, 
            ObjectStart, ObjectEnd,
            ArrayStart, ArrayEnd
        };
        
        Type type = Type::Null;  // 기본값으로 안전하게 초기화
        std::string value;       // 빈 문자열로 초기화
        size_t position = 0;     // 0으로 초기화
    };
    
    std::optional<Token> parse_next(const std::string& input, size_t& pos) {
        // 입력 검증
        if (pos >= input.size()) {
            return std::nullopt;  // 명시적으로 "토큰 없음"
        }
        
        // 공백 건너뛰기
        while (pos < input.size() && std::isspace(input[pos])) {
            ++pos;
        }
        
        if (pos >= input.size()) {
            return std::nullopt;
        }
        
        // 토큰 생성 - 모든 필드가 초기화됨
        Token token;
        token.position = pos;
        
        char current = input[pos];
        
        if (current == '{') {
            token.type = Token::Type::ObjectStart;
            ++pos;
        } else if (current == '}') {
            token.type = Token::Type::ObjectEnd;
            ++pos;
        } else if (current == '[') {
            token.type = Token::Type::ArrayStart;
            ++pos;
        } else if (current == ']') {
            token.type = Token::Type::ArrayEnd;
            ++pos;
        } else if (current == '"') {
            token.type = Token::Type::String;
            token.value = parse_string(input, pos);
        } else if (std::isdigit(current) || current == '-') {
            token.type = Token::Type::Number;
            token.value = parse_number(input, pos);
        } else {
            return std::nullopt;  // 유효하지 않은 문자
        }
        
        return token;  // 모든 경로에서 완전히 초기화된 토큰 반환
    }
    
private:
    std::string parse_string(const std::string& input, size_t& pos) {
        std::string result;  // 빈 문자열로 초기화
        ++pos;  // 시작 '"' 건너뛰기
        
        while (pos < input.size() && input[pos] != '"') {
            result += input[pos++];
        }
        
        if (pos < input.size()) {
            ++pos;  // 종료 '"' 건너뛰기
        }
        
        return result;  // 항상 유효한 문자열 반환
    }
    
    std::string parse_number(const std::string& input, size_t& pos) {
        std::string result;  // 빈 문자열로 초기화
        
        while (pos < input.size() && 
               (std::isdigit(input[pos]) || input[pos] == '.' || 
                input[pos] == '-' || input[pos] == 'e' || 
                input[pos] == 'E')) {
            result += input[pos++];
        }
        
        return result;  // 항상 유효한 문자열 반환
    }
};

// 사용 예시
void safe_json_parsing() {
    JsonParser parser;
    std::string json = R"({"name": "test", "value": 123})";
    size_t pos = 0;
    
    while (pos < json.size()) {
        auto token = parser.parse_next(json, pos);
        
        if (!token) {
            break;  // 더 이상 토큰 없음
        }
        
        // token은 항상 유효하고 완전히 초기화된 상태
        std::cout << "Token at " << token->position << '\n';
    }
}
```

```cpp
// 안전한 초기화를 위한 체크리스트
```

```cpp
// RAII 패턴으로 안전한 리소스 관리
class FileHandler {
    std::FILE* file = nullptr;  // 명시적 nullptr 초기화
    std::string filename;
    
public:
    explicit FileHandler(const std::string& name) 
        : filename(name) {
        file = std::fopen(name.c_str(), "r");
        if (!file) {
            throw std::runtime_error("파일 열기 실패: " + name);
        }
        // 생성자가 성공하면 file은 항상 유효함
    }
    
    ~FileHandler() {
        if (file) {
            std::fclose(file);
            file = nullptr;  // 방어적 코딩
        }
    }
    
    // 복사 금지
    FileHandler(const FileHandler&) = delete;
    FileHandler& operator=(const FileHandler&) = delete;
    
    // 이동 허용
    FileHandler(FileHandler&& other) noexcept 
        : file(other.file)
        , filename(std::move(other.filename)) {
        other.file = nullptr;  // 이전 객체 무효화
    }
    
    FileHandler& operator=(FileHandler&& other) noexcept {
        if (this != &other) {
            // 기존 리소스 정리
            if (file) {
                std::fclose(file);
            }
            
            // 새 리소스 획득
            file = other.file;
            filename = std::move(other.filename);
            
            // 이전 객체 무효화
            other.file = nullptr;
        }
        return *this;
    }
    
    std::string read_line() {
        if (!file) {
            throw std::runtime_error("파일이 열려있지 않음");
        }
        
        std::string line;
        char buffer[256]{};  // 버퍼 초기화
        
        if (std::fgets(buffer, sizeof(buffer), file)) {
            line = buffer;
        }
        
        return line;  // 항상 유효한 문자열 (빈 문자열 포함)
    }
};

// 사용
void safe_file_usage() {
    try {
        FileHandler handler("data.txt");
        // handler.file은 생성자 성공 시 항상 유효
        
        std::string line = handler.read_line();
        std::cout << line << '\n';
        
    } catch (const std::exception& e) {
        std::cerr << "오류: " << e.what() << '\n';
    }
    // 소멸자가 자동으로 파일 닫음
}
```
