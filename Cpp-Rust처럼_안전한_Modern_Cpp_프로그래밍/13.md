## 13.1 Saturation Arithmetic으로 오버플로우 방지

### 문제: 정수 오버플로우의 위험성

일반적인 정수 연산은 오버플로우가 발생하면 예측 불가능한 결과를 낳는다.

**안전하지 않은 코드:**

```cpp
#include <iostream>
#include <cstdint>

// 오디오 볼륨 증폭 함수 (취약한 버전)
int16_t amplify_audio_unsafe(int16_t sample, int gain) {
    return sample * gain;  // 오버플로우 발생 가능
}

int main() {
    int16_t sample = 20000;
    int gain = 2;
    
    int16_t result = amplify_audio_unsafe(sample, gain);
    std::cout << "원본 샘플: " << sample << "\n";
    std::cout << "증폭 결과: " << result << "\n";  // -25536 (오버플로우!)
    
    // 금융 계산에서의 오버플로우
    uint8_t account_balance = 250;
    uint8_t deposit = 10;
    
    uint8_t new_balance = account_balance + deposit;
    std::cout << "잔액: " << static_cast<int>(new_balance) << "\n";  // 4 (오버플로우!)
    
    return 0;
}
```

이 코드의 문제점은 다음과 같다.

- 오버플로우 발생 시 잘못된 음수 값이나 작은 값으로 wrap-around된다
- 금융, 오디오, 이미지 처리에서 심각한 논리적 오류를 초래한다
- 보안 취약점으로 이어질 수 있다

### 해결책: Saturation Arithmetic 구현

C++26은 아직 표준 saturation arithmetic을 제공하지 않지만, 안전한 구현을 만들 수 있다.

**안전한 코드:**

```cpp
#include <iostream>
#include <cstdint>
#include <limits>
#include <concepts>

// Saturation addition 템플릿
template<std::integral T>
constexpr T saturating_add(T a, T b) noexcept {
    if constexpr (std::is_unsigned_v<T>) {
        // unsigned 타입: 오버플로우 검사
        T result = a + b;
        if (result < a) {  // 오버플로우 발생
            return std::numeric_limits<T>::max();
        }
        return result;
    } else {
        // signed 타입: 양수/음수 오버플로우 모두 검사
        if (b > 0 && a > std::numeric_limits<T>::max() - b) {
            return std::numeric_limits<T>::max();
        }
        if (b < 0 && a < std::numeric_limits<T>::min() - b) {
            return std::numeric_limits<T>::min();
        }
        return a + b;
    }
}

// Saturation multiplication
template<std::integral T>
constexpr T saturating_mul(T a, T b) noexcept {
    if constexpr (std::is_unsigned_v<T>) {
        if (a == 0 || b == 0) return 0;
        if (a > std::numeric_limits<T>::max() / b) {
            return std::numeric_limits<T>::max();
        }
        return a * b;
    } else {
        // signed 곱셈의 오버플로우 검사는 더 복잡하다
        if (a == 0 || b == 0) return 0;
        
        if (a > 0 && b > 0) {
            if (a > std::numeric_limits<T>::max() / b) {
                return std::numeric_limits<T>::max();
            }
        } else if (a < 0 && b < 0) {
            if (a < std::numeric_limits<T>::max() / b) {
                return std::numeric_limits<T>::max();
            }
        } else if (a > 0 && b < 0) {
            if (b < std::numeric_limits<T>::min() / a) {
                return std::numeric_limits<T>::min();
            }
        } else {  // a < 0 && b > 0
            if (a < std::numeric_limits<T>::min() / b) {
                return std::numeric_limits<T>::min();
            }
        }
        return a * b;
    }
}

// Saturation subtraction
template<std::integral T>
constexpr T saturating_sub(T a, T b) noexcept {
    if constexpr (std::is_unsigned_v<T>) {
        if (a < b) {  // 언더플로우
            return 0;
        }
        return a - b;
    } else {
        if (b < 0 && a > std::numeric_limits<T>::max() + b) {
            return std::numeric_limits<T>::max();
        }
        if (b > 0 && a < std::numeric_limits<T>::min() + b) {
            return std::numeric_limits<T>::min();
        }
        return a - b;
    }
}

// 안전한 오디오 증폭
int16_t amplify_audio_safe(int16_t sample, int gain) {
    return saturating_mul(sample, static_cast<int16_t>(gain));
}

int main() {
    // 오디오 처리 예제
    int16_t sample = 20000;
    int gain = 2;
    
    int16_t result = amplify_audio_safe(sample, gain);
    std::cout << "원본 샘플: " << sample << "\n";
    std::cout << "안전한 증폭 결과: " << result << "\n";  // 32767 (최댓값으로 포화)
    
    // 금융 계산 예제
    uint8_t balance = 250;
    uint8_t deposit = 10;
    
    uint8_t new_balance = saturating_add(balance, deposit);
    std::cout << "안전한 잔액: " << static_cast<int>(new_balance) << "\n";  // 255 (최댓값)
    
    // 재고 관리 예제
    uint32_t inventory = 1000;
    uint32_t sale = 1200;
    
    uint32_t remaining = saturating_sub(inventory, sale);
    std::cout << "남은 재고: " << remaining << "\n";  // 0 (음수가 되지 않음)
    
    return 0;
}
```

### Saturation Arithmetic 타입 래퍼

재사용성을 높이기 위해 타입 래퍼를 만들 수 있다.

```cpp
#include <iostream>
#include <concepts>
#include <limits>

template<std::integral T>
class SaturatingInt {
    T value;
    
public:
    constexpr SaturatingInt() noexcept : value(0) {}
    constexpr explicit SaturatingInt(T val) noexcept : value(val) {}
    
    constexpr T get() const noexcept { return value; }
    
    constexpr SaturatingInt& operator+=(SaturatingInt other) noexcept {
        value = saturating_add(value, other.value);
        return *this;
    }
    
    constexpr SaturatingInt& operator-=(SaturatingInt other) noexcept {
        value = saturating_sub(value, other.value);
        return *this;
    }
    
    constexpr SaturatingInt& operator*=(SaturatingInt other) noexcept {
        value = saturating_mul(value, other.value);
        return *this;
    }
    
    constexpr SaturatingInt operator+(SaturatingInt other) const noexcept {
        return SaturatingInt(saturating_add(value, other.value));
    }
    
    constexpr SaturatingInt operator-(SaturatingInt other) const noexcept {
        return SaturatingInt(saturating_sub(value, other.value));
    }
    
    constexpr SaturatingInt operator*(SaturatingInt other) const noexcept {
        return SaturatingInt(saturating_mul(value, other.value));
    }
    
    constexpr bool operator==(SaturatingInt other) const noexcept = default;
    constexpr auto operator<=>(SaturatingInt other) const noexcept = default;
};

// 실제 사용 예제: 이미지 처리
void adjust_brightness_safe(uint8_t* pixels, size_t count, int adjustment) {
    for (size_t i = 0; i < count; ++i) {
        SaturatingInt<uint8_t> pixel(pixels[i]);
        SaturatingInt<uint8_t> adjust(static_cast<uint8_t>(
            adjustment > 0 ? adjustment : 0));
        
        if (adjustment > 0) {
            pixel += adjust;
        } else {
            pixel -= SaturatingInt<uint8_t>(static_cast<uint8_t>(-adjustment));
        }
        
        pixels[i] = pixel.get();
    }
}

int main() {
    // RGB 이미지 밝기 조정
    uint8_t image[] = {100, 200, 250, 10, 50};
    size_t pixel_count = 5;
    
    std::cout << "원본 픽셀: ";
    for (size_t i = 0; i < pixel_count; ++i) {
        std::cout << static_cast<int>(image[i]) << " ";
    }
    std::cout << "\n";
    
    adjust_brightness_safe(image, pixel_count, 50);
    
    std::cout << "밝게 조정 후: ";
    for (size_t i = 0; i < pixel_count; ++i) {
        std::cout << static_cast<int>(image[i]) << " ";
    }
    std::cout << "\n";  // 150 250 255 60 100 (255에서 포화)
    
    return 0;
}
```

## 13.2 `std::simd`를 이용한 안전한 벡터화

### 문제: 수동 SIMD의 위험성

전통적인 SIMD 프로그래밍은 플랫폼 의존적이고 타입 안전성이 부족하다.

**안전하지 않은 코드:**

```cpp
#include <iostream>
#include <immintrin.h>  // x86 SSE/AVX intrinsics

// 플랫폼 의존적이고 타입 불안전한 SIMD
void add_arrays_unsafe(const float* a, const float* b, float* result, size_t n) {
    // 정렬되지 않은 메모리 접근 가능
    // 크기가 8의 배수가 아닐 때 처리하지 않음
    for (size_t i = 0; i < n; i += 8) {
        __m256 va = _mm256_loadu_ps(&a[i]);  // 경계 검사 없음
        __m256 vb = _mm256_loadu_ps(&b[i]);
        __m256 vresult = _mm256_add_ps(va, vb);
        _mm256_storeu_ps(&result[i], vresult);
    }
}

int main() {
    float a[] = {1.0f, 2.0f, 3.0f, 4.0f, 5.0f};  // 크기가 8의 배수가 아님!
    float b[] = {10.0f, 20.0f, 30.0f, 40.0f, 50.0f};
    float result[5];
    
    add_arrays_unsafe(a, b, result, 5);  // 경계 오버런!
    
    return 0;
}
```

문제점은 다음과 같다.

- 플랫폼별로 다른 intrinsic을 사용해야 한다
- 경계 검사가 없어 버퍼 오버런이 발생할 수 있다
- 정렬되지 않은 메모리 접근 시 성능 저하 또는 크래시가 발생한다
- 타입 안전성이 보장되지 않는다

### 해결책: `std::simd` 사용

C++26의 `std::simd`는 타입 안전하고 이식 가능한 SIMD 프로그래밍을 제공한다.

**안전한 코드:**

```cpp
#include <iostream>
#include <experimental/simd>
#include <vector>
#include <span>

namespace stdx = std::experimental;

// 타입 안전하고 이식 가능한 SIMD 덧셈
template<typename T>
void add_arrays_safe(std::span<const T> a, std::span<const T> b, 
                     std::span<T> result) {
    if (a.size() != b.size() || a.size() != result.size()) {
        throw std::invalid_argument("배열 크기가 일치하지 않는다");
    }
    
    using simd_t = stdx::native_simd<T>;
    constexpr size_t simd_size = simd_t::size();
    
    size_t i = 0;
    
    // SIMD로 처리 가능한 부분
    for (; i + simd_size <= a.size(); i += simd_size) {
        simd_t va(&a[i], stdx::element_aligned);
        simd_t vb(&b[i], stdx::element_aligned);
        simd_t vresult = va + vb;
        vresult.copy_to(&result[i], stdx::element_aligned);
    }
    
    // 나머지 원소들 스칼라 처리
    for (; i < a.size(); ++i) {
        result[i] = a[i] + b[i];
    }
}

// 복잡한 연산: 가중 평균 계산
template<typename T>
void weighted_average_safe(std::span<const T> values, 
                          std::span<const T> weights,
                          std::span<T> result) {
    if (values.size() != weights.size() || values.size() != result.size()) {
        throw std::invalid_argument("배열 크기가 일치하지 않는다");
    }
    
    using simd_t = stdx::native_simd<T>;
    constexpr size_t simd_size = simd_t::size();
    
    size_t i = 0;
    
    for (; i + simd_size <= values.size(); i += simd_size) {
        simd_t v(&values[i], stdx::element_aligned);
        simd_t w(&weights[i], stdx::element_aligned);
        
        // 안전한 0 나눗셈 방지
        simd_t normalized = w / (w + simd_t(0.0001));
        simd_t weighted = v * normalized;
        
        weighted.copy_to(&result[i], stdx::element_aligned);
    }
    
    for (; i < values.size(); ++i) {
        T normalized_weight = weights[i] / (weights[i] + T(0.0001));
        result[i] = values[i] * normalized_weight;
    }
}

int main() {
    std::vector<float> a = {1.0f, 2.0f, 3.0f, 4.0f, 5.0f, 6.0f, 7.0f, 8.0f, 9.0f};
    std::vector<float> b = {10.0f, 20.0f, 30.0f, 40.0f, 50.0f, 60.0f, 70.0f, 80.0f, 90.0f};
    std::vector<float> result(a.size());
    
    add_arrays_safe<float>(a, b, result);
    
    std::cout << "SIMD 덧셈 결과: ";
    for (float val : result) {
        std::cout << val << " ";
    }
    std::cout << "\n";
    
    // 가중 평균 계산
    std::vector<float> values = {100.0f, 200.0f, 300.0f, 400.0f};
    std::vector<float> weights = {0.1f, 0.3f, 0.4f, 0.2f};
    std::vector<float> weighted_result(values.size());
    
    weighted_average_safe<float>(values, weights, weighted_result);
    
    std::cout << "가중 평균 결과: ";
    for (float val : weighted_result) {
        std::cout << val << " ";
    }
    std::cout << "\n";
    
    return 0;
}
```

### SIMD 마스킹을 통한 조건부 연산

조건부 연산도 안전하게 벡터화할 수 있다.

```cpp
#include <experimental/simd>
#include <vector>
#include <span>
#include <iostream>

namespace stdx = std::experimental;

// 조건부 클램핑: 값을 범위 내로 제한
template<typename T>
void clamp_array(std::span<T> data, T min_val, T max_val) {
    using simd_t = stdx::native_simd<T>;
    constexpr size_t simd_size = simd_t::size();
    
    simd_t vmin(min_val);
    simd_t vmax(max_val);
    
    size_t i = 0;
    
    for (; i + simd_size <= data.size(); i += simd_size) {
        simd_t v(&data[i], stdx::element_aligned);
        
        // 마스크를 사용한 조건부 연산
        auto too_low = v < vmin;
        auto too_high = v > vmax;
        
        // 조건에 맞는 원소만 업데이트
        stdx::where(too_low, v) = vmin;
        stdx::where(too_high, v) = vmax;
        
        v.copy_to(&data[i], stdx::element_aligned);
    }
    
    for (; i < data.size(); ++i) {
        if (data[i] < min_val) data[i] = min_val;
        if (data[i] > max_val) data[i] = max_val;
    }
}

// 이상치 제거 및 대체
template<typename T>
void remove_outliers(std::span<T> data, T threshold, T replacement) {
    using simd_t = stdx::native_simd<T>;
    constexpr size_t simd_size = simd_t::size();
    
    simd_t vthreshold(threshold);
    simd_t vreplacement(replacement);
    
    size_t i = 0;
    
    for (; i + simd_size <= data.size(); i += simd_size) {
        simd_t v(&data[i], stdx::element_aligned);
        
        // 절댓값이 임계값을 초과하는지 확인
        auto is_outlier = stdx::abs(v) > vthreshold;
        
        // 이상치를 대체값으로 교체
        stdx::where(is_outlier, v) = vreplacement;
        
        v.copy_to(&data[i], stdx::element_aligned);
    }
    
    for (; i < data.size(); ++i) {
        if (std::abs(data[i]) > threshold) {
            data[i] = replacement;
        }
    }
}

int main() {
    // 클램핑 예제
    std::vector<float> temperatures = {
        -50.0f, 10.0f, 25.0f, 45.0f, 100.0f, 15.0f, 30.0f, 80.0f
    };
    
    std::cout << "원본 온도: ";
    for (float temp : temperatures) {
        std::cout << temp << " ";
    }
    std::cout << "\n";
    
    clamp_array<float>(temperatures, 0.0f, 50.0f);
    
    std::cout << "클램핑 후: ";
    for (float temp : temperatures) {
        std::cout << temp << " ";
    }
    std::cout << "\n";  // 범위를 벗어난 값들이 0~50으로 제한됨
    
    // 이상치 제거 예제
    std::vector<float> sensor_data = {
        23.5f, 24.1f, 999.9f, 23.8f, -999.9f, 24.3f, 23.9f, 24.0f
    };
    
    std::cout << "\n센서 원본 데이터: ";
    for (float val : sensor_data) {
        std::cout << val << " ";
    }
    std::cout << "\n";
    
    remove_outliers<float>(sensor_data, 100.0f, 24.0f);
    
    std::cout << "이상치 제거 후: ";
    for (float val : sensor_data) {
        std::cout << val << " ";
    }
    std::cout << "\n";  // 999.9와 -999.9가 24.0으로 대체됨
    
    return 0;
}
```

## 13.3 부동소수점 연산의 정확성

### 문제: 부동소수점 정밀도 손실

부동소수점 연산은 반올림 오차와 정밀도 손실을 일으킨다.

**안전하지 않은 코드:**

```cpp
#include <iostream>
#include <iomanip>
#include <cmath>

// 정밀도 손실이 발생하는 금융 계산
double calculate_interest_unsafe(double principal, double rate, int years) {
    double amount = principal;
    for (int i = 0; i < years; ++i) {
        amount *= (1.0 + rate);  // 반복적인 곱셈으로 오차 누적
    }
    return amount - principal;
}

// 뺄셈 소거 문제
double catastrophic_cancellation() {
    double a = 1.0e16;
    double b = 1.0;
    double c = 1.0e16;
    
    return (a + b) - c;  // 0이 아닌 1을 기대하지만...
}

int main() {
    std::cout << std::setprecision(15);
    
    // 금융 계산 오차
    double principal = 1000.0;
    double rate = 0.05;  // 5%
    int years = 30;
    
    double interest = calculate_interest_unsafe(principal, rate, years);
    std::cout << "30년 후 이자: " << interest << "\n";
    
    // 뺄셈 소거
    double result = catastrophic_cancellation();
    std::cout << "뺄셈 소거 결과: " << result << "\n";  // 0 (정밀도 손실!)
    
    // 작은 수의 합 손실
    float sum = 0.0f;
    for (int i = 0; i < 10000000; ++i) {
        sum += 0.0001f;  // 작은 값 반복 더하기
    }
    std::cout << "작은 수 누적 합: " << sum << "\n";  // 정확하지 않음
    
    return 0;
}
```

문제점은 다음과 같다.

- 반복적인 연산으로 오차가 누적된다
- 크기가 매우 다른 수의 연산에서 정밀도가 손실된다
- 뺄셈 소거로 유효 숫자가 사라진다

### 해결책: Kahan 합산 알고리즘

정밀한 합산을 위해 Kahan 알고리즘을 사용한다.

**안전한 코드:**

```cpp
#include <iostream>
#include <iomanip>
#include <vector>
#include <span>
#include <cmath>
#include <numbers>

// Kahan 합산 알고리즘으로 정밀도 손실 방지
template<typename T>
class KahanAccumulator {
    T sum = 0;
    T compensation = 0;  // 손실된 하위 비트 보상
    
public:
    constexpr void add(T value) noexcept {
        T corrected = value - compensation;
        T new_sum = sum + corrected;
        compensation = (new_sum - sum) - corrected;
        sum = new_sum;
    }
    
    constexpr T get() const noexcept {
        return sum;
    }
    
    constexpr void reset() noexcept {
        sum = 0;
        compensation = 0;
    }
};

// 안전한 배열 합산
template<typename T>
T safe_sum(std::span<const T> values) {
    KahanAccumulator<T> acc;
    for (T val : values) {
        acc.add(val);
    }
    return acc.get();
}

// 안전한 평균 계산
template<typename T>
T safe_mean(std::span<const T> values) {
    if (values.empty()) return T(0);
    return safe_sum(values) / static_cast<T>(values.size());
}

// 안전한 분산 계산
template<typename T>
T safe_variance(std::span<const T> values) {
    if (values.size() < 2) return T(0);
    
    T mean = safe_mean(values);
    KahanAccumulator<T> acc;
    
    for (T val : values) {
        T diff = val - mean;
        acc.add(diff * diff);
    }
    
    return acc.get() / static_cast<T>(values.size() - 1);
}

int main() {
    std::cout << std::setprecision(15);
    
    // 작은 수 누적 합산 비교
    std::vector<float> small_values(10000000, 0.0001f);
    
    // 단순 합산
    float simple_sum = 0.0f;
    for (float val : small_values) {
        simple_sum += val;
    }
    
    // Kahan 합산
    float kahan_sum = safe_sum<float>(small_values);
    
    std::cout << "단순 합산: " << simple_sum << "\n";
    std::cout << "Kahan 합산: " << kahan_sum << "\n";
    std::cout << "예상 값: " << 1000.0f << "\n\n";
    
    // 통계 계산
    std::vector<double> measurements = {
        23.5, 24.1, 23.8, 24.3, 23.9, 24.0, 23.7, 24.2
    };
    
    double mean = safe_mean<double>(measurements);
    double variance = safe_variance<double>(measurements);
    double stddev = std::sqrt(variance);
    
    std::cout << "측정값 평균: " << mean << "\n";
    std::cout << "분산: " << variance << "\n";
    std::cout << "표준편차: " << stddev << "\n";
    
    return 0;
}
```

### 안전한 금융 계산

금융 계산에서는 정밀도가 특히 중요하다.

```cpp
#include <iostream>
#include <iomanip>
#include <cmath>
#include <numbers>

// 고정밀도 복리 이자 계산
double calculate_compound_interest_safe(double principal, double annual_rate, 
                                       int years, int compounds_per_year = 12) {
    // 더 정밀한 수식 사용: A = P(1 + r/n)^(nt)
    double rate_per_period = annual_rate / compounds_per_year;
    int total_periods = years * compounds_per_year;
    
    // log를 이용한 안전한 거듭제곱
    double log_factor = total_periods * std::log1p(rate_per_period);
    double final_amount = principal * std::exp(log_factor);
    
    return final_amount - principal;
}

// 연속 복리 계산
double calculate_continuous_compound(double principal, double annual_rate, 
                                    int years) {
    // A = Pe^(rt)
    return principal * std::expm1(annual_rate * years);  // expm1 = e^x - 1 (정밀)
}

// 대출 월 상환액 계산 (안전한 버전)
double calculate_monthly_payment(double principal, double annual_rate, 
                                int years) {
    double monthly_rate = annual_rate / 12.0;
    int num_payments = years * 12;
    
    if (monthly_rate == 0.0) {
        return principal / num_payments;
    }
    
    // M = P * [r(1+r)^n] / [(1+r)^n - 1]
    // log를 사용하여 안전하게 계산
    double log_factor = num_payments * std::log1p(monthly_rate);
    double power_term = std::exp(log_factor);
    
    return principal * (monthly_rate * power_term) / (power_term - 1.0);
}

// 현재가치 계산 (NPV)
double calculate_npv(std::span<const double> cash_flows, double discount_rate) {
    KahanAccumulator<double> npv;
    
    for (size_t t = 0; t < cash_flows.size(); ++t) {
        double discount_factor = std::pow(1.0 + discount_rate, -static_cast<double>(t));
        npv.add(cash_flows[t] * discount_factor);
    }
    
    return npv.get();
}

int main() {
    std::cout << std::setprecision(2) << std::fixed;
    
    // 복리 이자 비교
    double principal = 100000.0;
    double rate = 0.05;  // 5%
    int years = 30;
    
    double monthly_compound = calculate_compound_interest_safe(principal, rate, years, 12);
    double continuous_compound = calculate_continuous_compound(principal, rate, years);
    
    std::cout << "원금: $" << principal << "\n";
    std::cout << "월 복리 이자 (30년): $" << monthly_compound << "\n";
    std::cout << "연속 복리 이자 (30년): $" << continuous_compound << "\n\n";
    
    // 대출 상환액 계산
    double loan_amount = 300000.0;
    double loan_rate = 0.04;  // 4%
    int loan_years = 30;
    
    double monthly_payment = calculate_monthly_payment(loan_amount, loan_rate, loan_years);
    double total_payment = monthly_payment * loan_years * 12;
    double total_interest = total_payment - loan_amount;
    
    std::cout << "대출 금액: $" << loan_amount << "\n";
    std::cout << "월 상환액: $" << monthly_payment << "\n";
    std::cout << "총 상환액: $" << total_payment << "\n";
    std::cout << "총 이자: $" << total_interest << "\n\n";
    
    // NPV 계산
    std::vector<double> project_cash_flows = {
        -100000.0,  // 초기 투자
        30000.0,    // 1년차
        35000.0,    // 2년차
        40000.0,    // 3년차
        45000.0,    // 4년차
        50000.0     // 5년차
    };
    
    double discount_rate = 0.08;  // 8%
    double npv = calculate_npv(project_cash_flows, discount_rate);
    
    std::cout << "프로젝트 NPV (할인율 " << (discount_rate * 100) << "%): $" << npv << "\n";
    
    if (npv > 0) {
        std::cout << "프로젝트 실행 권장\n";
    } else {
        std::cout << "프로젝트 거부 권장\n";
    }
    
    return 0;
}
```

### 부동소수점 비교의 안전성

부동소수점 값을 비교할 때는 엡실론을 사용한다.

```cpp
#include <iostream>
#include <iomanip>
#include <cmath>
#include <limits>
#include <concepts>

// 상대 오차를 고려한 부동소수점 비교
template<std::floating_point T>
constexpr bool approximately_equal(T a, T b, 
                                  T rel_epsilon = std::numeric_limits<T>::epsilon() * 100) {
    // 절댓값이 작은 경우 절대 비교
    T abs_diff = std::abs(a - b);
    if (abs_diff <= rel_epsilon) {
        return true;
    }
    
    // 상대 오차 비교
    T larger = std::max(std::abs(a), std::abs(b));
    return abs_diff <= larger * rel_epsilon;
}

// 0과의 비교
template<std::floating_point T>
constexpr bool essentially_zero(T value, T epsilon = std::numeric_limits<T>::epsilon() * 100) {
    return std::abs(value) <= epsilon;
}

// 크기 비교 (a < b)
template<std::floating_point T>
constexpr bool definitely_less_than(T a, T b, T epsilon = std::numeric_limits<T>::epsilon() * 100) {
    return (b - a) > epsilon;
}

// 크기 비교 (a > b)
template<std::floating_point T>
constexpr bool definitely_greater_than(T a, T b, T epsilon = std::numeric_limits<T>::epsilon() * 100) {
    return (a - b) > epsilon;
}

// 안전한 나눗셈
template<std::floating_point T>
constexpr T safe_divide(T numerator, T denominator, T default_value = T(0)) {
    if (essentially_zero(denominator)) {
        return default_value;
    }
    return numerator / denominator;
}

// 기하 평균 계산 (안전 버전)
template<std::floating_point T>
T geometric_mean(std::span<const T> values) {
    if (values.empty()) return T(0);
    
    // 로그를 사용하여 오버플로우 방지
    KahanAccumulator<T> log_sum;
    
    for (T val : values) {
        if (val <= T(0)) {
            throw std::invalid_argument("기하 평균은 양수에만 적용된다");
        }
        log_sum.add(std::log(val));
    }
    
    T mean_log = log_sum.get() / static_cast<T>(values.size());
    return std::exp(mean_log);
}

int main() {
    std::cout << std::setprecision(15);
    
    // 부동소수점 비교 예제
    double a = 0.1 + 0.2;
    double b = 0.3;
    
    std::cout << "a = 0.1 + 0.2 = " << a << "\n";
    std::cout << "b = 0.3 = " << b << "\n";
    std::cout << "a == b: " << (a == b) << "\n";  // false!
    std::cout << "approximately_equal(a, b): " << approximately_equal(a, b) << "\n\n";
    
    // 안전한 0 비교
    double tiny = 1e-15;
    std::cout << "tiny = " << tiny << "\n";
    std::cout << "essentially_zero(tiny): " << essentially_zero(tiny) << "\n\n";
    
    // 안전한 나눗셈
    double numerator = 10.0;
    double almost_zero = 1e-20;
    
    double unsafe_result = numerator / almost_zero;  // 매우 큰 수
    double safe_result = safe_divide(numerator, almost_zero, 0.0);
    
    std::cout << "10.0 / " << almost_zero << ":\n";
    std::cout << "  안전하지 않은 결과: " << unsafe_result << "\n";
    std::cout << "  안전한 결과: " << safe_result << "\n\n";
    
    // 기하 평균
    std::vector<double> growth_rates = {1.05, 1.08, 1.03, 1.07, 1.06};
    double geom_mean = geometric_mean<double>(growth_rates);
    
    std::cout << "연간 성장률: ";
    for (double rate : growth_rates) {
        std::cout << (rate - 1.0) * 100 << "% ";
    }
    std::cout << "\n기하 평균 성장률: " << (geom_mean - 1.0) * 100 << "%\n";
    
    return 0;
}
```

## 실전 예제: 안전한 신호 처리 파이프라인

수치 계산의 모든 개념을 통합한 실전 예제를 살펴본다.

```cpp
#include <iostream>
#include <vector>
#include <span>
#include <cmath>
#include <numbers>
#include <experimental/simd>

namespace stdx = std::experimental;

// 디지털 신호 처리 파이프라인
class SafeSignalProcessor {
    static constexpr double PI = std::numbers::pi;
    
public:
    // 1. 입력 신호 정규화 (saturation 사용)
    static void normalize_signal(std::span<int16_t> signal) {
        int16_t max_val = 0;
        
        // 최댓값 찾기
        for (int16_t sample : signal) {
            int16_t abs_sample = std::abs(sample);
            if (abs_sample > max_val) {
                max_val = abs_sample;
            }
        }
        
        if (max_val == 0) return;
        
        // 정규화 (saturation 적용)
        int32_t scale = 32767;  // int16 최댓값
        for (size_t i = 0; i < signal.size(); ++i) {
            int32_t normalized = (static_cast<int32_t>(signal[i]) * scale) / max_val;
            signal[i] = saturating_clamp<int16_t>(normalized);
        }
    }
    
    // 2. SIMD를 사용한 FIR 필터
    static std::vector<float> apply_fir_filter(std::span<const float> input,
                                               std::span<const float> coefficients) {
        std::vector<float> output(input.size(), 0.0f);
        
        using simd_t = stdx::native_simd<float>;
        constexpr size_t simd_size = simd_t::size();
        
        // 각 출력 샘플 계산
        for (size_t n = 0; n < input.size(); ++n) {
            KahanAccumulator<float> acc;
            
            size_t k = 0;
            // SIMD로 계수 곱셈
            for (; k + simd_size <= coefficients.size() && k <= n; k += simd_size) {
                if (n - k >= simd_size - 1) {
                    simd_t coeff(&coefficients[k], stdx::element_aligned);
                    
                    // 입력 신호를 역순으로 로드
                    float reversed[simd_size];
                    for (size_t i = 0; i < simd_size; ++i) {
                        reversed[i] = input[n - k - i];
                    }
                    simd_t inp(reversed, stdx::element_aligned);
                    
                    simd_t product = coeff * inp;
                    
                    // 결과 누적
                    for (size_t i = 0; i < simd_size; ++i) {
                        acc.add(product[i]);
                    }
                }
            }
            
            // 나머지 스칼라 처리
            for (; k < coefficients.size() && k <= n; ++k) {
                acc.add(coefficients[k] * input[n - k]);
            }
            
            output[n] = acc.get();
        }
        
        return output;
    }
    
    // 3. 안전한 FFT 전처리
    static std::vector<std::complex<double>> safe_fft_preprocess(
        std::span<const float> signal) {
        
        std::vector<std::complex<double>> complex_signal(signal.size());
        
        // 윈도우 함수 적용 (Hann window)
        KahanAccumulator<double> energy;
        
        for (size_t i = 0; i < signal.size(); ++i) {
            double window = 0.5 * (1.0 - std::cos(2.0 * PI * i / (signal.size() - 1)));
            double windowed = signal[i] * window;
            
            complex_signal[i] = std::complex<double>(windowed, 0.0);
            energy.add(windowed * windowed);
        }
        
        // 에너지 정규화
        double total_energy = energy.get();
        if (!essentially_zero(total_energy)) {
            double norm_factor = 1.0 / std::sqrt(total_energy / signal.size());
            
            for (auto& sample : complex_signal) {
                sample *= norm_factor;
            }
        }
        
        return complex_signal;
    }
    
private:
    template<std::integral T>
    static constexpr T saturating_clamp(int32_t value) {
        if (value > std::numeric_limits<T>::max()) {
            return std::numeric_limits<T>::max();
        }
        if (value < std::numeric_limits<T>::min()) {
            return std::numeric_limits<T>::min();
        }
        return static_cast<T>(value);
    }
};

int main() {
    // 1. 신호 정규화 테스트
    std::vector<int16_t> audio_samples = {
        100, -150, 200, -250, 300, -350, 400, -450
    };
    
    std::cout << "원본 오디오 샘플: ";
    for (int16_t sample : audio_samples) {
        std::cout << sample << " ";
    }
    std::cout << "\n";
    
    SafeSignalProcessor::normalize_signal(audio_samples);
    
    std::cout << "정규화된 샘플: ";
    for (int16_t sample : audio_samples) {
        std::cout << sample << " ";
    }
    std::cout << "\n\n";
    
    // 2. FIR 저역 통과 필터
    std::vector<float> input_signal(100);
    for (size_t i = 0; i < input_signal.size(); ++i) {
        // 고주파 + 저주파 혼합 신호
        input_signal[i] = std::sin(2.0 * std::numbers::pi * 5.0 * i / 100.0) +
                         0.5f * std::sin(2.0 * std::numbers::pi * 50.0 * i / 100.0);
    }
    
    // 간단한 저역 통과 필터 계수
    std::vector<float> lpf_coefficients = {
        0.1f, 0.2f, 0.4f, 0.2f, 0.1f
    };
    
    auto filtered = SafeSignalProcessor::apply_fir_filter(input_signal, lpf_coefficients);
    
    std::cout << "필터링 완료: " << filtered.size() << " 샘플\n";
    std::cout << "처음 10개 샘플: ";
    for (size_t i = 0; i < 10; ++i) {
        std::cout << std::fixed << std::setprecision(3) << filtered[i] << " ";
    }
    std::cout << "\n\n";
    
    // 3. FFT 전처리
    std::vector<float> fft_input(64);
    for (size_t i = 0; i < fft_input.size(); ++i) {
        fft_input[i] = std::sin(2.0 * std::numbers::pi * 10.0 * i / 64.0);
    }
    
    auto preprocessed = SafeSignalProcessor::safe_fft_preprocess(fft_input);
    
    std::cout << "FFT 전처리 완료: " << preprocessed.size() << " 복소수 샘플\n";
    std::cout << "처음 5개 샘플: \n";
    for (size_t i = 0; i < 5; ++i) {
        std::cout << "  [" << i << "]: " 
                  << preprocessed[i].real() << " + " 
                  << preprocessed[i].imag() << "i\n";
    }
    
    return 0;
}
```

## 핵심 정리

이 장에서는 안전한 수치 계산을 위한 세 가지 핵심 기법을 다루었다.

**Saturation Arithmetic**는 정수 오버플로우를 방지하고 예측 가능한 동작을 보장한다. 오디오 처리, 이미지 처리, 금융 계산에서 필수적이다.

**`std::simd`**는 타입 안전하고 이식 가능한 벡터화를 제공한다. 플랫폼별 intrinsic을 추상화하고 경계 검사와 정렬을 자동으로 처리한다.

**부동소수점 안전성**은 Kahan 합산, 안전한 비교, 정밀한 수학 함수를 통해 달성된다. 금융, 과학 계산, 신호 처리에서 정확성을 보장한다.

다음 장에서는 메모리 정렬과 캐시 안전성을 다루며, 성능과 안전성을 모두 확보하는 방법을 학습한다.
