# 15장: 안전한 네트워크 프로그래밍

네트워크 프로그래밍은 버퍼 오버플로우, 리소스 누수, 동시성 문제 등 다양한 보안 취약점의 온상이다. C++26과 최신 표준 라이브러리는 이러한 위험을 줄이기 위한 여러 도구를 제공한다. 이 장에서는 비동기 I/O, 안전한 버퍼 관리, 실제 서버 구현을 통해 견고한 네트워크 애플리케이션을 작성하는 방법을 다룬다.

## 15.1 비동기 I/O와 에러 처리

### 문제: 블로킹 I/O와 부적절한 에러 처리

전통적인 블로킹 I/O는 확장성이 떨어지고 에러 처리가 복잡하다.

**안전하지 않은 코드:**

```cpp
#include <iostream>
#include <sys/socket.h>
#include <netinet/in.h>
#include <unistd.h>
#include <cstring>

// 블로킹 소켓 서버 (위험)
class UnsafeServer {
    int server_fd_;
    
public:
    UnsafeServer(int port) {
        server_fd_ = socket(AF_INET, SOCK_STREAM, 0);
        // 에러 검사 없음!
        
        sockaddr_in address{};
        address.sin_family = AF_INET;
        address.sin_addr.s_addr = INADDR_ANY;
        address.sin_port = htons(port);
        
        bind(server_fd_, (sockaddr*)&address, sizeof(address));
        // 실패 시 처리 없음!
        
        listen(server_fd_, 10);
        // 실패 시 처리 없음!
    }
    
    void handle_client() {
        char buffer[1024];  // 고정 크기 버퍼
        
        int client_fd = accept(server_fd_, nullptr, nullptr);
        // 에러 검사 없음!
        
        while (true) {
            // 버퍼 오버플로우 가능!
            ssize_t n = read(client_fd, buffer, sizeof(buffer));
            
            if (n <= 0) break;  // 에러와 EOF 구분 안 함
            
            // 에코 - 부분 쓰기 처리 안 함
            write(client_fd, buffer, n);
        }
        
        close(client_fd);
    }
    
    ~UnsafeServer() {
        close(server_fd_);  // 예외 발생 시 누수
    }
};

int main() {
    UnsafeServer server(8080);
    
    while (true) {
        server.handle_client();  // 한 번에 하나만 처리
    }
    
    return 0;
}
```

이 코드의 문제점은 다음과 같다.

- 시스템 호출 에러를 확인하지 않는다
- 블로킹으로 인해 한 번에 하나의 클라이언트만 처리 가능하다
- 버퍼 오버플로우에 취약하다
- 리소스 누수 가능성이 있다
- 부분 읽기/쓰기를 처리하지 않는다

### 해결책: RAII와 체계적인 에러 처리

안전한 소켓 래퍼와 에러 처리를 구현한다.

**안전한 코드:**

```cpp
#include <iostream>
#include <system_error>
#include <memory>
#include <span>
#include <vector>
#include <optional>
#include <expected>
#include <sys/socket.h>
#include <netinet/in.h>
#include <unistd.h>
#include <fcntl.h>
#include <cerrno>
#include <cstring>

// C++23 std::expected를 사용한 에러 처리
template<typename T>
using Result = std::expected<T, std::error_code>;

// RAII 소켓 래퍼
class Socket {
    int fd_;
    
    Socket(int fd) : fd_(fd) {}
    
public:
    Socket() : fd_(-1) {}
    
    ~Socket() {
        if (fd_ >= 0) {
            ::close(fd_);
        }
    }
    
    Socket(const Socket&) = delete;
    Socket& operator=(const Socket&) = delete;
    
    Socket(Socket&& other) noexcept : fd_(other.fd_) {
        other.fd_ = -1;
    }
    
    Socket& operator=(Socket&& other) noexcept {
        if (this != &other) {
            if (fd_ >= 0) {
                ::close(fd_);
            }
            fd_ = other.fd_;
            other.fd_ = -1;
        }
        return *this;
    }
    
    static Result<Socket> create(int domain, int type, int protocol) {
        int fd = ::socket(domain, type, protocol);
        if (fd < 0) {
            return std::unexpected(std::error_code(errno, std::system_category()));
        }
        return Socket(fd);
    }
    
    Result<void> bind(const sockaddr* addr, socklen_t addrlen) {
        if (::bind(fd_, addr, addrlen) < 0) {
            return std::unexpected(std::error_code(errno, std::system_category()));
        }
        return {};
    }
    
    Result<void> listen(int backlog) {
        if (::listen(fd_, backlog) < 0) {
            return std::unexpected(std::error_code(errno, std::system_category()));
        }
        return {};
    }
    
    Result<Socket> accept() {
        int client_fd = ::accept(fd_, nullptr, nullptr);
        if (client_fd < 0) {
            return std::unexpected(std::error_code(errno, std::system_category()));
        }
        return Socket(client_fd);
    }
    
    Result<size_t> read(std::span<std::byte> buffer) {
        ssize_t n = ::read(fd_, buffer.data(), buffer.size());
        if (n < 0) {
            return std::unexpected(std::error_code(errno, std::system_category()));
        }
        return static_cast<size_t>(n);
    }
    
    Result<size_t> write(std::span<const std::byte> data) {
        ssize_t n = ::write(fd_, data.data(), data.size());
        if (n < 0) {
            return std::unexpected(std::error_code(errno, std::system_category()));
        }
        return static_cast<size_t>(n);
    }
    
    Result<void> set_nonblocking(bool enable) {
        int flags = fcntl(fd_, F_GETFL, 0);
        if (flags < 0) {
            return std::unexpected(std::error_code(errno, std::system_category()));
        }
        
        if (enable) {
            flags |= O_NONBLOCK;
        } else {
            flags &= ~O_NONBLOCK;
        }
        
        if (fcntl(fd_, F_SETFL, flags) < 0) {
            return std::unexpected(std::error_code(errno, std::system_category()));
        }
        
        return {};
    }
    
    Result<void> set_reuse_addr(bool enable) {
        int opt = enable ? 1 : 0;
        if (setsockopt(fd_, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt)) < 0) {
            return std::unexpected(std::error_code(errno, std::system_category()));
        }
        return {};
    }
    
    int native_handle() const { return fd_; }
    bool is_valid() const { return fd_ >= 0; }
};

// 안전한 버퍼 읽기/쓰기 헬퍼
class BufferedReader {
    std::vector<std::byte> buffer_;
    size_t read_pos_ = 0;
    size_t write_pos_ = 0;
    
public:
    explicit BufferedReader(size_t capacity = 4096) 
        : buffer_(capacity) {}
    
    // 소켓에서 데이터 읽기
    Result<void> fill_from(Socket& socket) {
        if (write_pos_ >= buffer_.size()) {
            // 버퍼 가득 참
            compact();
        }
        
        auto available = std::span(buffer_).subspan(write_pos_);
        auto result = socket.read(available);
        
        if (!result) {
            return std::unexpected(result.error());
        }
        
        write_pos_ += *result;
        return {};
    }
    
    // 한 줄 읽기 (개행까지)
    std::optional<std::string> read_line() {
        for (size_t i = read_pos_; i < write_pos_; ++i) {
            if (buffer_[i] == std::byte('\n')) {
                std::string line;
                line.reserve(i - read_pos_);
                
                for (size_t j = read_pos_; j < i; ++j) {
                    line.push_back(static_cast<char>(buffer_[j]));
                }
                
                read_pos_ = i + 1;
                return line;
            }
        }
        
        return std::nullopt;
    }
    
    // 정확히 n바이트 읽기
    std::optional<std::vector<std::byte>> read_exact(size_t n) {
        if (write_pos_ - read_pos_ < n) {
            return std::nullopt;
        }
        
        std::vector<std::byte> result(buffer_.begin() + read_pos_,
                                     buffer_.begin() + read_pos_ + n);
        read_pos_ += n;
        return result;
    }
    
    size_t available() const {
        return write_pos_ - read_pos_;
    }
    
private:
    void compact() {
        if (read_pos_ > 0) {
            std::memmove(buffer_.data(), 
                        buffer_.data() + read_pos_,
                        write_pos_ - read_pos_);
            write_pos_ -= read_pos_;
            read_pos_ = 0;
        }
    }
};

// 완전한 쓰기를 보장하는 헬퍼
Result<void> write_all(Socket& socket, std::span<const std::byte> data) {
    size_t written = 0;
    
    while (written < data.size()) {
        auto remaining = data.subspan(written);
        auto result = socket.write(remaining);
        
        if (!result) {
            return std::unexpected(result.error());
        }
        
        written += *result;
    }
    
    return {};
}

// 안전한 서버
class SafeServer {
    Socket server_socket_;
    
public:
    static Result<SafeServer> create(uint16_t port) {
        auto socket_result = Socket::create(AF_INET, SOCK_STREAM, 0);
        if (!socket_result) {
            return std::unexpected(socket_result.error());
        }
        
        Socket socket = std::move(*socket_result);
        
        // 소켓 옵션 설정
        if (auto result = socket.set_reuse_addr(true); !result) {
            return std::unexpected(result.error());
        }
        
        // 바인드
        sockaddr_in address{};
        address.sin_family = AF_INET;
        address.sin_addr.s_addr = INADDR_ANY;
        address.sin_port = htons(port);
        
        if (auto result = socket.bind((sockaddr*)&address, sizeof(address)); !result) {
            return std::unexpected(result.error());
        }
        
        // 리슨
        if (auto result = socket.listen(128); !result) {
            return std::unexpected(result.error());
        }
        
        return SafeServer(std::move(socket));
    }
    
    Result<void> handle_client(Socket client) {
        BufferedReader reader;
        
        while (true) {
            // 데이터 읽기
            if (auto result = reader.fill_from(client); !result) {
                if (result.error() == std::errc::resource_unavailable_try_again) {
                    continue;
                }
                return std::unexpected(result.error());
            }
            
            // EOF 확인
            if (reader.available() == 0) {
                break;
            }
            
            // 에코 처리
            while (auto line = reader.read_line()) {
                std::cout << "받음: " << *line << "\n";
                
                // 응답 전송
                std::string response = "에코: " + *line + "\n";
                auto data = std::as_bytes(std::span(response));
                
                if (auto result = write_all(client, data); !result) {
                    return std::unexpected(result.error());
                }
            }
        }
        
        return {};
    }
    
    Result<Socket> accept() {
        return server_socket_.accept();
    }
    
private:
    explicit SafeServer(Socket socket) 
        : server_socket_(std::move(socket)) {}
};

int main() {
    auto server_result = SafeServer::create(8080);
    
    if (!server_result) {
        std::cerr << "서버 생성 실패: " 
                  << server_result.error().message() << "\n";
        return 1;
    }
    
    SafeServer server = std::move(*server_result);
    std::cout << "서버가 포트 8080에서 시작됨\n";
    
    while (true) {
        auto client_result = server.accept();
        
        if (!client_result) {
            std::cerr << "클라이언트 수락 실패: "
                      << client_result.error().message() << "\n";
            continue;
        }
        
        std::cout << "클라이언트 연결됨\n";
        
        if (auto result = server.handle_client(std::move(*client_result)); !result) {
            std::cerr << "클라이언트 처리 실패: "
                      << result.error().message() << "\n";
        }
        
        std::cout << "클라이언트 연결 종료\n";
    }
    
    return 0;
}
```

### 비동기 I/O를 위한 이벤트 루프

논블로킹 I/O와 이벤트 루프로 다중 클라이언트를 처리한다.

```cpp
#include <iostream>
#include <vector>
#include <unordered_map>
#include <memory>
#include <sys/epoll.h>

// 이벤트 타입
enum class EventType {
    Read,
    Write,
    Error,
    HangUp
};

// 이벤트 핸들러 인터페이스
class EventHandler {
public:
    virtual ~EventHandler() = default;
    virtual void on_event(EventType type) = 0;
    virtual int get_fd() const = 0;
};

// Epoll 기반 이벤트 루프
class EventLoop {
    int epoll_fd_;
    std::unordered_map<int, std::shared_ptr<EventHandler>> handlers_;
    bool running_ = false;
    
public:
    EventLoop() {
        epoll_fd_ = epoll_create1(0);
        if (epoll_fd_ < 0) {
            throw std::system_error(errno, std::system_category(), 
                                   "epoll_create1 실패");
        }
    }
    
    ~EventLoop() {
        if (epoll_fd_ >= 0) {
            ::close(epoll_fd_);
        }
    }
    
    EventLoop(const EventLoop&) = delete;
    EventLoop& operator=(const EventLoop&) = delete;
    
    Result<void> add_handler(std::shared_ptr<EventHandler> handler, 
                            uint32_t events = EPOLLIN | EPOLLET) {
        int fd = handler->get_fd();
        
        epoll_event ev{};
        ev.events = events;
        ev.data.fd = fd;
        
        if (epoll_ctl(epoll_fd_, EPOLL_CTL_ADD, fd, &ev) < 0) {
            return std::unexpected(std::error_code(errno, std::system_category()));
        }
        
        handlers_[fd] = std::move(handler);
        return {};
    }
    
    Result<void> remove_handler(int fd) {
        if (epoll_ctl(epoll_fd_, EPOLL_CTL_DEL, fd, nullptr) < 0) {
            return std::unexpected(std::error_code(errno, std::system_category()));
        }
        
        handlers_.erase(fd);
        return {};
    }
    
    void run() {
        running_ = true;
        constexpr int max_events = 64;
        epoll_event events[max_events];
        
        while (running_) {
            int n = epoll_wait(epoll_fd_, events, max_events, 1000);
            
            if (n < 0) {
                if (errno == EINTR) continue;
                std::cerr << "epoll_wait 실패: " 
                          << std::strerror(errno) << "\n";
                break;
            }
            
            for (int i = 0; i < n; ++i) {
                int fd = events[i].data.fd;
                uint32_t ev = events[i].events;
                
                auto it = handlers_.find(fd);
                if (it == handlers_.end()) continue;
                
                auto& handler = it->second;
                
                if (ev & EPOLLERR) {
                    handler->on_event(EventType::Error);
                } else if (ev & EPOLLHUP) {
                    handler->on_event(EventType::HangUp);
                    handlers_.erase(it);
                } else {
                    if (ev & EPOLLIN) {
                        handler->on_event(EventType::Read);
                    }
                    if (ev & EPOLLOUT) {
                        handler->on_event(EventType::Write);
                    }
                }
            }
        }
    }
    
    void stop() {
        running_ = false;
    }
};

// 클라이언트 연결 핸들러
class ClientHandler : public EventHandler {
    Socket socket_;
    BufferedReader reader_;
    EventLoop& event_loop_;
    
public:
    ClientHandler(Socket socket, EventLoop& loop)
        : socket_(std::move(socket)), event_loop_(loop) {
        
        // 논블로킹 모드 설정
        if (auto result = socket_.set_nonblocking(true); !result) {
            throw std::system_error(result.error());
        }
    }
    
    void on_event(EventType type) override {
        switch (type) {
            case EventType::Read:
                handle_read();
                break;
            case EventType::Error:
            case EventType::HangUp:
                std::cout << "클라이언트 연결 종료\n";
                event_loop_.remove_handler(socket_.native_handle());
                break;
            default:
                break;
        }
    }
    
    int get_fd() const override {
        return socket_.native_handle();
    }
    
private:
    void handle_read() {
        while (true) {
            auto result = reader_.fill_from(socket_);
            
            if (!result) {
                if (result.error() == std::errc::resource_unavailable_try_again) {
                    break;  // EAGAIN - 나중에 다시 시도
                }
                std::cerr << "읽기 오류: " << result.error().message() << "\n";
                event_loop_.remove_handler(socket_.native_handle());
                return;
            }
        }
        
        // 메시지 처리
        while (auto line = reader_.read_line()) {
            std::cout << "받음: " << *line << "\n";
            
            std::string response = "에코: " + *line + "\n";
            auto data = std::as_bytes(std::span(response));
            
            // 논블로킹 쓰기 (실제로는 쓰기 버퍼 필요)
            write_all(socket_, data);
        }
    }
};

// 서버 핸들러
class ServerHandler : public EventHandler {
    Socket& server_socket_;
    EventLoop& event_loop_;
    
public:
    ServerHandler(Socket& socket, EventLoop& loop)
        : server_socket_(socket), event_loop_(loop) {}
    
    void on_event(EventType type) override {
        if (type == EventType::Read) {
            accept_clients();
        }
    }
    
    int get_fd() const override {
        return server_socket_.native_handle();
    }
    
private:
    void accept_clients() {
        while (true) {
            auto client_result = server_socket_.accept();
            
            if (!client_result) {
                if (client_result.error() == std::errc::resource_unavailable_try_again) {
                    break;
                }
                std::cerr << "수락 오류: " 
                          << client_result.error().message() << "\n";
                return;
            }
            
            std::cout << "새 클라이언트 연결\n";
            
            auto handler = std::make_shared<ClientHandler>(
                std::move(*client_result), event_loop_);
            
            if (auto result = event_loop_.add_handler(handler); !result) {
                std::cerr << "핸들러 추가 실패: "
                          << result.error().message() << "\n";
            }
        }
    }
};

int main() {
    auto server_result = SafeServer::create(8080);
    
    if (!server_result) {
        std::cerr << "서버 생성 실패: "
                  << server_result.error().message() << "\n";
        return 1;
    }
    
    // 서버 소켓을 논블로킹으로 설정 (별도 접근 필요)
    // 실제 구현에서는 SafeServer에서 소켓에 접근할 방법 제공
    
    EventLoop event_loop;
    
    std::cout << "비동기 서버가 포트 8080에서 시작됨\n";
    event_loop.run();
    
    return 0;
}
```

## 15.2 버퍼 오버플로우 방지

### 문제: 고정 크기 버퍼의 위험성

고정 크기 버퍼는 버퍼 오버플로우의 주요 원인이다.

**안전하지 않은 코드:**

```cpp
#include <iostream>
#include <cstring>
#include <cstdint>

// 프로토콜 파서 (위험)
struct UnsafeProtocolParser {
    static constexpr size_t MAX_MESSAGE = 1024;
    
    // 길이 접두사 프로토콜: [4바이트 길이][메시지]
    bool parse_message(const char* data, size_t size, char* out_message) {
        if (size < 4) return false;
        
        // 길이 읽기 (엔디안 고려 안 함)
        uint32_t msg_len = *reinterpret_cast<const uint32_t*>(data);
        
        // 길이 검증 없음!
        if (size < 4 + msg_len) return false;
        
        // 버퍼 오버플로우 가능!
        memcpy(out_message, data + 4, msg_len);
        out_message[msg_len] = '\0';
        
        return true;
    }
    
    // HTTP 헤더 파싱 (위험)
    void parse_http_header(const char* header, char* method, char* path) {
        // 공백으로 분리 - 버퍼 크기 모름
        sscanf(header, "%s %s", method, path);  // 버퍼 오버플로우!
    }
};

int main() {
    UnsafeProtocolParser parser;
    
    // 악의적인 메시지
    char malicious_data[2048];
    *reinterpret_cast<uint32_t*>(malicious_data) = 2000;  // 큰 길이
    
    char buffer[1024];
    parser.parse_message(malicious_data, sizeof(malicious_data), buffer);
    // 버퍼 오버플로우 발생!
    
    return 0;
}
```

문제점은 다음과 같다.

- 입력 크기를 검증하지 않는다
- 고정 크기 버퍼에 임의 크기 데이터를 복사한다
- sscanf, strcpy 같은 안전하지 않은 함수를 사용한다

### 해결책: 동적 버퍼와 경계 검사

`std::span`과 동적 할당으로 안전한 버퍼 관리를 구현한다.

**안전한 코드:**

```cpp
#include <iostream>
#include <vector>
#include <span>
#include <string>
#include <string_view>
#include <optional>
#include <expected>
#include <cstring>
#include <bit>

// 안전한 프로토콜 파서
class SafeProtocolParser {
public:
    // 길이 제한
    static constexpr size_t MAX_MESSAGE_SIZE = 1024 * 1024;  // 1MB
    
    // 길이 접두사 메시지 파싱
    static Result<std::vector<std::byte>> parse_length_prefixed(
        std::span<const std::byte> data) {
        
        // 최소 크기 확인
        if (data.size() < sizeof(uint32_t)) {
            return std::unexpected(std::make_error_code(std::errc::message_size));
        }
        
        // 길이 읽기 (네트워크 바이트 순서)
        uint32_t msg_len;
        std::memcpy(&msg_len, data.data(), sizeof(uint32_t));
        msg_len = std::byteswap(msg_len);  // C++23
        
        // 길이 검증
        if (msg_len > MAX_MESSAGE_SIZE) {
            return std::unexpected(std::make_error_code(std::errc::message_size));
        }
        
        if (data.size() < sizeof(uint32_t) + msg_len) {
            return std::unexpected(std::make_error_code(std::errc::message_size));
        }
        
        // 안전한 복사
        std::vector<std::byte> message(msg_len);
        std::memcpy(message.data(), 
                   data.data() + sizeof(uint32_t), 
                   msg_len);
        
        return message;
    }
    
    // HTTP 요청 라인 파싱
    struct HttpRequest {
        std::string method;
        std::string path;
        std::string version;
    };
    
    static Result<HttpRequest> parse_http_request_line(std::string_view line) {
        // 최대 길이 확인
        if (line.size() > 8192) {
            return std::unexpected(std::make_error_code(std::errc::message_size));
        }
        
        HttpRequest request;
        
        // 첫 번째 공백 찾기
        size_t first_space = line.find(' ');
        if (first_space == std::string_view::npos) {
            return std::unexpected(std::make_error_code(std::errc::invalid_argument));
        }
        
        request.method = std::string(line.substr(0, first_space));
        
        // 두 번째 공백 찾기
        size_t second_space = line.find(' ', first_space + 1);
        if (second_space == std::string_view::npos) {
            return std::unexpected(std::make_error_code(std::errc::invalid_argument));
        }
        
        request.path = std::string(line.substr(first_space + 1, 
                                              second_space - first_space - 1));
        request.version = std::string(line.substr(second_space + 1));
        
        // 유효성 검증
        if (request.method.empty() || request.path.empty() || 
            request.version.empty()) {
            return std::unexpected(std::make_error_code(std::errc::invalid_argument));
        }
        
        return request;
    }
};

// 증분 버퍼 - 필요에 따라 자동 확장
class GrowingBuffer {
    std::vector<std::byte> buffer_;
    size_t write_pos_ = 0;
    size_t max_size_;
    
public:
    explicit GrowingBuffer(size_t initial_capacity = 4096,
                          size_t max_size = 10 * 1024 * 1024)  // 10MB
        : buffer_(initial_capacity), max_size_(max_size) {}
    
    Result<std::span<std::byte>> reserve(size_t additional) {
        size_t required = write_pos_ + additional;
        
        if (required > max_size_) {
            return std::unexpected(std::make_error_code(std::errc::not_enough_memory));
        }
        
        if (required > buffer_.size()) {
            size_t new_size = buffer_.size();
            while (new_size < required) {
                new_size *= 2;
                if (new_size > max_size_) {
                    new_size = max_size_;
                    break;
                }
            }
            
            buffer_.resize(new_size);
        }
        
        return std::span(buffer_).subspan(write_pos_, additional);
    }
    
    void commit(size_t n) {
        write_pos_ += n;
    }
    
    std::span<const std::byte> data() const {
        return std::span(buffer_).subspan(0, write_pos_);
    }
    
    void clear() {
        write_pos_ = 0;
    }
    
    size_t size() const { return write_pos_; }
    size_t capacity() const { return buffer_.size(); }
};

int main() {
    // 안전한 프로토콜 파싱
    std::vector<std::byte> packet(1024);
    
    // 헤더 작성: 메시지 길이
    uint32_t msg_len = 100;
    uint32_t network_len = std::byteswap(msg_len);
    std::memcpy(packet.data(), &network_len, sizeof(uint32_t));
    
    // 메시지 작성
    std::string message = "안전한 메시지 내용";
    std::memcpy(packet.data() + sizeof(uint32_t),
               message.data(), message.size());
    
    // 파싱
    auto result = SafeProtocolParser::parse_length_prefixed(packet);
    
    if (result) {
        std::cout << "메시지 파싱 성공: " << result->size() << " 바이트\n";
    } else {
        std::cerr << "파싱 실패: " << result.error().message() << "\n";
    }
    
    // HTTP 요청 파싱
    std::string_view request_line = "GET /index.html HTTP/1.1";
    auto http_result = SafeProtocolParser::parse_http_request_line(request_line);
    
    if (http_result) {
        std::cout << "메서드: " << http_result->method << "\n";
        std::cout << "경로: " << http_result->path << "\n";
        std::cout << "버전: " << http_result->version << "\n";
    }
    
    // 증분 버퍼
    GrowingBuffer buffer;
    
    for (int i = 0; i < 100; ++i) {
        auto space_result = buffer.reserve(100);
        if (space_result) {
            buffer.commit(100);
        }
    }
    
    std::cout << "\n버퍼 크기: " << buffer.size() << " 바이트\n";
    std::cout << "버퍼 용량: " << buffer.capacity() << " 바이트\n";
    
    return 0;
}
```

### 제로 카피 파싱

메모리 복사 없이 안전하게 데이터를 파싱한다.

```cpp
#include <iostream>
#include <span>
#include <string_view>
#include <vector>
#include <optional>

// 제로 카피 HTTP 헤더 파서
class HttpHeaderParser {
public:
    struct Header {
        std::string_view name;
        std::string_view value;
    };
    
    struct ParsedHeaders {
        std::string_view request_line;
        std::vector<Header> headers;
    };
    
    static std::optional<ParsedHeaders> parse(std::string_view data) {
        ParsedHeaders result;
        size_t pos = 0;
        
        // 요청 라인 찾기
        size_t line_end = data.find("\r\n", pos);
        if (line_end == std::string_view::npos) {
            return std::nullopt;
        }
        
        result.request_line = data.substr(pos, line_end - pos);
        pos = line_end + 2;
        
        // 헤더 파싱
        while (pos < data.size()) {
            line_end = data.find("\r\n", pos);
            if (line_end == std::string_view::npos) {
                break;
            }
            
            // 빈 줄 = 헤더 끝
            if (line_end == pos) {
                pos += 2;
                break;
            }
            
            std::string_view line = data.substr(pos, line_end - pos);
            pos = line_end + 2;
            
            // 헤더 이름과 값 분리
            size_t colon = line.find(':');
            if (colon == std::string_view::npos) {
                continue;
            }
            
            std::string_view name = line.substr(0, colon);
            std::string_view value = line.substr(colon + 1);
            
            // 앞뒤 공백 제거
            while (!value.empty() && value.front() == ' ') {
                value.remove_prefix(1);
            }
            while (!value.empty() && value.back() == ' ') {
                value.remove_suffix(1);
            }
            
            result.headers.push_back({name, value});
        }
        
        return result;
    }
};

// 청크 인코딩 파서 (제로 카피)
class ChunkedParser {
public:
    struct Chunk {
        size_t size;
        std::span<const std::byte> data;
    };
    
    static std::optional<Chunk> parse_chunk(std::span<const std::byte> data) {
        // 청크 크기 라인 찾기 (16진수)
        size_t i = 0;
        while (i < data.size() && 
               !(data[i] == std::byte('\r') && 
                 i + 1 < data.size() && 
                 data[i + 1] == std::byte('\n'))) {
            ++i;
        }
        
        if (i >= data.size() - 1) {
            return std::nullopt;
        }
        
        // 16진수 파싱
        std::string_view size_str(
            reinterpret_cast<const char*>(data.data()), i);
        
        size_t chunk_size = 0;
        for (char c : size_str) {
            chunk_size *= 16;
            if (c >= '0' && c <= '9') {
                chunk_size += c - '0';
            } else if (c >= 'a' && c <= 'f') {
                chunk_size += c - 'a' + 10;
            } else if (c >= 'A' && c <= 'F') {
                chunk_size += c - 'A' + 10;
            } else {
                return std::nullopt;
            }
        }
        
        // 청크 데이터 위치
        size_t data_start = i + 2;
        if (data_start + chunk_size + 2 > data.size()) {
            return std::nullopt;
        }
        
        return Chunk{
            chunk_size,
            data.subspan(data_start, chunk_size)
        };
    }
};

int main() {
    // HTTP 헤더 파싱 (제로 카피)
    std::string http_data = 
        "GET /api/users HTTP/1.1\r\n"
        "Host: example.com\r\n"
        "User-Agent: SafeClient/1.0\r\n"
        "Accept: application/json\r\n"
        "\r\n";
    
    auto parsed = HttpHeaderParser::parse(http_data);
    
    if (parsed) {
        std::cout << "요청: " << parsed->request_line << "\n\n";
        std::cout << "헤더:\n";
        for (const auto& header : parsed->headers) {
            std::cout << "  " << header.name << ": " 
                      << header.value << "\n";
        }
    }
    
    // 청크 파싱
    std::string chunk_data = "A\r\n0123456789\r\n";
    auto chunk_bytes = std::as_bytes(std::span(chunk_data));
    
    auto chunk = ChunkedParser::parse_chunk(chunk_bytes);
    if (chunk) {
        std::cout << "\n청크 크기: " << chunk->size << " 바이트\n";
    }
    
    return 0;
}
```

### 연결 풀과 리소스 관리

데이터베이스 연결 풀을 안전하게 구현한다.

```cpp
#include <iostream>
#include <memory>
#include <vector>
#include <mutex>
#include <condition_variable>
#include <chrono>
#include <optional>

// 연결 인터페이스
class Connection {
public:
    virtual ~Connection() = default;
    virtual bool is_alive() const = 0;
    virtual void reset() = 0;
};

// RAII 연결 핸들
template<typename T>
class ConnectionGuard {
    std::shared_ptr<T> connection_;
    std::function<void(std::shared_ptr<T>)> return_func_;
    
public:
    ConnectionGuard(std::shared_ptr<T> conn,
                   std::function<void(std::shared_ptr<T>)> return_func)
        : connection_(std::move(conn))
        , return_func_(std::move(return_func)) {}
    
    ~ConnectionGuard() {
        if (connection_ && return_func_) {
            return_func_(std::move(connection_));
        }
    }
    
    ConnectionGuard(const ConnectionGuard&) = delete;
    ConnectionGuard& operator=(const ConnectionGuard&) = delete;
    
    ConnectionGuard(ConnectionGuard&&) = default;
    ConnectionGuard& operator=(ConnectionGuard&&) = default;
    
    T* operator->() { return connection_.get(); }
    T& operator*() { return *connection_; }
};

// 연결 풀
template<typename T>
    requires std::derived_from<T, Connection>
class ConnectionPool {
    std::vector<std::shared_ptr<T>> connections_;
    std::mutex mutex_;
    std::condition_variable cv_;
    size_t max_connections_;
    size_t active_connections_ = 0;
    std::function<std::shared_ptr<T>()> factory_;
    
public:
    ConnectionPool(size_t max_connections,
                  std::function<std::shared_ptr<T>()> factory)
        : max_connections_(max_connections)
        , factory_(std::move(factory)) {}
    
    std::optional<ConnectionGuard<T>> acquire(
        std::chrono::milliseconds timeout = std::chrono::milliseconds(5000)) {
        
        std::unique_lock lock(mutex_);
        
        auto deadline = std::chrono::steady_clock::now() + timeout;
        
        while (true) {
            // 사용 가능한 연결 찾기
            if (!connections_.empty()) {
                auto conn = std::move(connections_.back());
                connections_.pop_back();
                
                // 연결 상태 확인
                if (conn->is_alive()) {
                    active_connections_++;
                    return ConnectionGuard<T>(
                        std::move(conn),
                        [this](std::shared_ptr<T> c) { this->release(std::move(c)); }
                    );
                }
                // 죽은 연결은 버림
                continue;
            }
            
            // 새 연결 생성 가능한지 확인
            if (active_connections_ < max_connections_) {
                lock.unlock();
                
                auto conn = factory_();
                if (!conn) {
                    return std::nullopt;
                }
                
                lock.lock();
                active_connections_++;
                
                return ConnectionGuard<T>(
                    std::move(conn),
                    [this](std::shared_ptr<T> c) { this->release(std::move(c)); }
                );
            }
            
            // 대기
            if (cv_.wait_until(lock, deadline) == std::cv_status::timeout) {
                return std::nullopt;
            }
        }
    }
    
    size_t active_count() const {
        std::lock_guard lock(mutex_);
        return active_connections_;
    }
    
    size_t available_count() const {
        std::lock_guard lock(mutex_);
        return connections_.size();
    }
    
private:
    void release(std::shared_ptr<T> conn) {
        std::lock_guard lock(mutex_);
        
        if (conn && conn->is_alive()) {
            conn->reset();
            connections_.push_back(std::move(conn));
        } else {
            active_connections_--;
        }
        
        cv_.notify_one();
    }
};

// 예제 데이터베이스 연결
class DbConnection : public Connection {
    bool alive_ = true;
    int connection_id_;
    
public:
    explicit DbConnection(int id) : connection_id_(id) {
        std::cout << "연결 " << connection_id_ << " 생성\n";
    }
    
    ~DbConnection() override {
        std::cout << "연결 " << connection_id_ << " 소멸\n";
    }
    
    bool is_alive() const override {
        return alive_;
    }
    
    void reset() override {
        // 연결 상태 초기화
        std::cout << "연결 " << connection_id_ << " 리셋\n";
    }
    
    void execute_query(const std::string& query) {
        std::cout << "연결 " << connection_id_ 
                  << "에서 쿼리 실행: " << query << "\n";
    }
};

int main() {
    // 연결 풀 생성
    int next_id = 0;
    ConnectionPool<DbConnection> pool(5, [&next_id]() {
        return std::make_shared<DbConnection>(++next_id);
    });
    
    std::cout << "=== 연결 풀 테스트 ===\n\n";
    
    // 연결 획득
    {
        auto conn1 = pool.acquire();
        if (conn1) {
            (*conn1)->execute_query("SELECT * FROM users");
        }
        
        std::cout << "활성 연결: " << pool.active_count() << "\n";
        std::cout << "대기 연결: " << pool.available_count() << "\n\n";
    }  // conn1 자동 반환
    
    std::cout << "활성 연결: " << pool.active_count() << "\n";
    std::cout << "대기 연결: " << pool.available_count() << "\n\n";
    
    // 다중 연결
    {
        std::vector<std::optional<ConnectionGuard<DbConnection>>> connections;
        
        for (int i = 0; i < 3; ++i) {
            connections.push_back(pool.acquire());
            if (connections.back()) {
                (*connections.back())->execute_query("INSERT INTO logs VALUES (...)");
            }
        }
        
        std::cout << "\n활성 연결: " << pool.active_count() << "\n";
        std::cout << "대기 연결: " << pool.available_count() << "\n";
    }
    
    std::cout << "\n최종 상태:\n";
    std::cout << "활성 연결: " << pool.active_count() << "\n";
    std::cout << "대기 연결: " << pool.available_count() << "\n";
```
