# 18장: 성능 프로파일링과 최적화

안전한 C++ 코드를 작성하는 것은 필수지만, 성능을 희생해서는 안 된다. "안전하지만 느린 코드"는 실전에서 채택되기 어렵다. 이 장에서는 안전성을 유지하면서도 최적의 성능을 달성하는 방법을 살펴본다. C++의 "zero-cost abstraction" 원칙에 따라, 안전한 추상화가 성능 오버헤드를 발생시키지 않도록 설계하고 측정하는 기법들을 다룬다.

## 18.1 안전성과 성능의 트레이드오프

### 18.1.1 경계 검사의 비용

**안전하지 않은 방법: 경계 검사 없음**

```cpp
#include <vector>
#include <chrono>
#include <iostream>

class Timer {
    std::chrono::high_resolution_clock::time_point start_;
public:
    Timer() : start_(std::chrono::high_resolution_clock::now()) {}
    
    double elapsed_ms() const {
        auto end = std::chrono::high_resolution_clock::now();
        return std::chrono::duration<double, std::milli>(end - start_).count();
    }
};

// 경계 검사 없는 접근
long long sum_unchecked(const std::vector<int>& vec) {
    long long sum = 0;
    for (size_t i = 0; i < vec.size(); ++i) {
        sum += vec[i];  // operator[] - 경계 검사 없음
    }
    return sum;
}

void benchmark_unchecked() {
    std::vector<int> data(10'000'000);
    for (size_t i = 0; i < data.size(); ++i) {
        data[i] = static_cast<int>(i % 1000);
    }
    
    Timer timer;
    volatile long long result = sum_unchecked(data);
    double elapsed = timer.elapsed_ms();
    
    std::cout << "Unchecked access: " << elapsed << " ms" << std::endl;
    std::cout << "Result: " << result << std::endl;
}
```

**비교: 경계 검사 있음**

```cpp
// 경계 검사 있는 접근
long long sum_checked(const std::vector<int>& vec) {
    long long sum = 0;
    for (size_t i = 0; i < vec.size(); ++i) {
        sum += vec.at(i);  // at() - 경계 검사 수행
    }
    return sum;
}

void benchmark_checked() {
    std::vector<int> data(10'000'000);
    for (size_t i = 0; i < data.size(); ++i) {
        data[i] = static_cast<int>(i % 1000);
    }
    
    Timer timer;
    volatile long long result = sum_checked(data);
    double elapsed = timer.elapsed_ms();
    
    std::cout << "Checked access: " << elapsed << " ms" << std::endl;
    std::cout << "Result: " << result << std::endl;
}
```

**안전한 방법: 컴파일러 최적화를 활용한 zero-cost 경계 검사**

```cpp
#include <ranges>
#include <numeric>

// 범위 기반 접근 - 컴파일러가 경계 검사 최적화
long long sum_range_based(const std::vector<int>& vec) {
    long long sum = 0;
    for (int val : vec) {  // 범위 기반 for - 안전하면서도 최적화 가능
        sum += val;
    }
    return sum;
}

// 알고리즘 활용 - 가장 최적화된 방법
long long sum_algorithm(const std::vector<int>& vec) {
    return std::accumulate(vec.begin(), vec.end(), 0LL);
}

// C++23 ranges 활용
long long sum_ranges(const std::vector<int>& vec) {
    return std::ranges::fold_left(vec, 0LL, std::plus<>{});
}

void benchmark_optimized() {
    std::vector<int> data(10'000'000);
    for (size_t i = 0; i < data.size(); ++i) {
        data[i] = static_cast<int>(i % 1000);
    }
    
    {
        Timer timer;
        volatile long long result = sum_range_based(data);
        std::cout << "Range-based: " << timer.elapsed_ms() << " ms" << std::endl;
    }
    
    {
        Timer timer;
        volatile long long result = sum_algorithm(data);
        std::cout << "Algorithm: " << timer.elapsed_ms() << " ms" << std::endl;
    }
    
    {
        Timer timer;
        volatile long long result = sum_ranges(data);
        std::cout << "Ranges: " << timer.elapsed_ms() << " ms" << std::endl;
    }
}
```

### 18.1.2 스마트 포인터 vs 원시 포인터

**안전하지 않은 방법: 원시 포인터**

```cpp
#include <memory>
#include <vector>

struct Node {
    int value;
    Node* next;
};

class LinkedListRaw {
    Node* head_ = nullptr;
    
public:
    void push_front(int value) {
        Node* new_node = new Node{value, head_};
        head_ = new_node;
    }
    
    ~LinkedListRaw() {
        while (head_) {
            Node* temp = head_;
            head_ = head_->next;
            delete temp;
        }
    }
    
    long long sum() const {
        long long total = 0;
        Node* current = head_;
        while (current) {
            total += current->value;
            current = current->next;
        }
        return total;
    }
};

void benchmark_raw_pointers() {
    LinkedListRaw list;
    
    Timer timer;
    for (int i = 0; i < 1'000'000; ++i) {
        list.push_front(i % 1000);
    }
    
    volatile long long result = list.sum();
    std::cout << "Raw pointers: " << timer.elapsed_ms() << " ms" << std::endl;
}
```

**안전한 방법: unique_ptr 사용**

```cpp
struct NodeSafe {
    int value;
    std::unique_ptr<NodeSafe> next;
};

class LinkedListSafe {
    std::unique_ptr<NodeSafe> head_;
    
public:
    void push_front(int value) {
        auto new_node = std::make_unique<NodeSafe>(value, std::move(head_));
        head_ = std::move(new_node);
    }
    
    long long sum() const {
        long long total = 0;
        NodeSafe* current = head_.get();
        while (current) {
            total += current->value;
            current = current->next.get();
        }
        return total;
    }
};

void benchmark_smart_pointers() {
    LinkedListSafe list;
    
    Timer timer;
    for (int i = 0; i < 1'000'000; ++i) {
        list.push_front(i % 1000);
    }
    
    volatile long long result = list.sum();
    std::cout << "Smart pointers: " << timer.elapsed_ms() << " ms" << std::endl;
}
```

**최적화된 방법: 캐시 친화적 설계**

```cpp
// 노드 기반 구조 대신 연속 메모리 사용
class VectorBased {
    std::vector<int> data_;
    
public:
    void push_front(int value) {
        data_.insert(data_.begin(), value);
    }
    
    void push_back(int value) {
        data_.push_back(value);
    }
    
    long long sum() const {
        return std::accumulate(data_.begin(), data_.end(), 0LL);
    }

void benchmark_vector_based() {

    list.data_.reserve(1'000'000);  // 사전 할당으로 재할당 방지

        list.push_back(i % 1000);  // push_back이 push_front보다 효율적
    }
    
    volatile long long result = list.sum();
    std::cout << "Vector-based: " << timer.elapsed_ms() << " ms" << std::endl;
}
```

### 18.1.3 예외 처리의 비용

**안전하지 않은 방법: 에러 코드 반환**

```cpp
#include <optional>
#include <cmath>

// 에러 코드 방식
std::optional<double> safe_sqrt_error_code(double x) noexcept {
    if (x < 0) {
        return std::nullopt;
    }
    return std::sqrt(x);
}

void benchmark_error_codes() {
    std::vector<double> inputs(1'000'000);
    for (size_t i = 0; i < inputs.size(); ++i) {
        inputs[i] = static_cast<double>(i);
    }
    
    Timer timer;
    double sum = 0;
    for (double x : inputs) {
        if (auto result = safe_sqrt_error_code(x)) {
            sum += *result;
        }
    }
    
    volatile double result = sum;
    std::cout << "Error codes: " << timer.elapsed_ms() << " ms" << std::endl;
}
```

**안전한 방법: 예외 사용**

```cpp
#include <stdexcept>

double safe_sqrt_exception(double x) {
    if (x < 0) {
        throw std::domain_error("Negative input");
    }
    return std::sqrt(x);
}

void benchmark_exceptions() {
    std::vector<double> inputs(1'000'000);
    for (size_t i = 0; i < inputs.size(); ++i) {
        inputs[i] = static_cast<double>(i);
    }
    
    Timer timer;
    double sum = 0;
    for (double x : inputs) {
        try {
            sum += safe_sqrt_exception(x);
        } catch (const std::domain_error&) {
            // 에러 처리
        }
    }
    
    volatile double result = sum;
    std::cout << "Exceptions (no throw): " << timer.elapsed_ms() << " ms" << std::endl;
}
```

**최적화된 방법: 조건부 예외 활용**

```cpp
// 정상 경로에서 예외를 던지지 않으면 오버헤드 최소
double safe_sqrt_optimized(double x) {
    // 사전조건을 caller가 보장하면 예외 경로 제거
    return std::sqrt(x);  // 빠른 경로
}

void benchmark_optimized_exceptions() {
    std::vector<double> inputs(1'000'000);
    for (size_t i = 0; i < inputs.size(); ++i) {
        inputs[i] = static_cast<double>(i);
    }
    
    Timer timer;
    double sum = 0;
    for (double x : inputs) {
        // 사전 검증으로 예외 경로 완전 제거
        if (x >= 0) {
            sum += safe_sqrt_optimized(x);
        }
    }
    
    volatile double result = sum;
    std::cout << "Optimized (no exception path): " << timer.elapsed_ms() << " ms" << std::endl;
}
```

## 18.2 Zero-cost abstraction 달성하기

### 18.2.1 인라인과 템플릿 최적화

**안전하지 않은 방법: 런타임 다형성**

```cpp
#include <vector>
#include <memory>

class ShapeBase {
public:
    virtual ~ShapeBase() = default;
    virtual double area() const = 0;
};

class Circle : public ShapeBase {
    double radius_;
public:
    explicit Circle(double r) : radius_(r) {}
    double area() const override {
        return 3.14159 * radius_ * radius_;
    }
};

class Rectangle : public ShapeBase {
    double width_, height_;
public:
    Rectangle(double w, double h) : width_(w), height_(h) {}
    double area() const override {
        return width_ * height_;
    }
};

void benchmark_virtual() {
    std::vector<std::unique_ptr<ShapeBase>> shapes;
    shapes.reserve(1'000'000);
    
    for (int i = 0; i < 500'000; ++i) {
        shapes.push_back(std::make_unique<Circle>(i % 100));
        shapes.push_back(std::make_unique<Rectangle>(i % 50, i % 50));
    }
    
    Timer timer;
    double total = 0;
    for (const auto& shape : shapes) {
        total += shape->area();  // 가상 함수 호출
    }
    
    volatile double result = total;
    std::cout << "Virtual functions: " << timer.elapsed_ms() << " ms" << std::endl;
}
```

**안전한 방법: std::variant와 std::visit**

```cpp
#include <variant>

struct CircleData {
    double radius;
    
    double area() const {
        return 3.14159 * radius * radius;
    }
};

struct RectangleData {
    double width, height;
    
    double area() const {
        return width * height;
    }
};

using Shape = std::variant<CircleData, RectangleData>;

void benchmark_variant() {
    std::vector<Shape> shapes;
    shapes.reserve(1'000'000);
    
    for (int i = 0; i < 500'000; ++i) {
        shapes.push_back(CircleData{static_cast<double>(i % 100)});
        shapes.push_back(RectangleData{
            static_cast<double>(i % 50),
            static_cast<double>(i % 50)
        });
    }
    
    Timer timer;
    double total = 0;
    for (const auto& shape : shapes) {
        total += std::visit([](const auto& s) { return s.area(); }, shape);
    }
    
    volatile double result = total;
    std::cout << "std::variant: " << timer.elapsed_ms() << " ms" << std::endl;
}
```

**최적화된 방법: 컴파일 타임 다형성**

```cpp
template<typename ShapeType>

    double total = 0;

        total += shape.area();  // 인라인 가능
    }
    return total;
}

void benchmark_compile_time() {

    circles.reserve(500'000);
    
    std::vector<RectangleData> rectangles;
    rectangles.reserve(500'000);

        });
    }
    
    Timer timer;
    double total = 0;
    total += calculate_total_area(circles);
    total += calculate_total_area(rectangles);
    
    volatile double result = total;
    std::cout << "Compile-time polymorphism: " << timer.elapsed_ms() << " ms" << std::endl;
}
```

### 18.2.3 constexpr의 성능 이점

### 18.2.4 이동 의미론 최적화

**활용 예: 복사 vs 이동**

```cpp
#include <string>
#include <vector>

class DataContainer {
    std::vector<std::string> data_;
    
public:
    explicit DataContainer(size_t size) {

void benchmark_copy() {
    std::vector<DataContainer> containers;
    
    Timer timer;
    
    DataContainer source(1000);
    for (int i = 0; i < 1000; ++i) {
        containers.push_back(source);  // 복사
    }
    
    std::cout << "Copy: " << timer.elapsed_ms() << " ms" << std::endl;
}

void benchmark_move() {
    std::vector<DataContainer> containers;
    
    Timer timer;
    
    for (int i = 0; i < 1000; ++i) {
        containers.push_back(DataContainer(1000));  // 이동
    }
    
    std::cout << "Move: " << timer.elapsed_ms() << " ms" << std::endl;
}

void benchmark_emplace() {
    std::vector<DataContainer> containers;
    
    Timer timer;
    
    for (int i = 0; i < 1000; ++i) {

}
```

## 18.3 실제 벤치마킹과 측정

### 18.3.1 마이크로벤치마킹 프레임워크

**활용 예: 정확한 벤치마킹**

```cpp
#include <chrono>
#include <vector>
#include <numeric>
#include <algorithm>
#include <iostream>
#include <iomanip>

class Benchmark {
public:
    struct Result {
        std::string name;
        double mean_ms;
        double min_ms;
        double max_ms;
        double stddev_ms;
    };
    
private:
    std::string name_;
    std::vector<double> measurements_;
    size_t iterations_;
    size_t warmup_iterations_;
    
    static double calculate_mean(const std::vector<double>& values) {
        return std::accumulate(values.begin(), values.end(), 0.0) / values.size();
    }
    
    static double calculate_stddev(const std::vector<double>& values, double mean) {
        double variance = 0.0;
        for (double val : values) {
            double diff = val - mean;
            variance += diff * diff;
        }
        return std::sqrt(variance / values.size());
    }
    
public:
    explicit Benchmark(std::string name, size_t iterations = 100, size_t warmup = 10)
        : name_(std::move(name)), iterations_(iterations), warmup_iterations_(warmup) {
        measurements_.reserve(iterations);
    }
    
    template<typename Fn>
    Result run(Fn&& fn) {
        measurements_.clear();
        
        // Warmup
        for (size_t i = 0; i < warmup_iterations_; ++i) {
            fn();
        }
        
        // Actual measurements
        for (size_t i = 0; i < iterations_; ++i) {
            auto start = std::chrono::high_resolution_clock::now();
            fn();
            auto end = std::chrono::high_resolution_clock::now();
            
            double elapsed = std::chrono::duration<double, std::milli>(end - start).count();
            measurements_.push_back(elapsed);
        }
        
        double mean = calculate_mean(measurements_);
        double stddev = calculate_stddev(measurements_, mean);
        
        return Result{
            name_,
            mean,
            *std::min_element(measurements_.begin(), measurements_.end()),
            *std::max_element(measurements_.begin(), measurements_.end()),
            stddev
        };
    }
    
    static void print_result(const Result& result) {
        std::cout << std::fixed << std::setprecision(3);
        std::cout << result.name << ":\n";
        std::cout << "  Mean:   " << std::setw(10) << result.mean_ms << " ms\n";
        std::cout << "  Min:    " << std::setw(10) << result.min_ms << " ms\n";
        std::cout << "  Max:    " << std::setw(10) << result.max_ms << " ms\n";
        std::cout << "  StdDev: " << std::setw(10) << result.stddev_ms << " ms\n";
    }
};

// 사용 예
void demonstrate_benchmarking() {
    const size_t N = 1'000'000;
    std::vector<int> data(N);
    std::iota(data.begin(), data.end(), 0);
    
    // 벤치마크 1: std::accumulate
    {
        Benchmark bench("std::accumulate");
        auto result = bench.run([&data]() {
            volatile long long sum = std::accumulate(data.begin(), data.end(), 0LL);
```

```cpp
#include <cstddef>
#include <new>

// 메모리 할당 추적
class MemoryTracker {
    static inline size_t total_allocated_ = 0;
    static inline size_t total_deallocated_ = 0;
    static inline size_t current_usage_ = 0;
    static inline size_t peak_usage_ = 0;
    
public:
    static void allocate(size_t size) {
        total_allocated_ += size;
        current_usage_ += size;
        if (current_usage_ > peak_usage_) {
            peak_usage_ = current_usage_;
        }
    }
    
    static void deallocate(size_t size) {
        total_deallocated_ += size;
        current_usage_ -= size;
    }
    
    static void reset() {
        total_allocated_ = 0;
        total_deallocated_ = 0;
        current_usage_ = 0;
        peak_usage_ = 0;
    }
    
    static void print_stats() {
        std::cout << "Memory Statistics:\n";
        std::cout << "  Total allocated:   " << total_allocated_ << " bytes\n";
        std::cout << "  Total deallocated: " << total_deallocated_ << " bytes\n";
        std::cout << "  Current usage:     " << current_usage_ << " bytes\n";
        std::cout << "  Peak usage:        " << peak_usage_ << " bytes\n";
    }
};

// 커스텀 allocator
template<typename T>
class TrackingAllocator {
public:
    using value_type = T;
    
    TrackingAllocator() = default;
    
    template<typename U>
    TrackingAllocator(const TrackingAllocator<U>&) noexcept {}
    
    T* allocate(size_t n) {
        size_t bytes = n * sizeof(T);
        MemoryTracker::allocate(bytes);
        return static_cast<T*>(::operator new(bytes));
    }
    
    void deallocate(T* p, size_t n) noexcept {
        size_t bytes = n * sizeof(T);
        MemoryTracker::deallocate(bytes);
        ::operator delete(p);
    }
    
    template<typename U>
    bool operator==(const TrackingAllocator<U>&) const noexcept {
        return true;
    }
    
    template<typename U>
    bool operator!=(const TrackingAllocator<U>&) const noexcept {
        return false;
    }
};

void benchmark_memory_usage() {
    MemoryTracker::reset();
    
    {
        std::vector<int, TrackingAllocator<int>> vec;
        
        for (int i = 0; i < 1'000'000; ++i) {
            vec.push_back(i);
        }
        
        std::cout << "After building vector:\n";
        MemoryTracker::print_stats();
    }
    
    std::cout << "\nAfter vector destroyed:\n";
    MemoryTracker::print_stats();
}
```

```cpp
std::vector<int> parse_csv_stringstream(const std::string& input) {
    std::vector<int> result;
    std::stringstream ss(input);
    std::string token;
    
    while (std::getline(ss, token, ',')) {
        result.push_back(std::stoi(token));
    }
    
    return result;
}

// 방법 2: string_view와 find 사용
std::vector<int> parse_csv_string_view(std::string_view input) {
    std::vector<int> result;
    size_t pos = 0;
    
    while (pos < input.size()) {
        size_t comma = input.find(',', pos);
        if (comma == std::string_view::npos) {
            comma = input.size();
        }
        
        auto token = input.substr(pos, comma - pos);
        result.push_back(std::stoi(std::string(token)));
        
        pos = comma + 1;
    }
    
    return result;
}

// 방법 3: std::from_chars 사용 (C++17, 가장 빠름)
std::vector<int> parse_csv_from_chars(std::string_view input) {
    std::vector<int> result;
    const char* ptr = input.data();
    const char* end = ptr + input.size();
    
    while (ptr < end) {
        int value;
        auto [next_ptr, ec] = std::from_chars(ptr, end, value);
        
        if (ec == std::errc{}) {
            result.push_back(value);
        }
        
        // 다음 숫자로 이동 (쉼표 건너뛰기)
        ptr = next_ptr;
        if (ptr < end && *ptr == ',') {
            ++ptr;
        }
    }
    
    return result;
}
```

**활용 예: 종합 성능 최적화**

```cpp
#include <span>
#include <ranges>

// 최적화 전: 여러 문제점
class DataProcessorSlow {
    std::vector<int> data_;
    
public:
    void process(const std::vector<int>& input) {
        data_.clear();
        
        // 문제 1: 불필요한 복사
        for (int val : input) {
            // 문제 2: 범위 검사 오버헤드
            if (val > 0) {
                // 문제 3: 여러 번 push_back (재할당 발생 가능)
                data_.push_back(val * 2);
            }
        }
        
        // 문제 4: 불필요한 정렬 (이미 정렬되어 있을 수 있음)
        std::sort(data_.begin(), data_.end());
    }
    
    int sum() const {
        int total = 0;
        // 문제 5: 범위 검사 오버헤드
        for (size_t i = 0; i < data_.size(); ++i) {
            total += data_.at(i);
        }
        return total;
    }
};

// 최적화 후
class DataProcessorFast {
    std::vector<int> data_;
    
public:
    void process(std::span<const int> input) {
        // 예상 크기 사전 할당
        data_.clear();
        data_.reserve(input.size());
        
        // 범위 기반 for + 인라인 최적화
        for (int val : input) {
            if (val > 0) {
                data_.push_back(val * 2);
            }
        }
        
        // 필요한 경우에만 정렬
        if (!std::ranges::is_sorted(data_)) {
            std::ranges::sort(data_);
        }
    }
    
    int sum() const {
        // 알고리즘 사용 (최적화된 구현)
        return std::accumulate(data_.begin(), data_.end(), 0);
    }
};

// C++23 ranges를 활용한 최적화
class DataProcessorRanges {
public:
    int process_and_sum(std::span<const int> input) {
        // 지연 평가로 중간 벡터 생성 없음
        auto positive_doubled = input 
            | std::views::filter([](int x) { return x > 0; })
            | std::views::transform([](int x) { return x * 2; });
        
        return std::ranges::fold_left(positive_doubled, 0, std::plus<>{});
    }
};

void benchmark_optimization_comparison() {
    std::vector<int> input(1'000'000);
    std::iota(input.begin(), input.end(), -500'000);
    
    {
        Benchmark bench("Slow version");
        auto result = bench.run([&input]() {
            DataProcessorSlow processor;
            processor.process(input);
            volatile int sum = processor.sum();
        });
        Benchmark::print_result(result);
    }
    
    {
        Benchmark bench("Fast version");
        auto result = bench.run([&input]() {
            DataProcessorFast processor;
            processor.process(input);
            volatile int sum = processor.sum();
        });
        Benchmark::print_result(result);
    }
    
    {
        Benchmark bench("Ranges version");
        auto result = bench.run([&input]() {
            DataProcessorRanges processor;
            volatile int sum = processor.process_and_sum(input);
        });
        Benchmark::print_result(result);
    }
}
```

## 정리

이 장에서는 안전한 C++ 코드를 작성하면서도 최적의 성능을 달성하는 방법을 살펴보았다. 안전성과 성능은 상충하는 목표가 아니며, 올바른 추상화와 최적화 기법을 사용하면 둘 다 달성할 수 있다.

핵심 교훈은 다음과 같다.

첫째, 경계 검사, 스마트 포인터, 예외 처리 등의 안전 기능은 올바르게 사용하면 성능 오버헤드가 최소화되거나 제거된다. 둘째, 컴파일 타임 다형성, constexpr, 이동 의미론 등을 활용하여 zero-cost abstraction을 달성할 수 있다. 셋째, 정확한 벤치마킹을 통해 실제 성능 병목을 식별하고 측정 기반으로 최적화해야 한다. 넷째, 메모리 레이아웃, 캐시 친화성, 알고리즘 선택이 성능에 큰 영향을 미친다. 다섯째, C++26의 최신 기능들(ranges, views 등)을 활용하면 안전하면서도 효율적인 코드를 작성할 수 있다.

"먼저 올바르게 만들고, 그 다음 빠르게 만들어라"는 원칙을 따르되, C++의 zero-cost abstraction 원칙에 따라 안전성이 곧 성능이 될 수 있도록 설계한다. 이 책에서 다룬 모든 기법들을 종합하여 Rust 수준의 안전성과 C++ 고유의 성능을 동시에 달성할 수 있다.
