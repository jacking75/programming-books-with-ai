## 12.1 C++26 정적 리플렉션 기초

### 전통적인 메타프로그래밍의 안전성 문제

C++에서 타입 정보에 접근하는 전통적인 방법들은 런타임 오버헤드와 타입 안전성 문제를 내포하고 있다.

```cpp
#include <typeinfo>
#include <string>
#include <iostream>
#include <any>

// 안전하지 않은 예 1: typeid를 사용한 런타임 타입 검사
class UnsafeTypeInspection {
public:
    template<typename T>
    static std::string get_type_name(const T& value) {
        // 런타임 오버헤드, 플랫폼 의존적 이름
        return typeid(value).name();  // 디맹글링 필요
    }
    
    // std::any를 사용한 타입 소거 - 안전하지 않음
    static void process_any(std::any value) {
        if (value.type() == typeid(int)) {
            int i = std::any_cast<int>(value);  // 런타임 검사
            std::cout << "정수: " << i << "\n";
        } else if (value.type() == typeid(std::string)) {
            auto s = std::any_cast<std::string>(value);
            std::cout << "문자열: " << s << "\n";
        }
        // 모든 타입을 수동으로 처리해야 함
    }
};

// 안전하지 않은 예 2: 매크로 기반 리플렉션
#define REFLECT_MEMBER(type, name) \
    { #name, offsetof(type, name), sizeof(((type*)0)->name) }

struct OldStyleReflection {
    int id;
    std::string name;
    double value;
    
    // 수동으로 멤버 정보 유지 - 오류 가능성 높음
    static constexpr auto members = {
        REFLECT_MEMBER(OldStyleReflection, id),
        REFLECT_MEMBER(OldStyleReflection, name),
        REFLECT_MEMBER(OldStyleReflection, value)
    };
};
```

이러한 접근 방식의 문제점은 다음과 같다.

1. **런타임 오버헤드**: `typeid`와 `std::any`는 런타임 타입 검사를 수행한다
2. **수동 유지보수**: 구조체가 변경되면 리플렉션 코드도 수동으로 업데이트해야 한다
3. **타입 안전성 부족**: 잘못된 캐스팅이나 타입 불일치를 컴파일 타임에 감지하지 못한다
4. **플랫폼 의존성**: 타입 이름이 컴파일러마다 다르다

### C++26 정적 리플렉션 소개

C++26의 정적 리플렉션은 컴파일 타임에 타입 정보에 안전하게 접근할 수 있는 기능을 제공한다.

```cpp
#include <experimental/reflect>
#include <iostream>
#include <string_view>

namespace refl = std::experimental::reflect;

// 안전한 예: 정적 리플렉션 사용
struct Person {
    int id;
    std::string name;
    double salary;
};

// 컴파일 타임 타입 정보 추출
template<typename T>
consteval auto get_member_count() {
    return refl::get_member_count(reflexpr(T));
}

// 컴파일 타임에 멤버 이름 추출
template<typename T, size_t Index>
consteval std::string_view get_member_name() {
    using members = refl::get_data_members_t<reflexpr(T)>;
    using member = refl::get_element_t<Index, members>;
    return refl::get_name_v<member>;
}

void demonstrate_static_reflection() {
    // 컴파일 타임에 멤버 개수 확인
    static_assert(get_member_count<Person>() == 3);
    
    // 컴파일 타임에 멤버 이름 확인
    constexpr auto first_member = get_member_name<Person, 0>();
    static_assert(first_member == "id");
    
    std::cout << "Person 구조체 멤버:\n";
    std::cout << "  " << get_member_name<Person, 0>() << "\n";
    std::cout << "  " << get_member_name<Person, 1>() << "\n";
    std::cout << "  " << get_member_name<Person, 2>() << "\n";
}
```

### 타입 안전한 리플렉션 API

C++26 리플렉션은 완전한 타입 안전성을 보장한다.

```cpp
#include <experimental/reflect>
#include <type_traits>
#include <iostream>

namespace refl = std::experimental::reflect;

// 구조체 정의
struct Employee {
    int employee_id;
    std::string full_name;
    double annual_salary;
    bool is_active;
};

// 안전한 멤버 접근
template<typename T>
class SafeReflector {
public:
    // 컴파일 타임에 멤버 타입 추출
    template<size_t Index>
    using member_type = typename refl::get_type_t<
        refl::get_element_t<
            Index, 
            refl::get_data_members_t<reflexpr(T)>
        >
    >;
    
    // 컴파일 타임 타입 검증
    template<size_t Index, typename Expected>
    static constexpr bool is_member_type_v = 
        std::is_same_v<member_type<Index>, Expected>;
    
    // 멤버 개수 확인
    static constexpr size_t member_count = 
        refl::get_size_v<refl::get_data_members_t<reflexpr(T)>>;
    
    // 안전한 멤버 이름 추출
    template<size_t Index>
    static constexpr std::string_view get_name() {
        using members = refl::get_data_members_t<reflexpr(T)>;
        using member = refl::get_element_t<Index, members>;
        return refl::get_name_v<member>;
    }
};

void demonstrate_type_safety() {
    using Reflector = SafeReflector<Employee>;
    
    // 컴파일 타임 검증
    static_assert(Reflector::member_count == 4);
    static_assert(Reflector::is_member_type_v<0, int>);
    static_assert(Reflector::is_member_type_v<1, std::string>);
    static_assert(Reflector::is_member_type_v<2, double>);
    static_assert(Reflector::is_member_type_v<3, bool>);
    
    // 잘못된 타입 검증은 컴파일 오류
    // static_assert(Reflector::is_member_type_v<0, std::string>);  // 실패!
    
    std::cout << "타입 안전성 검증 완료\n";
}
```

### 열거형 리플렉션

열거형 값을 안전하게 다루는 방법이다.

```cpp
#include <experimental/reflect>
#include <iostream>
#include <string_view>
#include <optional>

namespace refl = std::experimental::reflect;

// 열거형 정의
enum class Status {
    Pending,
    Processing,
    Completed,
    Failed
};

// 안전한 열거형 유틸리티
template<typename Enum>
class EnumReflector {
public:
    // 열거형 값을 문자열로 변환
    static constexpr std::string_view to_string(Enum value) {
        using enum_info = reflexpr(Enum);
        using enumerators = refl::get_enumerators_t<enum_info>;
        
        return []<size_t... Is>(
            Enum val, 
            std::index_sequence<Is...>
        ) -> std::string_view {
            std::string_view result = "Unknown";
            
            (([&] {
                using enumerator = refl::get_element_t<Is, enumerators>;
                if (refl::get_constant_v<enumerator> == val) {
                    result = refl::get_name_v<enumerator>;
                }
            }()), ...);
            
            return result;
        }(value, std::make_index_sequence<
            refl::get_size_v<enumerators>
        >{});
    }
    
    // 문자열을 열거형 값으로 변환
    static constexpr std::optional<Enum> from_string(std::string_view name) {
        using enum_info = reflexpr(Enum);
        using enumerators = refl::get_enumerators_t<enum_info>;
        
        return []<size_t... Is>(
            std::string_view n,
            std::index_sequence<Is...>
        ) -> std::optional<Enum> {
            std::optional<Enum> result;
            
            (([&] {
                using enumerator = refl::get_element_t<Is, enumerators>;
                if (refl::get_name_v<enumerator> == n) {
                    result = refl::get_constant_v<enumerator>;
                }
            }()), ...);
            
            return result;
        }(name, std::make_index_sequence<
            refl::get_size_v<enumerators>
        >{});
    }
    
    // 모든 열거형 값 순회
    template<typename Func>
    static constexpr void for_each(Func&& func) {
        using enum_info = reflexpr(Enum);
        using enumerators = refl::get_enumerators_t<enum_info>;
        
        []<size_t... Is>(
            Func&& f,
            std::index_sequence<Is...>
        ) {
            (([&] {
                using enumerator = refl::get_element_t<Is, enumerators>;
                f(
                    refl::get_constant_v<enumerator>,
                    refl::get_name_v<enumerator>
                );
            }()), ...);
        }(std::forward<Func>(func), std::make_index_sequence<
            refl::get_size_v<enumerators>
        >{});
    }
};

void demonstrate_enum_reflection() {
    using StatusReflector = EnumReflector<Status>;
    
    // 열거형을 문자열로 변환
    Status status = Status::Processing;
    std::cout << "상태: " << StatusReflector::to_string(status) << "\n";
    
    // 문자열을 열거형으로 변환
    auto parsed = StatusReflector::from_string("Completed");
    if (parsed) {
        std::cout << "파싱된 상태: " 
                  << StatusReflector::to_string(*parsed) << "\n";
    }
    
    // 모든 열거형 값 출력
    std::cout << "모든 상태 값:\n";
    StatusReflector::for_each([](Status s, std::string_view name) {
        std::cout << "  " << name << " = " 
                  << static_cast<int>(s) << "\n";
    });
}
```

## 12.2 컴파일 타임 타입 검증

### 구조체 레이아웃 검증

컴파일 타임에 구조체의 레이아웃을 검증하여 안전성을 보장한다.

```cpp
#include <experimental/reflect>
#include <type_traits>
#include <cstddef>

namespace refl = std::experimental::reflect;

// 검증이 필요한 구조체
struct Message {
    uint32_t header;
    uint16_t type;
    uint16_t flags;
    uint64_t timestamp;
    char payload[256];
};

// 컴파일 타임 레이아웃 검증
template<typename T>
class LayoutValidator {
public:
    // 멤버 개수 검증
    template<size_t Expected>
    static constexpr bool validate_member_count() {
        using members = refl::get_data_members_t<reflexpr(T)>;
        return refl::get_size_v<members> == Expected;
    }
    
    // 특정 멤버의 타입 검증
    template<size_t Index, typename ExpectedType>
    static constexpr bool validate_member_type() {
        using members = refl::get_data_members_t<reflexpr(T)>;
        using member = refl::get_element_t<Index, members>;
        using actual_type = refl::get_type_t<member>;
        return std::is_same_v<actual_type, ExpectedType>;
    }
    
    // 구조체 크기 검증
    template<size_t ExpectedSize>
    static constexpr bool validate_size() {
        return sizeof(T) == ExpectedSize;
    }
    
    // 정렬 요구사항 검증
    template<size_t ExpectedAlignment>
    static constexpr bool validate_alignment() {
        return alignof(T) == ExpectedAlignment;
    }
};

// 컴파일 타임 검증 수행
void validate_message_layout() {
    using Validator = LayoutValidator<Message>;
    
    // 멤버 개수 확인
    static_assert(Validator::validate_member_count<5>(), 
                  "Message 구조체는 5개 멤버를 가져야 한다");
    
    // 각 멤버 타입 확인
    static_assert(Validator::validate_member_type<0, uint32_t>(),
                  "첫 번째 멤버는 uint32_t여야 한다");
    static_assert(Validator::validate_member_type<1, uint16_t>(),
                  "두 번째 멤버는 uint16_t여야 한다");
    static_assert(Validator::validate_member_type<2, uint16_t>(),
                  "세 번째 멤버는 uint16_t여야 한다");
    static_assert(Validator::validate_member_type<3, uint64_t>(),
                  "네 번째 멤버는 uint64_t여야 한다");
    static_assert(Validator::validate_member_type<4, char[256]>(),
                  "다섯 번째 멤버는 char[256]이어야 한다");
    
    std::cout << "Message 레이아웃 검증 완료\n";
}
```

### 타입 특성 자동 추론

리플렉션을 사용하여 타입 특성을 자동으로 추론한다.

```cpp
#include <experimental/reflect>
#include <type_traits>
#include <iostream>

namespace refl = std::experimental::reflect;

// 타입 특성 추론 유틸리티
template<typename T>
class TypeTraits {
public:
    // 모든 멤버가 기본 생성 가능한지 확인
    static constexpr bool all_members_default_constructible() {
        using members = refl::get_data_members_t<reflexpr(T)>;
        
        return []<size_t... Is>(std::index_sequence<Is...>) {
            return (std::is_default_constructible_v<
                refl::get_type_t<
                    refl::get_element_t<Is, members>
                >
            > && ...);
        }(std::make_index_sequence<refl::get_size_v<members>>{});
    }
    
    // 모든 멤버가 복사 가능한지 확인
    static constexpr bool all_members_copyable() {
        using members = refl::get_data_members_t<reflexpr(T)>;
        
        return []<size_t... Is>(std::index_sequence<Is...>) {
            return (std::is_copy_constructible_v<
                refl::get_type_t<
                    refl::get_element_t<Is, members>
                >
            > && ...);
        }(std::make_index_sequence<refl::get_size_v<members>>{});
    }
    
    // 모든 멤버가 이동 가능한지 확인
    static constexpr bool all_members_movable() {
        using members = refl::get_data_members_t<reflexpr(T)>;
        
        return []<size_t... Is>(std::index_sequence<Is...>) {
            return (std::is_move_constructible_v<
                refl::get_type_t<
                    refl::get_element_t<Is, members>
                >
            > && ...);
        }(std::make_index_sequence<refl::get_size_v<members>>{});
    }
    
    // POD 타입 여부 확인
    static constexpr bool is_pod() {
        return std::is_standard_layout_v<T> && std::is_trivial_v<T>;
    }
};

// 테스트용 구조체
struct SimplePOD {
    int x;
    double y;
};

struct NonPOD {
    std::string text;
    int value;
};

void demonstrate_type_traits() {
    using PODTraits = TypeTraits<SimplePOD>;
    using NonPODTraits = TypeTraits<NonPOD>;
    
    static_assert(PODTraits::all_members_default_constructible());
    static_assert(PODTraits::all_members_copyable());
    static_assert(PODTraits::all_members_movable());
    static_assert(PODTraits::is_pod());
    
    static_assert(NonPODTraits::all_members_default_constructible());
    static_assert(NonPODTraits::all_members_copyable());
    static_assert(NonPODTraits::all_members_movable());
    static_assert(!NonPODTraits::is_pod());
    
    std::cout << "타입 특성 검증 완료\n";
}
```

### 제약 조건 검증

컴파일 타임에 타입이 특정 제약 조건을 만족하는지 검증한다.

```cpp
#include <experimental/reflect>
#include <concepts>
#include <type_traits>
#include <string>

namespace refl = std::experimental::reflect;

// 제약 조건 정의
template<typename T>
concept Serializable = requires {
    // 모든 멤버가 기본 타입이거나 문자열이어야 함
    requires []<size_t... Is>(std::index_sequence<Is...>) {
        using members = refl::get_data_members_t<reflexpr(T)>;
        
        return ((
            std::is_arithmetic_v<
                refl::get_type_t<refl::get_element_t<Is, members>>
            > ||
            std::is_same_v<
                refl::get_type_t<refl::get_element_t<Is, members>>,
                std::string
            >
        ) && ...);
    }(std::make_index_sequence<
        refl::get_size_v<refl::get_data_members_t<reflexpr(T)>>
    >{});
};

// 제약 조건을 만족하는 구조체
struct ValidData {
    int id;
    std::string name;
    double value;
};

// 제약 조건을 만족하지 않는 구조체
struct InvalidData {
    int id;
    std::vector<int> numbers;  // 복잡한 타입
};

// 제약 조건 검증 함수
template<typename T>
    requires Serializable<T>
void process_serializable(const T& data) {
    std::cout << "직렬화 가능한 데이터 처리\n";
}

void demonstrate_constraints() {
    ValidData valid{1, "test", 3.14};
    process_serializable(valid);  // OK
    
    // InvalidData invalid{1, {1, 2, 3}};
    // process_serializable(invalid);  // 컴파일 오류!
}
```

## 12.3 안전한 직렬화/역직렬화

### 컴파일 타임 직렬화 생성

리플렉션을 사용하여 타입 안전한 직렬화 코드를 자동 생성한다.

```cpp
#include <experimental/reflect>
#include <string>
#include <sstream>
#include <iostream>
#include <type_traits>

namespace refl = std::experimental::reflect;

// 안전한 직렬화 엔진
template<typename T>
class SafeSerializer {
public:
    // JSON 직렬화
    static std::string to_json(const T& obj) {
        std::ostringstream oss;
        oss << "{";
        
        serialize_members(obj, oss);
        
        oss << "}";
        return oss.str();
    }
    
private:
    // 멤버 직렬화
    static void serialize_members(const T& obj, std::ostringstream& oss) {
        using members = refl::get_data_members_t<reflexpr(T)>;
        constexpr size_t count = refl::get_size_v<members>;
        
        [&]<size_t... Is>(std::index_sequence<Is...>) {
            bool first = true;
            
            (([&] {
                if (!first) oss << ",";
                first = false;
                
                using member = refl::get_element_t<Is, members>;
                using member_type = refl::get_type_t<member>;
                constexpr auto name = refl::get_name_v<member>;
                
                oss << "\"" << name << "\":";
                
                // 멤버 포인터를 통한 안전한 접근
                const auto& value = get_member_value<Is>(obj);
                serialize_value(value, oss);
            }()), ...);
        }(std::make_index_sequence<count>{});
    }
    
    // 멤버 값 추출
    template<size_t Index>
    static decltype(auto) get_member_value(const T& obj) {
        using members = refl::get_data_members_t<reflexpr(T)>;
        using member = refl::get_element_t<Index, members>;
        using pointer = refl::get_pointer_t<member>;
        
        return obj.*pointer::value;
    }
    
    // 값 직렬화 (타입별 처리)
    template<typename U>
    static void serialize_value(const U& value, std::ostringstream& oss) {
        if constexpr (std::is_arithmetic_v<U>) {
            oss << value;
        } else if constexpr (std::is_same_v<U, std::string>) {
            oss << "\"" << value << "\"";
        } else if constexpr (std::is_same_v<U, bool>) {
            oss << (value ? "true" : "false");
        }
    }
};

// 테스트용 구조체
struct Product {
    int id;
    std::string name;
    double price;
    bool available;
};

void demonstrate_serialization() {
    Product product{
        42,
        "Laptop",
        999.99,
        true
    };
    
    std::string json = SafeSerializer<Product>::to_json(product);
    std::cout << "JSON: " << json << "\n";
}
```

### 타입 안전한 역직렬화

컴파일 타임 검증을 통한 안전한 역직렬화를 구현한다.

```cpp
#include <experimental/reflect>
#include <string>
#include <string_view>
#include <optional>
#include <charconv>
#include <iostream>

namespace refl = std::experimental::reflect;

// 안전한 역직렬화 엔진
template<typename T>
class SafeDeserializer {
public:
    // 간단한 키-값 파싱 (실제로는 JSON 파서 사용)
    static std::optional<T> from_key_value(std::string_view input) {
        T result{};
        
        bool success = deserialize_members(result, input);
        
        return success ? std::make_optional(result) : std::nullopt;
    }
    
private:
    // 멤버 역직렬화
    static bool deserialize_members(T& obj, std::string_view input) {
        using members = refl::get_data_members_t<reflexpr(T)>;
        constexpr size_t count = refl::get_size_v<members>;
        
        return [&]<size_t... Is>(std::index_sequence<Is...>) {
            return (deserialize_member<Is>(obj, input) && ...);
        }(std::make_index_sequence<count>{});
    }
    
    // 개별 멤버 역직렬화
    template<size_t Index>
    static bool deserialize_member(T& obj, std::string_view input) {
        using members = refl::get_data_members_t<reflexpr(T)>;
        using member = refl::get_element_t<Index, members>;
        using member_type = refl::get_type_t<member>;
        constexpr auto name = refl::get_name_v<member>;
        
        // 멤버 이름 찾기
        auto key = std::string(name) + "=";
        auto pos = input.find(key);
        
        if (pos == std::string_view::npos) {
            return false;
        }
        
        // 값 추출
        auto value_start = pos + key.size();
        auto value_end = input.find(',', value_start);
        if (value_end == std::string_view::npos) {
            value_end = input.size();
        }
        
        auto value_str = input.substr(value_start, value_end - value_start);
        
        // 타입에 맞게 파싱
        return parse_value<member_type>(
            get_member_reference<Index>(obj),
            value_str
        );
    }
    
    // 멤버 참조 획득
    template<size_t Index>
    static decltype(auto) get_member_reference(T& obj) {
        using members = refl::get_data_members_t<reflexpr(T)>;
        using member = refl::get_element_t<Index, members>;
        using pointer = refl::get_pointer_t<member>;
        
        return obj.*pointer::value;
    }
    
    // 값 파싱 (타입별 처리)
    template<typename U>
    static bool parse_value(U& target, std::string_view str) {
        if constexpr (std::is_integral_v<U>) {
            auto result = std::from_chars(
                str.data(), 
                str.data() + str.size(), 
                target
            );
            return result.ec == std::errc{};
        } else if constexpr (std::is_floating_point_v<U>) {
            auto result = std::from_chars(
                str.data(), 
                str.data() + str.size(), 
                target
            );
            return result.ec == std::errc{};
        } else if constexpr (std::is_same_v<U, std::string>) {
            target = std::string(str);
            return true;
        } else if constexpr (std::is_same_v<U, bool>) {
            if (str == "true") {
                target = true;
                return true;
            } else if (str == "false") {
                target = false;
                return true;
            }
            return false;
        }
        
        return false;
    }
};

// 테스트용 구조체
struct Config {
    int port;
    std::string host;
    bool ssl_enabled;
    double timeout;
};

void demonstrate_deserialization() {
    std::string input = "port=8080,host=localhost,ssl_enabled=true,timeout=30.5";
    
    auto config = SafeDeserializer<Config>::from_key_value(input);
    
    if (config) {
        std::cout << "역직렬화 성공:\n";
        std::cout << "  포트: " << config->port << "\n";
        std::cout << "  호스트: " << config->host << "\n";
        std::cout << "  SSL: " << (config->ssl_enabled ? "활성" : "비활성") << "\n";
        std::cout << "  타임아웃: " << config->timeout << "\n";
    } else {
        std::cout << "역직렬화 실패\n";
    }
}
```

### 스키마 검증

컴파일 타임에 데이터 스키마를 검증한다.

```cpp
#include <experimental/reflect>
#include <type_traits>
#include <concepts>
#include <iostream>

namespace refl = std::experimental::reflect;

// 스키마 제약 조건
template<typename T>
concept ValidDatabaseSchema = requires {
    // 기본 키가 존재해야 함
    requires []<size_t... Is>(std::index_sequence<Is...>) {
        using members = refl::get_data_members_t<reflexpr(T)>;
        
        return ((
            refl::get_name_v<refl::get_element_t<Is, members>> == "id"
        ) || ...);
    }(std::make_index_sequence<
        refl::get_size_v<refl::get_data_members_t<reflexpr(T)>>
    >{});
    
    // 모든 멤버가 직렬화 가능해야 함
    requires []<size_t... Is>(std::index_sequence<Is...>) {
        using members = refl::get_data_members_t<reflexpr(T)>;
        
        return ((
            std::is_arithmetic_v<
                refl::get_type_t<refl::get_element_t<Is, members>>
            > ||
            std::is_same_v<
                refl::get_type_t<refl::get_element_t<Is, members>>,
                std::string
            >
        ) && ...);
    }(std::make_index_sequence<
        refl::get_size_v<refl::get_data_members_t<reflexpr(T)>>
    >{});
};

// 유효한 스키마
struct User {
    int id;
    std::string username;
    std::string email;
    int age;
};

// 무효한 스키마 (id 없음)
struct InvalidSchema1 {
    std::string name;
    int value;
};

// 무효한 스키마 (직렬화 불가능한 멤버)
struct InvalidSchema2 {
    int id;
    std::vector<int> data;
};

// 스키마 검증 함수
template<typename T>
    requires ValidDatabaseSchema<T>
class DatabaseTable {
    std::vector<T> records;
    
public:
    void insert(const T& record) {
        records.push_back(record);
        std::cout << "레코드 삽입 완료\n";
    }
    
    size_t size() const {
        return records.size();
    }
};

void demonstrate_schema_validation() {
    DatabaseTable<User> users;  // OK
    users.insert({1, "alice", "alice@example.com", 30});
    
    // DatabaseTable<InvalidSchema1> invalid1;  // 컴파일 오류!
    // DatabaseTable<InvalidSchema2> invalid2;  // 컴파일 오류!
    
    std::cout << "스키마 검증 완료\n";
}
```

### 실전 응용: ORM 시스템

리플렉션을 활용한 타입 안전한 ORM 시스템 구현이다.

```cpp
#include <experimental/reflect>
#include <string>
#include <vector>
#include <sstream>
#include <iostream>
#include <optional>

namespace refl = std::experimental::reflect;

// 간단한 ORM 시스템
template<typename T>
    requires ValidDatabaseSchema<T>
class ORM {
public:
    // CREATE TABLE 문 생성
    static std::string generate_create_table() {
        std::ostringstream oss;
        oss << "CREATE TABLE " << get_table_name() << " (";
        
        using members = refl::get_data_members_t<reflexpr(T)>;
        constexpr size_t count = refl::get_size_v<members>;
        
        [&]<size_t... Is>(std::index_sequence<Is...>) {
            bool first = true;
            
            (([&] {
                if (!first) oss << ", ";
                first = false;
                
                using member = refl::get_element_t<Is, members>;
                using member_type = refl::get_type_t<member>;
                constexpr auto name = refl::get_name_v<member>;
                
                oss << name << " " << get_sql_type<member_type>();
                
                if constexpr (name == "id") {
                    oss << " PRIMARY KEY";
                }
            }()), ...);
        }(std::make_index_sequence<count>{});
        
        oss << ")";
        return oss.str();
    }
    
    // INSERT 문 생성
    static std::string generate_insert(const T& obj) {
        std::ostringstream oss;
        oss << "INSERT INTO " << get_table_name() << " (";
        
        using members = refl::get_data_members_t<reflexpr(T)>;
        constexpr size_t count = refl::get_size_v<members>;
        
        // 컬럼 이름
        [&]<size_t... Is>(std::index_sequence<Is...>) {
            bool first = true;
            
            (([&] {
                if (!first) oss << ", ";
                first = false;
                
                using member = refl::get_element_t<Is, members>;
                oss << refl::get_name_v<member>;
            }()), ...);
        }(std::make_index_sequence<count>{});
        
        oss << ") VALUES (";
        
        // 값
        [&]<size_t... Is>(std::index_sequence<Is...>) {
            bool first = true;
            
            (([&] {
                if (!first) oss << ", ";
                first = false;
                
                using members_t = refl::get_data_members_t<reflexpr(T)>;
                using member = refl::get_element_t<Is, members_t>;
                using member_type = refl::get_type_t<member>;
                using pointer = refl::get_pointer_t<member>;
                
                const auto& value = obj.*pointer::value;
                
                if constexpr (std::is_same_v<member_type, std::string>) {
                    oss << "'" << value << "'";
                } else {
                    oss << value;
                }
            }()), ...);
        }(std::make_index_sequence<count>{});
        
        oss << ")";
        return oss.str();
    }
    
    // SELECT 문 생성
    static std::string generate_select() {
        std::ostringstream oss;
        oss << "SELECT ";
        
        using members = refl::get_data_members_t<reflexpr(T)>;
        constexpr size_t count = refl::get_size_v<members>;
        
        [&]<size_t... Is>(std::index_sequence<Is...>) {
            bool first = true;
            
            (([&] {
                if (!first) oss << ", ";
                first = false;
                
                using member = refl::get_element_t<Is, members>;
                oss << refl::get_name_v<member>;
            }()), ...);
        }(std::make_index_sequence<count>{});
        
        oss << " FROM " << get_table_name();
        return oss.str();
    }
    
private:
    // 테이블 이름 추출
    static std::string get_table_name() {
        return std::string(refl::get_name_v<reflexpr(T)>);
    }
    
    // SQL 타입 매핑
    template<typename U>
    static std::string get_sql_type() {
        if constexpr (std::is_same_v<U, int>) {
            return "INTEGER";
        } else if constexpr (std::is_same_v<U, std::string>) {
            return "TEXT";
        } else if constexpr (std::is_same_v<U, double>) {
            return "REAL";
        } else if constexpr (std::is_same_v<U, bool>) {
            return "BOOLEAN";
        }
        return "UNKNOWN";
    }
};

// 엔티티 정의
struct Article {
    int id;
    std::string title;
    std::string content;
    int author_id;
};

void demonstrate_orm() {
    using ArticleORM = ORM<Article>;
    
    // CREATE TABLE 문 생성
    std::cout << "테이블 생성 쿼리:\n";
    std::cout << ArticleORM::generate_create_table() << "\n\n";
    
    // INSERT 문 생성
    Article article{
        1,
        "C++26 리플렉션",
        "리플렉션은 강력한 도구다",
        100
    };
    
    std::cout << "삽입 쿼리:\n";
    std::cout << ArticleORM::generate_insert(article) << "\n\n";
    
    // SELECT 문 생성
    std::cout << "조회 쿼리:\n";
    std::cout << ArticleORM::generate_select() << "\n";
}
```
