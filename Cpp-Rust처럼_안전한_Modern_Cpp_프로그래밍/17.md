# 17장: 단위 테스트와 속성 기반 테스트

안전한 C++ 코드를 작성했다면 그 안전성을 검증해야 한다. 단순히 "컴파일이 되니까 안전하다"는 가정은 위험하다. 체계적인 테스트를 통해 코드의 안전성을 입증하고, 예상치 못한 경계 조건에서도 올바르게 동작함을 보장해야 한다. 이 장에서는 타입 안전한 테스트 프레임워크 구축부터 Fuzzing, Contract 기반 테스트까지 실전에서 사용할 수 있는 테스트 기법들을 살펴본다.

## 17.1 타입 안전한 테스트 프레임워크

### 17.1.1 기존 테스트 프레임워크의 한계

**안전하지 않은 방법: 타입 안전성이 없는 테스트**

```cpp
#include <iostream>
#include <cstring>

// 전통적인 assert 기반 테스트
void test_old_style() {
    // 문제 1: 타입 정보 손실
    int expected = 42;
    double actual = 42.5;
    
    // 암묵적 변환으로 인한 잘못된 비교
    if (expected == actual) {  // 42 == 42로 변환됨
        std::cout << "PASS (but wrong!)" << std::endl;
    }
    
    // 문제 2: 에러 메시지 부족
    const char* str1 = "hello";
    const char* str2 = "hello";
    
    // 포인터 비교 (의도하지 않음)
    if (str1 == str2) {  // 구현 의존적
        std::cout << "PASS (maybe)" << std::endl;
    }
    
    // 문제 3: 예외 테스트 어려움
    try {
        throw std::runtime_error("error");
        std::cout << "FAIL: no exception" << std::endl;
    } catch (...) {
        // 어떤 예외인지 확인 불가
        std::cout << "PASS (but what exception?)" << std::endl;
    }
}

int main() {
    test_old_style();
    return 0;
}
```

**안전한 방법: 타입 안전한 테스트 프레임워크**

```cpp
#include <iostream>
#include <stdexcept>
#include <string>
#include <string_view>
#include <source_location>
#include <concepts>
#include <type_traits>

// 타입 안전한 테스트 프레임워크
namespace test {

// 테스트 결과
struct TestResult {
    bool passed;
    std::string message;
    std::source_location location;
};

// 테스트 컨텍스트
class TestContext {
private:
    std::string test_name_;
    int passed_count_ = 0;
    int failed_count_ = 0;
    
public:
    explicit TestContext(std::string_view name) : test_name_(name) {
        std::cout << "\n[TEST] " << test_name_ << std::endl;
    }
    
    ~TestContext() {
        std::cout << "[RESULT] " << test_name_ << ": "
                  << passed_count_ << " passed, "
                  << failed_count_ << " failed" << std::endl;
    }
    
    void record(const TestResult& result) {
        if (result.passed) {
            ++passed_count_;
            std::cout << "  ✓ PASS";
        } else {
            ++failed_count_;
            std::cout << "  ✗ FAIL: " << result.message;
            std::cout << "\n    at " << result.location.file_name()
                     << ":" << result.location.line();
        }
        std::cout << std::endl;
    }
    
    bool all_passed() const { return failed_count_ == 0; }
};

// 동등성 비교 - 타입이 정확히 일치해야 함
template<typename T>
    requires std::equality_comparable<T>
TestResult equal(const T& expected, const T& actual,
                std::source_location loc = std::source_location::current()) {
    if (expected == actual) {
        return {true, "", loc};
    }
    
    return {
        false,
        std::string("Expected: ") + std::to_string(expected) +
        ", Actual: " + std::to_string(actual),
        loc
    };
}

// 문자열 특수화
TestResult equal(std::string_view expected, std::string_view actual,
                std::source_location loc = std::source_location::current()) {
    if (expected == actual) {
        return {true, "", loc};
    }
    
    return {
        false,
        std::string("Expected: \"") + std::string(expected) +
        "\", Actual: \"" + std::string(actual) + "\"",
        loc
    };
}

// 부동소수점 비교 - 오차 허용
template<std::floating_point T>
TestResult approximately_equal(T expected, T actual, T epsilon = T(1e-6),
                              std::source_location loc = std::source_location::current()) {
    if (std::abs(expected - actual) <= epsilon) {
        return {true, "", loc};
    }
    
    return {
        false,
        std::string("Expected: ") + std::to_string(expected) +
        ", Actual: " + std::to_string(actual) +
        ", Diff: " + std::to_string(std::abs(expected - actual)),
        loc
    };
}

// 예외 테스트 - 특정 예외 타입 검증
template<typename ExceptionType, typename Callable>
    requires std::derived_from<ExceptionType, std::exception>
TestResult throws(Callable&& fn,
                 std::source_location loc = std::source_location::current()) {
    try {
        fn();
        return {false, "Expected exception but none was thrown", loc};
    } catch (const ExceptionType& e) {
        return {
            true,
            std::string("Correctly threw: ") + e.what(),
            loc
        };
    } catch (const std::exception& e) {
        return {
            false,
            std::string("Wrong exception type: ") + e.what(),
            loc
        };
    } catch (...) {
        return {false, "Unknown exception type", loc};
    }
}

// 조건 검사
TestResult is_true(bool condition, std::string_view message = "",
                  std::source_location loc = std::source_location::current()) {
    if (condition) {
        return {true, "", loc};
    }
    return {false, std::string(message), loc};
}

} // namespace test

// 실제 사용 예
void demonstrate_type_safe_tests() {
    test::TestContext ctx("Type Safe Tests");
    
    // 정수 비교 - 타입이 정확히 일치
    ctx.record(test::equal(42, 42));
    
    // 컴파일 에러를 방지 - int와 double 직접 비교 불가
    // ctx.record(test::equal(42, 42.0));  // 컴파일 에러
    
    // 명시적 변환 필요
    ctx.record(test::equal(42.0, static_cast<double>(42)));
    
    // 부동소수점 비교
    ctx.record(test::approximately_equal(3.14159, 3.14160, 0.0001));
    
    // 문자열 비교 - 포인터가 아닌 내용 비교
    ctx.record(test::equal("hello", std::string("hello")));
    
    // 예외 테스트 - 정확한 예외 타입 검증
    ctx.record(test::throws<std::runtime_error>([]() {
        throw std::runtime_error("test error");
    }));
    
    // 조건 검사
    ctx.record(test::is_true(42 > 0, "42 should be positive"));
}

int main() {
    demonstrate_type_safe_tests();
    return 0;
}
```

**안전한 방법: RAII 기반 테스트 픽스처**

```cpp
#include <iostream>
#include <fstream>
#include <memory>
#include <filesystem>
#include <string>
#include <source_location>

namespace test {

// 테스트 픽스처 베이스 클래스
class TestFixture {
protected:
    virtual void setup() {}
    virtual void teardown() {}
    
public:
    TestFixture() {
        try {
            setup();
        } catch (...) {
            teardown();
            throw;
        }
    }
    
    virtual ~TestFixture() {
        teardown();
    }
    
    // 복사/이동 금지
    TestFixture(const TestFixture&) = delete;
    TestFixture& operator=(const TestFixture&) = delete;
    TestFixture(TestFixture&&) = delete;
    TestFixture& operator=(TestFixture&&) = delete;
};

// 임시 파일 픽스처
class TempFileFixture : public TestFixture {
protected:
    std::filesystem::path temp_path_;
    
    void setup() override {
        temp_path_ = std::filesystem::temp_directory_path() / 
                     std::tmpnam(nullptr);
    }
    
    void teardown() override {
        if (std::filesystem::exists(temp_path_)) {
            std::filesystem::remove(temp_path_);
        }
    }
    
public:
    const std::filesystem::path& path() const { return temp_path_; }
    
    void write_file(const std::string& content) {
        std::ofstream file(temp_path_);
        file << content;
    }
    
    std::string read_file() {
        std::ifstream file(temp_path_);
        return std::string(
            std::istreambuf_iterator<char>(file),
            std::istreambuf_iterator<char>()
        );
    }
};

// 임시 디렉토리 픽스처
class TempDirFixture : public TestFixture {
protected:
    std::filesystem::path temp_dir_;
    
    void setup() override {
        temp_dir_ = std::filesystem::temp_directory_path() / 
                    std::tmpnam(nullptr);
        std::filesystem::create_directory(temp_dir_);
    }
    
    void teardown() override {
        if (std::filesystem::exists(temp_dir_)) {
            std::filesystem::remove_all(temp_dir_);
        }
    }
    
public:
    const std::filesystem::path& dir() const { return temp_dir_; }
    
    std::filesystem::path create_file(const std::string& name) {
        auto path = temp_dir_ / name;
        std::ofstream file(path);
        return path;
    }
};

} // namespace test

// 실제 사용 예
void test_with_temp_file() {
    test::TestContext ctx("File Operations");
    test::TempFileFixture fixture;
    
    // 파일 쓰기 테스트
    const std::string test_data = "Hello, World!";
    fixture.write_file(test_data);
    
    ctx.record(test::equal(
        test_data,
        fixture.read_file()
    ));
    
    // 픽스처 소멸자에서 자동으로 정리됨
}

void test_with_temp_dir() {
    test::TestContext ctx("Directory Operations");
    test::TempDirFixture fixture;
    
    // 여러 파일 생성
    auto file1 = fixture.create_file("test1.txt");
    auto file2 = fixture.create_file("test2.txt");
    
    ctx.record(test::is_true(
        std::filesystem::exists(file1),
        "File 1 should exist"
    ));
    
    ctx.record(test::is_true(
        std::filesystem::exists(file2),
        "File 2 should exist"
    ));
    
    // 픽스처 소멸자에서 디렉토리와 모든 파일 정리
}

int main() {
    test_with_temp_file();
    test_with_temp_dir();
    return 0;
}
```

### 17.1.3 매개변수화된 테스트

**활용 예: 타입 안전한 매개변수화 테스트**

```cpp
#include <iostream>
#include <vector>
#include <tuple>
#include <string>
#include <concepts>

namespace test {

// 매개변수화된 테스트를 위한 데이터 세트
template<typename... Args>
class TestDataSet {
private:
    std::vector<std::tuple<Args...>> data_;
    
public:
    void add(Args... args) {
        data_.emplace_back(args...);
    }
    
    template<typename Fn>
    void run(TestContext& ctx, Fn&& test_fn) {
        int case_num = 0;
        for (const auto& test_case : data_) {
            ++case_num;
            std::cout << "  Case " << case_num << ": ";
            std::apply(test_fn, test_case);
        }
    }
};

} // namespace test

// 실제 사용 예: 수학 함수 테스트
int safe_divide(int a, int b) {
    if (b == 0) {
        throw std::domain_error("Division by zero");
    }
    return a / b;
}

void test_division_parameterized() {
    test::TestContext ctx("Parameterized Division Tests");
    
    // 테스트 데이터 세트 정의
    test::TestDataSet<int, int, int> normal_cases;
    normal_cases.add(10, 2, 5);
    normal_cases.add(15, 3, 5);
    normal_cases.add(20, 4, 5);
    normal_cases.add(-10, 2, -5);
    normal_cases.add(0, 5, 0);
    
    // 정상 케이스 실행
    normal_cases.run(ctx, [&ctx](int a, int b, int expected) {
        ctx.record(test::equal(expected, safe_divide(a, b)));
    });
    
    // 예외 케이스
    test::TestDataSet<int, int> error_cases;
    error_cases.add(10, 0);
    error_cases.add(0, 0);
    error_cases.add(-5, 0);
    
    error_cases.run(ctx, [&ctx](int a, int b) {
        ctx.record(test::throws<std::domain_error>([a, b]() {
            safe_divide(a, b);
        }));
    });
}

// 문자열 처리 함수 테스트
std::string to_upper(std::string_view str) {
    std::string result;
    result.reserve(str.size());
    
    for (char c : str) {
        if (c >= 'a' && c <= 'z') {
            result += static_cast<char>(c - 'a' + 'A');
        } else {
            result += c;
        }
    }
    
    return result;
}

void test_string_conversion_parameterized() {
    test::TestContext ctx("Parameterized String Tests");
    
    test::TestDataSet<std::string_view, std::string_view> cases;
    cases.add("hello", "HELLO");
    cases.add("World", "WORLD");
    cases.add("123", "123");
    cases.add("Hello123", "HELLO123");
    cases.add("", "");
    
    cases.run(ctx, [&ctx](std::string_view input, std::string_view expected) {
        ctx.record(test::equal(expected, to_upper(input)));
    });
}

int main() {
    test_division_parameterized();
    test_string_conversion_parameterized();
    return 0;
}
```

## 17.2 Fuzzing을 통한 경계 조건 테스트

Fuzzing은 무작위 또는 변형된 입력을 생성하여 프로그램의 예상치 못한 동작을 찾아내는 기법이다.

### 17.2.1 기본 Fuzzing 구현

**안전하지 않은 방법: 경계 조건을 놓친 코드**

```cpp
#include <string>
#include <cstdint>

// 취약한 파싱 함수
struct Config {
    uint16_t port;
    std::string host;
};

Config parse_config(const std::string& input) {
    Config cfg;
    
    // 단순한 파싱 (취약함)
    auto colon_pos = input.find(':');
    cfg.host = input.substr(0, colon_pos);
    
    std::string port_str = input.substr(colon_pos + 1);
    cfg.port = static_cast<uint16_t>(std::stoi(port_str));  // 오버플로우 가능
    
    return cfg;
}

// 수동 테스트만으로는 모든 경계 조건을 찾기 어려움
void manual_test() {
    parse_config("localhost:8080");  // 정상
    parse_config("0.0.0.0:80");      // 정상
    // 하지만 다음과 같은 경우는?
    // - parse_config(":8080")
    // - parse_config("localhost:")
    // - parse_config("localhost:99999")
    // - parse_config("")
    // - parse_config(":")
}
```

**안전한 방법: Fuzzing으로 경계 조건 발견**

```cpp
#include <string>
#include <cstdint>
#include <optional>
#include <random>
#include <iostream>
#include <stdexcept>

// 안전한 파싱 함수
struct Config {
    uint16_t port;
    std::string host;
};

std::optional<Config> safe_parse_config(const std::string& input) {
    auto colon_pos = input.find(':');
    
    // 검증 1: 콜론이 있는가?
    if (colon_pos == std::string::npos) {
        return std::nullopt;
    }
    
    // 검증 2: 호스트가 비어있지 않은가?
    if (colon_pos == 0) {
        return std::nullopt;
    }
    
    // 검증 3: 포트 문자열이 있는가?
    if (colon_pos == input.size() - 1) {
        return std::nullopt;
    }
    
    Config cfg;
    cfg.host = input.substr(0, colon_pos);
    
    std::string port_str = input.substr(colon_pos + 1);
    
    // 검증 4: 포트가 숫자인가?
    try {
        size_t pos;
        int port_int = std::stoi(port_str, &pos);
        
        // 검증 5: 모든 문자를 소비했는가?
        if (pos != port_str.size()) {
            return std::nullopt;
        }
        
        // 검증 6: 유효한 포트 범위인가?
        if (port_int < 0 || port_int > 65535) {
            return std::nullopt;
        }
        
        cfg.port = static_cast<uint16_t>(port_int);
        
    } catch (const std::exception&) {
        return std::nullopt;
    }
    
    return cfg;
}

// Fuzzer 구현
class ConfigFuzzer {
private:
    std::mt19937 rng_;
    std::uniform_int_distribution<char> char_dist_;
    std::uniform_int_distribution<size_t> length_dist_;
    
public:
    ConfigFuzzer() : rng_(std::random_device{}()),
                     char_dist_(0, 127),
                     length_dist_(0, 100) {}
    
    std::string generate_random_input() {
        size_t length = length_dist_(rng_);
        std::string result;
        result.reserve(length);
        
        for (size_t i = 0; i < length; ++i) {
            result += char_dist_(rng_);
        }
        
        return result;
    }
    
    std::string generate_interesting_input() {
        std::uniform_int_distribution<int> strategy_dist(0, 9);
        
        switch (strategy_dist(rng_)) {
            case 0: return "";
            case 1: return ":";
            case 2: return "::";
            case 3: return "localhost:";
            case 4: return ":8080";
            case 5: return "host:99999";
            case 6: return "host:port";
            case 7: return std::string(1000, 'a') + ":8080";
            case 8: return "host:" + std::string(100, '9');
            case 9: return "host:0";
            default: return generate_random_input();
        }
    }
    
    void fuzz(size_t iterations) {
        test::TestContext ctx("Fuzzing Config Parser");
        
        size_t crash_count = 0;
        size_t error_count = 0;
        
        for (size_t i = 0; i < iterations; ++i) {
            std::string input;
            
            if (i % 2 == 0) {
                input = generate_interesting_input();
            } else {
                input = generate_random_input();
            }
            
            try {
                auto result = safe_parse_config(input);
                
                // 결과가 있으면 유효성 검증
                if (result) {
                    if (result->host.empty()) {
                        std::cout << "Invalid: empty host for input: " 
                                  << input << std::endl;
                        ++error_count;
                    }
                }
                
            } catch (const std::exception& e) {
                std::cout << "Crash on input: " << input 
                          << " - " << e.what() << std::endl;
                ++crash_count;
            } catch (...) {
                std::cout << "Unknown crash on input: " << input << std::endl;
                ++crash_count;
            }
        }
        
        std::cout << "\nFuzzing results:" << std::endl;
        std::cout << "  Total iterations: " << iterations << std::endl;
        std::cout << "  Crashes: " << crash_count << std::endl;
        std::cout << "  Logic errors: " << error_count << std::endl;
        
        ctx.record(test::is_true(
            crash_count == 0,
            "No crashes should occur"
        ));
    }
};

int main() {
    ConfigFuzzer fuzzer;
    fuzzer.fuzz(10000);
    
    return 0;
}
```

### 17.2.2 LibFuzzer 통합

**활용 예: LibFuzzer를 사용한 Fuzzing**

```cpp
// fuzz_target.cpp
// 컴파일: clang++ -std=c++26 -g -fsanitize=fuzzer,address fuzz_target.cpp
#include <cstdint>
#include <cstddef>
#include <string>
#include <optional>

// 테스트할 함수 (이전 예제의 safe_parse_config)
struct Config {
    uint16_t port;
    std::string host;
};

std::optional<Config> safe_parse_config(const std::string& input) {
    auto colon_pos = input.find(':');
    
    if (colon_pos == std::string::npos) {
        return std::nullopt;
    }
    
    if (colon_pos == 0 || colon_pos == input.size() - 1) {
        return std::nullopt;
    }
    
    Config cfg;
    cfg.host = input.substr(0, colon_pos);
    
    std::string port_str = input.substr(colon_pos + 1);
    
    try {
        size_t pos;
        int port_int = std::stoi(port_str, &pos);
        
        if (pos != port_str.size()) {
            return std::nullopt;
        }
        
        if (port_int < 0 || port_int > 65535) {
            return std::nullopt;
        }
        
        cfg.port = static_cast<uint16_t>(port_int);
        
    } catch (...) {
        return std::nullopt;
    }
    
    return cfg;
}

// LibFuzzer 진입점
extern "C" int LLVMFuzzerTestOneInput(const uint8_t* data, size_t size) {
    // 입력 데이터를 문자열로 변환
    std::string input(reinterpret_cast<const char*>(data), size);
    
    // 함수 테스트 - 크래시가 발생하면 LibFuzzer가 감지
    auto result = safe_parse_config(input);
    
    // 추가 검증: 결과가 있으면 유효성 확인
    if (result) {
        // 호스트가 비어있지 않아야 함
        if (result->host.empty()) {
            __builtin_trap();  // 논리 오류 감지
        }
        
        // 포트가 원본 문자열에 있어야 함
        std::string port_str = std::to_string(result->port);
        if (input.find(port_str) == std::string::npos) {
            __builtin_trap();
        }
    }
    
    return 0;
}
```

### 17.2.3 구조화된 Fuzzing

**활용 예: 복잡한 데이터 구조 Fuzzing**

```cpp
#include <cstdint>
#include <cstddef>
#include <vector>
#include <span>
#include <optional>

// 바이너리 프로토콜 파서
struct Message {
    uint8_t type;
    uint16_t length;
    std::vector<uint8_t> payload;
};

std::optional<Message> parse_message(std::span<const uint8_t> data) {
    // 최소 헤더 크기 검증
    if (data.size() < 3) {
        return std::nullopt;
    }
    
    Message msg;
    msg.type = data[0];
    
    // 길이 읽기 (big-endian)
    msg.length = static_cast<uint16_t>((data[1] << 8) | data[2]);
    
    // 페이로드 크기 검증
    if (data.size() < 3 + msg.length) {
        return std::nullopt;
    }
    
    // 페이로드 복사
    msg.payload.assign(
        data.begin() + 3,
        data.begin() + 3 + msg.length
    );
    
    return msg;
}

// 구조화된 Fuzzer 입력 생성
class StructuredFuzzer {
public:
    static std::vector<uint8_t> generate_valid_message(
        const uint8_t* data, size_t size
    ) {
        if (size < 3) {
            return {};
        }
        
        std::vector<uint8_t> message;
        
        // 타입
        message.push_back(data[0]);
        
        // 길이 (나머지 데이터 크기로 제한)
        uint16_t length = static_cast<uint16_t>(
            std::min(size - 3, size_t(256))
        );
        message.push_back(static_cast<uint8_t>(length >> 8));
        message.push_back(static_cast<uint8_t>(length & 0xFF));
        
        // 페이로드
        for (size_t i = 0; i < length && i + 3 < size; ++i) {
            message.push_back(data[3 + i]);
        }
        
        return message;
    }
};

extern "C" int LLVMFuzzerTestOneInput(const uint8_t* data, size_t size) {
    // 원본 데이터로 테스트
    {
        std::span<const uint8_t> input(data, size);
        auto result = parse_message(input);
        
        // 결과 검증
        if (result) {
            // 페이로드 크기가 헤더의 길이와 일치해야 함
            if (result->payload.size() != result->length) {
                __builtin_trap();
            }
        }
    }
    
    // 구조화된 입력으로 테스트
    {
        auto structured_input = StructuredFuzzer::generate_valid_message(data, size);
        if (!structured_input.empty()) {
            auto result = parse_message(structured_input);
            
            // 구조화된 입력은 항상 파싱되어야 함
            if (!result) {
                __builtin_trap();
            }
        }
    }
    
    return 0;
}
```

### 17.3.1 Contract 기본 사용법

**안전하지 않은 방법: 암묵적인 가정**

```cpp
#include <vector>
#include <algorithm>

// 암묵적인 가정들이 있는 함수
int binary_search(const std::vector<int>& sorted_vec, int target) {
    // 가정 1: 벡터가 정렬되어 있음 (검증 안 함)
    // 가정 2: 벡터가 비어있지 않음 (검증 안 함)
    
    int left = 0;
    int right = static_cast<int>(sorted_vec.size()) - 1;
    
    while (left <= right) {
        int mid = left + (right - left) / 2;
        
        if (sorted_vec[mid] == target) {
            return mid;
        } else if (sorted_vec[mid] < target) {
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }
    
    return -1;
}
```

**안전한 방법: Contract를 사용한 명시적 검증**

```cpp
#include <vector>
#include <algorithm>
#include <cassert>
#include <ranges>

// Contract 검증 헬퍼
namespace contracts {

template<typename T>
bool is_sorted(const std::vector<T>& vec) {
    return std::ranges::is_sorted(vec);
}

template<typename T>
bool is_valid_index(const std::vector<T>& vec, int index) {
    return index >= -1 && index < static_cast<int>(vec.size());
}

} // namespace contracts

// Contract가 명시된 함수
int safe_binary_search(const std::vector<int>& sorted_vec, int target) {
    // 사전조건: 벡터가 정렬되어 있어야 함
    assert(contracts::is_sorted(sorted_vec) && 
           "Precondition: vector must be sorted");
    
    int left = 0;
    int right = static_cast<int>(sorted_vec.size()) - 1;
    int result = -1;
    
    while (left <= right) {
        int mid = left + (right - left) / 2;
        
        // 불변조건: mid는 유효한 인덱스
        assert(mid >= 0 && mid < static_cast<int>(sorted_vec.size()) &&
               "Invariant: mid must be valid index");
        
        if (sorted_vec[mid] == target) {
            result = mid;
            break;
        } else if (sorted_vec[mid] < target) {
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }
    
    // 사후조건: 결과가 유효한 범위
    assert(contracts::is_valid_index(sorted_vec, result) &&
           "Postcondition: result must be valid index or -1");
    
    // 사후조건: 결과가 -1이 아니면 실제로 찾은 값
    assert((result == -1 || sorted_vec[result] == target) &&
           "Postcondition: if found, must equal target");
    
    return result;
}

// 테스트
void test_binary_search_with_contracts() {
    test::TestContext ctx("Binary Search Contracts");
    
    // 정상 케이스
    {
        std::vector<int> sorted_vec = {1, 3, 5, 7, 9};
        ctx.record(test::equal(2, safe_binary_search(sorted_vec, 5)));
        ctx.record(test::equal(-1, safe_binary_search(sorted_vec, 4)));
    }
    
    // 빈 벡터
    {
        std::vector<int> empty_vec;
        ctx.record(test::equal(-1, safe_binary_search(empty_vec, 5)));
    }
    
    // Contract 위반 테스트
    {
        std::vector<int> unsorted_vec = {5, 3, 1, 9, 7};
        
        // 디버그 빌드에서는 assertion 실패
        // 릴리스 빌드에서는 정의되지 않은 동작
        // safe_binary_search(unsorted_vec, 5);  // Assert 실패
    }
}
```

### 17.3.2 클래스 불변조건

**활용 예: 클래스 불변조건 검증**

```cpp
#include <vector>
#include <cassert>
#include <algorithm>

// 우선순위 큐 구현
template<typename T>
class PriorityQueue {
private:
    std::vector<T> heap_;
    
    // 불변조건 검증
    bool check_invariants() const {
        // 힙 속성 검증
        for (size_t i = 0; i < heap_.size(); ++i) {
            size_t left = 2 * i + 1;
            size_t right = 2 * i + 2;
            
            if (left < heap_.size() && heap_[i] < heap_[left]) {
                return false;
            }
            
            if (right < heap_.size() && heap_[i] < heap_[right]) {
                return false;
            }
        }
        
        return true;
    }
    
    void heapify_up(size_t index) {
        while (index > 0) {
            size_t parent = (index - 1) / 2;
            
            if (heap_[parent] >= heap_[index]) {
                break;
            }
            
            std::swap(heap_[parent], heap_[index]);
            index = parent;
        }
        
        // 불변조건 검증
        assert(check_invariants() && "Invariant violated after heapify_up");
    }
    
    void heapify_down(size_t index) {
        while (true) {
            size_t largest = index;
            size_t left = 2 * index + 1;
            size_t right = 2 * index + 2;
            
            if (left < heap_.size() && heap_[left] > heap_[largest]) {
                largest = left;
            }
            
            if (right < heap_.size() && heap_[right] > heap_[largest]) {
                largest = right;
            }
            
            if (largest == index) {
                break;
            }
            
            std::swap(heap_[index], heap_[largest]);
            index = largest;
        }
        
        // 불변조건 검증
        assert(check_invariants() && "Invariant violated after heapify_down");
    }
    
public:
    void push(const T& value) {
        heap_.push_back(value);
        heapify_up(heap_.size() - 1);
        
        // 사후조건: 크기 증가
        assert(heap_.size() > 0 && "Postcondition: size must increase");
    }
    
    T pop() {
        // 사전조건: 비어있지 않음
        assert(!heap_.empty() && "Precondition: queue must not be empty");
        
        T result = heap_[0];
        heap_[0] = heap_.back();
        heap_.pop_back();
        
        if (!heap_.empty()) {
            heapify_down(0);
        }
        
        // 사후조건: 불변조건 유지
        assert(check_invariants() && "Postcondition: invariants must hold");
        
        return result;
    }
    
    const T& top() const {
        // 사전조건: 비어있지 않음
        assert(!heap_.empty() && "Precondition: queue must not be empty");
        
        return heap_[0];
    }
    
    bool empty() const {
        return heap_.empty();
    }
    
    size_t size() const {
        return heap_.size();
    }
    
    // 테스트용: 불변조건 외부 검증
    bool verify_invariants() const {
        return check_invariants();
    }
};

// 테스트
void test_priority_queue_invariants() {
    test::TestContext ctx("Priority Queue Invariants");
    
    PriorityQueue<int> pq;
    
    // 삽입 테스트
    std::vector<int> values = {3, 1, 4, 1, 5, 9, 2, 6, 5};
    for (int val : values) {
        pq.push(val);
        ctx.record(test::is_true(
            pq.verify_invariants(),
            "Invariants should hold after push"
        ));
    }
    
    // 삭제 테스트
    std::vector<int> expected_order = {9, 6, 5, 5, 4, 3, 2, 1, 1};
    for (int expected : expected_order) {
        ctx.record(test::equal(expected, pq.top()));
        pq.pop();
        
        if (!pq.empty()) {
            ctx.record(test::is_true(
                pq.verify_invariants(),
                "Invariants should hold after pop"
            ));
        }
    }
    
    ctx.record(test::is_true(pq.empty(), "Queue should be empty"));
}
```

### 17.3.3 속성 기반 테스트

속성 기반 테스트는 코드가 만족해야 할 속성을 정의하고, 다양한 입력에 대해 이 속성이 유지되는지 검증한다.

**활용 예: 정렬 알고리즘의 속성 검증**

```cpp
#include <vector>
#include <algorithm>
#include <random>
#include <ranges>

// 정렬 알고리즘의 속성들
namespace sort_properties {

// 속성 1: 출력이 정렬되어 있어야 함
template<typename T>
bool is_sorted_property(const std::vector<T>& output) {
    return std::ranges::is_sorted(output);
}

// 속성 2: 출력의 크기가 입력과 같아야 함
template<typename T>
bool size_preserved_property(
    const std::vector<T>& input,
    const std::vector<T>& output
) {
    return input.size() == output.size();
}

// 속성 3: 출력이 입력의 순열이어야 함 (같은 원소들)
template<typename T>
bool permutation_property(
    const std::vector<T>& input,
    const std::vector<T>& output
) {
    auto sorted_input = input;
    auto sorted_output = output;
    
    std::ranges::sort(sorted_input);
    std::ranges::sort(sorted_output);
    
    return sorted_input == sorted_output;
}

// 속성 4: 안정 정렬의 경우, 같은 값의 상대적 순서 유지
template<typename T>
struct IndexedValue {
    T value;
    size_t original_index;
    
    bool operator<(const IndexedValue& other) const {
        return value < other.value;
    }
};

template<typename T>
bool stability_property(
    const std::vector<T>& input,
    const std::vector<T>& output
) {
    // 원본 인덱스와 함께 저장
    std::vector<IndexedValue<T>> indexed_input;
    for (size_t i = 0; i < input.size(); ++i) {
        indexed_input.push_back({input[i], i});
    }
    
    std::vector<IndexedValue<T>> indexed_output;
    for (size_t i = 0; i < output.size(); ++i) {
        indexed_output.push_back({output[i], 
                                  static_cast<size_t>(-1)});
    }
    
    // 출력에서 원본 인덱스 찾기
    size_t input_idx = 0;
    for (auto& out : indexed_output) {
        while (input_idx < indexed_input.size()) {
            if (indexed_input[input_idx].value == out.value) {
                out.original_index = indexed_input[input_idx].original_index;
                ++input_idx;
                break;
            }
            ++input_idx;
        }
    }
    
    // 같은 값끼리 원본 순서 확인
    for (size_t i = 1; i < indexed_output.size(); ++i) {
        if (indexed_output[i].value == indexed_output[i-1].value) {
            if (indexed_output[i].original_index < 
                indexed_output[i-1].original_index) {
                return false;  // 순서가 바뀜
            }
        }
    }
    
    return true;
}

} // namespace sort_properties

// 속성 기반 테스트 러너
class PropertyBasedTester {
private:
    std::mt19937 rng_;
    
    template<typename T>
    std::vector<T> generate_random_vector(size_t size, T min_val, T max_val) {
        std::uniform_int_distribution<T> dist(min_val, max_val);
        
        std::vector<T> result;
        result.reserve(size);
        
        for (size_t i = 0; i < size; ++i) {
            result.push_back(dist(rng_));
        }
        
        return result;
    }
    
public:
    PropertyBasedTester() : rng_(std::random_device{}()) {}
    
    template<typename SortFn>
    void test_sort_properties(
        SortFn sort_fn,
        const std::string& name,
        size_t num_tests = 100
    ) {
        test::TestContext ctx("Property-Based: " + name);
        
        for (size_t test_num = 0; test_num < num_tests; ++test_num) {
            // 다양한 크기의 입력 생성
            size_t size = test_num < 10 ? test_num : (test_num % 100) + 1;
            auto input = generate_random_vector<int>(size, -1000, 1000);
            
            // 정렬 수행
            auto output = input;
            sort_fn(output);
            
            // 속성 검증
            ctx.record(test::is_true(
                sort_properties::is_sorted_property(output),
                "Output must be sorted"
            ));
            
            ctx.record(test::is_true(
                sort_properties::size_preserved_property(input, output),
                "Size must be preserved"
            ));
            
            ctx.record(test::is_true(
                sort_properties::permutation_property(input, output),
                "Output must be permutation of input"
            ));
        }
    }
};

// 테스트할 정렬 함수들
void std_sort(std::vector<int>& vec) {
    std::ranges::sort(vec);
}

void std_stable_sort(std::vector<int>& vec) {
    std::ranges::stable_sort(vec);
}

// 실행
int main() {
    PropertyBasedTester tester;
    
    tester.test_sort_properties(std_sort, "std::ranges::sort", 100);
    tester.test_sort_properties(std_stable_sort, "std::ranges::stable_sort", 100);
    
    return 0;
}
```

### 17.3.4 통합 테스트 예시

```cpp
#include <vector>

#include <string>
#include <algorithm>

// 테스트할 클래스: 동적 배열
template<typename T>
class DynamicArray {
private:

        // 불변조건: 내부 벡터는 항상 유효

        return value;

    // 2. 속성 기반 테스트
    {

            std::vector<int> reference;  // 참조 구현
            
            size_t operations = size_dist(rng);
            
            for (size_t op = 0; op < operations; ++op) {
                int value = value_dist(rng);
                
                arr.push_back(value);
                reference.push_back(value);
                
                // 속성 1: 크기 일치
                ctx.record(test::equal(
                    reference.size(),
                    arr.size()
                ));
                
                // 속성 2: 내용 일치
                bool contents_match = true;
                for (size_t i = 0; i < reference.size(); ++i) {
                    if (arr.at(i).value() != reference[i]) {
                        contents_match = false;
                        break;
                    }
                }
                ctx.record(test::is_true(
                    contents_match,
                    "Contents must match reference"
                ));
            }
        }
    }
    
    // 3. Fuzzing 테스트
    {
        test::TestContext ctx("Fuzzing Tests: DynamicArray");
        
        std::mt19937 rng(std::random_device{}());
        std::uniform_int_distribution<int> op_dist(0, 3);
        std::uniform_int_distribution<int> value_dist(-1000, 1000);
        std::uniform_int_distribution<size_t> index_dist(0, 1000);
        
        size_t crash_count = 0;
        
        for (int test_num = 0; test_num < 1000; ++test_num) {
            DynamicArray<int> arr;

                for (int op_num = 0; op_num < 100; ++op_num) {
                    int operation = op_dist(rng);
                    
                    switch (operation) {
                        case 0:  // push_back
                            arr.push_back(value_dist(rng));
                            break;
                            
                        case 1:  // pop_back
                            arr.pop_back();
                            break;
                            
                        case 2:  // at
                            arr.at(index_dist(rng));
                            break;
                            
                        case 3:  // clear
                            arr.clear();
                            break;
                    }
                }
            } catch (const std::exception& e) {
                ++crash_count;
                std::cout << "Crash in test " << test_num 
                         << ": " << e.what() << std::endl;
            } catch (...) {
                ++crash_count;
                std::cout << "Unknown crash in test " << test_num << std::endl;
            }
        }
        
        ctx.record(test::is_true(
            crash_count == 0,
            "No crashes should occur"
        ));
    }
```
