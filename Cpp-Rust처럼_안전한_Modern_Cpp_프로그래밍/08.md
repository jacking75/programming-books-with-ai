# 8장: std::execution으로 안전한 비동기 프로그래밍

## 8.1 기존 std::async의 한계
C++11에서 도입된 `std::async`는 비동기 프로그래밍의 첫 걸음이었지만, 여러 근본적인 문제를 가지고 있다. 가장 큰 문제는 타입 안전성 부족, 에러 처리의 어려움, 그리고 조합 가능성의 결여다.

### std::async의 안전하지 않은 사용

```cpp
#include <future>
#include <iostream>
#include <vector>
#include <stdexcept>

// 문제 1: 예외 처리가 복잡하고 누락되기 쉽다
int unsafe_computation(int x) {
    if (x < 0) {
        throw std::runtime_error("Negative input");
    }
    return x * x;
}

void demonstrate_async_problems() {
    // 예외를 제대로 처리하지 않으면 프로그램이 종료될 수 있다
    auto future1 = std::async(std::launch::async, []() {
        throw std::runtime_error("Unhandled exception");
        return 42;
    });
    
    // future의 소멸자에서 예외가 발생하면 std::terminate 호출
    // future1이 스코프를 벗어날 때 위험!
    
    // 문제 2: 여러 비동기 작업의 조합이 어렵다
    auto future2 = std::async(std::launch::async, []() { return 10; });
    auto future3 = std::async(std::launch::async, []() { return 20; });
    
    // 두 결과를 조합하려면 명시적으로 get()을 호출해야 함
    int result1 = future2.get();  // 블로킹
    int result2 = future3.get();  // 블로킹
    int combined = result1 + result2;
    
    // 문제 3: 취소 메커니즘이 없다
    auto long_running = std::async(std::launch::async, []() {
        for (int i = 0; i < 1000000; ++i) {
            // 작업을 중단할 방법이 없다
        }
        return 0;
    });
    
    // 더 이상 결과가 필요 없어도 작업을 취소할 수 없다
}
```

이러한 문제들은 코드의 안전성과 유지보수성을 크게 해친다. 특히 예외 처리 누락은 프로그램 전체를 중단시킬 수 있는 심각한 버그로 이어진다.

### 에러 처리의 복잡성

```cpp
#include <future>
#include <vector>
#include <exception>
#include <iostream>

class UnsafeAsyncPipeline {
public:
    // 여러 단계의 비동기 작업을 체이닝하기 어렵다
    static int process_data(int input) {
        auto stage1 = std::async(std::launch::async, [input]() {
            if (input < 0) throw std::runtime_error("Stage 1 failed");
            return input * 2;
        });
        
        try {
            int intermediate = stage1.get();
            
            auto stage2 = std::async(std::launch::async, [intermediate]() {
                if (intermediate > 100) throw std::runtime_error("Stage 2 failed");
                return intermediate + 10;
            });
            
            return stage2.get();
            
        } catch (const std::exception& e) {
            // 에러 처리가 중첩되고 복잡해진다
            std::cerr << "Pipeline error: " << e.what() << std::endl;
            throw;
        }
    }
    
    // 여러 작업을 병렬로 실행하고 하나라도 실패하면 전체 취소
    static std::vector<int> process_batch(const std::vector<int>& inputs) {
        std::vector<std::future<int>> futures;
        
        for (int input : inputs) {
            futures.push_back(std::async(std::launch::async, [input]() {
                return input * input;
            }));
        }
        
        std::vector<int> results;
        for (auto& future : futures) {
            try {
                results.push_back(future.get());
            } catch (const std::exception& e) {
                // 하나가 실패해도 나머지 작업을 취소할 수 없다
                std::cerr << "Task failed: " << e.what() << std::endl;
                // 이미 시작된 다른 작업들은 계속 실행된다
                throw;
            }
        }
        
        return results;
    }
};
```

위 코드는 에러 처리가 산재되어 있고, 부분적인 실패를 우아하게 처리하기 어렵다. 또한 이미 시작된 작업을 취소할 방법이 없어 리소스가 낭비된다.
  

## 8.2 Sender/Receiver 모델 이해
C++26의 `std::execution`은 Sender/Receiver 모델을 도입하여 이러한 문제들을 해결한다. 이 모델은 비동기 작업을 타입 안전하게 표현하고, 에러 전파를 명시적으로 처리하며, 작업의 조합과 취소를 지원한다.

### 기본 개념: Sender와 Receiver

```cpp
#include <execution>
#include <iostream>

// Sender: 비동기 작업을 나타내는 객체
// Receiver: 작업 완료 시 호출되는 핸들러

namespace ex = std::execution;

// 간단한 sender 생성
auto create_simple_sender() {
    // just는 값을 즉시 반환하는 sender를 생성한다
    return ex::just(42);
}

// Receiver 정의
struct my_receiver {
    // 성공 시 호출
    void set_value(int value) {
        std::cout << "Received value: " << value << std::endl;
    }
    
    // 에러 시 호출
    void set_error(std::exception_ptr error) {
        try {
            std::rethrow_exception(error);
        } catch (const std::exception& e) {
            std::cout << "Received error: " << e.what() << std::endl;
        }
    }
    
    // 취소 시 호출
    void set_stopped() {
        std::cout << "Operation was cancelled" << std::endl;
    }
};

void demonstrate_sender_receiver() {
    auto sender = create_simple_sender();
    
    // sender를 receiver와 연결하고 실행
    auto state = ex::connect(sender, my_receiver{});
    ex::start(state);
}
```

Sender/Receiver 모델의 핵심은 세 가지 완료 채널이다:
1. `set_value`: 작업이 성공적으로 완료됨
2. `set_error`: 작업이 예외와 함께 실패함
3. `set_stopped`: 작업이 취소됨

이 세 채널을 통해 모든 가능한 결과를 타입 안전하게 처리할 수 있다.

### 타입 안전한 비동기 체인

```cpp
#include <execution>
#include <stdexcept>
#include <string>

namespace ex = std::execution;

// then을 사용한 안전한 체이닝
auto safe_async_chain() {
    return ex::just(10)
        | ex::then([](int x) {
            // 첫 번째 변환: int -> int
            if (x < 0) {
                throw std::invalid_argument("Negative value");
            }
            return x * 2;
        })
        | ex::then([](int x) {
            // 두 번째 변환: int -> string
            return std::to_string(x) + " units";
        })
        | ex::then([](const std::string& s) {
            // 세 번째 변환: string -> size_t
            return s.length();
        });
}

// 타입이 컴파일 타임에 추론되고 검증된다
void use_safe_chain() {
    auto sender = safe_async_chain();
    
    // Receiver는 size_t를 받는다 (마지막 then의 반환 타입)
    struct result_receiver {
        void set_value(size_t length) {
            std::cout << "Final length: " << length << std::endl;
        }
        
        void set_error(std::exception_ptr e) {
            try {
                std::rethrow_exception(e);
            } catch (const std::exception& ex) {
                std::cout << "Error: " << ex.what() << std::endl;
            }
        }
        
        void set_stopped() {
            std::cout << "Stopped" << std::endl;
        }
    };
    
    auto state = ex::connect(sender, result_receiver{});
    ex::start(state);
}
```

파이프 연산자 `|`를 사용하면 비동기 작업을 선형적으로 조합할 수 있다. 각 단계의 타입은 컴파일 타임에 검증되므로, 잘못된 타입 변환은 컴파일 에러로 잡힌다.

### 에러 전파와 처리

```cpp
#include <execution>
#include <exception>
#include <variant>

namespace ex = std::execution;

// upon_error로 에러 처리
auto handle_errors() {
    return ex::just(42)
        | ex::then([](int x) {
            if (x > 40) {
                throw std::runtime_error("Value too large");
            }
            return x;
        })
        | ex::upon_error([](std::exception_ptr e) {
            // 에러를 복구하고 기본값 반환
            std::cout << "Error occurred, using default value" << std::endl;
            return ex::just(0);
        });
}

// let_error로 더 복잡한 에러 처리
auto advanced_error_handling(int input) {
    return ex::just(input)
        | ex::then([](int x) {
            if (x < 0) {
                throw std::invalid_argument("Negative input");
            }
            if (x > 100) {
                throw std::out_of_range("Input too large");
            }
            return x * x;
        })
        | ex::let_error([](std::exception_ptr e) {
            try {
                std::rethrow_exception(e);
            } catch (const std::invalid_argument&) {
                // 음수는 0으로 처리
                return ex::just(0);
            } catch (const std::out_of_range&) {
                // 너무 큰 값은 최대값으로 제한
                return ex::just(10000);
            } catch (...) {
                // 다른 에러는 재발생
                throw;
            }
        });
}

// 여러 작업 중 하나라도 실패하면 전체 실패
auto fail_fast_operations() {
    auto task1 = ex::just(10) | ex::then([](int x) { return x * 2; });
    auto task2 = ex::just(20) | ex::then([](int x) {
        if (x > 15) throw std::runtime_error("Task 2 failed");
        return x * 3;
    });
    auto task3 = ex::just(30) | ex::then([](int x) { return x * 4; });
    
    // when_all은 모든 작업이 성공해야 성공
    return ex::when_all(task1, task2, task3)
        | ex::then([](int r1, int r2, int r3) {
            return r1 + r2 + r3;
        })
        | ex::upon_error([](std::exception_ptr e) {
            std::cout << "One of the tasks failed" << std::endl;
            return ex::just(-1);
        });
}
```

에러 처리가 명시적이고 타입 안전하다. `upon_error`는 에러를 다른 값으로 변환하고, `let_error`는 에러에 따라 다른 sender를 반환할 수 있다.

### 다양한 완료 시그니처 처리

```cpp
#include <execution>
#include <variant>
#include <string>

namespace ex = std::execution;

// 여러 가능한 값 타입을 반환하는 sender
auto multi_value_sender(int choice) {
    if (choice == 1) {
        return ex::just(42);
    } else if (choice == 2) {
        return ex::just(std::string("hello"));
    } else {
        return ex::just(3.14);
    }
}

// std::variant를 사용한 다형성 처리
auto polymorphic_handling() {
    using result_t = std::variant<int, std::string, double>;
    
    return ex::just(1)
        | ex::then([](int choice) -> result_t {
            if (choice == 1) return 42;
            if (choice == 2) return std::string("hello");
            return 3.14;
        })
        | ex::then([](const result_t& value) {
            return std::visit([](auto&& v) {
                using T = std::decay_t<decltype(v)>;
                if constexpr (std::is_same_v<T, int>) {
                    return std::to_string(v) + " (int)";
                } else if constexpr (std::is_same_v<T, std::string>) {
                    return v + " (string)";
                } else {
                    return std::to_string(v) + " (double)";
                }
            }, value);
        });
}
```
  

## 8.3 실제 활용 예제

### 파이프라인 구성

```cpp
#include <execution>
#include <vector>
#include <string>
#include <fstream>
#include <algorithm>

namespace ex = std::execution;

// 데이터 처리 파이프라인
struct DataProcessor {
    // 파일 읽기 -> 파싱 -> 변환 -> 집계
    static auto create_pipeline(const std::string& filename) {
        return ex::just(filename)
            | ex::then([](const std::string& file) {
                // 1단계: 파일 읽기 (I/O 작업)
                std::ifstream ifs(file);
                if (!ifs) {
                    throw std::runtime_error("Cannot open file");
                }
                
                std::string content;
                std::string line;
                while (std::getline(ifs, line)) {
                    content += line + "\n";
                }
                return content;
            })
            | ex::then([](const std::string& content) {
                // 2단계: 파싱 (CPU 집약적)
                std::vector<int> numbers;
                std::istringstream iss(content);
                int num;
                while (iss >> num) {
                    numbers.push_back(num);
                }
                return numbers;
            })
            | ex::then([](const std::vector<int>& numbers) {
                // 3단계: 변환
                std::vector<int> squared;
                squared.reserve(numbers.size());
                for (int n : numbers) {
                    squared.push_back(n * n);
                }
                return squared;
            })
            | ex::then([](const std::vector<int>& squared) {
                // 4단계: 집계
                int sum = 0;
                for (int n : squared) {
                    sum += n;
                }
                return sum;
            })
            | ex::upon_error([](std::exception_ptr e) {
                try {
                    std::rethrow_exception(e);
                } catch (const std::exception& ex) {
                    std::cerr << "Pipeline error: " << ex.what() << std::endl;
                }
                return ex::just(-1);
            });
    }
};

// 병렬 파이프라인: 여러 파일을 동시에 처리
auto parallel_file_processing(const std::vector<std::string>& files) {
    std::vector<decltype(DataProcessor::create_pipeline(""))> senders;
    
    for (const auto& file : files) {
        senders.push_back(DataProcessor::create_pipeline(file));
    }
    
    // 모든 파일을 병렬로 처리
    return ex::when_all(senders.begin(), senders.end())
        | ex::then([](const std::vector<int>& results) {
            int total = 0;
            for (int r : results) {
                if (r >= 0) {  // -1은 에러
                    total += r;
                }
            }
            return total;
        });
}
```

파이프라인은 각 단계가 독립적이며, 타입 안전성이 보장된다. 에러는 자동으로 전파되어 `upon_error`에서 중앙 집중식으로 처리된다.

### 취소와 타임아웃 처리

```cpp
#include <execution>
#include <stop_token>
#include <chrono>

namespace ex = std::execution;

// 취소 가능한 비동기 작업
auto cancellable_operation(std::stop_token stop_token) {
    return ex::just()
        | ex::then([stop_token]() {
            int result = 0;
            for (int i = 0; i < 1000000; ++i) {
                // 주기적으로 취소 요청 확인
                if (stop_token.stop_requested()) {
                    throw std::runtime_error("Operation cancelled");
                }
                result += i;
            }
            return result;
        });
}

// 타임아웃을 지원하는 작업
auto operation_with_timeout(std::chrono::milliseconds timeout) {
    auto work = ex::just()
        | ex::then([]() {
            // 시간이 오래 걸리는 작업
            std::this_thread::sleep_for(std::chrono::seconds(5));
            return 42;
        });
    
    auto timer = ex::schedule_after(
        ex::get_scheduler(),
        timeout
    ) | ex::then([]() -> int {
        throw std::runtime_error("Timeout");
    });
    
    // 둘 중 먼저 완료되는 것을 반환
    return ex::when_any(work, timer)
        | ex::then([](auto&& result) {
            return result;
        })
        | ex::upon_error([](std::exception_ptr e) {
            try {
                std::rethrow_exception(e);
            } catch (const std::runtime_error& ex) {
                std::cout << "Operation timed out" << std::endl;
            }
            return ex::just(-1);
        });
}

// stop_source를 사용한 명시적 취소
class CancellableTask {
    std::stop_source stop_source;
    
public:
    auto create_task() {
        auto token = stop_source.get_token();
        
        return ex::just()
            | ex::then([token]() {
                for (int i = 0; i < 100; ++i) {
                    if (token.stop_requested()) {
                        std::cout << "Task cancelled at iteration " << i << std::endl;
                        throw std::runtime_error("Cancelled");
                    }
                    
                    // 실제 작업 수행
                    std::this_thread::sleep_for(std::chrono::milliseconds(10));
                }
                return 100;
            })
            | ex::upon_stopped([]() {
                std::cout << "Task was stopped" << std::endl;
                return ex::just(0);
            });
    }
    
    void cancel() {
        stop_source.request_stop();
    }
};
```

취소 메커니즘은 `std::stop_token`을 통해 협력적으로 작동한다. 작업은 주기적으로 취소 요청을 확인하고, 요청이 있으면 작업을 중단한다.

### 실용적인 HTTP 클라이언트 예제

```cpp
#include <execution>
#include <string>
#include <chrono>
#include <optional>

namespace ex = std::execution;

// HTTP 응답 구조체
struct HttpResponse {
    int status_code;
    std::string body;
    std::map<std::string, std::string> headers;
};

// 안전한 HTTP 클라이언트
class SafeHttpClient {
public:
    // 재시도 로직을 포함한 GET 요청
    static auto get_with_retry(
        const std::string& url,
        int max_retries = 3,
        std::chrono::milliseconds timeout = std::chrono::seconds(10)
    ) {
        return ex::just(url)
            | ex::let_value([=](const std::string& u) {
                return attempt_request(u, max_retries, timeout);
            })
            | ex::upon_error([](std::exception_ptr e) {
                try {
                    std::rethrow_exception(e);
                } catch (const std::exception& ex) {
                    std::cerr << "Request failed: " << ex.what() << std::endl;
                }
                return ex::just(HttpResponse{0, "", {}});
            });
    }
    
private:
    static auto attempt_request(
        const std::string& url,
        int retries_left,
        std::chrono::milliseconds timeout
    ) {
        auto request = ex::just()
            | ex::then([url]() {
                // 실제 HTTP 요청 수행 (예시)
                // 실제로는 비동기 I/O 라이브러리 사용
                HttpResponse response;
                response.status_code = 200;
                response.body = "Success";
                
                // 시뮬레이션: 가끔 실패
                if (std::rand() % 10 < 3) {
                    throw std::runtime_error("Network error");
                }
                
                return response;
            });
        
        // 타임아웃 추가
        auto timed_request = ex::when_any(
            request,
            ex::schedule_after(ex::get_scheduler(), timeout)
                | ex::then([]() -> HttpResponse {
                    throw std::runtime_error("Request timeout");
                })
        );
        
        // 재시도 로직
        return timed_request
            | ex::let_error([=](std::exception_ptr e) {
                if (retries_left > 0) {
                    std::cout << "Retrying... (" << retries_left 
                             << " attempts left)" << std::endl;
                    
                    // 지수 백오프
                    auto delay = std::chrono::milliseconds(
                        100 * (1 << (3 - retries_left))
                    );
                    
                    return ex::schedule_after(ex::get_scheduler(), delay)
                        | ex::let_value([=]() {
                            return attempt_request(url, retries_left - 1, timeout);
                        });
                } else {
                    std::rethrow_exception(e);
                }
            });
    }
};

// 여러 URL을 병렬로 가져오기
auto fetch_multiple_urls(const std::vector<std::string>& urls) {
    std::vector<decltype(SafeHttpClient::get_with_retry(""))> senders;
    
    for (const auto& url : urls) {
        senders.push_back(SafeHttpClient::get_with_retry(url));
    }
    
    return ex::when_all(senders.begin(), senders.end())
        | ex::then([](const std::vector<HttpResponse>& responses) {
            std::vector<HttpResponse> successful;
            for (const auto& resp : responses) {
                if (resp.status_code >= 200 && resp.status_code < 300) {
                    successful.push_back(resp);
                }
            }
            return successful;
        });
}
```

### 복잡한 워크플로우 조합

```cpp
#include <execution>
#include <variant>

namespace ex = std::execution;

// 여러 비동기 작업을 조합한 복잡한 워크플로우
class WorkflowOrchestrator {
public:
    // 조건부 분기를 포함한 워크플로우
    static auto conditional_workflow(int input) {
        return ex::just(input)
            | ex::then([](int x) {
                return x > 50 ? "high" : "low";
            })
            | ex::let_value([](const std::string& category) {
                if (category == "high") {
                    // 높은 값은 복잡한 처리
                    return ex::just(category)
                        | ex::then([](const std::string&) {
                            return heavy_computation();
                        });
                } else {
                    // 낮은 값은 간단한 처리
                    return ex::just(category)
                        | ex::then([](const std::string&) {
                            return light_computation();
                        });
                }
            });
    }
    
    // 순차적 종속성이 있는 작업들
    static auto dependent_tasks(int seed) {
        return ex::just(seed)
            | ex::then([](int s) {
                // 첫 번째 작업
                return fetch_data(s);
            })
            | ex::let_value([](int data) {
                // 두 번째 작업 (첫 번째 결과에 의존)
                return ex::just(data)
                    | ex::then([](int d) {
                        return process_data(d);
                    });
            })
            | ex::let_value([](const std::string& processed) {
                // 세 번째 작업 (두 번째 결과에 의존)
                return ex::just(processed)
                    | ex::then([](const std::string& p) {
                        return save_result(p);
                    });
            });
    }
    
    // 병렬 작업 후 결과 조합
    static auto parallel_aggregation(const std::vector<int>& inputs) {
        std::vector<decltype(ex::just(0) | ex::then([](int x) { return x * x; }))> tasks;
        
        for (int input : inputs) {
            tasks.push_back(
                ex::just(input)
                | ex::then([](int x) {
                    return x * x;
                })
            );
        }
        
        return ex::when_all(tasks.begin(), tasks.end())
            | ex::then([](const std::vector<int>& results) {
                // 모든 결과를 집계
                int sum = 0;
                int count = 0;
                for (int r : results) {
                    sum += r;
                    count++;
                }
                return count > 0 ? sum / count : 0;
            });
    }
    
private:
    static int heavy_computation() {
        // 무거운 계산 시뮬레이션
        int result = 0;
        for (int i = 0; i < 1000000; ++i) {
            result += i;
        }
        return result;
    }
    
    static int light_computation() {
        return 42;
    }
    
    static int fetch_data(int seed) {
        return seed * 2;
    }
    
    static std::string process_data(int data) {
        return "Processed: " + std::to_string(data);
    }
    
    static bool save_result(const std::string& result) {
        std::cout << result << std::endl;
        return true;
    }
};
```

### 스케줄러를 활용한 스레드 풀 관리

```cpp
#include <execution>
#include <thread>

namespace ex = std::execution;

// 커스텀 스케줄러로 특정 스레드에서 실행
class ThreadPoolScheduler {
    struct thread_pool {
        // 간단한 스레드 풀 구현
        std::vector<std::thread> threads;
        // 실제로는 작업 큐와 동기화 필요
    };
    
    thread_pool pool;
    
public:
    auto schedule() {
        // 스케줄러는 sender를 반환한다
        return ex::just()
            | ex::then([this]() {
                // 스레드 풀에서 실행
                return 0;
            });
    }
    
    // 특정 스케줄러에서 작업 실행
    template<typename Sender>
    auto on(Sender&& sender) {
        return std::forward<Sender>(sender)
            | ex::transfer(*this);
    }
};

// CPU 집약적 작업과 I/O 작업을 다른 스레드 풀에서 실행
auto optimized_workflow() {
    ThreadPoolScheduler cpu_pool;
    ThreadPoolScheduler io_pool;
    
    return io_pool.on(
        ex::just()
        | ex::then([]() {
            // I/O 작업: 파일 읽기
            return "data from file";
        })
    )
    | cpu_pool.on(
        ex::then([](const std::string& data) {
            // CPU 작업: 데이터 처리
            return data.length();
        })
    )
    | io_pool.on(
        ex::then([](size_t length) {
            // I/O 작업: 결과 저장
            std::cout << "Saved result: " << length << std::endl;
            return true;
        })
    );
}
```

`std::execution`의 스케줄러 추상화를 통해 작업을 특정 실행 컨텍스트에서 수행할 수 있다. 이는 I/O 바운드 작업과 CPU 바운드 작업을 적절히 분리하여 최적의 성능을 달성하는 데 유용하다.

### 에러 복구와 폴백 전략

```cpp
#include <execution>
#include <variant>
#include <optional>

namespace ex = std::execution;

// 여러 백업 전략을 시도하는 resilient한 시스템
class ResilientService {
public:
    // 주 서비스 실패 시 백업 서비스 시도
    static auto fetch_with_fallback(const std::string& key) {
        return try_primary_service(key)
            | ex::upon_error([=](std::exception_ptr e1) {
                std::cout << "Primary service failed, trying secondary..." << std::endl;
                
                return try_secondary_service(key)
                    | ex::upon_error([=](std::exception_ptr e2) {
                        std::cout << "Secondary service failed, trying cache..." << std::endl;
                        
                        return try_cache(key)
                            | ex::upon_error([](std::exception_ptr e3) {
                                std::cout << "All services failed, using default" << std::endl;
                                return ex::just(std::string("default_value"));
                            });
                    });
            });
    }
    
    // 여러 소스에서 동시에 가져와서 가장 빠른 응답 사용
    static auto fetch_fastest(const std::string& key) {
        auto source1 = try_primary_service(key);
        auto source2 = try_secondary_service(key);
        auto source3 = try_cache(key);
        
        // 가장 먼저 완료되는 것을 반환
        return ex::when_any(source1, source2, source3)
            | ex::then([](const std::string& result) {
                std::cout << "Got fastest response: " << result << std::endl;
                return result;
            })
            | ex::upon_error([](std::exception_ptr e) {
                // 모두 실패한 경우에만 여기 도달
                return ex::just(std::string("all_failed"));
            });
    }
    
private:
    static auto try_primary_service(const std::string& key) {
        return ex::just(key)
            | ex::then([](const std::string& k) -> std::string {
                // 시뮬레이션: 가끔 실패
                if (std::rand() % 2 == 0) {
                    throw std::runtime_error("Primary service error");
                }
                return "primary:" + k;
            });
    }
    
    static auto try_secondary_service(const std::string& key) {
        return ex::just(key)
            | ex::then([](const std::string& k) -> std::string {
                if (std::rand() % 3 == 0) {
                    throw std::runtime_error("Secondary service error");
                }
                return "secondary:" + k;
            });
    }
    
    static auto try_cache(const std::string& key) {
        return ex::just(key)
            | ex::then([](const std::string& k) -> std::string {
                return "cache:" + k;
            });
    }
};
```

Sender/Receiver 모델의 진정한 힘은 복잡한 비동기 로직을 선언적이고 타입 안전하게 표현할 수 있다는 점이다. 에러 처리, 재시도, 타임아웃, 폴백 같은 복잡한 패턴을 간결하게 구현할 수 있으며, 모든 것이 컴파일 타임에 검증된다. 이는 `std::async`의 한계를 완전히 극복하고, Rust의 `async/await`에 필적하는 안전성과 표현력을 제공한다.
