# Day 6: 게임 콘텐츠 시스템 구현
Day 6에서는 실제 온라인 게임에서 자주 사용되는 핵심 콘텐츠 시스템을 구현한다. 친구 시스템, 우편함 시스템, 던전 보상 시스템을 통해 실무에서 바로 활용 가능한 패턴을 배운다.

## 6.1 친구 시스템 구현
친구 시스템은 소셜 기능의 핵심이다. 유저 간의 관계를 효율적으로 관리하고 조회하는 방법을 알아본다.

### 6.1.1 친구 관계 테이블 설계
친구 관계는 양방향 관계이지만, 데이터베이스에서는 단방향으로 저장하는 것이 일반적이다. 친구 관계의 상태(요청, 수락, 차단 등)도 함께 관리한다.

```sql
-- 친구 관계 테이블
CREATE TABLE friend_relations (
    relation_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    from_user_id BIGINT NOT NULL,
    to_user_id BIGINT NOT NULL,
    status ENUM('PENDING', 'ACCEPTED', 'BLOCKED') NOT NULL DEFAULT 'PENDING',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    -- 복합 인덱스: 중복 관계 방지 및 조회 성능 향상
    UNIQUE KEY uk_friend_pair (from_user_id, to_user_id),
    
    -- 역방향 조회를 위한 인덱스
    KEY idx_to_user_status (to_user_id, status),
    
    -- 외래키 제약조건
    FOREIGN KEY (from_user_id) REFERENCES users(user_id) ON DELETE CASCADE,
    FOREIGN KEY (to_user_id) REFERENCES users(user_id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
```

친구 관계의 핵심 포인트는 다음과 같다.

**양방향 관계 처리 전략**
```
User A → User B (PENDING)    # A가 B에게 친구 요청
User B → User A (ACCEPTED)   # B가 수락하면 양방향 레코드 생성

또는

User A ↔ User B (ACCEPTED)   # 하나의 레코드로 양방향 표현 (정규화된 방식)
```

정규화된 방식을 사용하면 데이터 중복이 없지만, 양방향 조회 시 `OR` 조건이 필요하다. 반정규화 방식(양방향 레코드)을 사용하면 조회가 간단하지만 데이터가 2배로 증가한다.

### 6.1.2 친구 요청/수락/거절 로직
C#에서 SQLKata를 이용해 친구 시스템의 핵심 기능을 구현한다.

```csharp
using SqlKata;
using SqlKata.Execution;
using MySqlConnector;
using System;
using System.Threading.Tasks;
using System.Collections.Generic;

public class FriendService
{
    private readonly QueryFactory _db;

    public FriendService(MySqlConnection connection)
    {
        var compiler = new SqlKata.Compilers.MySqlCompiler();
        _db = new QueryFactory(connection, compiler);
    }

    // 친구 요청 보내기
    public async Task<bool> SendFriendRequestAsync(long fromUserId, long toUserId)
    {
        // 자기 자신에게는 요청 불가
        if (fromUserId == toUserId)
        {
            throw new InvalidOperationException("자기 자신에게 친구 요청을 보낼 수 없다.");
        }

        // 이미 관계가 존재하는지 확인
        var existingRelation = await _db.Query("friend_relations")
            .Where("from_user_id", fromUserId)
            .Where("to_user_id", toUserId)
            .FirstOrDefaultAsync<FriendRelation>();

        if (existingRelation != null)
        {
            throw new InvalidOperationException("이미 친구 관계가 존재한다.");
        }

        // 역방향 관계 확인 (상대방이 먼저 요청했는지)
        var reverseRelation = await _db.Query("friend_relations")
            .Where("from_user_id", toUserId)
            .Where("to_user_id", fromUserId)
            .FirstOrDefaultAsync<FriendRelation>();

        if (reverseRelation != null)
        {
            // 상대방이 이미 요청했다면 자동으로 수락 처리
            return await AcceptFriendRequestAsync(toUserId, fromUserId);
        }

        // 새로운 친구 요청 생성
        var relationId = await _db.Query("friend_relations").InsertGetIdAsync<long>(new
        {
            from_user_id = fromUserId,
            to_user_id = toUserId,
            status = "PENDING"
        });

        return relationId > 0;
    }

    // 친구 요청 수락
    public async Task<bool> AcceptFriendRequestAsync(long fromUserId, long toUserId)
    {
        // 트랜잭션 시작
        using var transaction = await _db.Connection.BeginTransactionAsync();
        
        try
        {
            // 요청 상태 확인
            var request = await _db.Query("friend_relations")
                .Where("from_user_id", fromUserId)
                .Where("to_user_id", toUserId)
                .Where("status", "PENDING")
                .FirstOrDefaultAsync<FriendRelation>();

            if (request == null)
            {
                throw new InvalidOperationException("처리할 친구 요청이 없다.");
            }

            // 요청 상태를 ACCEPTED로 변경
            await _db.Query("friend_relations")
                .Where("relation_id", request.RelationId)
                .UpdateAsync(new { status = "ACCEPTED" });

            // 양방향 관계 생성 (역방향 레코드 추가)
            await _db.Query("friend_relations").InsertAsync(new
            {
                from_user_id = toUserId,
                to_user_id = fromUserId,
                status = "ACCEPTED"
            });

            await transaction.CommitAsync();
            return true;
        }
        catch
        {
            await transaction.RollbackAsync();
            throw;
        }
    }

    // 친구 요청 거절
    public async Task<bool> RejectFriendRequestAsync(long fromUserId, long toUserId)
    {
        var affected = await _db.Query("friend_relations")
            .Where("from_user_id", fromUserId)
            .Where("to_user_id", toUserId)
            .Where("status", "PENDING")
            .DeleteAsync();

        return affected > 0;
    }

    // 친구 삭제
    public async Task<bool> RemoveFriendAsync(long userId, long friendUserId)
    {
        using var transaction = await _db.Connection.BeginTransactionAsync();
        
        try
        {
            // 양방향 레코드 모두 삭제
            await _db.Query("friend_relations")
                .Where(q => q
                    .Where(w => w.Where("from_user_id", userId).Where("to_user_id", friendUserId))
                    .OrWhere(w => w.Where("from_user_id", friendUserId).Where("to_user_id", userId))
                )
                .DeleteAsync();

            await transaction.CommitAsync();
            return true;
        }
        catch
        {
            await transaction.RollbackAsync();
            throw;
        }
    }
}

public class FriendRelation
{
    public long RelationId { get; set; }
    public long FromUserId { get; set; }
    public long ToUserId { get; set; }
    public string Status { get; set; }
    public DateTime CreatedAt { get; set; }
    public DateTime UpdatedAt { get; set; }
}
```

### 6.1.3 친구 목록 조회
친구 목록 조회는 JOIN을 활용하여 친구의 상세 정보까지 함께 가져온다.

```csharp
public class FriendListService
{
    private readonly QueryFactory _db;

    public FriendListService(MySqlConnection connection)
    {
        var compiler = new SqlKata.Compilers.MySqlCompiler();
        _db = new QueryFactory(connection, compiler);
    }

    // 수락된 친구 목록 조회
    public async Task<List<FriendInfo>> GetFriendsAsync(long userId)
    {
        var friends = await _db.Query("friend_relations as fr")
            .Join("users as u", "fr.to_user_id", "u.user_id")
            .Where("fr.from_user_id", userId)
            .Where("fr.status", "ACCEPTED")
            .Select("u.user_id", "u.username", "u.level", "fr.created_at as friend_since")
            .GetAsync<FriendInfo>();

        return friends.ToList();
    }

    // 받은 친구 요청 목록 조회
    public async Task<List<FriendRequestInfo>> GetPendingRequestsAsync(long userId)
    {
        var requests = await _db.Query("friend_relations as fr")
            .Join("users as u", "fr.from_user_id", "u.user_id")
            .Where("fr.to_user_id", userId)
            .Where("fr.status", "PENDING")
            .Select("fr.relation_id", "u.user_id", "u.username", "u.level", "fr.created_at")
            .OrderByDesc("fr.created_at")
            .GetAsync<FriendRequestInfo>();

        return requests.ToList();
    }

    // 보낸 친구 요청 목록 조회
    public async Task<List<FriendRequestInfo>> GetSentRequestsAsync(long userId)
    {
        var requests = await _db.Query("friend_relations as fr")
            .Join("users as u", "fr.to_user_id", "u.user_id")
            .Where("fr.from_user_id", userId)
            .Where("fr.status", "PENDING")
            .Select("fr.relation_id", "u.user_id", "u.username", "u.level", "fr.created_at")
            .OrderByDesc("fr.created_at")
            .GetAsync<FriendRequestInfo>();

        return requests.ToList();
    }
}

public class FriendInfo
{
    public long UserId { get; set; }
    public string Username { get; set; }
    public int Level { get; set; }
    public DateTime FriendSince { get; set; }
}

public class FriendRequestInfo
{
    public long RelationId { get; set; }
    public long UserId { get; set; }
    public string Username { get; set; }
    public int Level { get; set; }
    public DateTime CreatedAt { get; set; }
}
```
  

## 6.2 우편함(메일) 시스템 구현
우편함 시스템은 게임 내에서 유저 간 또는 운영자가 유저에게 메시지와 아이템을 전달하는 핵심 기능이다.

### 6.2.1 우편함 테이블 설계
우편함은 메일 본문과 첨부 아이템을 분리하여 설계한다.

```sql
-- 메일 테이블
CREATE TABLE mails (
    mail_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    sender_id BIGINT NULL,  -- NULL이면 시스템 메일
    receiver_id BIGINT NOT NULL,
    title VARCHAR(100) NOT NULL,
    content TEXT NOT NULL,
    mail_type ENUM('NORMAL', 'SYSTEM', 'REWARD') NOT NULL DEFAULT 'NORMAL',
    is_read BOOLEAN NOT NULL DEFAULT FALSE,
    is_claimed BOOLEAN NOT NULL DEFAULT FALSE,  -- 보상 수령 여부
    expire_at DATETIME NULL,  -- 만료 시간
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    read_at DATETIME NULL,
    claimed_at DATETIME NULL,
    
    KEY idx_receiver_read (receiver_id, is_read),
    KEY idx_receiver_expire (receiver_id, expire_at),
    KEY idx_created (created_at),
    
    FOREIGN KEY (sender_id) REFERENCES users(user_id) ON DELETE SET NULL,
    FOREIGN KEY (receiver_id) REFERENCES users(user_id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- 메일 첨부 아이템 테이블
CREATE TABLE mail_attachments (
    attachment_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    mail_id BIGINT NOT NULL,
    item_id INT NOT NULL,
    quantity INT NOT NULL DEFAULT 1,
    
    KEY idx_mail (mail_id),
    
    FOREIGN KEY (mail_id) REFERENCES mails(mail_id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
```

메일 시스템의 구조는 다음과 같다.

```
┌─────────────────────────────────────────────────────┐
│                    Mail System                       │
├─────────────────────────────────────────────────────┤
│                                                      │
│  ┌──────────┐         ┌──────────────────────┐     │
│  │  Sender  │────────▶│       Mails          │     │
│  │  (User)  │         │  - title             │     │
│  └──────────┘         │  - content           │     │
│                       │  - is_read           │     │
│                       │  - is_claimed        │     │
│                       └──────────┬───────────┘     │
│                                  │                  │
│                                  │ 1:N              │
│  ┌──────────┐                   │                  │
│  │ Receiver │◀──────────────────┘                  │
│  │  (User)  │         ┌──────────────────────┐     │
│  └──────────┘         │  Mail Attachments    │     │
│                       │  - item_id           │     │
│                       │  - quantity          │     │
│                       └──────────────────────┘     │
│                                                      │
└─────────────────────────────────────────────────────┘
```

### 6.2.2 메일 발송 및 조회
메일 발송과 조회 기능을 구현한다.

```csharp
using SqlKata;
using SqlKata.Execution;
using MySqlConnector;
using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading.Tasks;

public class MailService
{
    private readonly QueryFactory _db;

    public MailService(MySqlConnection connection)
    {
        var compiler = new SqlKata.Compilers.MySqlCompiler();
        _db = new QueryFactory(connection, compiler);
    }

    // 메일 발송 (아이템 첨부 가능)
    public async Task<long> SendMailAsync(
        long? senderId,  // NULL이면 시스템 메일
        long receiverId, 
        string title, 
        string content,
        string mailType = "NORMAL",
        DateTime? expireAt = null,
        List<MailAttachment> attachments = null)
    {
        using var transaction = await _db.Connection.BeginTransactionAsync();
        
        try
        {
            // 메일 본문 삽입
            var mailId = await _db.Query("mails").InsertGetIdAsync<long>(new
            {
                sender_id = senderId,
                receiver_id = receiverId,
                title = title,
                content = content,
                mail_type = mailType,
                expire_at = expireAt
            });

            // 첨부 아이템이 있다면 삽입
            if (attachments != null && attachments.Any())
            {
                var attachmentData = attachments.Select(a => new
                {
                    mail_id = mailId,
                    item_id = a.ItemId,
                    quantity = a.Quantity
                }).ToArray();

                await _db.Query("mail_attachments").InsertAsync(attachmentData);
            }

            await transaction.CommitAsync();
            return mailId;
        }
        catch
        {
            await transaction.RollbackAsync();
            throw;
        }
    }

    // 메일함 조회
    public async Task<List<MailInfo>> GetMailboxAsync(long userId, bool includeRead = true)
    {
        var query = _db.Query("mails as m")
            .LeftJoin("users as u", "m.sender_id", "u.user_id")
            .Where("m.receiver_id", userId)
            .Where(q => q.WhereNull("m.expire_at").OrWhere("m.expire_at", ">", DateTime.Now))
            .Select(
                "m.mail_id",
                "m.sender_id",
                "u.username as sender_name",
                "m.title",
                "m.content",
                "m.mail_type",
                "m.is_read",
                "m.is_claimed",
                "m.created_at",
                "m.expire_at"
            )
            .OrderByDesc("m.created_at");

        if (!includeRead)
        {
            query.Where("m.is_read", false);
        }

        var mails = await query.GetAsync<MailInfo>();
        var mailList = mails.ToList();

        // 각 메일의 첨부 아이템 조회
        if (mailList.Any())
        {
            var mailIds = mailList.Select(m => m.MailId).ToArray();
            var attachments = await _db.Query("mail_attachments")
                .WhereIn("mail_id", mailIds)
                .GetAsync<MailAttachment>();

            var attachmentDict = attachments
                .GroupBy(a => a.MailId)
                .ToDictionary(g => g.Key, g => g.ToList());

            foreach (var mail in mailList)
            {
                if (attachmentDict.TryGetValue(mail.MailId, out var items))
                {
                    mail.Attachments = items;
                }
            }
        }

        return mailList;
    }

    // 메일 읽기
    public async Task<bool> ReadMailAsync(long userId, long mailId)
    {
        var affected = await _db.Query("mails")
            .Where("mail_id", mailId)
            .Where("receiver_id", userId)
            .Where("is_read", false)
            .UpdateAsync(new
            {
                is_read = true,
                read_at = DateTime.Now
            });

        return affected > 0;
    }

    // 메일 삭제
    public async Task<bool> DeleteMailAsync(long userId, long mailId)
    {
        // 보상이 미수령된 메일은 삭제 불가
        var mail = await _db.Query("mails")
            .Where("mail_id", mailId)
            .Where("receiver_id", userId)
            .FirstOrDefaultAsync<dynamic>();

        if (mail == null)
        {
            return false;
        }

        // 첨부 아이템이 있는데 수령하지 않았다면 삭제 불가
        var hasAttachments = await _db.Query("mail_attachments")
            .Where("mail_id", mailId)
            .ExistsAsync();

        if (hasAttachments && !mail.is_claimed)
        {
            throw new InvalidOperationException("첨부 아이템을 수령하지 않은 메일은 삭제할 수 없다.");
        }

        var affected = await _db.Query("mails")
            .Where("mail_id", mailId)
            .Where("receiver_id", userId)
            .DeleteAsync();

        return affected > 0;
    }
}

public class MailInfo
{
    public long MailId { get; set; }
    public long? SenderId { get; set; }
    public string SenderName { get; set; }
    public string Title { get; set; }
    public string Content { get; set; }
    public string MailType { get; set; }
    public bool IsRead { get; set; }
    public bool IsClaimed { get; set; }
    public DateTime CreatedAt { get; set; }
    public DateTime? ExpireAt { get; set; }
    public List<MailAttachment> Attachments { get; set; } = new List<MailAttachment>();
}

public class MailAttachment
{
    public long AttachmentId { get; set; }
    public long MailId { get; set; }
    public int ItemId { get; set; }
    public int Quantity { get; set; }
}
```

### 6.2.3 아이템 첨부 메일 보상 수령
첨부된 아이템을 수령하는 기능을 구현한다. 보상 수령은 트랜잭션으로 처리하여 데이터 일관성을 보장한다.

```csharp
public class MailRewardService
{
    private readonly QueryFactory _db;

    public MailRewardService(MySqlConnection connection)
    {
        var compiler = new SqlKata.Compilers.MySqlCompiler();
        _db = new QueryFactory(connection, compiler);
    }

    // 메일 보상 수령
    public async Task<List<MailAttachment>> ClaimMailRewardAsync(long userId, long mailId)
    {
        using var transaction = await _db.Connection.BeginTransactionAsync();
        
        try
        {
            // 메일 확인 및 잠금
            var mail = await _db.Query("mails")
                .Where("mail_id", mailId)
                .Where("receiver_id", userId)
                .Where("is_claimed", false)
                .ForUpdate()  // 행 잠금
                .FirstOrDefaultAsync<dynamic>();

            if (mail == null)
            {
                throw new InvalidOperationException("수령 가능한 메일이 없다.");
            }

            // 첨부 아이템 조회
            var attachments = await _db.Query("mail_attachments")
                .Where("mail_id", mailId)
                .GetAsync<MailAttachment>();

            var attachmentList = attachments.ToList();

            if (!attachmentList.Any())
            {
                throw new InvalidOperationException("첨부 아이템이 없다.");
            }

            // 인벤토리에 아이템 추가
            foreach (var attachment in attachmentList)
            {
                await AddItemToInventoryAsync(userId, attachment.ItemId, attachment.Quantity);
            }

            // 메일 수령 처리
            await _db.Query("mails")
                .Where("mail_id", mailId)
                .UpdateAsync(new
                {
                    is_claimed = true,
                    claimed_at = DateTime.Now
                });

            await transaction.CommitAsync();
            return attachmentList;
        }
        catch
        {
            await transaction.RollbackAsync();
            throw;
        }
    }

    // 인벤토리에 아이템 추가 (간단한 예시)
    private async Task AddItemToInventoryAsync(long userId, int itemId, int quantity)
    {
        // 기존 아이템이 있는지 확인
        var existing = await _db.Query("user_inventory")
            .Where("user_id", userId)
            .Where("item_id", itemId)
            .FirstOrDefaultAsync<dynamic>();

        if (existing != null)
        {
            // 수량 증가
            await _db.Query("user_inventory")
                .Where("user_id", userId)
                .Where("item_id", itemId)
                .IncrementAsync("quantity", quantity);
        }
        else
        {
            // 새 아이템 추가
            await _db.Query("user_inventory").InsertAsync(new
            {
                user_id = userId,
                item_id = itemId,
                quantity = quantity
            });
        }
    }

    // 모든 보상 일괄 수령
    public async Task<int> ClaimAllRewardsAsync(long userId)
    {
        var mails = await _db.Query("mails as m")
            .Join("mail_attachments as ma", "m.mail_id", "ma.mail_id")
            .Where("m.receiver_id", userId)
            .Where("m.is_claimed", false)
            .Select("m.mail_id")
            .Distinct()
            .GetAsync<dynamic>();

        var mailIds = mails.Select(m => (long)m.mail_id).ToList();
        var claimedCount = 0;

        foreach (var mailId in mailIds)
        {
            try
            {
                await ClaimMailRewardAsync(userId, mailId);
                claimedCount++;
            }
            catch
            {
                // 개별 실패는 무시하고 계속 진행
                continue;
            }
        }

        return claimedCount;
    }
}
```

## 6.3 던전 보상 시스템 구현
던전 클리어 시 보상을 지급하는 시스템을 구현한다. 확률형 보상과 고정 보상을 모두 처리한다.

### 6.3.1 던전 및 보상 테이블 설계
던전 정보, 클리어 기록, 보상 테이블을 설계한다.

```sql
-- 던전 정보 테이블
CREATE TABLE dungeons (
    dungeon_id INT AUTO_INCREMENT PRIMARY KEY,
    dungeon_name VARCHAR(100) NOT NULL,
    dungeon_level INT NOT NULL,
    entry_cost INT NOT NULL DEFAULT 0,  -- 입장 비용 (스태미나 등)
    max_daily_entries INT NOT NULL DEFAULT 3,  -- 일일 입장 제한
    is_active BOOLEAN NOT NULL DEFAULT TRUE,
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- 던전 보상 테이블
CREATE TABLE dungeon_rewards (
    reward_id INT AUTO_INCREMENT PRIMARY KEY,
    dungeon_id INT NOT NULL,
    item_id INT NOT NULL,
    min_quantity INT NOT NULL DEFAULT 1,
    max_quantity INT NOT NULL DEFAULT 1,
    drop_rate DECIMAL(5,2) NOT NULL,  -- 드롭 확률 (0.00 ~ 100.00)
    reward_type ENUM('FIXED', 'RANDOM') NOT NULL DEFAULT 'RANDOM',
    
    KEY idx_dungeon (dungeon_id),
    
    FOREIGN KEY (dungeon_id) REFERENCES dungeons(dungeon_id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- 던전 클리어 기록 테이블
CREATE TABLE dungeon_clears (
    clear_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT NOT NULL,
    dungeon_id INT NOT NULL,
    clear_time INT NOT NULL,  -- 클리어 시간 (초)
    score INT NOT NULL DEFAULT 0,
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    clear_date DATE NOT NULL,  -- 일일 입장 제한 체크용
    
    KEY idx_user_dungeon_date (user_id, dungeon_id, clear_date),
    KEY idx_dungeon_score (dungeon_id, score DESC),
    
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE,
    FOREIGN KEY (dungeon_id) REFERENCES dungeons(dungeon_id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- 획득한 보상 기록 테이블
CREATE TABLE dungeon_reward_logs (
    log_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    clear_id BIGINT NOT NULL,
    item_id INT NOT NULL,
    quantity INT NOT NULL,
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    
    KEY idx_clear (clear_id),
    KEY idx_user_item (user_id, item_id),
    
    FOREIGN KEY (clear_id) REFERENCES dungeon_clears(clear_id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
```

던전 보상 시스템의 흐름은 다음과 같다.

```
┌──────────────────────────────────────────────────────┐
│           Dungeon Reward System Flow                 │
└──────────────────────────────────────────────────────┘

1. 입장 체크
   ├─ 일일 입장 횟수 확인
   ├─ 입장 비용 차감
   └─ 던전 시작

2. 클리어
   ├─ 클리어 기록 저장
   └─ 보상 계산 시작

3. 보상 계산

       ├─ drop_rate와 비교

4. 보상 지급
   ├─ 인벤토리에 아이템 추가
   └─ 보상 로그 기록

5. 완료
```

### 6.3.2 던전 입장 및 클리어 처리
던전 입장 가능 여부를 확인하고 클리어를 처리한다.

```csharp
using SqlKata;
using SqlKata.Execution;
using MySqlConnector;
using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading.Tasks;

public class DungeonService
{
    private readonly QueryFactory _db;
    private readonly Random _random = new Random();

    public DungeonService(MySqlConnection connection)
    {
        var compiler = new SqlKata.Compilers.MySqlCompiler();
        _db = new QueryFactory(connection, compiler);
    }

    // 던전 입장 가능 여부 확인
    public async Task<bool> CanEnterDungeonAsync(long userId, int dungeonId)
    {
        // 던전 정보 조회
        var dungeon = await _db.Query("dungeons")
            .Where("dungeon_id", dungeonId)
            .Where("is_active", true)
            .FirstOrDefaultAsync<DungeonInfo>();

        if (dungeon == null)
        {
            throw new InvalidOperationException("존재하지 않거나 비활성화된 던전이다.");
        }

        // 오늘 입장 횟수 확인
        var today = DateTime.Today;
        var entryCount = await _db.Query("dungeon_clears")
            .Where("user_id", userId)
            .Where("dungeon_id", dungeonId)
            .Where("clear_date", today)
            .CountAsync<int>();

        if (entryCount >= dungeon.MaxDailyEntries)
        {
            return false;
        }

        // 입장 비용 확인 (스태미나 등)
        // 실제로는 user_resources 테이블을 확인
        var userStamina = await GetUserStaminaAsync(userId);
        if (userStamina < dungeon.EntryCost)
        {
            return false;
        }

        return true;
    }

    // 던전 클리어 처리
    public async Task<DungeonClearResult> ClearDungeonAsync(
        long userId, 
        int dungeonId, 
        int clearTime, 
        int score)
    {
        if (!await CanEnterDungeonAsync(userId, dungeonId))
        {
            throw new InvalidOperationException("던전에 입장할 수 없다.");
        }

        using var transaction = await _db.Connection.BeginTransactionAsync();
        
        try
        {
            // 입장 비용 차감
            var dungeon = await _db.Query("dungeons")
                .Where("dungeon_id", dungeonId)
                .FirstOrDefaultAsync<DungeonInfo>();

            await ConsumeStaminaAsync(userId, dungeon.EntryCost);

            // 클리어 기록 저장
            var clearId = await _db.Query("dungeon_clears").InsertGetIdAsync<long>(new
            {
                user_id = userId,
                dungeon_id = dungeonId,
                clear_time = clearTime,
                score = score,
                clear_date = DateTime.Today
            });

            // 보상 계산 및 지급
            var rewards = await CalculateAndGrantRewardsAsync(userId, dungeonId, clearId);

            await transaction.CommitAsync();

            return new DungeonClearResult
            {
                ClearId = clearId,
                Score = score,
                ClearTime = clearTime,
                Rewards = rewards
            };
        }
        catch
        {
            await transaction.RollbackAsync();
            throw;
        }
    }

    // 보상 계산 및 지급
    private async Task<List<RewardItem>> CalculateAndGrantRewardsAsync(
        long userId, 
        int dungeonId, 
        long clearId)
    {
        // 던전 보상 목록 조회
        var rewardConfigs = await _db.Query("dungeon_rewards")
            .Where("dungeon_id", dungeonId)
            .GetAsync<DungeonRewardConfig>();

        var grantedRewards = new List<RewardItem>();

        foreach (var config in rewardConfigs)
        {
            bool shouldGrant = false;
            int quantity = 0;

            if (config.RewardType == "FIXED")
            {
                // 고정 보상은 항상 지급
                shouldGrant = true;
                quantity = config.MinQuantity;
            }
            else // RANDOM
            {
                // 확률 체크
                var roll = _random.NextDouble() * 100;
                if (roll <= (double)config.DropRate)
                {
                    shouldGrant = true;
                    // 수량 랜덤 결정
                    quantity = _random.Next(config.MinQuantity, config.MaxQuantity + 1);
                }
            }

            if (shouldGrant && quantity > 0)
            {
                // 인벤토리에 아이템 추가
                await AddItemToInventoryAsync(userId, config.ItemId, quantity);

                // 보상 로그 기록
                await _db.Query("dungeon_reward_logs").InsertAsync(new
                {
                    clear_id = clearId,
                    item_id = config.ItemId,
                    quantity = quantity
                });

                grantedRewards.Add(new RewardItem
                {
                    ItemId = config.ItemId,
                    Quantity = quantity
                });
            }
        }

        return grantedRewards;
    }

    // 인벤토리에 아이템 추가
    private async Task AddItemToInventoryAsync(long userId, int itemId, int quantity)
    {
        var existing = await _db.Query("user_inventory")
            .Where("user_id", userId)
            .Where("item_id", itemId)
            .FirstOrDefaultAsync<dynamic>();

        if (existing != null)
        {
            await _db.Query("user_inventory")
                .Where("user_id", userId)
                .Where("item_id", itemId)
                .IncrementAsync("quantity", quantity);
        }
        else
        {
            await _db.Query("user_inventory").InsertAsync(new
            {
                user_id = userId,
                item_id = itemId,
                quantity = quantity
            });
        }
    }

    // 스태미나 조회 (간단한 예시)
    private async Task<int> GetUserStaminaAsync(long userId)
    {
        var result = await _db.Query("user_resources")
            .Where("user_id", userId)
            .Where("resource_type", "STAMINA")
            .FirstOrDefaultAsync<dynamic>();

        return result?.amount ?? 0;
    }

    // 스태미나 소비
    private async Task ConsumeStaminaAsync(long userId, int amount)
    {
        await _db.Query("user_resources")
            .Where("user_id", userId)
            .Where("resource_type", "STAMINA")
            .DecrementAsync("amount", amount);
    }
}
```

### 6.3.3 던전 통계 및 랭킹 조회
던전별 통계와 랭킹을 조회하는 기능을 구현한다.

```csharp
public class DungeonStatisticsService
{
    private readonly QueryFactory _db;

    public DungeonStatisticsService(MySqlConnection connection)
    {
        var compiler = new SqlKata.Compilers.MySqlCompiler();
        _db = new QueryFactory(connection, compiler);
    }

    // 유저의 던전별 클리어 통계
    public async Task<List<UserDungeonStats>> GetUserDungeonStatsAsync(long userId)
    {
        var stats = await _db.Query("dungeon_clears as dc")
            .Join("dungeons as d", "dc.dungeon_id", "d.dungeon_id")
            .Where("dc.user_id", userId)
            .GroupBy("dc.dungeon_id", "d.dungeon_name")
            .Select(
                "dc.dungeon_id",
                "d.dungeon_name",
                _db.Query().SelectRaw("COUNT(*) as clear_count"),
                _db.Query().SelectRaw("MAX(dc.score) as best_score"),
                _db.Query().SelectRaw("MIN(dc.clear_time) as best_time"),
                _db.Query().SelectRaw("AVG(dc.score) as avg_score")
            )
            .GetAsync<UserDungeonStats>();

        return stats.ToList();
    }

    // 던전별 랭킹 조회 (점수 기준)
    public async Task<List<DungeonRanking>> GetDungeonRankingAsync(int dungeonId, int limit = 100)
    {
        var rankings = await _db.Query("dungeon_clears as dc")
            .Join("users as u", "dc.user_id", "u.user_id")
            .Where("dc.dungeon_id", dungeonId)
            .GroupBy("dc.user_id", "u.username")
            .Select(
                "dc.user_id",
                "u.username",
                _db.Query().SelectRaw("MAX(dc.score) as best_score"),
                _db.Query().SelectRaw("MIN(dc.clear_time) as best_time"),
                _db.Query().SelectRaw("MAX(dc.created_at) as last_clear")
            )
            .OrderByDesc("best_score")
            .ThenBy("best_time")
            .Limit(limit)
            .GetAsync<DungeonRanking>();

        var rankingList = rankings.ToList();

        // 순위 부여
        for (int i = 0; i < rankingList.Count; i++)
        {
            rankingList[i].Rank = i + 1;
        }

        return rankingList;
    }

    // 오늘의 던전 클리어 현황
    public async Task<List<TodayDungeonStatus>> GetTodayDungeonStatusAsync(long userId)
    {
        var today = DateTime.Today;

        var status = await _db.Query("dungeons as d")
            .LeftJoin("dungeon_clears as dc", j => j
                .On("d.dungeon_id", "dc.dungeon_id")
                .On("dc.user_id", userId)
                .On("dc.clear_date", today))
            .Where("d.is_active", true)
            .GroupBy("d.dungeon_id", "d.dungeon_name", "d.max_daily_entries")
            .Select(
                "d.dungeon_id",
                "d.dungeon_name",
                "d.max_daily_entries",
                _db.Query().SelectRaw("COUNT(dc.clear_id) as today_clears"),
                _db.Query().SelectRaw("d.max_daily_entries - COUNT(dc.clear_id) as remaining_entries")
            )
            .GetAsync<TodayDungeonStatus>();

        return status.ToList();
    }

    // 보상 획득 통계
    public async Task<List<RewardStats>> GetRewardStatsAsync(long userId, int dungeonId)
    {
        var stats = await _db.Query("dungeon_reward_logs as drl")
            .Join("dungeon_clears as dc", "drl.clear_id", "dc.clear_id")
            .Where("dc.user_id", userId)
            .Where("dc.dungeon_id", dungeonId)
            .GroupBy("drl.item_id")
            .Select(
                "drl.item_id",
                _db.Query().SelectRaw("SUM(drl.quantity) as total_quantity"),
                _db.Query().SelectRaw("COUNT(DISTINCT drl.clear_id) as drop_count")
            )
            .GetAsync<RewardStats>();

        return stats.ToList();
    }
}
```

## Day 6 정리
Day 6에서는 실제 게임에서 핵심적으로 사용되는 세 가지 콘텐츠 시스템을 구현했다.

**친구 시스템**에서는 양방향 관계를 데이터베이스에서 효율적으로 관리하는 방법을 배웠다. 친구 요청, 수락, 거절의 상태 전환을 트랜잭션으로 안전하게 처리하고, JOIN을 활용하여 친구 목록을 조회하는 방법을 익혔다.

**우편함 시스템**에서는 메시지와 아이템 첨부를 분리하여 설계하고, 트랜잭션을 통해 보상 수령의 원자성을 보장하는 방법을 다뤘다. 만료 시간 처리와 읽음/안읽음 상태 관리도 함께 구현했다.

**던전 보상 시스템**에서는 일일 입장 제한, 확률형 보상 계산, 보상 로그 기록까지 종합적인 게임 콘텐츠 로직을 구현했다. 특히 확률 계산과 난수 생성을 통한 보상 지급은 많은 게임 콘텐츠에 응용할 수 있는 패턴이다.

이 세 가지 시스템은 대부분의 온라인 게임에서 기본적으로 제공하는 기능이며, 이를 확장하면 길드 시스템, 거래소 시스템, 레이드 시스템 등 더 복잡한 콘텐츠도 구현할 수 있다. 트랜잭션 처리, JOIN을 통한 데이터 조회, 인덱스 설계는 모든 게임 콘텐츠 구현의 기초가 된다.
