## Day 8-10: Result<T, E> & Option<T>

### 🎯 **C++ 예외처리 vs Rust Result의 철학적 차이**
C++에서 10년간 개발하면서 다음과 같은 예외처리의 문제점들을 경험해보았을 것이다:

```cpp
// C++: 예외의 숨겨진 위험들
class GameServerCpp {
public:
    // 예외를 던질 수 있는지 시그니처에서 알 수 없음
    User* authenticateUser(const std::string& token) {
        if (token.empty()) {
            throw std::invalid_argument("Empty token");        // 런타임에 발견
        }
        
        if (!validateToken(token)) {
            throw std::runtime_error("Invalid token");         // 예상치 못한 예외
        }
        
        User* user = database.findUser(token);
        if (!user) {
            throw std::runtime_error("User not found");        // 또 다른 예외
        }
        
        return user;  // 성공 시에만 도달
    }
    
    void problematicUsage() {
        try {
            User* user = authenticateUser("abc123");
            // 수많은 코드들...
            
            processUser(user);  // 여기서도 예외가 발생할 수 있음
            
        } catch (const std::invalid_argument& e) {
            std::cerr << "Argument error: " << e.what() << std::endl;
        } catch (const std::runtime_error& e) {
            std::cerr << "Runtime error: " << e.what() << std::endl;
        } catch (...) {
            std::cerr << "Unknown error occurred" << std::endl;    // 🚨 예상치 못한 예외
        }
    }
};
```

**C++ 예외처리의 문제점:**
1. **숨겨진 제어 흐름**: 함수가 예외를 던질 수 있는지 시그니처에서 알 수 없음
2. **성능 오버헤드**: 스택 언와인딩 비용
3. **예외 안전성**: 리소스 누수 위험
4. **강제성 부족**: 예외 처리를 강제하지 않음

### 🔥 **Rust의 혁신적 접근: Result<T, E>**
Rust는 이 모든 문제를 타입 시스템으로 해결한다:

```rust
// Rust: 명시적이고 안전한 오류 처리
use std::collections::HashMap;

#[derive(Debug, Clone)]
pub enum AuthError {
    EmptyToken,
    InvalidToken,
    UserNotFound,
    DatabaseError(String),
}

pub struct GameServer {
    users: HashMap<String, User>,
}

impl GameServer {
    // 반환 타입에서 실패 가능성이 명확히 드러남
    pub fn authenticate_user(&self, token: &str) -> Result<&User, AuthError> {
        if token.is_empty() {
            return Err(AuthError::EmptyToken);          // 명시적 오류 반환
        }
        
        if !self.validate_token(token) {
            return Err(AuthError::InvalidToken);        // 타입 안전한 오류
        }
        
        self.users.get(token)
            .ok_or(AuthError::UserNotFound)             // Option을 Result로 변환
    }
    
    pub fn safe_usage(&self) {
        match self.authenticate_user("abc123") {
            Ok(user) => {
                println!("Welcome, {}", user.id);
                // 성공 시 처리
            }
            Err(AuthError::EmptyToken) => {
                println!("Please provide a token");      // 구체적 오류 처리
            }
            Err(AuthError::InvalidToken) => {
                println!("Invalid token provided");
            }
            Err(AuthError::UserNotFound) => {
                println!("User not found");
            }
            Err(AuthError::DatabaseError(msg)) => {
                println!("Database error: {}", msg);
            }
        }
    }
}
```

// 실제 ChatServer 메서드에서의 활용
impl UserManager {

        self.session_map.insert(session_id, index);
        self.id_map.insert(id, index);

### 🔍 **? 연산자: 오류 전파의 마법**
C++에서는 오류를 상위로 전파하기 위해 복잡한 코드가 필요하다:

```cpp
// C++: 복잡한 오류 전파
class NetworkManagerCpp {
public:
    enum class ErrorCode { Success, ConnectionFailed, SendFailed, ReceiveFailed };
    
    ErrorCode sendMessage(const std::string& msg) {
        ErrorCode result = connect();
        if (result != ErrorCode::Success) {
            return result;  // 오류 전파
        }
        
        result = authenticate();
        if (result != ErrorCode::Success) {
            return result;  // 또 다른 오류 전파
        }
        
        result = send(msg);
        if (result != ErrorCode::Success) {
            return result;  // 계속 반복...
        }
        
        return ErrorCode::Success;
    }
};
```

Rust의 ? 연산자는 이를 극적으로 간소화한다:  

```rust
// Rust: ? 연산자로 간결한 오류 전파
#[derive(Debug)]
pub enum NetworkError {
    ConnectionFailed,
    AuthenticationFailed,
    SendFailed,
    ReceiveFailed,
}

impl NetworkManager {
    pub async fn send_message(&mut self, msg: &str) -> Result<(), NetworkError> {
        self.connect().await?;          // 실패시 즉시 반환
        self.authenticate().await?;     // 실패시 즉시 반환  
        self.send(msg).await?;          // 실패시 즉시 반환
        
        Ok(())  // 모든 단계 성공
    }
    
    async fn connect(&mut self) -> Result<(), NetworkError> {
        // 연결 로직...
        if self.is_connected {
            Ok(())
        } else {
            Err(NetworkError::ConnectionFailed)
        }
    }
    
    async fn authenticate(&self) -> Result<(), NetworkError> {
        // 인증 로직...
        if self.is_authenticated {
            Ok(())
        } else {
            Err(NetworkError::AuthenticationFailed)
        }
    }
    
    async fn send(&self, _msg: &str) -> Result<(), NetworkError> {
        // 전송 로직...
        if self.can_send {
            Ok(())
        } else {
            Err(NetworkError::SendFailed)
        }
    }
}
```

### 💡 **Option<T>: Null의 안전한 대안**
C++의 nullptr는 "Billion Dollar Mistake"라고 불린다. Rust는 이를 어떻게 해결했을까?  

```cpp
// C++: nullptr의 위험성
class UserStoreCpp {
    std::unordered_map<int, User> users;
    
public:
    User* getUser(int id) {
        auto it = users.find(id);
        return it != users.end() ? &it->second : nullptr;
    }
    
    void dangerousUsage() {
        User* user = getUser(999);  // nullptr 반환 가능
        
        // 수십 줄의 코드...
        
        std::cout << user->name;    // 💥 Segmentation Fault!
    }
};
```

```rust
// Rust: Option으로 안전하게
use std::collections::HashMap;

pub struct UserStore {
    users: HashMap<i32, User>,
}

impl UserStore {
    pub fn get_user(&self, id: i32) -> Option<&User> {
        self.users.get(&id)  // Option<&User> 반환
    }
    
    pub fn safe_usage(&self) {
        // 패턴 1: match 사용
        match self.get_user(999) {
            Some(user) => println!("User: {}", user.name),
            None => println!("User not found"),
        }
        
        // 패턴 2: if let 사용
        if let Some(user) = self.get_user(999) {
            println!("Found user: {}", user.name);
        } else {
            println!("User not found");
        }
        
        // 패턴 3: 함수형 접근
        let user_name = self.get_user(999)
            .map(|user| &user.name)                // Some이면 변환
            .unwrap_or(&"Unknown".to_string());    // None이면 기본값
        
        println!("User name: {}", user_name);
        
        // 패턴 4: 오류 전파 (?와 함께)
        let user = self.get_user(999).ok_or("User not found")?;
        println!("User: {}", user.name);
    }
}
```

// Rust: 트레이트 기반 추상화
use serde::{Serialize, Deserialize};

trait GameSerializable {
    fn to_bytes(&self) -> Vec<u8>;
    fn from_bytes(data: &[u8]) -> Result<Self, String> 
    where 
        Self: Sized;  // Self 타입의 크기를 컴파일 타임에 알아야 함
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct User {
    pub name: String,
    pub level: i32,
}

impl GameSerializable for User {
    fn to_bytes(&self) -> Vec<u8> {
        // serde_json을 활용한 직렬화
        serde_json::to_vec(self).unwrap_or_default()
    }
    
    fn from_bytes(data: &[u8]) -> Result<Self, String> {
        serde_json::from_slice(data)
            .map_err(|e| format!("Failed to deserialize User: {}", e))
    }
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Inventory {
    pub items: Vec<String>,
}

impl GameSerializable for Inventory {
    fn to_bytes(&self) -> Vec<u8> {
        serde_json::to_vec(self).unwrap_or_default()
    }
    
    fn from_bytes(data: &[u8]) -> Result<Self, String> {
        serde_json::from_slice(data)
            .map_err(|e| format!("Failed to deserialize Inventory: {}", e))
    }
}

pub fn serialize_packet<T: Serialize>(packet: &T) -> Result<Bytes, serde_json::Error> {
    let json = serde_json::to_vec(packet)?;  // Serialize 트레이트 활용
    Ok(Bytes::from(json))
}

pub fn deserialize_packet<T: for<'de> Deserialize<'de>>(
    data: &[u8]
) -> Result<T, serde_json::Error> {
    serde_json::from_slice(data)  // Deserialize 트레이트 활용
}

// 실제 사용 예시
impl ChatServer {
    async fn handle_login(&self, session_id: Uuid, body: Bytes) -> ServerResult<()> {
        // 제네릭 함수로 역직렬화
        let req: LoginReq = deserialize_packet(&body)
            .map_err(|_| ServerError::UnassignedError)?;
        
        let response = match self.user_manager.add_user(session_id, req.id).await {
            Ok(_) => LoginRes {
                error_code: ServerError::None.code(),
            },
            Err(e) => LoginRes {
                error_code: e.code(),
            },
        };

        // 제네릭 함수로 직렬화
        let body = serialize_packet(&response)
            .map_err(|_| ServerError::UnassignedError)?;
        let packet = Packet::new(PacketId::LoginInRes, body);
        
        self.network.send_packet(session_id, packet).await
            .map_err(|_| ServerError::UnassignedError)?;

        Ok(())
    }
}
