# 9장. 고급 해결 기법
복잡한 데이터 구조를 다룰 때 직접 참조만으로는 빌림 검사기를 만족시키기 어려운 경우가 많다. 이 장에서는 게임 엔진, 에디터, 시뮬레이터 같은 복잡한 시스템에서 사용되는 고급 기법들을 다룬다. 이 기법들은 빌림 규칙을 우회하지 않으면서도 복잡한 관계를 표현할 수 있게 해준다.

## 9.1 인덱스 타입으로 간접 참조하기
직접 참조 대신 인덱스를 사용하면 빌림 문제의 상당 부분을 해결할 수 있다. 이는 C++의 포인터 대신 배열 인덱스를 사용하는 것과 유사하지만, Rust에서는 타입 안전성까지 확보할 수 있다.

### 문제 상황: 그래프 구조의 직접 참조
먼저 직접 참조로 그래프를 구현하려는 시도를 보자.

```rust
// 컴파일되지 않는 코드
struct Node {
    value: i32,
    neighbors: Vec<&Node>, // 라이프타임 문제 발생
}

struct Graph {
    nodes: Vec<Node>,
}
```

이 코드는 라이프타임과 가변성 문제로 실용적이지 않다.

### 해결책: 인덱스 기반 접근

```rust
// 타입 안전 인덱스
#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
struct NodeId(usize);

#[derive(Debug)]
struct Node {
    value: i32,
    neighbors: Vec<NodeId>,
}

struct Graph {
    nodes: Vec<Node>,
}

impl Graph {
    fn new() -> Self {
        Graph { nodes: Vec::new() }
    }
    
    fn add_node(&mut self, value: i32) -> NodeId {
        let id = NodeId(self.nodes.len());
        self.nodes.push(Node {
            value,
            neighbors: Vec::new(),
        });
        id
    }
    
    fn add_edge(&mut self, from: NodeId, to: NodeId) {
        if let Some(node) = self.nodes.get_mut(from.0) {
            node.neighbors.push(to);
        }
    }
    
    fn get_node(&self, id: NodeId) -> Option<&Node> {
        self.nodes.get(id.0)
    }
    
    fn get_node_mut(&mut self, id: NodeId) -> Option<&mut Node> {
        self.nodes.get_mut(id.0)
    }
    
    // 노드와 그 이웃들을 동시에 접근
    fn process_node_with_neighbors(&mut self, id: NodeId) {
        let neighbor_ids: Vec<NodeId> = self.nodes
            .get(id.0)
            .map(|n| n.neighbors.clone())
            .unwrap_or_default();
        
        if let Some(node) = self.nodes.get_mut(id.0) {
            println!("Processing node {} with value {}", id.0, node.value);
            node.value += 1;
        }
        
        for neighbor_id in neighbor_ids {
            if let Some(neighbor) = self.nodes.get(neighbor_id.0) {
                println!("  Neighbor {} has value {}", neighbor_id.0, neighbor.value);
            }
        }
    }
}

fn main() {
    let mut graph = Graph::new();
    
    let n0 = graph.add_node(10);
    let n1 = graph.add_node(20);
    let n2 = graph.add_node(30);
    
    graph.add_edge(n0, n1);
    graph.add_edge(n0, n2);
    graph.add_edge(n1, n2);
    
    graph.process_node_with_neighbors(n0);
}
```

### 더 안전한 패턴: NewType과 제네릭
인덱스의 타입 안전성을 더 강화할 수 있다.

```rust
use std::marker::PhantomData;

// 제네릭 인덱스 타입
#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
struct Id<T> {
    index: usize,
    _phantom: PhantomData<T>,
}

impl<T> Id<T> {
    fn new(index: usize) -> Self {
        Id {
            index,
            _phantom: PhantomData,
        }
    }
    
    fn index(&self) -> usize {
        self.index
    }
}

// 각 엔티티 타입 정의
struct Entity {
    name: String,
    health: i32,
    inventory_id: Option<Id<Inventory>>,
}

struct Inventory {
    items: Vec<String>,
    owner_id: Id<Entity>,
}

struct World {
    entities: Vec<Entity>,
    inventories: Vec<Inventory>,
}

impl World {
    fn new() -> Self {
        World {
            entities: Vec::new(),
            inventories: Vec::new(),
        }
    }
    
    fn create_entity(&mut self, name: String, health: i32) -> Id<Entity> {
        let id = Id::new(self.entities.len());
        self.entities.push(Entity {
            name,
            health,
            inventory_id: None,
        });
        id
    }
    
    fn create_inventory(&mut self, owner_id: Id<Entity>) -> Id<Inventory> {
        let id = Id::new(self.inventories.len());
        self.inventories.push(Inventory {
            items: Vec::new(),
            owner_id,
        });
        
        if let Some(entity) = self.entities.get_mut(owner_id.index()) {
            entity.inventory_id = Some(id);
        }
        
        id
    }
    
    fn add_item_to_inventory(&mut self, inv_id: Id<Inventory>, item: String) {
        if let Some(inventory) = self.inventories.get_mut(inv_id.index()) {
            inventory.items.push(item);
        }
    }
    
    // 엔티티와 인벤토리를 함께 처리
    fn process_entity_with_inventory(&mut self, entity_id: Id<Entity>) {
        let inventory_id = self.entities
            .get(entity_id.index())
            .and_then(|e| e.inventory_id);
        
        if let Some(inv_id) = inventory_id {
            // 인벤토리 정보 읽기
            let item_count = self.inventories
                .get(inv_id.index())
                .map(|inv| inv.items.len())
                .unwrap_or(0);
            
            // 엔티티 수정
            if let Some(entity) = self.entities.get_mut(entity_id.index()) {
                println!("{} has {} items", entity.name, item_count);
                entity.health += item_count as i32;
            }
        }
    }
}

fn main() {
    let mut world = World::new();
    
    let player = world.create_entity("Player".to_string(), 100);
    let inventory = world.create_inventory(player);
    
    world.add_item_to_inventory(inventory, "Sword".to_string());
    world.add_item_to_inventory(inventory, "Potion".to_string());
    
    world.process_entity_with_inventory(player);
}
```

### 실전 예제: 엔티티-컴포넌트 시스템
게임 엔진에서 사용하는 ECS 패턴을 간단히 구현해본다.

```rust
use std::collections::HashMap;
use std::marker::PhantomData;

#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
struct EntityId(usize);

// 컴포넌트 타입 마커
trait Component: 'static {}

// 컴포넌트 저장소
struct ComponentStorage<T: Component> {
    components: HashMap<EntityId, T>,
}

impl<T: Component> ComponentStorage<T> {
    fn new() -> Self {
        ComponentStorage {
            components: HashMap::new(),
        }
    }
    
    fn insert(&mut self, entity: EntityId, component: T) {
        self.components.insert(entity, component);
    }
    
    fn get(&self, entity: EntityId) -> Option<&T> {
        self.components.get(&entity)
    }
    
    fn get_mut(&mut self, entity: EntityId) -> Option<&mut T> {
        self.components.get_mut(&entity)
    }
    
    fn remove(&mut self, entity: EntityId) -> Option<T> {
        self.components.remove(&entity)
    }
}

// 컴포넌트 정의
#[derive(Debug, Clone)]
struct Position {
    x: f32,
    y: f32,
}

impl Component for Position {}

#[derive(Debug, Clone)]
struct Velocity {
    dx: f32,
    dy: f32,
}

impl Component for Velocity {}

#[derive(Debug, Clone)]
struct Health {
    current: i32,
    max: i32,
}

impl Component for Health {}

// 월드
struct World {
    next_entity_id: usize,
    positions: ComponentStorage<Position>,
    velocities: ComponentStorage<Velocity>,
    healths: ComponentStorage<Health>,
}

impl World {
    fn new() -> Self {
        World {
            next_entity_id: 0,
            positions: ComponentStorage::new(),
            velocities: ComponentStorage::new(),
            healths: ComponentStorage::new(),
        }
    }
    
    fn create_entity(&mut self) -> EntityId {
        let id = EntityId(self.next_entity_id);
        self.next_entity_id += 1;
        id
    }
    
    fn add_position(&mut self, entity: EntityId, x: f32, y: f32) {
        self.positions.insert(entity, Position { x, y });
    }
    
    fn add_velocity(&mut self, entity: EntityId, dx: f32, dy: f32) {
        self.velocities.insert(entity, Velocity { dx, dy });
    }
    
    fn add_health(&mut self, entity: EntityId, current: i32, max: i32) {
        self.healths.insert(entity, Health { current, max });
    }
    
    // 시스템: 물리 업데이트
    fn update_physics(&mut self, dt: f32) {
        let entities_to_update: Vec<EntityId> = self.velocities
            .components
            .keys()
            .copied()
            .collect();
        
        for entity in entities_to_update {
            if let Some(vel) = self.velocities.get(entity) {
                let dx = vel.dx * dt;
                let dy = vel.dy * dt;
                
                if let Some(pos) = self.positions.get_mut(entity) {
                    pos.x += dx;
                    pos.y += dy;
                    println!("Entity {:?} moved to ({}, {})", entity, pos.x, pos.y);
                }
            }
        }
    }
    
    // 시스템: 충돌 감지
    fn check_collisions(&mut self) {
        let entities: Vec<EntityId> = self.positions
            .components
            .keys()
            .copied()
            .collect();
        
        for i in 0..entities.len() {
            for j in (i + 1)..entities.len() {
                let (e1, e2) = (entities[i], entities[j]);
                
                let collision = {
                    let pos1 = self.positions.get(e1);
                    let pos2 = self.positions.get(e2);
                    
                    if let (Some(p1), Some(p2)) = (pos1, pos2) {
                        let dx = p1.x - p2.x;
                        let dy = p1.y - p2.y;
                        let distance = (dx * dx + dy * dy).sqrt();
                        distance < 1.0 // 충돌 반경
                    } else {
                        false
                    }
                };
                
                if collision {
                    println!("Collision between {:?} and {:?}", e1, e2);
                    
                    // 체력 감소
                    if let Some(health) = self.healths.get_mut(e1) {
                        health.current -= 10;
                    }
                    if let Some(health) = self.healths.get_mut(e2) {
                        health.current -= 10;
                    }
                }
            }
        }
    }
}

fn main() {
    let mut world = World::new();
    
    let entity1 = world.create_entity();
    world.add_position(entity1, 0.0, 0.0);
    world.add_velocity(entity1, 1.0, 0.0);
    world.add_health(entity1, 100, 100);
    
    let entity2 = world.create_entity();
    world.add_position(entity2, 0.5, 0.0);
    world.add_velocity(entity2, -0.5, 0.0);
    world.add_health(entity2, 100, 100);
    
    for frame in 0..5 {
        println!("\n=== Frame {} ===", frame);
        world.update_physics(0.1);
        world.check_collisions();
    }
}
```

## 9.2 아레나 할당자 패턴
아레나(Arena) 패턴은 모든 객체를 하나의 컨테이너에 저장하고 인덱스로 참조하는 방식이다. 이를 통해 라이프타임 문제를 완전히 제거할 수 있다.

### 기본 아레나 구현

```rust
#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
struct Handle(usize);

struct Arena<T> {
    items: Vec<T>,
}

impl<T> Arena<T> {
    fn new() -> Self {
        Arena { items: Vec::new() }
    }
    
    fn insert(&mut self, item: T) -> Handle {
        let handle = Handle(self.items.len());
        self.items.push(item);
        handle
    }
    
    fn get(&self, handle: Handle) -> Option<&T> {
        self.items.get(handle.0)
    }
    
    fn get_mut(&mut self, handle: Handle) -> Option<&mut T> {
        self.items.get_mut(handle.0)
    }
    
    fn len(&self) -> usize {
        self.items.len()
    }
    
    // 여러 항목을 동시에 가변 참조
    fn get_two_mut(&mut self, h1: Handle, h2: Handle) -> Option<(&mut T, &mut T)> {
        if h1.0 == h2.0 {
            return None;
        }
        
        let (i1, i2) = (h1.0, h2.0);
        let (min, max) = if i1 < i2 { (i1, i2) } else { (i2, i1) };
        
        let (left, right) = self.items.split_at_mut(max);
        
        if i1 < i2 {
            Some((&mut left[i1], &mut right[0]))
        } else {
            Some((&mut right[0], &mut left[i2]))
        }
    }
}

// 사용 예제: 씬 그래프
#[derive(Debug)]
struct SceneNode {
    name: String,
    transform: Transform,
    parent: Option<Handle>,
    children: Vec<Handle>,
}

#[derive(Debug, Clone)]
struct Transform {
    position: (f32, f32, f32),
    rotation: (f32, f32, f32),
    scale: (f32, f32, f32),
}

impl Transform {
    fn identity() -> Self {
        Transform {
            position: (0.0, 0.0, 0.0),
            rotation: (0.0, 0.0, 0.0),
            scale: (1.0, 1.0, 1.0),
        }
    }
}

struct Scene {
    nodes: Arena<SceneNode>,
    root: Option<Handle>,
}

impl Scene {
    fn new() -> Self {
        Scene {
            nodes: Arena::new(),
            root: None,
        }
    }
    
    fn create_node(&mut self, name: String, parent: Option<Handle>) -> Handle {
        let handle = self.nodes.insert(SceneNode {
            name,
            transform: Transform::identity(),
            parent,
            children: Vec::new(),
        });
        
        if let Some(parent_handle) = parent {
            if let Some(parent_node) = self.nodes.get_mut(parent_handle) {
                parent_node.children.push(handle);
            }
        } else if self.root.is_none() {
            self.root = Some(handle);
        }
        
        handle
    }
    
    fn set_position(&mut self, handle: Handle, x: f32, y: f32, z: f32) {
        if let Some(node) = self.nodes.get_mut(handle) {
            node.transform.position = (x, y, z);
        }
    }
    
    fn get_world_position(&self, handle: Handle) -> (f32, f32, f32) {
        let mut pos = (0.0, 0.0, 0.0);
        let mut current = Some(handle);
        
        while let Some(h) = current {
            if let Some(node) = self.nodes.get(h) {
                pos.0 += node.transform.position.0;
                pos.1 += node.transform.position.1;
                pos.2 += node.transform.position.2;
                current = node.parent;
            } else {
                break;
            }
        }
        
        pos
    }
    
    fn print_hierarchy(&self, handle: Handle, depth: usize) {
        if let Some(node) = self.nodes.get(handle) {
            let indent = "  ".repeat(depth);
            println!("{}Node: {} at {:?}", indent, node.name, node.transform.position);
            
            for child in &node.children {
                self.print_hierarchy(*child, depth + 1);
            }
        }
    }
}

fn main() {
    let mut scene = Scene::new();
    
    let root = scene.create_node("Root".to_string(), None);
    scene.set_position(root, 0.0, 0.0, 0.0);
    
    let child1 = scene.create_node("Child1".to_string(), Some(root));
    scene.set_position(child1, 1.0, 0.0, 0.0);
    
    let child2 = scene.create_node("Child2".to_string(), Some(root));
    scene.set_position(child2, -1.0, 0.0, 0.0);
    
    let grandchild = scene.create_node("Grandchild".to_string(), Some(child1));
    scene.set_position(grandchild, 0.0, 1.0, 0.0);
    
    println!("Scene hierarchy:");
    scene.print_hierarchy(root, 0);
    
    println!("\nWorld position of grandchild: {:?}", 
             scene.get_world_position(grandchild));
}
```

### 삭제 기능이 있는 아레나

```rust
#[derive(Debug, Clone, Copy, PartialEq, Eq)]
enum Slot<T> {
    Occupied(T),
    Vacant,
}

struct ArenaWithDeletion<T> {
    slots: Vec<Slot<T>>,
    free_list: Vec<usize>,
}

impl<T> ArenaWithDeletion<T> {
    fn new() -> Self {
        ArenaWithDeletion {
            slots: Vec::new(),
            free_list: Vec::new(),
        }
    }
    
    fn insert(&mut self, item: T) -> Handle {
        if let Some(index) = self.free_list.pop() {
            self.slots[index] = Slot::Occupied(item);
            Handle(index)
        } else {
            let handle = Handle(self.slots.len());
            self.slots.push(Slot::Occupied(item));
            handle
        }
    }
    
    fn remove(&mut self, handle: Handle) -> Option<T> {
        if let Some(slot) = self.slots.get_mut(handle.0) {
            match std::mem::replace(slot, Slot::Vacant) {
                Slot::Occupied(item) => {
                    self.free_list.push(handle.0);
                    Some(item)
                }
                Slot::Vacant => None,
            }
        } else {
            None
        }
    }
    
    fn get(&self, handle: Handle) -> Option<&T> {
        match self.slots.get(handle.0)? {
            Slot::Occupied(item) => Some(item),
            Slot::Vacant => None,
        }
    }
    
    fn get_mut(&mut self, handle: Handle) -> Option<&mut T> {
        match self.slots.get_mut(handle.0)? {
            Slot::Occupied(item) => Some(item),
            Slot::Vacant => None,
        }
    }
    
    fn iter(&self) -> impl Iterator<Item = (Handle, &T)> {
        self.slots
            .iter()
            .enumerate()
            .filter_map(|(i, slot)| match slot {
                Slot::Occupied(item) => Some((Handle(i), item)),
                Slot::Vacant => None,
            })
    }
}

fn main() {
    let mut arena = ArenaWithDeletion::new();
    
    let h1 = arena.insert("First");
    let h2 = arena.insert("Second");
    let h3 = arena.insert("Third");
    
    println!("Before removal:");
    for (handle, item) in arena.iter() {
        println!("  {:?}: {}", handle, item);
    }
    
    arena.remove(h2);
    
    println!("\nAfter removing h2:");
    for (handle, item) in arena.iter() {
        println!("  {:?}: {}", handle, item);
    }
    
    let h4 = arena.insert("Fourth"); // h2의 슬롯 재사용
    
    println!("\nAfter adding h4:");
    for (handle, item) in arena.iter() {
        println!("  {:?}: {}", handle, item);
    }
}
```

## 9.3 슬롯맵과 세대별 인덱스
슬롯맵은 삭제된 핸들의 재사용을 감지할 수 있는 고급 아레나다. 세대(generation) 카운터를 사용해 해제 후 재사용(use-after-free) 버그를 방지한다.

### 슬롯맵 구현

```rust
#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
struct GenerationalHandle {
    index: usize,
    generation: u64,
}

#[derive(Debug)]
struct Slot<T> {
    value: Option<T>,
    generation: u64,
}

struct SlotMap<T> {
    slots: Vec<Slot<T>>,
    free_list: Vec<usize>,
}

impl<T> SlotMap<T> {
    fn new() -> Self {
        SlotMap {
            slots: Vec::new(),
            free_list: Vec::new(),
        }
    }
    
    fn insert(&mut self, value: T) -> GenerationalHandle {
        if let Some(index) = self.free_list.pop() {
            let slot = &mut self.slots[index];
            slot.generation += 1;
            slot.value = Some(value);
            
            GenerationalHandle {
                index,
                generation: slot.generation,
            }
        } else {
            let index = self.slots.len();
            self.slots.push(Slot {
                value: Some(value),
                generation: 0,
            });
            
            GenerationalHandle {
                index,
                generation: 0,
            }
        }
    }
    
    fn remove(&mut self, handle: GenerationalHandle) -> Option<T> {
        let slot = self.slots.get_mut(handle.index)?;
        
        if slot.generation == handle.generation {
            self.free_list.push(handle.index);
            slot.value.take()
        } else {
            None // 세대가 맞지 않음 - 이미 삭제된 핸들
        }
    }
    
    fn get(&self, handle: GenerationalHandle) -> Option<&T> {
        let slot = self.slots.get(handle.index)?;
        
        if slot.generation == handle.generation {
            slot.value.as_ref()
        } else {
            None
        }
    }
    
    fn get_mut(&mut self, handle: GenerationalHandle) -> Option<&mut T> {
        let slot = self.slots.get_mut(handle.index)?;
        
        if slot.generation == handle.generation {
            slot.value.as_mut()
        } else {
            None
        }
    }
    
    fn is_valid(&self, handle: GenerationalHandle) -> bool {
        self.slots
            .get(handle.index)
            .map(|slot| slot.generation == handle.generation)
            .unwrap_or(false)
    }
}

// 사용 예제
#[derive(Debug)]
struct Particle {
    position: (f32, f32),
    velocity: (f32, f32),
    lifetime: f32,
}

struct ParticleSystem {
    particles: SlotMap<Particle>,
}

impl ParticleSystem {
    fn new() -> Self {
        ParticleSystem {
            particles: SlotMap::new(),
        }
    }
    
    fn spawn(&mut self, x: f32, y: f32, dx: f32, dy: f32) -> GenerationalHandle {
        self.particles.insert(Particle {
            position: (x, y),
            velocity: (dx, dy),
            lifetime: 1.0,
        })
    }
    
    fn update(&mut self, dt: f32) -> Vec<GenerationalHandle> {
        let mut to_remove = Vec::new();
        
        // 모든 파티클의 핸들 수집
        let handles: Vec<GenerationalHandle> = self.particles
            .slots
            .iter()
            .enumerate()
            .filter_map(|(i, slot)| {
                if slot.value.is_some() {
                    Some(GenerationalHandle {
                        index: i,
                        generation: slot.generation,
                    })
                } else {
                    None
                }
            })
            .collect();
        
        for handle in handles {
            if let Some(particle) = self.particles.get_mut(handle) {
                particle.position.0 += particle.velocity.0 * dt;
                particle.position.1 += particle.velocity.1 * dt;
                particle.lifetime -= dt;
                
                if particle.lifetime <= 0.0 {
                    to_remove.push(handle);
                }
            }
        }
        
        for handle in &to_remove {
            self.particles.remove(*handle);
        }
        
        to_remove
    }
    
    fn get_position(&self, handle: GenerationalHandle) -> Option<(f32, f32)> {
        self.particles.get(handle).map(|p| p.position)
    }
}

fn main() {
    let mut system = ParticleSystem::new();
    
    let p1 = system.spawn(0.0, 0.0, 1.0, 0.0);
    let p2 = system.spawn(0.0, 0.0, 0.0, 1.0);
    
    println!("Created particles");
    
    for frame in 0..15 {
        let removed = system.update(0.1);
        
        println!("\nFrame {}", frame);
        if let Some(pos) = system.get_position(p1) {
            println!("  P1: {:?}", pos);
        } else {
            println!("  P1: expired");
        }
        
        if let Some(pos) = system.get_position(p2) {
            println!("  P2: {:?}", pos);
        } else {
            println!("  P2: expired");
        }
        
        if !removed.is_empty() {
            println!("  Removed {} particles", removed.len());
        }
    }
    
    // 삭제된 핸들 사용 시도
    println!("\nTrying to access removed particles:");
    println!("P1 valid: {}", system.particles.is_valid(p1));
    println!("P2 valid: {}", system.particles.is_valid(p2));
}
```

### 실전 예제: 리소스 관리자
게임 엔진의 텍스처, 메시, 사운드 같은 리소스를 관리하는 시스템이다.

```rust
use std::collections::HashMap;

#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
struct ResourceHandle {
    index: usize,
    generation: u64,
}

#[derive(Debug)]
enum ResourceData {
    Texture { width: u32, height: u32, data: Vec<u8> },
    Mesh { vertices: usize, indices: usize },
    Audio { duration: f32, channels: u8 },
}

#[derive(Debug)]
struct Resource {
    name: String,
    data: ResourceData,
    ref_count: usize,
}

struct ResourceManager {
    resources: SlotMap<Resource>,
    name_to_handle: HashMap<String, ResourceHandle>,
}

impl ResourceManager {
    fn new() -> Self {
        ResourceManager {
            resources: SlotMap::new(),
            name_to_handle: HashMap::new(),
        }
    }
    
    fn load_texture(&mut self, name: String, width: u32, height: u32) -> ResourceHandle {
        if let Some(&handle) = self.name_to_handle.get(&name) {
            if self.resources.is_valid(handle) {
                if let Some(res) = self.resources.get_mut(handle) {
                    res.ref_count += 1;
                }
                return handle;
            }
        }
        
        let data = vec![0u8; (width * height * 4) as usize];
        let handle = self.resources.insert(Resource {
            name: name.clone(),
            data: ResourceData::Texture { width, height, data },
            ref_count: 1,
        });
        
        self.name_to_handle.insert(name, handle);
        
        println!("Loaded texture: {:?}", handle);
        handle
    }
    
    fn acquire(&mut self, handle: ResourceHandle) -> bool {
        if let Some(res) = self.resources.get_mut(handle) {
            res.ref_count += 1;
            true
        } else {
            false
        }
    }
    
    fn release(&mut self, handle: ResourceHandle) {
        if let Some(res) = self.resources.get_mut(handle) {
            if res.ref_count > 0 {
                res.ref_count -= 1;
                
                if res.ref_count == 0 {
                    println!("Unloading resource: {}", res.name);
                    let name = res.name.clone();
                    self.resources.remove(handle);
                    self.name_to_handle.remove(&name);
                }
            }
        }
    }
    
    fn get_info(&self, handle: ResourceHandle) -> Option<String> {
        self.resources.get(handle).map(|res| {
            format!("{} (refs: {})", res.name, res.ref_count)
        })
    }
}

fn main() {
    let mut manager = ResourceManager::new();
    
    let tex1 = manager.load_texture("wall.png".to_string(), 256, 256);
    let tex2 = manager.load_texture("floor.png".to_string(), 512, 512);
    let tex3 = manager.load_texture("wall.png".to_string(), 256, 256); // 재사용
    
    println!("\nLoaded resources:");
    println!("  tex1: {:?}", manager.get_info(tex1));
    println!("  tex2: {:?}", manager.get_info(tex2));
    println!("  tex3: {:?}", manager.get_info(tex3));
    
    manager.release(tex1);
    println!("\nAfter releasing tex1:");
    println!("  tex1: {:?}", manager.get_info(tex1));
    
    manager.release(tex3);
    println!("\nAfter releasing tex3:");
    println!("  tex3 valid: {}", manager.resources.is_valid(tex3));
}
```
  

## 9.4 unsafe를 최소화하면서 복잡한 구조 만들기
때로는 `unsafe`가 필요하지만, 안전한 인터페이스로 감싸서 사용 범위를 최소화할 수 있다.

### 이중 연결 리스트 구현

```rust
use std::ptr::NonNull;
use std::marker::PhantomData;

struct Node<T> {
    data: T,
    prev: Option<NonNull<Node<T>>>,
    next: Option<NonNull<Node<T>>>,
}

pub struct LinkedList<T> {
    head: Option<NonNull<Node<T>>>,
    tail: Option<NonNull<Node<T>>>,
    len: usize,
    _marker: PhantomData<Box<Node<T>>>,
}

impl<T> LinkedList<T> {
    pub fn new() -> Self {
        LinkedList {
            head: None,
            tail: None,
            len: 0,
            _marker: PhantomData,
        }
    }
    
    pub fn push_back(&mut self, data: T) {
        let mut node = Box::new(Node {
            data,
            prev: self.tail,
            next: None,
        });
        
        let node_ptr = NonNull::new(Box::into_raw(node)).unwrap();
        
        match self.tail {
            None => self.head = Some(node_ptr),
            Some(mut tail) => unsafe {
                tail.as_mut().next = Some(node_ptr);
            },
        }
        
        self.tail = Some(node_ptr);
        self.len += 1;
    }
    
    pub fn push_front(&mut self, data: T) {
        let mut node = Box::new(Node {
            data,
            prev: None,
            next: self.head,
        });
        
        let node_ptr = NonNull::new(Box::into_raw(node)).unwrap();
        
        match self.head {
            None => self.tail = Some(node_ptr),
            Some(mut head) => unsafe {
                head.as_mut().prev = Some(node_ptr);
            },
        }
        
        self.head = Some(node_ptr);
        self.len += 1;
    }
    
    pub fn pop_front(&mut self) -> Option<T> {
        self.head.map(|node_ptr| {
            let node = unsafe { Box::from_raw(node_ptr.as_ptr()) };
            
            self.head = node.next;
            
            match self.head {
                None => self.tail = None,
                Some(mut head) => unsafe {
                    head.as_mut().prev = None;
                },
            }
            
            self.len -= 1;
            node.data
        })
    }
    
    pub fn len(&self) -> usize {
        self.len
    }
    
    pub fn is_empty(&self) -> bool {
        self.len == 0
    }
}

impl<T> Drop for LinkedList<T> {
    fn drop(&mut self) {
        while self.pop_front().is_some() {}
    }
}

// 안전한 반복자
pub struct Iter<'a, T> {
    current: Option<NonNull<Node<T>>>,
    _marker: PhantomData<&'a T>,
}

impl<T> LinkedList<T> {
    pub fn iter(&self) -> Iter<T> {
        Iter {
            current: self.head,
            _marker: PhantomData,
        }
    }
}

impl<'a, T> Iterator for Iter<'a, T> {
    type Item = &'a T;
    
    fn next(&mut self) -> Option<Self::Item> {
        self.current.map(|node_ptr| {
            let node = unsafe { node_ptr.as_ref() };
            self.current = node.next;
            &node.data
        })
    }
}

fn main() {
    let mut list = LinkedList::new();
    
    list.push_back(1);
    list.push_back(2);
    list.push_back(3);
    list.push_front(0);
    
    println!("List length: {}", list.len());
    
    println!("Elements:");
    for item in list.iter() {
        println!("  {}", item);
    }
    
    while let Some(val) = list.pop_front() {
        println!("Popped: {}", val);
    }
}
```

### 침입형 연결 리스트 (Intrusive List)
메모리 할당을 최소화하는 고성능 패턴이다.

```rust
use std::ptr::NonNull;
use std::marker::PhantomData;

// 연결 리스트 노드로 사용될 수 있는 타입의 트레이트
pub trait IntrusiveNode: Sized {
    fn links(&mut self) -> &mut Links<Self>;
}

pub struct Links<T> {
    prev: Option<NonNull<T>>,
    next: Option<NonNull<T>>,
}

impl<T> Links<T> {
    pub fn new() -> Self {
        Links {
            prev: None,
            next: None,
        }
    }
}

pub struct IntrusiveList<T: IntrusiveNode> {
    head: Option<NonNull<T>>,
    tail: Option<NonNull<T>>,
    len: usize,
    _marker: PhantomData<Box<T>>,
}

impl<T: IntrusiveNode> IntrusiveList<T> {
    pub fn new() -> Self {
        IntrusiveList {
            head: None,
            tail: None,
            len: 0,
            _marker: PhantomData,
        }
    }
    
    pub unsafe fn push_back(&mut self, ptr: NonNull<T>) {
        let links = ptr.as_ref().links() as *const Links<T> as *mut Links<T>;
        (*links).prev = self.tail;
        (*links).next = None;
        
        match self.tail {
            None => self.head = Some(ptr),
            Some(mut tail) => {
                let tail_links = tail.as_mut().links();
                tail_links.next = Some(ptr);
            },
        }
        
        self.tail = Some(ptr);
        self.len += 1;
    }
    
    pub fn len(&self) -> usize {
        self.len
    }
}

// 사용 예제
struct Task {
    id: u32,
    priority: u32,
    links: Links<Task>,
}

impl IntrusiveNode for Task {
    fn links(&mut self) -> &mut Links<Self> {
        &mut self.links
    }
}

impl Task {
    fn new(id: u32, priority: u32) -> Box<Self> {
        Box::new(Task {
            id,
            priority,
            links: Links::new(),
        })
    }
}

fn main() {
    let mut scheduler = IntrusiveList::new();
    
    let task1 = Task::new(1, 10);
    let task2 = Task::new(2, 20);
    let task3 = Task::new(3, 15);
    
    unsafe {
        scheduler.push_back(NonNull::new(Box::into_raw(task1)).unwrap());
        scheduler.push_back(NonNull::new(Box::into_raw(task2)).unwrap());
        scheduler.push_back(NonNull::new(Box::into_raw(task3)).unwrap());
    }
    
    println!("Scheduled {} tasks", scheduler.len());
}
```

### 안전한 래퍼로 unsafe 캡슐화

```rust
// 내부적으로 unsafe를 사용하지만 외부 API는 안전함
pub struct SafeVec<T> {
    ptr: NonNull<T>,
    len: usize,
    capacity: usize,
    _marker: PhantomData<T>,
}

impl<T> SafeVec<T> {
    pub fn new() -> Self {
        SafeVec {
            ptr: NonNull::dangling(),
            len: 0,
            capacity: 0,
            _marker: PhantomData,
        }
    }
    
    pub fn with_capacity(capacity: usize) -> Self {
        if capacity == 0 {
            return Self::new();
        }
        
        let layout = std::alloc::Layout::array::<T>(capacity).unwrap();
        let ptr = unsafe {
            let ptr = std::alloc::alloc(layout) as *mut T;
            NonNull::new(ptr).expect("Allocation failed")
        };
        
        SafeVec {
            ptr,
            len: 0,
            capacity,
            _marker: PhantomData,
        }
    }
    
    pub fn push(&mut self, value: T) {
        if self.len == self.capacity {
            self.grow();
        }
        
        unsafe {
            std::ptr::write(self.ptr.as_ptr().add(self.len), value);
        }
        self.len += 1;
    }
    
    pub fn get(&self, index: usize) -> Option<&T> {
        if index < self.len {
            unsafe { Some(&*self.ptr.as_ptr().add(index)) }
        } else {
            None
        }
    }
    
    pub fn len(&self) -> usize {
        self.len
    }
    
    fn grow(&mut self) {
        let new_capacity = if self.capacity == 0 {
            1
        } else {
            self.capacity * 2
        };
        
        let new_layout = std::alloc::Layout::array::<T>(new_capacity).unwrap();
        
        let new_ptr = if self.capacity == 0 {
            unsafe { std::alloc::alloc(new_layout) as *mut T }
        } else {
            let old_layout = std::alloc::Layout::array::<T>(self.capacity).unwrap();
            unsafe {
                std::alloc::realloc(
                    self.ptr.as_ptr() as *mut u8,
                    old_layout,
                    new_layout.size()
                ) as *mut T
            }
        };
        
        self.ptr = NonNull::new(new_ptr).expect("Reallocation failed");
        self.capacity = new_capacity;
    }
}

impl<T> Drop for SafeVec<T> {
    fn drop(&mut self) {
        if self.capacity > 0 {
            for i in 0..self.len {
                unsafe {
                    std::ptr::drop_in_place(self.ptr.as_ptr().add(i));
                }
            }
            
            let layout = std::alloc::Layout::array::<T>(self.capacity).unwrap();
            unsafe {
                std::alloc::dealloc(self.ptr.as_ptr() as *mut u8, layout);
            }
        }
    }
}

fn main() {
    let mut vec = SafeVec::new();
    
    vec.push(10);
    vec.push(20);
    vec.push(30);
    
    println!("Vector length: {}", vec.len());
    
    for i in 0..vec.len() {
        if let Some(val) = vec.get(i) {
            println!("  [{}] = {}", i, val);
        }
    }
}
```

이 장에서 다룬 기법들은 복잡한 데이터 구조를 안전하게 구현하는 방법을 보여준다. 인덱스 기반 참조는 가장 간단하면서도 효과적인 방법이고, 아레나와 슬롯맵은 성능과 안전성을 모두 확보할 수 있다. `unsafe`는 필요할 때만 사용하되, 반드시 안전한 인터페이스로 감싸야 한다. 이러한 패턴들을 익히면 Rust의 빌림 검사기와 싸우지 않고 협력하는 방법을 터득하게 된다.
