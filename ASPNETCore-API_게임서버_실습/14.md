# 14장. 캐시 전략 구현 (실습)

## 14.1 캐시 전략 개요

게임 서버에서 캐시는 성능 최적화의 핵심 요소다. 모든 데이터를 매번 MySQL에서 조회하면 응답 시간이 느려지고 데이터베이스에 부하가 집중된다. Redis를 활용한 캐싱 전략을 통해 읽기 성능을 크게 향상시킬 수 있다.

이 장에서는 수집형 RPG 게임에서 자주 사용되는 캐시 패턴들을 실습한다. 마스터 데이터 캐싱, Look-Aside 패턴, Write-Through 패턴을 구현하고, 적절한 만료 시간 설정과 캐시 무효화 전략을 학습한다.

## 14.2 마스터 데이터 캐싱

마스터 데이터는 게임의 기본 설정 정보로, 아이템 정보, 캐릭터 스탯, 던전 설정 등이 해당된다. 이 데이터는 자주 변경되지 않고 모든 사용자가 동일하게 참조하므로 캐싱하기에 이상적이다.

### 14.2.3 마스터 데이터 캐싱 서비스 구현

`Services/MasterDataCacheService.cs` 파일을 생성한다.

```csharp
using CloudStructures;
using CloudStructures.Structures;
using SqlKata.Execution;
using GameServer.Models;
using System.Text.Json;

namespace GameServer.Services
{
    public class MasterDataCacheService
    {
        private readonly QueryFactory _queryFactory;
        private readonly RedisConnection _redis;
        private readonly ILogger<MasterDataCacheService> _logger;

        // 캐시 키 상수
        private const string ITEM_MASTER_KEY = "master:items";
        private const string ITEM_MASTER_BY_ID_KEY = "master:item:{0}";

        // 캐시 만료 시간 (24시간)
        private static readonly TimeSpan CacheExpiration = TimeSpan.FromHours(24);

        public MasterDataCacheService(
            QueryFactory queryFactory,
            RedisConnection redis,
            ILogger<MasterDataCacheService> logger)
        {
            _queryFactory = queryFactory;
            _redis = redis;
            _logger = logger;
        }

        /// <summary>
        /// 모든 아이템 마스터 데이터 조회 (캐시 우선)
        /// </summary>
        public async Task<List<ItemMaster>> GetAllItemsAsync()
        {
            try
            {
                // 1. Redis 캐시에서 조회 시도
                var cache = new RedisString<string>(_redis, ITEM_MASTER_KEY, CacheExpiration);
                var cachedData = await cache.GetAsync();

                if (cachedData.HasValue)
                {
                    _logger.LogInformation("아이템 마스터 데이터를 캐시에서 조회");
                    var items = JsonSerializer.Deserialize<List<ItemMaster>>(cachedData.Value);
                    return items ?? new List<ItemMaster>();
                }

                // 2. 캐시 미스 시 MySQL에서 조회
                _logger.LogInformation("캐시 미스 - MySQL에서 아이템 마스터 데이터 조회");
                var itemsFromDb = await _queryFactory.Query("item_master")
                    .OrderBy("item_id")
                    .GetAsync<ItemMaster>();

                var itemList = itemsFromDb.ToList();

                // 3. 조회한 데이터를 Redis에 캐싱
                if (itemList.Any())
                {
                    var jsonData = JsonSerializer.Serialize(itemList);
                    await cache.SetAsync(jsonData);
                    _logger.LogInformation($"{itemList.Count}개의 아이템 마스터 데이터를 캐싱");
                }

                return itemList;
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "아이템 마스터 데이터 조회 실패");
                return new List<ItemMaster>();
            }
        }

        /// <summary>
        /// 특정 아이템 마스터 데이터 조회 (캐시 우선)
        /// </summary>
        public async Task<ItemMaster?> GetItemByIdAsync(int itemId)
        {
            try
            {
                // 1. Redis 캐시에서 조회 시도
                var cacheKey = string.Format(ITEM_MASTER_BY_ID_KEY, itemId);
                var cache = new RedisString<string>(_redis, cacheKey, CacheExpiration);
                var cachedData = await cache.GetAsync();

                if (cachedData.HasValue)
                {
                    _logger.LogInformation($"아이템 {itemId} 데이터를 캐시에서 조회");
                    return JsonSerializer.Deserialize<ItemMaster>(cachedData.Value);
                }

                // 2. 캐시 미스 시 MySQL에서 조회
                _logger.LogInformation($"캐시 미스 - MySQL에서 아이템 {itemId} 데이터 조회");
                var item = await _queryFactory.Query("item_master")
                    .Where("item_id", itemId)
                    .FirstOrDefaultAsync<ItemMaster>();

                // 3. 조회한 데이터를 Redis에 캐싱
                if (item != null)
                {
                    var jsonData = JsonSerializer.Serialize(item);
                    await cache.SetAsync(jsonData);
                    _logger.LogInformation($"아이템 {itemId} 데이터를 캐싱");
                }

                return item;
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, $"아이템 {itemId} 데이터 조회 실패");
                return null;
            }
        }

        /// <summary>
        /// 아이템 마스터 캐시 무효화
        /// </summary>
        public async Task<bool> InvalidateItemCacheAsync()
        {
            try
            {
                var cache = new RedisString<string>(_redis, ITEM_MASTER_KEY, null);
                await cache.DeleteAsync();
                _logger.LogInformation("아이템 마스터 캐시 무효화 완료");
                return true;
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "아이템 마스터 캐시 무효화 실패");
                return false;
            }
        }

        /// <summary>
        /// 특정 아이템 캐시 무효화
        /// </summary>
        public async Task<bool> InvalidateItemCacheAsync(int itemId)
        {
            try
            {
                var cacheKey = string.Format(ITEM_MASTER_BY_ID_KEY, itemId);
                var cache = new RedisString<string>(_redis, cacheKey, null);
                await cache.DeleteAsync();
                _logger.LogInformation($"아이템 {itemId} 캐시 무효화 완료");
                return true;
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, $"아이템 {itemId} 캐시 무효화 실패");
                return false;
            }
        }
    }
}
```

## 14.3 Look-Aside 캐시 패턴

Look-Aside 패턴은 가장 일반적인 캐시 패턴이다. 애플리케이션이 먼저 캐시를 확인하고, 캐시에 데이터가 없으면 데이터베이스에서 조회한 후 캐시에 저장한다.

### 14.3.1 사용자 프로필 캐싱 구현

`Services/UserCacheService.cs` 파일을 생성한다.

```csharp
using CloudStructures;
using CloudStructures.Structures;
using SqlKata.Execution;
using GameServer.Models;
using System.Text.Json;

namespace GameServer.Services
{
    public class UserCacheService
    {
        private readonly QueryFactory _queryFactory;
        private readonly RedisConnection _redis;
        private readonly ILogger<UserCacheService> _logger;

        private const string USER_PROFILE_KEY = "user:profile:{0}";
        private static readonly TimeSpan CacheExpiration = TimeSpan.FromMinutes(30);

        public UserCacheService(
            QueryFactory queryFactory,
            RedisConnection redis,
            ILogger<UserCacheService> logger)
        {
            _queryFactory = queryFactory;
            _redis = redis;
            _logger = logger;
        }

        /// <summary>
        /// Look-Aside 패턴을 사용한 사용자 프로필 조회
        /// </summary>
        public async Task<UserProfile?> GetUserProfileAsync(long userId)
        {
            try
            {
                // 1. 캐시 조회
                var cacheKey = string.Format(USER_PROFILE_KEY, userId);
                var cache = new RedisString<string>(_redis, cacheKey, CacheExpiration);
                var cachedData = await cache.GetAsync();

                if (cachedData.HasValue)
                {
                    _logger.LogInformation($"사용자 {userId} 프로필을 캐시에서 조회");
                    return JsonSerializer.Deserialize<UserProfile>(cachedData.Value);
                }

                // 2. 캐시 미스 - DB 조회
                _logger.LogInformation($"캐시 미스 - 사용자 {userId} 프로필을 DB에서 조회");
                var profile = await _queryFactory.Query("users")
                    .Select("user_id", "email", "nickname", "level", "exp", "gold", "created_at")
                    .Where("user_id", userId)
                    .FirstOrDefaultAsync<UserProfile>();

                // 3. 조회 결과를 캐시에 저장
                if (profile != null)
                {
                    var jsonData = JsonSerializer.Serialize(profile);
                    await cache.SetAsync(jsonData);
                    _logger.LogInformation($"사용자 {userId} 프로필을 캐싱");
                }

                return profile;
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, $"사용자 {userId} 프로필 조회 실패");
                return null;
            }
        }

        /// <summary>
        /// 캐시 무효화
        /// </summary>
        public async Task InvalidateUserProfileAsync(long userId)
        {
            try
            {
                var cacheKey = string.Format(USER_PROFILE_KEY, userId);
                var cache = new RedisString<string>(_redis, cacheKey, null);
                await cache.DeleteAsync();
                _logger.LogInformation($"사용자 {userId} 프로필 캐시 무효화");
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, $"사용자 {userId} 프로필 캐시 무효화 실패");
            }
        }
    }

    public class UserProfile
    {
        public long UserId { get; set; }
        public string Email { get; set; }
        public string Nickname { get; set; }
        public int Level { get; set; }
        public int Exp { get; set; }
        public int Gold { get; set; }
        public DateTime CreatedAt { get; set; }
    }
}
```

## 14.4 Write-Through 캐시 패턴

Write-Through 패턴은 데이터를 쓸 때 캐시와 데이터베이스를 동시에 업데이트한다. 이 방식은 캐시와 DB의 일관성을 보장하지만, 쓰기 지연이 발생할 수 있다.

### 14.4.1 사용자 경험치 업데이트 구현

`Services/UserExpService.cs` 파일을 생성한다.

```csharp
using CloudStructures;
using CloudStructures.Structures;
using SqlKata.Execution;
using System.Text.Json;
using GameServer.Models;

namespace GameServer.Services
{
    public class UserExpService
    {
        private readonly QueryFactory _queryFactory;
        private readonly RedisConnection _redis;
        private readonly UserCacheService _userCache;
        private readonly ILogger<UserExpService> _logger;

        private const string USER_EXP_KEY = "user:exp:{0}";
        private static readonly TimeSpan CacheExpiration = TimeSpan.FromMinutes(30);

        public UserExpService(
            QueryFactory queryFactory,
            RedisConnection redis,
            UserCacheService userCache,
            ILogger<UserExpService> logger)
        {
            _queryFactory = queryFactory;
            _redis = redis;
            _userCache = userCache;
            _logger = logger;
        }

        /// <summary>
        /// Write-Through 패턴을 사용한 경험치 추가
        /// </summary>
        public async Task<(bool success, int newLevel, int newExp)> AddExpAsync(long userId, int expAmount)
        {
            try
            {
                // 1. 현재 사용자 정보 조회
                var profile = await _userCache.GetUserProfileAsync(userId);
                if (profile == null)
                {
                    _logger.LogWarning($"사용자 {userId}를 찾을 수 없음");
                    return (false, 0, 0);
                }

                int currentLevel = profile.Level;
                int currentExp = profile.Exp;
                int newExp = currentExp + expAmount;
                int newLevel = currentLevel;

                // 2. 레벨업 계산 (간단한 예시: 100 경험치당 1레벨)
                while (newExp >= 100 && newLevel < 100)
                {
                    newExp -= 100;
                    newLevel++;
                }

                // 3. DB 업데이트
                var affectedRows = await _queryFactory.Query("users")
                    .Where("user_id", userId)
                    .UpdateAsync(new
                    {
                        level = newLevel,
                        exp = newExp,
                        updated_at = DateTime.Now
                    });

                if (affectedRows == 0)
                {
                    _logger.LogError($"사용자 {userId} 경험치 업데이트 실패");
                    return (false, 0, 0);
                }

                // 4. 캐시 업데이트 (Write-Through)
                profile.Level = newLevel;
                profile.Exp = newExp;

                var cacheKey = string.Format("user:profile:{0}", userId);
                var cache = new RedisString<string>(_redis, cacheKey, CacheExpiration);
                var jsonData = JsonSerializer.Serialize(profile);
                await cache.SetAsync(jsonData);

                _logger.LogInformation($"사용자 {userId} 경험치 업데이트 완료 - Level: {newLevel}, Exp: {newExp}");
                return (true, newLevel, newExp);
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, $"사용자 {userId} 경험치 추가 실패");
                return (false, 0, 0);
            }
        }

        /// <summary>
        /// 경험치 조회 (캐시 우선)
        /// </summary>
        public async Task<(int level, int exp)?> GetExpAsync(long userId)
        {
            try
            {
                var profile = await _userCache.GetUserProfileAsync(userId);
                if (profile == null)
                {
                    return null;
                }

                return (profile.Level, profile.Exp);
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, $"사용자 {userId} 경험치 조회 실패");
                return null;
            }
        }
    }
}
```

## 14.5 캐시 만료 시간 설정 전략

캐시 만료 시간은 데이터의 특성에 따라 다르게 설정해야 한다.

### 14.5.1 만료 시간 설정 가이드

`Services/CacheExpirationConfig.cs` 파일을 생성한다.

```csharp
namespace GameServer.Services

    /// 캐시 만료 시간 설정 상수

    public static class CacheExpirationConfig
    {

        public static readonly TimeSpan GameStateExpiration = TimeSpan.FromMinutes(10);

        // 랭킹 데이터 - 주기적으로 갱신
        public static readonly TimeSpan RankingExpiration = TimeSpan.FromMinutes(5);

        // 임시 데이터 (던전 입장 상태 등)
        public static readonly TimeSpan TemporaryDataExpiration = TimeSpan.FromMinutes(15);
```

### 14.5.2 동적 만료 시간 설정

특정 조건에 따라 만료 시간을 동적으로 조정할 수 있다.

```csharp
public class DynamicCacheService
{
    private readonly RedisConnection _redis;

    public DynamicCacheService(RedisConnection redis)
    {
        _redis = redis;
    }

    /// <summary>
    /// 데이터 중요도에 따른 만료 시간 계산
    /// </summary>
    private TimeSpan GetExpirationTime(string dataType, int importance)
    {
        return dataType switch
        {
            "master" => CacheExpirationConfig.MasterDataExpiration,
            "profile" when importance > 5 => TimeSpan.FromHours(1), // 중요 사용자는 더 오래 캐싱
            "profile" => CacheExpirationConfig.UserProfileExpiration,
            "ranking" => CacheExpirationConfig.RankingExpiration,
            _ => TimeSpan.FromMinutes(10)
        };
    }

    /// <summary>
    /// 시간대별 만료 시간 조정 (피크 시간에는 더 오래 캐싱)
    /// </summary>
    private TimeSpan GetTimeBasedExpiration(TimeSpan baseExpiration)
    {
        var hour = DateTime.Now.Hour;
        
        // 피크 시간 (저녁 6시 ~ 자정)
        if (hour >= 18 && hour < 24)
        {
            return TimeSpan.FromTicks(baseExpiration.Ticks * 2); // 2배 연장
        }
        
        return baseExpiration;
    }
}
```

## 14.6 캐시 무효화 전략

캐시된 데이터가 변경되었을 때 적절히 무효화하는 전략이 필요하다.

### 14.6.1 캐시 무효화 서비스 구현

`Services/CacheInvalidationService.cs` 파일을 생성한다.

```csharp
using CloudStructures;
using CloudStructures.Structures;

namespace GameServer.Services
{
    public class CacheInvalidationService
    {
        private readonly RedisConnection _redis;
        private readonly ILogger<CacheInvalidationService> _logger;

        public CacheInvalidationService(
            RedisConnection redis,
            ILogger<CacheInvalidationService> logger)
        {
            _redis = redis;
            _logger = logger;
        }

        /// <summary>
        /// 패턴 매칭으로 여러 캐시 한번에 삭제
        /// </summary>
        public async Task InvalidateCacheByPatternAsync(string pattern)
        {
            try
            {
                var db = _redis.GetConnection().GetDatabase();
                var server = _redis.GetConnection().GetServer(_redis.GetConnection().GetEndPoints()[0]);
                
                var keys = server.Keys(pattern: pattern);
                var keyArray = keys.ToArray();

                if (keyArray.Length > 0)
                {
                    await db.KeyDeleteAsync(keyArray);
                    _logger.LogInformation($"패턴 '{pattern}'에 매칭되는 {keyArray.Length}개 캐시 삭제");
                }
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, $"패턴 '{pattern}' 캐시 삭제 실패");
            }
        }

        /// <summary>
        /// 사용자 관련 모든 캐시 무효화
        /// </summary>
        public async Task InvalidateUserCachesAsync(long userId)
        {
            try
            {
                var patterns = new[]
                {
                    $"user:profile:{userId}",
                    $"user:exp:{userId}",
                    $"user:inventory:{userId}*",
                    $"user:characters:{userId}*",
                    $"user:mail:{userId}*"
                };

                foreach (var pattern in patterns)
                {
                    await InvalidateCacheByPatternAsync(pattern);
                }

                _logger.LogInformation($"사용자 {userId}의 모든 캐시 무효화 완료");
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, $"사용자 {userId} 캐시 무효화 실패");
            }
        }

        /// <summary>
        /// 마스터 데이터 캐시 전체 갱신
        /// </summary>
        public async Task RefreshMasterDataCacheAsync()
        {
            try
            {
                await InvalidateCacheByPatternAsync("master:*");
                _logger.LogInformation("마스터 데이터 캐시 전체 무효화 완료");
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "마스터 데이터 캐시 무효화 실패");
            }
        }

        /// <summary>
        /// 시간 기반 자동 캐시 정리
        /// </summary>
        public async Task CleanupExpiredCachesAsync()
        {
            try
            {
                // Redis는 자동으로 만료된 키를 제거하지만,
                // 명시적으로 정리가 필요한 경우 사용
                var db = _redis.GetConnection().GetDatabase();
                var server = _redis.GetConnection().GetServer(_redis.GetConnection().GetEndPoints()[0]);
                
                var keys = server.Keys(pattern: "temp:*");
                var now = DateTime.Now;
                int deletedCount = 0;

                foreach (var key in keys)
                {
                    var ttl = await db.KeyTimeToLiveAsync(key);
                    if (ttl == null || ttl.Value.TotalSeconds < 0)
                    {
                        await db.KeyDeleteAsync(key);
                        deletedCount++;
                    }
                }

                _logger.LogInformation($"만료된 임시 캐시 {deletedCount}개 정리 완료");
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "캐시 정리 실패");
            }
        }
    }
}
```

## 14.8 캐시 성능 모니터링

캐시의 히트율과 성능을 모니터링하는 기능을 구현한다.

### 14.8.1 캐시 통계 서비스

`Services/CacheStatisticsService.cs` 파일을 생성한다.

```csharp
using CloudStructures;
using CloudStructures.Structures;

namespace GameServer.Services
{
    public class CacheStatisticsService
    {
        private readonly RedisConnection _redis;
        private readonly ILogger<CacheStatisticsService> _logger;

        private const string CACHE_HIT_KEY = "stats:cache:hit";
        private const string CACHE_MISS_KEY = "stats:cache:miss";

        public CacheStatisticsService(
            RedisConnection redis,
            ILogger<CacheStatisticsService> logger)
        {
            _redis = redis;
            _logger = logger;
        }

        /// <summary>
        /// 캐시 히트 기록
        /// </summary>
        public async Task RecordCacheHitAsync()
        {
            try
            {
                var counter = new RedisString<long>(_redis, CACHE_HIT_KEY, null);
                await counter.IncrementAsync();
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "캐시 히트 기록 실패");
            }
        }

        /// <summary>
        /// 캐시 미스 기록
        /// </summary>
        public async Task RecordCacheMissAsync()
        {
            try
            {
                var counter = new RedisString<long>(_redis, CACHE_MISS_KEY, null);
                await counter.IncrementAsync();
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "캐시 미스 기록 실패");
            }
        }

        /// <summary>
        /// 캐시 히트율 조회
        /// </summary>
        public async Task<(long hits, long misses, double hitRate)> GetCacheStatisticsAsync()
        {
            try
            {
                var hitCounter = new RedisString<long>(_redis, CACHE_HIT_KEY, null);
                var missCounter = new RedisString<long>(_redis, CACHE_MISS_KEY, null);

                var hits = await hitCounter.GetAsync();
                var misses = await missCounter.GetAsync();

                long hitCount = hits.HasValue ? hits.Value : 0;
                long missCount = misses.HasValue ? misses.Value : 0;
                long total = hitCount + missCount;

                double hitRate = total > 0 ? (double)hitCount / total * 100 : 0;

                return (hitCount, missCount, hitRate);
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "캐시 통계 조회 실패");
                return (0, 0, 0);
            }
        }

        /// <summary>
        /// 캐시 통계 초기화
        /// </summary>
        public async Task ResetStatisticsAsync()
        {
            try
            {
                var hitCounter = new RedisString<long>(_redis, CACHE_HIT_KEY, null);
                var missCounter = new RedisString<long>(_redis, CACHE_MISS_KEY, null);

                await hitCounter.DeleteAsync();
                await missCounter.DeleteAsync();

                _logger.LogInformation("캐시 통계 초기화 완료");
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "캐시 통계 초기화 실패");
            }
        }
    }
}
```

## 14.9 정리

이번 장에서는 게임 서버에서 필수적인 캐시 전략들을 실습했다. 마스터 데이터 캐싱으로 불변 데이터를 효율적으로 관리하고, Look-Aside 패턴으로 읽기 성능을 최적화했다. Write-Through 패턴을 통해 데이터 일관성을 유지하면서 캐시를 업데이트하는 방법도 배웠다.

또한 데이터 특성에 맞는 캐시 만료 시간 설정, 캐시 무효화 전략, 캐시 워밍업, 성능 모니터링까지 실무에서 필요한 다양한 캐시 관리 기법을 구현했다.

다음 장에서는 이러한 캐시 전략을 활용하여 캐릭터 시스템을 구현하고, 인증된 사용자가 자신의 캐릭터를 안전하게 관리할 수 있도록 한다.
