# 15장. 캐릭터 시스템 (실습)

## 15.1 캐릭터 시스템 개요

수집형 RPG 게임의 핵심은 다양한 캐릭터를 수집하고 육성하는 것이다. 이번 장에서는 사용자가 여러 캐릭터를 소유하고 관리할 수 있는 시스템을 구현한다. 캐릭터 생성, 조회, 강화, 레벨업 등의 기능을 만들면서 인증된 사용자만 자신의 캐릭터를 관리할 수 있도록 권한 검증도 함께 구현한다.

캐릭터 시스템은 마스터 데이터와 사용자 소유 데이터로 분리하여 설계한다. 캐릭터 마스터 데이터는 모든 사용자가 공유하는 기본 정보이고, 사용자 캐릭터 데이터는 각 사용자가 소유한 캐릭터의 개별 정보다.

각 사용자가 소유한 캐릭터 정보를 저장하는 테이블을 생성한다.

```sql
CREATE TABLE user_characters (
    user_character_id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id BIGINT NOT NULL,
    character_id INT NOT NULL,
    level INT NOT NULL DEFAULT 1,
    exp INT NOT NULL DEFAULT 0,
    enhancement_level INT NOT NULL DEFAULT 0 COMMENT '강화 단계 (0~10)',
    hp INT NOT NULL,
    attack INT NOT NULL,
    defense INT NOT NULL,
    speed INT NOT NULL,
    is_main BOOLEAN DEFAULT FALSE COMMENT '대표 캐릭터 여부',
    acquired_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_user_id (user_id),
    INDEX idx_character_id (character_id),
    INDEX idx_user_main (user_id, is_main),
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE,
    FOREIGN KEY (character_id) REFERENCES character_master(character_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='사용자 소유 캐릭터';
```

## 15.4 캐릭터 마스터 캐싱 서비스

캐릭터 마스터 데이터는 자주 조회되므로 Redis에 캐싱한다.

`Services/CharacterMasterCacheService.cs` 파일을 생성한다.

```csharp
using CloudStructures;
using CloudStructures.Structures;
using SqlKata.Execution;
using GameServer.Models;
using System.Text.Json;

namespace GameServer.Services
{
    public class CharacterMasterCacheService
    {
        private readonly QueryFactory _queryFactory;
        private readonly RedisConnection _redis;
        private readonly ILogger<CharacterMasterCacheService> _logger;

        private const string CHAR_MASTER_ALL_KEY = "master:characters";
        private const string CHAR_MASTER_BY_ID_KEY = "master:character:{0}";
        private static readonly TimeSpan CacheExpiration = TimeSpan.FromHours(24);

        public CharacterMasterCacheService(
            QueryFactory queryFactory,
            RedisConnection redis,
            ILogger<CharacterMasterCacheService> logger)
        {
            _queryFactory = queryFactory;
            _redis = redis;
            _logger = logger;
        }

        /// <summary>
        /// 모든 캐릭터 마스터 데이터 조회
        /// </summary>
        public async Task<List<CharacterMaster>> GetAllCharactersAsync()
        {
            try
            {
                // 캐시 조회
                var cache = new RedisString<string>(_redis, CHAR_MASTER_ALL_KEY, CacheExpiration);
                var cachedData = await cache.GetAsync();

                if (cachedData.HasValue)
                {
                    _logger.LogInformation("캐릭터 마스터 데이터를 캐시에서 조회");
                    var characters = JsonSerializer.Deserialize<List<CharacterMaster>>(cachedData.Value);
                    return characters ?? new List<CharacterMaster>();
                }

                // DB 조회
                _logger.LogInformation("캐시 미스 - DB에서 캐릭터 마스터 데이터 조회");
                var charactersFromDb = await _queryFactory.Query("character_master")
                    .OrderBy("character_id")
                    .GetAsync<CharacterMaster>();

                var characterList = charactersFromDb.ToList();

                // 캐싱
                if (characterList.Any())
                {
                    var jsonData = JsonSerializer.Serialize(characterList);
                    await cache.SetAsync(jsonData);
                    _logger.LogInformation($"{characterList.Count}개의 캐릭터 마스터 데이터 캐싱 완료");
                }

                return characterList;
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "캐릭터 마스터 데이터 조회 실패");
                return new List<CharacterMaster>();
            }
        }

        /// <summary>
        /// 특정 캐릭터 마스터 데이터 조회
        /// </summary>
        public async Task<CharacterMaster?> GetCharacterByIdAsync(int characterId)
        {
            try
            {
                var cacheKey = string.Format(CHAR_MASTER_BY_ID_KEY, characterId);
                var cache = new RedisString<string>(_redis, cacheKey, CacheExpiration);
                var cachedData = await cache.GetAsync();

                if (cachedData.HasValue)
                {
                    return JsonSerializer.Deserialize<CharacterMaster>(cachedData.Value);
                }

                var character = await _queryFactory.Query("character_master")
                    .Where("character_id", characterId)
                    .FirstOrDefaultAsync<CharacterMaster>();

                if (character != null)
                {
                    var jsonData = JsonSerializer.Serialize(character);
                    await cache.SetAsync(jsonData);
                }

                return character;
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, $"캐릭터 {characterId} 마스터 데이터 조회 실패");
                return null;
            }
        }
    }
}
```

## 15.5 캐릭터 생성 API

### 15.5.1 캐릭터 서비스 구현

`Services/CharacterService.cs` 파일을 생성한다.

```csharp
using SqlKata.Execution;
using GameServer.Models;

namespace GameServer.Services
{
    public class CharacterService
    {
        private readonly QueryFactory _queryFactory;
        private readonly CharacterMasterCacheService _characterMasterCache;
        private readonly ILogger<CharacterService> _logger;

        public CharacterService(
            QueryFactory queryFactory,
            CharacterMasterCacheService characterMasterCache,
            ILogger<CharacterService> logger)
        {
            _queryFactory = queryFactory;
            _characterMasterCache = characterMasterCache;
            _logger = logger;
        }

        /// <summary>
        /// 캐릭터 생성 (뽑기 등으로 획득)
        /// </summary>
        public async Task<UserCharacter?> CreateCharacterAsync(long userId, int characterId)
        {
            try
            {
                // 마스터 데이터 조회
                var master = await _characterMasterCache.GetCharacterByIdAsync(characterId);
                if (master == null)
                {
                    _logger.LogWarning($"존재하지 않는 캐릭터 ID: {characterId}");
                    return null;
                }

                // 사용자가 캐릭터를 하나도 소유하지 않은 경우 대표 캐릭터로 설정
                var existingCount = await _queryFactory.Query("user_characters")
                    .Where("user_id", userId)
                    .CountAsync<int>();

                bool isMain = existingCount == 0;

                // 캐릭터 생성
                var newCharacterId = await _queryFactory.Query("user_characters")
                    .InsertGetIdAsync<long>(new
                    {
                        user_id = userId,
                        character_id = characterId,
                        level = 1,
                        exp = 0,
                        enhancement_level = 0,
                        hp = master.BaseHp,
                        attack = master.BaseAttack,
                        defense = master.BaseDefense,
                        speed = master.BaseSpeed,
                        is_main = isMain,
                        acquired_at = DateTime.Now
                    });

                // 생성된 캐릭터 조회
                var newCharacter = await _queryFactory.Query("user_characters")
                    .Where("user_character_id", newCharacterId)
                    .FirstOrDefaultAsync<UserCharacter>();

                _logger.LogInformation($"사용자 {userId}가 캐릭터 {characterId} 획득");
                return newCharacter;
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, $"캐릭터 생성 실패 - UserId: {userId}, CharacterId: {characterId}");
                return null;
            }
        }

        /// <summary>
        /// 특정 사용자 캐릭터 조회
        /// </summary>
        public async Task<UserCharacter?> GetUserCharacterAsync(long userId, long userCharacterId)
        {
            try
            {
                var character = await _queryFactory.Query("user_characters")
                    .Where("user_character_id", userCharacterId)
                    .Where("user_id", userId)
                    .FirstOrDefaultAsync<UserCharacter>();

                return character;
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, $"사용자 캐릭터 조회 실패 - UserCharacterId: {userCharacterId}");
                return null;
            }
        }

        /// <summary>
        /// 사용자의 모든 캐릭터 조회 (마스터 데이터 포함)
        /// </summary>
        public async Task<List<UserCharacterDetail>> GetUserCharactersAsync(long userId)
        {
            try
            {
                var characters = await _queryFactory.Query("user_characters as uc")
                    .Join("character_master as cm", "uc.character_id", "cm.character_id")
                    .Where("uc.user_id", userId)
                    .Select(
                        "uc.user_character_id",
                        "uc.user_id",
                        "uc.level",
                        "uc.exp",
                        "uc.enhancement_level",
                        "uc.hp",
                        "uc.attack",
                        "uc.defense",
                        "uc.speed",
                        "uc.is_main",
                        "uc.acquired_at",
                        "cm.character_id",
                        "cm.character_name",
                        "cm.character_type",
                        "cm.grade",
                        "cm.max_level",
                        "cm.description",
                        "cm.skill_description"
                    )
                    .OrderByDesc("uc.is_main")
                    .OrderByDesc("cm.grade")
                    .OrderByDesc("uc.level")
                    .GetAsync<UserCharacterDetail>();

                return characters.ToList();
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, $"사용자 캐릭터 목록 조회 실패 - UserId: {userId}");
                return new List<UserCharacterDetail>();
            }
        }
    }
}
```

```csharp
using Microsoft.AspNetCore.Mvc;
using GameServer.Services;

namespace GameServer.Controllers
{
    [ApiController]
    [Route("api/[controller]")]
    public class CharacterController : ControllerBase
    {
        private readonly CharacterService _characterService;
        private readonly CharacterMasterCacheService _characterMasterCache;
        private readonly ILogger<CharacterController> _logger;

        public CharacterController(
            CharacterService characterService,
            CharacterMasterCacheService characterMasterCache,
            ILogger<CharacterController> logger)
        {
            _characterService = characterService;
            _characterMasterCache = characterMasterCache;
            _logger = logger;
        }

        /// <summary>
        /// 캐릭터 마스터 데이터 전체 조회
        /// </summary>
        [HttpPost("master/list")]
        public async Task<GetCharacterMasterListResponse> GetCharacterMasterList()
        {
            var characters = await _characterMasterCache.GetAllCharactersAsync();
            return new GetCharacterMasterListResponse
            {
                Result = ErrorCode.None,
                Characters = characters.ToList()
            };
        }

        /// <summary>
        /// 캐릭터 생성 (뽑기 시뮬레이션)
        /// </summary>
        [HttpPost("create")]
        public async Task<CreateCharacterResponse> CreateCharacter([FromBody] CreateCharacterRequest request)
        {
            // TODO: 실제로는 인증 미들웨어를 통해 검증된 UserId 사용
            if (request.UserId <= 0 || request.CharacterId <= 0)
            {
                return new CreateCharacterResponse
                {
                    Result = ErrorCode.InvalidRequest,
                    Message = "유효하지 않은 요청"
                };
            }

            var character = await _characterService.CreateCharacterAsync(request.UserId, request.CharacterId);
            
            if (character == null)
            {
                return new CreateCharacterResponse
                {
                    Result = ErrorCode.CharacterCreateFailed,
                    Message = "캐릭터 생성 실패"
                };
            }

            return new CreateCharacterResponse
            {
                Result = ErrorCode.None,
                Message = "캐릭터 획득 완료",
                UserCharacterId = character.UserCharacterId,
                Character = character
            };
        }

        /// <summary>
        /// 사용자 소유 캐릭터 목록 조회
        /// </summary>
        [HttpPost("list")]
        public async Task<GetUserCharactersResponse> GetUserCharacters([FromBody] GetUserCharactersRequest request)
        {
            if (request.UserId <= 0)
            {
                return new GetUserCharactersResponse
                {
                    Result = ErrorCode.InvalidRequest,
                    Message = "유효하지 않은 사용자 ID"
                };
            }

            var characters = await _characterService.GetUserCharactersAsync(request.UserId);
            
            return new GetUserCharactersResponse
            {
                Result = ErrorCode.None,
                TotalCount = characters.Count,
                Characters = characters
            };
        }

        /// <summary>
        /// 특정 사용자 캐릭터 상세 조회
        /// </summary>
        [HttpPost("detail")]
        public async Task<GetCharacterDetailResponse> GetCharacterDetail([FromBody] GetCharacterDetailRequest request)
        {
            if (request.UserId <= 0 || request.UserCharacterId <= 0)
            {
                return new GetCharacterDetailResponse
                {
                    Result = ErrorCode.InvalidRequest,
                    Message = "유효하지 않은 요청"
                };
            }

            var character = await _characterService.GetUserCharacterAsync(request.UserId, request.UserCharacterId);
            
            if (character == null)
            {
                return new GetCharacterDetailResponse
                {
                    Result = ErrorCode.CharacterNotFound,
                    Message = "캐릭터를 찾을 수 없음"
                };
            }

            // 마스터 데이터도 함께 조회
            var master = await _characterMasterCache.GetCharacterByIdAsync(character.CharacterId);

            return new GetCharacterDetailResponse
            {
                Result = ErrorCode.None,
                Character = character,
                Master = master
            };
        }
    }

    public class CreateCharacterRequest
    {
        public long UserId { get; set; }
        public string Token { get; set; }
        public int CharacterId { get; set; }
    }

    public class GetUserCharactersRequest
    {
        public long UserId { get; set; }
        public string Token { get; set; }
    }

    public class GetCharacterDetailRequest
    {
        public long UserId { get; set; }
        public string Token { get; set; }
        public long UserCharacterId { get; set; }
    }

    public class GetCharacterMasterListResponse
    {

        public List<CharacterMaster> Characters { get; set; } = new();
    }

    public class CreateCharacterResponse
    {

        public UserCharacter? Character { get; set; }
    }

    public class GetUserCharactersResponse
    {
        public ErrorCode Result { get; set; }

        public int TotalCount { get; set; }

    }

    public class GetCharacterDetailResponse
    {
        public ErrorCode Result { get; set; }

        public UserCharacter? Character { get; set; }
```

### 15.6.1 레벨업 서비스 구현

`CharacterService.cs`에 레벨업 관련 메서드를 추가한다.

```csharp
/// <summary>
/// 캐릭터 경험치 추가 및 레벨업
/// </summary>
public async Task<(bool success, int newLevel, int newExp, bool leveledUp)> AddExpAsync(
    long userId, long userCharacterId, int expAmount)
{
    try
    {
        // 캐릭터 조회
        var character = await GetUserCharacterAsync(userId, userCharacterId);
        if (character == null)
        {
            _logger.LogWarning($"캐릭터를 찾을 수 없음 - UserCharacterId: {userCharacterId}");
            return (false, 0, 0, false);
        }

        // 마스터 데이터 조회 (최대 레벨 확인)
        var master = await _characterMasterCache.GetCharacterByIdAsync(character.CharacterId);
        if (master == null)
        {
            return (false, 0, 0, false);
        }

        // 이미 최대 레벨인 경우
        if (character.Level >= master.MaxLevel)
        {
            _logger.LogInformation($"캐릭터가 최대 레벨에 도달 - UserCharacterId: {userCharacterId}");
            return (true, character.Level, character.Exp, false);
        }

        int currentLevel = character.Level;
        int currentExp = character.Exp;
        int newExp = currentExp + expAmount;
        int newLevel = currentLevel;
        bool leveledUp = false;

        // 레벨업 계산 (레벨당 필요 경험치: 레벨 * 100)
        while (newLevel < master.MaxLevel)
        {
            int requiredExp = newLevel * 100;
            
            if (newExp >= requiredExp)
            {
                newExp -= requiredExp;
                newLevel++;
                leveledUp = true;
            }
            else
            {
                break;
            }
        }

        // 스탯 계산 (레벨업 시 기본 스탯 증가)
        int newHp = master.BaseHp + ((newLevel - 1) * 10);
        int newAttack = master.BaseAttack + ((newLevel - 1) * 2);
        int newDefense = master.BaseDefense + ((newLevel - 1) * 1);
        int newSpeed = master.BaseSpeed + ((newLevel - 1) * 1);

        // 강화 레벨에 따른 추가 스탯 적용
        newHp += character.EnhancementLevel * 15;
        newAttack += character.EnhancementLevel * 3;
        newDefense += character.EnhancementLevel * 2;
        newSpeed += character.EnhancementLevel * 1;

        // DB 업데이트
        var affectedRows = await _queryFactory.Query("user_characters")
            .Where("user_character_id", userCharacterId)
            .Where("user_id", userId)
            .UpdateAsync(new
            {
                level = newLevel,
                exp = newExp,
                hp = newHp,
                attack = newAttack,
                defense = newDefense,
                speed = newSpeed,
                updated_at = DateTime.Now
            });

        if (affectedRows == 0)
        {
            _logger.LogError($"캐릭터 경험치 업데이트 실패 - UserCharacterId: {userCharacterId}");
            return (false, 0, 0, false);
        }

        _logger.LogInformation(
            $"캐릭터 경험치 추가 - UserCharacterId: {userCharacterId}, Level: {currentLevel} -> {newLevel}");
        
        return (true, newLevel, newExp, leveledUp);
    }
    catch (Exception ex)
    {
        _logger.LogError(ex, $"캐릭터 경험치 추가 실패 - UserCharacterId: {userCharacterId}");
        return (false, 0, 0, false);
    }
}
```

```csharp
/// <summary>
/// 캐릭터 강화
/// </summary>
public async Task<(bool success, int newEnhancementLevel, int newHp, int newAttack, int newDefense, int newSpeed)> 
    EnhanceCharacterAsync(long userId, long userCharacterId)
{
    try
    {
        // 캐릭터 조회
        var character = await GetUserCharacterAsync(userId, userCharacterId);
        if (character == null)
        {
            return (false, 0, 0, 0, 0, 0);
        }

        // 최대 강화 레벨 체크
        const int MAX_ENHANCEMENT = 10;
        if (character.EnhancementLevel >= MAX_ENHANCEMENT)
        {
            _logger.LogWarning($"캐릭터가 최대 강화 레벨에 도달 - UserCharacterId: {userCharacterId}");
            return (false, character.EnhancementLevel, character.Hp, character.Attack, character.Defense, character.Speed);
        }

        // 강화 비용 계산 (강화 레벨에 따라 증가)
        int enhancementCost = (character.EnhancementLevel + 1) * 1000;

        // 사용자 골드 확인 및 차감
        var userGold = await _queryFactory.Query("users")
            .Where("user_id", userId)
            .Select("gold")
            .FirstOrDefaultAsync<int>();

        if (userGold < enhancementCost)
        {
            _logger.LogWarning($"골드 부족 - UserId: {userId}, Required: {enhancementCost}, Has: {userGold}");
            return (false, 0, 0, 0, 0, 0);
        }

        // 마스터 데이터 조회
        var master = await _characterMasterCache.GetCharacterByIdAsync(character.CharacterId);
        if (master == null)
        {
            return (false, 0, 0, 0, 0, 0);
        }

        // 강화 성공 확률 계산 (간단한 예시)
        Random random = new Random();
        int successRate = 100 - (character.EnhancementLevel * 5); // 강화 레벨이 높을수록 확률 감소
        bool isSuccess = random.Next(100) < successRate;

        if (!isSuccess)
        {
            // 강화 실패 시 골드만 차감
            await _queryFactory.Query("users")
                .Where("user_id", userId)
                .UpdateAsync(new { gold = userGold - enhancementCost });

            _logger.LogInformation($"캐릭터 강화 실패 - UserCharacterId: {userCharacterId}");
            return (false, character.EnhancementLevel, character.Hp, character.Attack, character.Defense, character.Speed);
        }

        // 강화 성공 시 스탯 재계산
        int newEnhancementLevel = character.EnhancementLevel + 1;

        // 레벨에 따른 기본 스탯
        int baseHp = master.BaseHp + ((character.Level - 1) * 10);
        int baseAttack = master.BaseAttack + ((character.Level - 1) * 2);
        int baseDefense = master.BaseDefense + ((character.Level - 1) * 1);
        int baseSpeed = master.BaseSpeed + ((character.Level - 1) * 1);

        // 강화에 따른 추가 스탯
        int newHp = baseHp + (newEnhancementLevel * 15);
        int newAttack = baseAttack + (newEnhancementLevel * 3);
        int newDefense = baseDefense + (newEnhancementLevel * 2);
        int newSpeed = baseSpeed + (newEnhancementLevel * 1);

        // 트랜잭션 시작
        using var transaction = _queryFactory.Connection.BeginTransaction();
        try
        {
            // 캐릭터 강화 레벨 및 스탯 업데이트
            await _queryFactory.Query("user_characters")
                .Where("user_character_id", userCharacterId)
                .Where("user_id", userId)
                .UpdateAsync(new
                {
                    enhancement_level = newEnhancementLevel,
                    hp = newHp,
                    attack = newAttack,
                    defense = newDefense,
                    speed = newSpeed,
                    updated_at = DateTime.Now
                });

            // 사용자 골드 차감
            await _queryFactory.Query("users")
                .Where("user_id", userId)
                .UpdateAsync(new { gold = userGold - enhancementCost });

            transaction.Commit();

            _logger.LogInformation(
                $"캐릭터 강화 성공 - UserCharacterId: {userCharacterId}, Enhancement: {newEnhancementLevel}");

            return (true, newEnhancementLevel, newHp, newAttack, newDefense, newSpeed);
        }
        catch
        {
            transaction.Rollback();
            throw;
        }
    }
    catch (Exception ex)
    {
        _logger.LogError(ex, $"캐릭터 강화 실패 - UserCharacterId: {userCharacterId}");
        return (false, 0, 0, 0, 0, 0);
    }
}
```
