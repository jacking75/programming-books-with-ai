# 16장. 아이템 및 인벤토리 시스템 (실습)

이번 장에서는 수집형 RPG 게임의 핵심 요소인 아이템과 인벤토리 시스템을 구현한다. 아이템 마스터 데이터를 관리하고, 사용자별 인벤토리를 데이터베이스와 Redis를 활용하여 효율적으로 처리하는 방법을 학습한다.

## 16.1 아이템 시스템 설계

아이템 시스템은 크게 두 가지 영역으로 나뉜다. 첫째는 게임에 존재하는 모든 아이템의 정보를 담은 마스터 데이터이고, 둘째는 각 사용자가 실제로 소유한 아이템 정보를 담은 인벤토리 데이터다.

마스터 데이터는 서버 시작 시 한 번 로딩하여 메모리에 캐싱하고, 인벤토리 데이터는 사용자별로 MySQL에 저장하되 Redis를 통해 빠르게 조회할 수 있도록 구성한다.

## 16.2 데이터베이스 테이블 설계

먼저 아이템 마스터 데이터 테이블을 생성한다. 이 테이블은 게임에 등장하는 모든 아이템의 기본 정보를 저장한다.

```sql
CREATE TABLE item_master (
    item_id INT PRIMARY KEY,
    item_name VARCHAR(100) NOT NULL,
    item_type TINYINT NOT NULL COMMENT '1:무기, 2:방어구, 3:소비, 4:재료',
    rarity TINYINT NOT NULL COMMENT '1:일반, 2:고급, 3:희귀, 4:영웅, 5:전설',
    max_stack INT NOT NULL DEFAULT 1,
    attack_power INT NOT NULL DEFAULT 0,
    defense_power INT NOT NULL DEFAULT 0,
    hp_recovery INT NOT NULL DEFAULT 0,
    sell_price INT NOT NULL DEFAULT 0,
    description VARCHAR(500),
    is_tradable BOOLEAN NOT NULL DEFAULT true,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

다음으로 사용자의 인벤토리 정보를 저장할 테이블을 생성한다.

```sql
CREATE TABLE user_inventory (
    inventory_id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id BIGINT NOT NULL,
    item_id INT NOT NULL,
    item_count INT NOT NULL DEFAULT 1,
    is_equipped BOOLEAN NOT NULL DEFAULT false,
    acquired_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_user_id (user_id),
    INDEX idx_user_item (user_id, item_id),
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE,
    FOREIGN KEY (item_id) REFERENCES item_master(item_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

마스터 데이터를 로딩하고 관리하는 서비스를 작성한다.

```csharp
// Services/MasterDataService.cs
using GameServer.Models;
using SqlKata.Execution;
using System.Collections.Concurrent;

namespace GameServer.Services
{
    public class MasterDataService
    {
        private readonly QueryFactory _queryFactory;
        private ConcurrentDictionary<int, ItemMaster> _itemMasterData;

        public MasterDataService(QueryFactory queryFactory)
        {
            _queryFactory = queryFactory;
            _itemMasterData = new ConcurrentDictionary<int, ItemMaster>();
        }

        public async Task LoadAllMasterDataAsync()
        {
            await LoadItemMasterDataAsync();
        }

        private async Task LoadItemMasterDataAsync()
        {
            var items = await _queryFactory.Query("item_master")
                .GetAsync<ItemMaster>();

            _itemMasterData.Clear();
            foreach (var item in items)
            {
                _itemMasterData.TryAdd(item.ItemId, item);
            }

            Console.WriteLine($"[MasterData] Loaded {_itemMasterData.Count} items");
        }

        public ItemMaster GetItem(int itemId)
        {
            _itemMasterData.TryGetValue(itemId, out var item);
            return item;
        }

        public bool IsValidItem(int itemId)
        {
            return _itemMasterData.ContainsKey(itemId);
        }

        public IEnumerable<ItemMaster> GetAllItems()
        {
            return _itemMasterData.Values;
        }
    }
}
```

Program.cs에서 서비스를 등록하고 서버 시작 시 마스터 데이터를 로딩한다.


## 16.5 인벤토리 Repository 구현

인벤토리 데이터를 MySQL에서 조회하고 수정하는 Repository를 구현한다.

```csharp
// Repositories/InventoryRepository.cs
using GameServer.Models;
using SqlKata.Execution;

namespace GameServer.Repositories
{
    public class InventoryRepository
    {
        private readonly QueryFactory _queryFactory;

        public InventoryRepository(QueryFactory queryFactory)
        {
            _queryFactory = queryFactory;
        }

        public async Task<List<UserInventory>> GetUserInventoryAsync(long userId)
        {
            var items = await _queryFactory.Query("user_inventory")
                .Where("user_id", userId)
                .OrderBy("inventory_id")
                .GetAsync<UserInventory>();

            return items.ToList();
        }

        public async Task<UserInventory> GetInventoryItemAsync(long userId, long inventoryId)
        {
            var item = await _queryFactory.Query("user_inventory")
                .Where("user_id", userId)
                .Where("inventory_id", inventoryId)
                .FirstOrDefaultAsync<UserInventory>();

            return item;
        }

        public async Task<UserInventory> GetInventoryByItemIdAsync(long userId, int itemId)
        {
            var item = await _queryFactory.Query("user_inventory")
                .Where("user_id", userId)
                .Where("item_id", itemId)
                .FirstOrDefaultAsync<UserInventory>();

            return item;
        }

        public async Task<long> AddItemAsync(long userId, int itemId, int itemCount)
        {
            var inventoryId = await _queryFactory.Query("user_inventory")
                .InsertGetIdAsync<long>(new
                {
                    user_id = userId,
                    item_id = itemId,
                    item_count = itemCount,
                    is_equipped = false,
                    acquired_at = DateTime.Now
                });

            return inventoryId;
        }

        public async Task<bool> UpdateItemCountAsync(long inventoryId, int itemCount)
        {
            var affected = await _queryFactory.Query("user_inventory")
                .Where("inventory_id", inventoryId)
                .UpdateAsync(new { item_count = itemCount });

            return affected > 0;
        }

        public async Task<bool> DeleteItemAsync(long inventoryId)
        {
            var affected = await _queryFactory.Query("user_inventory")
                .Where("inventory_id", inventoryId)
                .DeleteAsync();

            return affected > 0;
        }

        public async Task<bool> SetEquipStatusAsync(long userId, long inventoryId, bool isEquipped)
        {
            var affected = await _queryFactory.Query("user_inventory")
                .Where("user_id", userId)
                .Where("inventory_id", inventoryId)
                .UpdateAsync(new { is_equipped = isEquipped });

            return affected > 0;
        }

        public async Task<bool> UnequipItemsByTypeAsync(long userId, byte itemType)
        {
            // 특정 타입의 모든 장착 아이템을 해제한다
            var affected = await _queryFactory.Query("user_inventory")
                .Join("item_master", "user_inventory.item_id", "item_master.item_id")
                .Where("user_inventory.user_id", userId)
                .Where("item_master.item_type", itemType)
                .Where("user_inventory.is_equipped", true)
                .UpdateAsync(new { is_equipped = false });

            return affected > 0;
        }
    }
}
```

## 16.6 Redis를 이용한 인벤토리 캐싱

사용자의 인벤토리 데이터를 Redis에 캐싱하여 빠른 조회를 지원하는 서비스를 구현한다.

```csharp
// Services/InventoryCacheService.cs
using CloudStructures;
using CloudStructures.Structures;
using GameServer.Models;
using System.Text.Json;

namespace GameServer.Services
{
    public class InventoryCacheService
    {
        private readonly RedisConnection _redisConnection;
        private readonly TimeSpan _cacheExpiry = TimeSpan.FromMinutes(30);

        public InventoryCacheService(RedisConnection redisConnection)
        {
            _redisConnection = redisConnection;
        }

        private string GetInventoryCacheKey(long userId)
        {
            return $"inventory:{userId}";
        }

        public async Task<List<UserInventory>> GetInventoryAsync(long userId)
        {
            var redis = new RedisString<string>(_redisConnection, GetInventoryCacheKey(userId), _cacheExpiry);
            var cachedData = await redis.GetAsync();

            if (cachedData.HasValue)
            {
                return JsonSerializer.Deserialize<List<UserInventory>>(cachedData.Value);
            }

            return null;
        }

        public async Task SetInventoryAsync(long userId, List<UserInventory> inventory)
        {
            var redis = new RedisString<string>(_redisConnection, GetInventoryCacheKey(userId), _cacheExpiry);
            var json = JsonSerializer.Serialize(inventory);
            await redis.SetAsync(json, _cacheExpiry);
        }

        public async Task InvalidateInventoryAsync(long userId)
        {
            var redis = new RedisString<string>(_redisConnection, GetInventoryCacheKey(userId), _cacheExpiry);
            await redis.DeleteAsync();
        }

        public async Task UpdateItemInCacheAsync(long userId, UserInventory updatedItem)
        {
            var inventory = await GetInventoryAsync(userId);
            if (inventory == null)
            {
                return;
            }

            var existingItem = inventory.FirstOrDefault(i => i.InventoryId == updatedItem.InventoryId);
            if (existingItem != null)
            {
                inventory.Remove(existingItem);
            }

            if (updatedItem.ItemCount > 0)
            {
                inventory.Add(updatedItem);
            }

            await SetInventoryAsync(userId, inventory);
        }

        public async Task AddItemToCacheAsync(long userId, UserInventory newItem)
        {
            var inventory = await GetInventoryAsync(userId);
            if (inventory != null)
            {
                inventory.Add(newItem);
                await SetInventoryAsync(userId, inventory);
            }
        }
    }
}
```

## 16.7 인벤토리 서비스 구현

인벤토리 관련 비즈니스 로직을 처리하는 서비스를 구현한다.

```csharp
// Services/InventoryService.cs
using GameServer.DTOs;
using GameServer.Models;
using GameServer.Repositories;

namespace GameServer.Services
{
    public class InventoryService
    {
        private readonly InventoryRepository _inventoryRepository;
        private readonly InventoryCacheService _cacheService;
        private readonly MasterDataService _masterDataService;

        public InventoryService(
            InventoryRepository inventoryRepository,
            InventoryCacheService cacheService,
            MasterDataService masterDataService)
        {
            _inventoryRepository = inventoryRepository;
            _cacheService = cacheService;
            _masterDataService = masterDataService;
        }

        public async Task<GetInventoryResponse> GetInventoryAsync(long userId)
        {
            // 캐시에서 먼저 조회한다
            var cachedInventory = await _cacheService.GetInventoryAsync(userId);
            List<UserInventory> inventory;

            if (cachedInventory != null)
            {
                inventory = cachedInventory;
            }
            else
            {
                // 캐시 미스 시 DB에서 조회하고 캐싱한다
                inventory = await _inventoryRepository.GetUserInventoryAsync(userId);
                await _cacheService.SetInventoryAsync(userId, inventory);
            }

            var items = new List<InventoryItem>();
            foreach (var invItem in inventory)
            {
                var masterItem = _masterDataService.GetItem(invItem.ItemId);
                if (masterItem != null)
                {
                    items.Add(new InventoryItem
                    {
                        InventoryId = invItem.InventoryId,
                        ItemId = invItem.ItemId,
                        ItemName = masterItem.ItemName,
                        ItemType = masterItem.ItemType,
                        Rarity = masterItem.Rarity,
                        ItemCount = invItem.ItemCount,
                        IsEquipped = invItem.IsEquipped,
                        AttackPower = masterItem.AttackPower,
                        DefensePower = masterItem.DefensePower
                    });
                }
            }

            return new GetInventoryResponse
            {
                Result = 0,
                Items = items
            };
        }

        public async Task<AcquireItemResponse> AcquireItemAsync(long userId, int itemId, int itemCount)
        {
            // 아이템이 존재하는지 확인한다
            var masterItem = _masterDataService.GetItem(itemId);
            if (masterItem == null)
            {
                return new AcquireItemResponse { Result = 1 }; // 존재하지 않는 아이템
            }

            // 획득 수량이 유효한지 확인한다
            if (itemCount <= 0)
            {
                return new AcquireItemResponse { Result = 2 }; // 잘못된 수량
            }

            // 스택 가능 아이템인 경우 기존 인벤토리에서 찾는다
            var existingItem = await _inventoryRepository.GetInventoryByItemIdAsync(userId, itemId);

            long inventoryId;
            int totalCount;

            if (existingItem != null && masterItem.MaxStack > 1)
            {
                // 기존 아이템에 수량을 추가한다
                totalCount = existingItem.ItemCount + itemCount;

                // 최대 스택 수량을 초과하는지 확인한다
                if (totalCount > masterItem.MaxStack)
                {
                    return new AcquireItemResponse { Result = 3 }; // 최대 스택 초과
                }

                await _inventoryRepository.UpdateItemCountAsync(existingItem.InventoryId, totalCount);
                inventoryId = existingItem.InventoryId;

                // 캐시 업데이트
                existingItem.ItemCount = totalCount;
                await _cacheService.UpdateItemInCacheAsync(userId, existingItem);
            }
            else
            {
                // 새로운 아이템을 추가한다
                inventoryId = await _inventoryRepository.AddItemAsync(userId, itemId, itemCount);
                totalCount = itemCount;

                // 캐시에 추가
                var newItem = new UserInventory
                {
                    InventoryId = inventoryId,
                    UserId = userId,
                    ItemId = itemId,
                    ItemCount = itemCount,
                    IsEquipped = false,
                    AcquiredAt = DateTime.Now
                };
                await _cacheService.AddItemToCacheAsync(userId, newItem);
            }

            return new AcquireItemResponse
            {
                Result = 0,
                InventoryId = inventoryId,
                TotalCount = totalCount
            };
        }

        public async Task<UseItemResponse> UseItemAsync(long userId, long inventoryId, int useCount)
        {
            // 인벤토리 아이템을 조회한다
            var item = await _inventoryRepository.GetInventoryItemAsync(userId, inventoryId);
            if (item == null)
            {
                return new UseItemResponse { Result = 1 }; // 아이템을 찾을 수 없음
            }

            // 사용 수량이 유효한지 확인한다
            if (useCount <= 0 || useCount > item.ItemCount)
            {
                return new UseItemResponse { Result = 2 }; // 잘못된 수량
            }

            // 마스터 데이터에서 아이템 정보를 확인한다
            var masterItem = _masterDataService.GetItem(item.ItemId);
            if (masterItem == null)
            {
                return new UseItemResponse { Result = 3 }; // 존재하지 않는 아이템
            }

            // 소비 아이템만 사용 가능하다
            if (masterItem.ItemType != (byte)ItemType.Consumable)
            {
                return new UseItemResponse { Result = 4 }; // 사용할 수 없는 아이템
            }

            int remainingCount = item.ItemCount - useCount;

            if (remainingCount > 0)
            {
                // 수량을 감소시킨다
                await _inventoryRepository.UpdateItemCountAsync(inventoryId, remainingCount);
                item.ItemCount = remainingCount;
                await _cacheService.UpdateItemInCacheAsync(userId, item);
            }
            else
            {
                // 아이템을 삭제한다
                await _inventoryRepository.DeleteItemAsync(inventoryId);
                item.ItemCount = 0;
                await _cacheService.UpdateItemInCacheAsync(userId, item);
            }

            return new UseItemResponse
            {
                Result = 0,
                RemainingCount = remainingCount
            };
        }

        public async Task<EquipItemResponse> EquipItemAsync(long userId, long inventoryId)
        {
            // 인벤토리 아이템을 조회한다
            var item = await _inventoryRepository.GetInventoryItemAsync(userId, inventoryId);
            if (item == null)
            {
                return new EquipItemResponse { Result = 1 }; // 아이템을 찾을 수 없음
            }

            // 마스터 데이터에서 아이템 정보를 확인한다
            var masterItem = _masterDataService.GetItem(item.ItemId);
            if (masterItem == null)
            {
                return new EquipItemResponse { Result = 2 }; // 존재하지 않는 아이템
            }

            // 장착 가능한 아이템인지 확인한다
            if (masterItem.ItemType != (byte)ItemType.Weapon && masterItem.ItemType != (byte)ItemType.Armor)
            {
                return new EquipItemResponse { Result = 3 }; // 장착할 수 없는 아이템
            }

            // 이미 장착된 아이템인지 확인한다
            if (item.IsEquipped)
            {
                return new EquipItemResponse { Result = 4 }; // 이미 장착됨
            }

            // 같은 타입의 기존 장착 아이템을 해제한다
            await _inventoryRepository.UnequipItemsByTypeAsync(userId, masterItem.ItemType);

            // 아이템을 장착한다
            await _inventoryRepository.SetEquipStatusAsync(userId, inventoryId, true);

            // 캐시를 무효화한다
            await _cacheService.InvalidateInventoryAsync(userId);

            return new EquipItemResponse { Result = 0 };
        }

        public async Task<EquipItemResponse> UnequipItemAsync(long userId, long inventoryId)
        {
            // 인벤토리 아이템을 조회한다
            var item = await _inventoryRepository.GetInventoryItemAsync(userId, inventoryId);
            if (item == null)
            {
                return new EquipItemResponse { Result = 1 }; // 아이템을 찾을 수 없음
            }

            // 장착되지 않은 아이템인지 확인한다
            if (!item.IsEquipped)
            {
                return new EquipItemResponse { Result = 2 }; // 장착되지 않음
            }

            // 아이템을 해제한다
            await _inventoryRepository.SetEquipStatusAsync(userId, inventoryId, false);

            // 캐시를 무효화한다
            await _cacheService.InvalidateInventoryAsync(userId);

            return new EquipItemResponse { Result = 0 };
        }
    }
}
```

```csharp
    [Route("api/[controller]")]
```

## 16.12 소유권 검증의 중요성

인벤토리 시스템에서 가장 중요한 보안 요소는 아이템 소유권 검증이다. 모든 인벤토리 관련 API에서 요청한 사용자가 해당 아이템의 실제 소유자인지 반드시 확인해야 한다.

Repository의 모든 메서드에서 userId를 조건으로 포함하는 이유가 바로 이것이다. 만약 inventoryId만으로 아이템을 조회하거나 수정한다면, 악의적인 사용자가 다른 사람의 inventoryId를 추측하여 부정한 방법으로 아이템을 조작할 수 있다.

```csharp
// 올바른 방법 - userId와 inventoryId를 모두 확인한다
var item = await _queryFactory.Query("user_inventory")
    .Where("user_id", userId)
    .Where("inventory_id", inventoryId)
    .FirstOrDefaultAsync<UserInventory>();

// 잘못된 방법 - inventoryId만 확인한다 (보안 취약)
var item = await _queryFactory.Query("user_inventory")
    .Where("inventory_id", inventoryId)
    .FirstOrDefaultAsync<UserInventory>();
```
