# 22장. 랭킹 시스템 (실습)

## 22.1 개요

랭킹 시스템은 수집형 RPG 게임에서 사용자들의 경쟁심을 유발하고 게임 플레이를 촉진하는 핵심 기능이다. 이번 장에서는 Redis Sorted Set을 활용하여 실시간 랭킹 시스템을 구현하고, MySQL과 데이터를 동기화하는 방법을 학습한다.

랭킹 시스템의 주요 특징은 다음과 같다.

**실시간성**: 게임 내 점수가 변경되는 즉시 랭킹에 반영된다.

**확장성**: 수백만 명의 사용자 데이터를 효율적으로 처리한다.

**다양성**: 전체 랭킹, 친구 랭킹, 구간별 랭킹 등 다양한 형태를 제공한다.

Redis Sorted Set은 스코어 기반으로 자동 정렬되는 자료구조로, 랭킹 시스템 구현에 최적화되어 있다. O(log N) 시간 복잡도로 삽입, 조회, 순위 계산이 가능하여 대용량 데이터를 빠르게 처리한다.

## 22.2 랭킹 테이블 설계

먼저 MySQL에 랭킹 데이터를 저장할 테이블을 생성한다. 이 테이블은 주기적인 백업과 보상 지급을 위해 사용된다.

```sql
-- 전체 랭킹 테이블
CREATE TABLE user_ranking (
    user_id BIGINT NOT NULL,
    score BIGINT NOT NULL DEFAULT 0,
    ranking INT NOT NULL DEFAULT 0,
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (user_id),
    INDEX idx_score (score DESC),
    INDEX idx_ranking (ranking),
    INDEX idx_updated_at (updated_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- 랭킹 히스토리 테이블 (시즌별 기록)
CREATE TABLE ranking_history (
    id BIGINT AUTO_INCREMENT,
    season_id INT NOT NULL,
    user_id BIGINT NOT NULL,
    score BIGINT NOT NULL,
    ranking INT NOT NULL,
    reward_received TINYINT(1) NOT NULL DEFAULT 0,
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id),
    UNIQUE KEY uk_season_user (season_id, user_id),
    INDEX idx_season_ranking (season_id, ranking),
    INDEX idx_user_season (user_id, season_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- 랭킹 보상 마스터 데이터
CREATE TABLE ranking_reward_master (
    id INT AUTO_INCREMENT,
    season_id INT NOT NULL,
    rank_start INT NOT NULL,
    rank_end INT NOT NULL,
    reward_type VARCHAR(50) NOT NULL,
    reward_id INT NOT NULL,
    reward_count INT NOT NULL,
    PRIMARY KEY (id),
    INDEX idx_season_rank (season_id, rank_start, rank_end)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
```

테이블 구조를 살펴보면, `user_ranking` 테이블은 현재 시즌의 실시간 랭킹을 저장한다. `score` 컬럼에 내림차순 인덱스를 생성하여 빠른 조회를 지원한다. `ranking_history` 테이블은 시즌이 종료될 때 최종 랭킹을 기록하고, 보상 지급 여부를 관리한다. `ranking_reward_master` 테이블은 순위 구간별 보상 정보를 저장한다.

## 22.4 랭킹 서비스 구현

Redis Sorted Set을 활용한 랭킹 서비스를 구현한다.

```csharp
// Services/RankingService.cs
using CloudStructures;
using CloudStructures.Structures;
using SqlKata.Execution;
using System.Data;

namespace GameAPIServer.Services;

public class RankingService
{
    private readonly RedisConnection _redis;
    private readonly QueryFactory _queryFactory;
    private readonly ILogger<RankingService> _logger;
    
    // Redis 키 상수
    private const string RANKING_KEY = "ranking:global";
    private const string RANKING_SEASON_KEY_PREFIX = "ranking:season:";
    
    public RankingService(
        RedisConnection redis,
        QueryFactory queryFactory,
        ILogger<RankingService> logger)
    {
        _redis = redis;
        _queryFactory = queryFactory;
        _logger = logger;
    }
    
    // 랭킹 점수 업데이트
    public async Task<bool> UpdateScoreAsync(long userId, long score)
    {
        try
        {
            var sortedSet = new RedisSortedSet<long>(_redis, RANKING_KEY, null);
            
            // Sorted Set에 점수 업데이트 (이미 존재하면 갱신)
            var success = await sortedSet.AddAsync(userId, score);
            
            if (success)
            {
                // MySQL에도 비동기로 업데이트
                _ = Task.Run(async () =>
                {
                    try
                    {
                        await UpdateScoreInDatabaseAsync(userId, score);
                    }
                    catch (Exception ex)
                    {
                        _logger.LogError(ex, "MySQL 랭킹 업데이트 실패. UserId: {UserId}", userId);
                    }
                });
                
                _logger.LogInformation("랭킹 점수 업데이트 성공. UserId: {UserId}, Score: {Score}", userId, score);
                return true;
            }
            
            return false;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "랭킹 점수 업데이트 실패. UserId: {UserId}", userId);
            return false;
        }
    }
    
    // MySQL에 랭킹 점수 업데이트
    private async Task UpdateScoreInDatabaseAsync(long userId, long score)
    {
        var affectedRows = await _queryFactory.Query("user_ranking")
            .InsertAsync(new
            {
                user_id = userId,
                score = score,
                updated_at = DateTime.UtcNow
            });
        
        if (affectedRows == 0)
        {
            // 이미 존재하면 업데이트
            await _queryFactory.Query("user_ranking")
                .Where("user_id", userId)
                .UpdateAsync(new
                {
                    score = score,
                    updated_at = DateTime.UtcNow
                });
        }
    }
    
    // 특정 사용자의 랭킹 조회
    public async Task<int?> GetUserRankingAsync(long userId)
    {
        try
        {
            var sortedSet = new RedisSortedSet<long>(_redis, RANKING_KEY, null);
            
            // Sorted Set은 기본적으로 오름차순이므로 REVRANK 사용 (내림차순 순위)
            var rank = await sortedSet.RankAsync(userId, order: Order.Descending);
            
            if (rank.HasValue)
            {
                // 순위는 0부터 시작하므로 1을 더함
                return (int)rank.Value + 1;
            }
            
            return null;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "사용자 랭킹 조회 실패. UserId: {UserId}", userId);
            return null;
        }
    }
    
    // 전체 랭킹 조회 (상위 N명)
    public async Task<List<UserRankingInfo>> GetTopRankingsAsync(int count = 100)
    {
        try
        {
            var sortedSet = new RedisSortedSet<long>(_redis, RANKING_KEY, null);
            
            // 상위 count명의 데이터 조회 (점수 포함)
            var topUsers = await sortedSet.RangeByRankWithScoresAsync(0, count - 1, Order.Descending);
            
            if (topUsers == null || topUsers.Length == 0)
            {
                return new List<UserRankingInfo>();
            }
            
            // UserId 목록 추출
            var userIds = topUsers.Select(x => x.Value).ToList();
            
            // MySQL에서 사용자 정보 조회
            var users = await _queryFactory.Query("user_account")
                .WhereIn("user_id", userIds)
                .Select("user_id", "nickname")
                .GetAsync<dynamic>();
            
            var userDict = users.ToDictionary(
                u => (long)u.user_id,
                u => (string)u.nickname
            );
            
            // 랭킹 정보 생성
            var rankings = new List<UserRankingInfo>();
            for (int i = 0; i < topUsers.Length; i++)
            {
                var entry = topUsers[i];
                rankings.Add(new UserRankingInfo
                {
                    UserId = entry.Value,
                    Nickname = userDict.GetValueOrDefault(entry.Value, "Unknown"),
                    Score = (long)entry.Score,
                    Ranking = i + 1
                });
            }
            
            return rankings;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "전체 랭킹 조회 실패");
            return new List<UserRankingInfo>();
        }
    }
    
    // 특정 사용자 주변 랭킹 조회
    public async Task<List<UserRankingInfo>> GetRankingsAroundUserAsync(long userId, int range = 5)
    {
        try
        {
            var sortedSet = new RedisSortedSet<long>(_redis, RANKING_KEY, null);
            
            // 사용자의 현재 순위 조회
            var userRank = await sortedSet.RankAsync(userId, order: Order.Descending);
            
            if (!userRank.HasValue)
            {
                return new List<UserRankingInfo>();
            }
            
            // 사용자 위아래 range명씩 조회
            var startRank = Math.Max(0, userRank.Value - range);
            var endRank = userRank.Value + range;
            
            var rangeUsers = await sortedSet.RangeByRankWithScoresAsync(startRank, endRank, Order.Descending);
            
            if (rangeUsers == null || rangeUsers.Length == 0)
            {
                return new List<UserRankingInfo>();
            }
            
            var userIds = rangeUsers.Select(x => x.Value).ToList();
            
            var users = await _queryFactory.Query("user_account")
                .WhereIn("user_id", userIds)
                .Select("user_id", "nickname")
                .GetAsync<dynamic>();
            
            var userDict = users.ToDictionary(
                u => (long)u.user_id,
                u => (string)u.nickname
            );
            
            var rankings = new List<UserRankingInfo>();
            for (int i = 0; i < rangeUsers.Length; i++)
            {
                var entry = rangeUsers[i];
                rankings.Add(new UserRankingInfo
                {
                    UserId = entry.Value,
                    Nickname = userDict.GetValueOrDefault(entry.Value, "Unknown"),
                    Score = (long)entry.Score,
                    Ranking = (int)(startRank + i + 1)
                });
            }
            
            return rankings;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "사용자 주변 랭킹 조회 실패. UserId: {UserId}", userId);
            return new List<UserRankingInfo>();
        }
    }
    
    // 친구 랭킹 조회
    public async Task<List<UserRankingInfo>> GetFriendRankingsAsync(long userId, List<long> friendIds)
    {
        try
        {
            var sortedSet = new RedisSortedSet<long>(_redis, RANKING_KEY, null);
            
            // 자신과 친구들의 점수 조회
            var allUserIds = new List<long> { userId };
            allUserIds.AddRange(friendIds);
            
            var rankings = new List<UserRankingInfo>();
            
            foreach (var id in allUserIds)
            {
                var score = await sortedSet.ScoreAsync(id);
                var rank = await sortedSet.RankAsync(id, order: Order.Descending);
                
                if (score.HasValue && rank.HasValue)
                {
                    rankings.Add(new UserRankingInfo
                    {
                        UserId = id,
                        Score = (long)score.Value,
                        Ranking = (int)rank.Value + 1
                    });
                }
            }
            
            // 점수 기준 내림차순 정렬
            rankings = rankings.OrderByDescending(r => r.Score).ToList();
            
            // 사용자 정보 조회
            var users = await _queryFactory.Query("user_account")
                .WhereIn("user_id", allUserIds)
                .Select("user_id", "nickname")
                .GetAsync<dynamic>();
            
            var userDict = users.ToDictionary(
                u => (long)u.user_id,
                u => (string)u.nickname
            );
            
            // 닉네임 설정
            foreach (var ranking in rankings)
            {
                ranking.Nickname = userDict.GetValueOrDefault(ranking.UserId, "Unknown");
            }
            
            return rankings;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "친구 랭킹 조회 실패. UserId: {UserId}", userId);
            return new List<UserRankingInfo>();
        }
    }
}
```

`RankingService`는 Redis Sorted Set의 특성을 활용하여 효율적인 랭킹 조회를 제공한다. `UpdateScoreAsync` 메서드는 Redis에 점수를 업데이트하고, 백그라운드에서 MySQL에도 동기화한다. 이를 통해 빠른 응답 속도를 유지하면서도 데이터 안정성을 확보한다.

`GetTopRankingsAsync` 메서드는 상위 N명의 랭킹을 조회한다. `RangeByRankWithScoresAsync`를 사용하여 한 번의 Redis 호출로 필요한 데이터를 가져온다. `GetRankingsAroundUserAsync` 메서드는 특정 사용자 주변의 랭킹을 조회하여, 사용자가 자신의 위치를 쉽게 파악할 수 있도록 한다.

## 22.5 랭킹 API 구현

랭킹 조회 및 업데이트를 위한 API를 구현한다.

```csharp
// Controllers/RankingController.cs
using Microsoft.AspNetCore.Mvc;
using GameAPIServer.Models;
using GameAPIServer.Services;

namespace GameAPIServer.Controllers;

[ApiController]
[Route("api/[controller]")]
public class RankingController : ControllerBase
{
    private readonly RankingService _rankingService;
    private readonly FriendService _friendService;
    private readonly ILogger<RankingController> _logger;
    
    public RankingController(
        RankingService rankingService,
        FriendService friendService,
        ILogger<RankingController> logger)
    {
        _rankingService = rankingService;
        _friendService = friendService;
        _logger = logger;
    }
    
    // 랭킹 점수 업데이트
    [HttpPost("update")]
    public async Task<UpdateScoreResponse> UpdateScore([FromBody] UpdateScoreRequest request)
    {
        var userId = GetUserIdFromToken();
        if (userId == null)
        {
            return new UpdateScoreResponse { Result = ErrorCode.Unauthorized, Message = "인증 실패" };
        }
        
        if (request.Score < 0)
        {
            return new UpdateScoreResponse { Result = ErrorCode.InvalidRequest, Message = "점수는 0 이상이어야 한다" };
        }
        
        var success = await _rankingService.UpdateScoreAsync(userId.Value, request.Score);
        
        if (!success)
        {
            return new UpdateScoreResponse { Result = ErrorCode.RankingUpdateFailed, Message = "랭킹 점수 업데이트 실패" };
        }
        
        // 업데이트 후 현재 랭킹 조회
        var ranking = await _rankingService.GetUserRankingAsync(userId.Value);
        
        return new UpdateScoreResponse
        {
            Result = ErrorCode.None,
            Message = "랭킹 점수 업데이트 성공",
            Score = request.Score,
            Ranking = ranking ?? 0
        };
    }
    
    // 전체 랭킹 조회
    [HttpPost("top")]
    public async Task<GetTopRankingsResponse> GetTopRankings([FromBody] GetTopRankingsRequest request)
    {
        var userId = GetUserIdFromToken();
        if (userId == null)
        {
            return new GetTopRankingsResponse { Result = ErrorCode.Unauthorized, Message = "인증 실패" };
        }
        
        var count = Math.Min(request.Count, 100); // 최대 100명까지
        var rankings = await _rankingService.GetTopRankingsAsync(count);
        
        // 내 랭킹 정보 추가
        var myRanking = await _rankingService.GetUserRankingAsync(userId.Value);
        
        return new GetTopRankingsResponse
        {
            Result = ErrorCode.None,
            Rankings = rankings,
            MyRanking = myRanking ?? 0
        };
    }
    
    // 내 주변 랭킹 조회
    [HttpPost("around-me")]
    public async Task<GetRankingsAroundResponse> GetRankingsAroundMe([FromBody] GetRankingsAroundRequest request)
    {
        var userId = GetUserIdFromToken();
        if (userId == null)
        {
            return new GetRankingsAroundResponse { Result = ErrorCode.Unauthorized, Message = "인증 실패" };
        }
        
        var range = Math.Min(request.Range, 20); // 최대 위아래 20명까지
        var rankings = await _rankingService.GetRankingsAroundUserAsync(userId.Value, range);
        
        return new GetRankingsAroundResponse { Result = ErrorCode.None, Rankings = rankings };
    }
    
    // 친구 랭킹 조회
    [HttpPost("friends")]
    public async Task<GetFriendRankingsResponse> GetFriendRankings()
    {
        var userId = GetUserIdFromToken();
        if (userId == null)
        {
            return new GetFriendRankingsResponse { Result = ErrorCode.Unauthorized, Message = "인증 실패" };
        }
        
        // 친구 목록 조회
        var friends = await _friendService.GetFriendListAsync(userId.Value);
        var friendIds = friends.Select(f => f.FriendUserId).ToList();
        
        // 친구 랭킹 조회
        var rankings = await _rankingService.GetFriendRankingsAsync(userId.Value, friendIds);
        
        return new GetFriendRankingsResponse { Result = ErrorCode.None, Rankings = rankings };
    }
    
    // 내 랭킹 정보 조회
    [HttpPost("my-rank")]
    public async Task<GetMyRankingResponse> GetMyRanking()
    {
        var userId = GetUserIdFromToken();
        if (userId == null)
        {
            return new GetMyRankingResponse { Result = ErrorCode.Unauthorized, Message = "인증 실패" };
        }
        
        var ranking = await _rankingService.GetUserRankingAsync(userId.Value);
        
        if (ranking == null)
        {
            return new GetMyRankingResponse
            {
                Result = ErrorCode.None,
                Ranking = 0,
                Message = "랭킹 정보가 없다"
            };
        }
        
        return new GetMyRankingResponse { Result = ErrorCode.None, Ranking = ranking.Value };
    }
    
    private long? GetUserIdFromToken()
    {
        if (HttpContext.Items.TryGetValue("UserId", out var userIdObj))
        {
            return (long)userIdObj;
        }
        return null;
    }
}
```

API 컨트롤러는 다양한 랭킹 조회 기능을 제공한다. 전체 랭킹, 내 주변 랭킹, 친구 랭킹 등을 별도의 엔드포인트로 구분하여 클라이언트가 필요한 데이터만 요청할 수 있도록 한다.

## 22.6 시즌 랭킹 시스템

랭킹 시스템에 시즌 개념을 추가하여 주기적으로 랭킹을 초기화하고 보상을 지급하는 기능을 구현한다.

```csharp
// Services/RankingSeasonService.cs
using CloudStructures;
using CloudStructures.Structures;
using SqlKata.Execution;
using System.Transactions;

namespace GameAPIServer.Services;

public class RankingSeasonService
{
    private readonly RedisConnection _redis;
    private readonly QueryFactory _queryFactory;
    private readonly ILogger<RankingSeasonService> _logger;
    
    public RankingSeasonService(
        RedisConnection redis,
        QueryFactory queryFactory,
        ILogger<RankingSeasonService> logger)
    {
        _redis = redis;
        _queryFactory = queryFactory;
        _logger = logger;
    }
    
    // 시즌 종료 및 보상 지급
    public async Task<bool> EndSeasonAndDistributeRewardsAsync(int seasonId)
    {
        try
        {
            _logger.LogInformation("시즌 {SeasonId} 종료 및 보상 지급 시작", seasonId);
            
            // 1. 현재 랭킹 데이터를 히스토리에 저장
            var saved = await SaveRankingHistoryAsync(seasonId);
            if (!saved)
            {
                _logger.LogError("랭킹 히스토리 저장 실패");
                return false;
            }
            
            // 2. 보상 지급
            var distributed = await DistributeSeasonRewardsAsync(seasonId);
            if (!distributed)
            {
                _logger.LogError("시즌 보상 지급 실패");
                return false;
            }
            
            // 3. 현재 시즌 랭킹 초기화
            var cleared = await ClearCurrentSeasonRankingAsync();
            if (!cleared)
            {
                _logger.LogError("현재 시즌 랭킹 초기화 실패");
                return false;
            }
            
            _logger.LogInformation("시즌 {SeasonId} 종료 및 보상 지급 완료", seasonId);
            return true;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "시즌 종료 처리 실패. SeasonId: {SeasonId}", seasonId);
            return false;
        }
    }
    
    // 현재 랭킹을 히스토리에 저장
    private async Task<bool> SaveRankingHistoryAsync(int seasonId)
    {
        try
        {
            var sortedSet = new RedisSortedSet<long>(_redis, "ranking:global", null);
            
            // 전체 랭킹 데이터 조회
            var allUsers = await sortedSet.RangeByRankWithScoresAsync(0, -1, Order.Descending);
            
            if (allUsers == null || allUsers.Length == 0)
            {
                _logger.LogWarning("저장할 랭킹 데이터가 없다");
                return true;
            }
            
            var historyRecords = new List<object>();
            for (int i = 0; i < allUsers.Length; i++)
            {
                historyRecords.Add(new
                {
                    season_id = seasonId,
                    user_id = allUsers[i].Value,
                    score = (long)allUsers[i].Score,
                    ranking = i + 1,
                    reward_received = false,
                    created_at = DateTime.UtcNow
                });
            }
            
            // 배치 삽입
            await _queryFactory.Query("ranking_history")
                .InsertAsync(historyRecords);
            
            _logger.LogInformation("{Count}개의 랭킹 히스토리 저장 완료", historyRecords.Count);
            return true;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "랭킹 히스토리 저장 실패");
            return false;
        }
    }
    
    // 시즌 보상 지급
    private async Task<bool> DistributeSeasonRewardsAsync(int seasonId)
    {
        try
        {
            // 보상 마스터 데이터 조회
            var rewardConfigs = await _queryFactory.Query("ranking_reward_master")
                .Where("season_id", seasonId)
                .OrderBy("rank_start")
                .GetAsync<RankingReward>();
            
            if (!rewardConfigs.Any())
            {
                _logger.LogWarning("시즌 {SeasonId}에 대한 보상 설정이 없다", seasonId);
                return true;
            }
            
            // 보상 대상자 조회
            var recipients = await _queryFactory.Query("ranking_history")
                .Where("season_id", seasonId)
                .Where("reward_received", false)
                .OrderBy("ranking")
                .GetAsync<RankingHistory>();
            
            var rewardCount = 0;
            
            foreach (var recipient in recipients)
            {
                // 해당 순위에 맞는 보상 찾기
                var reward = rewardConfigs.FirstOrDefault(r =>
                    recipient.Ranking >= r.RankStart &&
                    recipient.Ranking <= r.RankEnd);
                
                if (reward == null)
                {
                    continue;
                }
                
                // 우편함으로 보상 발송
                var mailSent = await SendRewardMailAsync(
                    recipient.UserId,
                    seasonId,
                    recipient.Ranking,
                    reward);
                
                if (mailSent)
                {
                    // 보상 지급 완료 표시
                    await _queryFactory.Query("ranking_history")
                        .Where("season_id", seasonId)
                        .Where("user_id", recipient.UserId)
                        .UpdateAsync(new { reward_received = true });
                    
                    rewardCount++;
                }
            }
            
            _logger.LogInformation("{Count}명에게 시즌 보상 지급 완료", rewardCount);
            return true;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "시즌 보상 지급 실패");
            return false;
        }
    }
    
    // 보상 우편 발송
    private async Task<bool> SendRewardMailAsync(
        long userId,
        int seasonId,
        int ranking,
        RankingReward reward)
    {
        try
        {
            var mailId = await _queryFactory.Query("mail")
                .InsertGetIdAsync<long>(new
                {
                    receiver_id = userId,
                    sender_name = "시스템",
                    title = $"시즌 {seasonId} 랭킹 보상",
                    content = $"축하한다! {ranking}등을 달성하여 보상을 받는다.",
                    item_id = reward.RewardId,
                    item_count = reward.RewardCount,
                    has_item = true,
                    is_read = false,
                    is_received = false,
                    expire_at = DateTime.UtcNow.AddDays(30),
                    created_at = DateTime.UtcNow
                });
            
            return mailId > 0;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "보상 우편 발송 실패. UserId: {UserId}", userId);
            return false;
        }
    }
    
    // 현재 시즌 랭킹 초기화
    private async Task<bool> ClearCurrentSeasonRankingAsync()
    {
        try
        {
            // Redis 랭킹 데이터 삭제
            var sortedSet = new RedisSortedSet<long>(_redis, "ranking:global", null);
            await sortedSet.DeleteAsync();
            
            // MySQL 현재 랭킹 테이블 초기화
            await _queryFactory.Query("user_ranking").DeleteAsync();
            
            _logger.LogInformation("현재 시즌 랭킹 초기화 완료");
            return true;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "랭킹 초기화 실패");
            return false;
        }
    }
}
```

`RankingSeasonService`는 시즌 종료 시 필요한 일련의 작업을 처리한다. 먼저 현재 랭킹 데이터를 히스토리 테이블에 저장하고, 순위에 따라 보상을 지급한다. 보상은 우편함 시스템을 통해 전달되며, 사용자는 우편함에서 보상을 수령한다. 마지막으로 현재 시즌 랭킹을 초기화하여 새로운 시즌을 시작할 준비를 한다.

## 22.7 MySQL과 Redis 동기화

장기간 운영 시 Redis와 MySQL 간 데이터 불일치가 발생할 수 있다. 주기적으로 동기화하는 배치 작업을 구현한다.

```csharp
// Services/RankingSyncService.cs
using CloudStructures;
using CloudStructures.Structures;
using SqlKata.Execution;

namespace GameAPIServer.Services;

public class RankingSyncService
{
    private readonly RedisConnection _redis;
    private readonly QueryFactory _queryFactory;
    private readonly ILogger<RankingSyncService> _logger;
    
    public RankingSyncService(
        RedisConnection redis,
        QueryFactory queryFactory,
        ILogger<RankingSyncService> logger)
    {
        _redis = redis;
        _queryFactory = queryFactory;
        _logger = logger;
    }
    
    // MySQL 데이터를 Redis로 동기화
    public async Task<bool> SyncFromMySQLToRedisAsync()
    {
        try
        {
            _logger.LogInformation("MySQL → Redis 랭킹 동기화 시작");
            
            // MySQL에서 전체 랭킹 데이터 조회
            var rankings = await _queryFactory.Query("user_ranking")
                .OrderByDesc("score")
                .GetAsync<RankingEntry>();
            
            if (!rankings.Any())
            {
                _logger.LogWarning("동기화할 랭킹 데이터가 없다");
                return true;
            }
            
            var sortedSet = new RedisSortedSet<long>(_redis, "ranking:global", null);
            
            // 기존 Redis 데이터 삭제
            await sortedSet.DeleteAsync();
            
            // 배치로 Redis에 데이터 추가
            var members = rankings.Select(r =>
                new SortedSetEntry<long>(r.UserId, r.Score)).ToArray();
            
            await sortedSet.AddAsync(members);
            
            _logger.LogInformation("MySQL → Redis 랭킹 동기화 완료. {Count}개 레코드", rankings.Count());
            return true;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "MySQL → Redis 랭킹 동기화 실패");
            return false;
        }
    }
    
    // Redis 데이터를 MySQL로 동기화
    public async Task<bool> SyncFromRedisToMySQLAsync()
    {
        try
        {
            _logger.LogInformation("Redis → MySQL 랭킹 동기화 시작");
            
            var sortedSet = new RedisSortedSet<long>(_redis, "ranking:global", null);
            
            // Redis에서 전체 랭킹 데이터 조회
            var redisData = await sortedSet.RangeByRankWithScoresAsync(0, -1, Order.Descending);
            
            if (redisData == null || redisData.Length == 0)
            {
                _logger.LogWarning("동기화할 Redis 랭킹 데이터가 없다");
                return true;
            }
            
            // MySQL 테이블 초기화 후 삽입
            await _queryFactory.Query("user_ranking").DeleteAsync();
            
            var records = new List<object>();
            for (int i = 0; i < redisData.Length; i++)
            {
                records.Add(new
                {
                    user_id = redisData[i].Value,
                    score = (long)redisData[i].Score,
                    ranking = i + 1,
                    updated_at = DateTime.UtcNow
                });
            }
            
            // 배치 삽입
            await _queryFactory.Query("user_ranking").InsertAsync(records);
            
            _logger.LogInformation("Redis → MySQL 랭킹 동기화 완료. {Count}개 레코드", records.Count);
            return true;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Redis → MySQL 랭킹 동기화 실패");
            return false;
        }
    }
    
    // 불일치 데이터 검증 및 수정
    public async Task<bool> ValidateAndFixDataAsync()
    {
        try
        {
            _logger.LogInformation("랭킹 데이터 검증 시작");
            
            var sortedSet = new RedisSortedSet<long>(_redis, "ranking:global", null);
            
            // MySQL 데이터 조회
            var mysqlRankings = await _queryFactory.Query("user_ranking")
                .GetAsync<RankingEntry>();
            
            var mysqlDict = mysqlRankings.ToDictionary(r => r.UserId, r => r.Score);
            
            var inconsistencies = 0;
            
            // 각 사용자별로 Redis와 MySQL 비교
            foreach (var mysql in mysqlRankings)
            {
                var redisScore = await sortedSet.ScoreAsync(mysql.UserId);
                
                if (!redisScore.HasValue)
                {
                    // Redis에 없는 경우 추가
                    await sortedSet.AddAsync(mysql.UserId, mysql.Score);
                    inconsistencies++;
                    _logger.LogWarning("Redis에 없는 데이터 추가. UserId: {UserId}", mysql.UserId);
                }
                else if ((long)redisScore.Value != mysql.Score)
                {
                    // 점수가 다른 경우 Redis 기준으로 수정 (Redis가 더 최신)
                    await _queryFactory.Query("user_ranking")
                        .Where("user_id", mysql.UserId)
                        .UpdateAsync(new
                        {
                            score = (long)redisScore.Value,
                            updated_at = DateTime.UtcNow
                        });
                    
                    inconsistencies++;
                    _logger.LogWarning("불일치 데이터 수정. UserId: {UserId}, MySQL: {MySQL}, Redis: {Redis}",
                        mysql.UserId, mysql.Score, (long)redisScore.Value);
                }
            }
            
            _logger.LogInformation("랭킹 데이터 검증 완료. 불일치 건수: {Count}", inconsistencies);
            return true;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "랭킹 데이터 검증 실패");
            return false;
        }
    }
}
```

동기화 서비스는 양방향 동기화를 지원한다. `SyncFromMySQLToRedisAsync`는 서버 재시작 시 MySQL 데이터를 Redis로 복원할 때 사용한다. `SyncFromRedisToMySQLAsync`는 주기적으로 Redis의 최신 데이터를 MySQL에 백업할 때 사용한다. `ValidateAndFixDataAsync`는 데이터 불일치를 감지하고 자동으로 수정한다.

## 22.8 배경 작업 설정

주기적인 동기화와 시즌 종료 작업을 위한 배경 서비스를 구현한다.

```csharp
// Services/RankingBackgroundService.cs
namespace GameAPIServer.Services;

public class RankingBackgroundService : BackgroundService
{
    private readonly IServiceProvider _serviceProvider;
    private readonly ILogger<RankingBackgroundService> _logger;
    
    public RankingBackgroundService(
        IServiceProvider serviceProvider,
        ILogger<RankingBackgroundService> logger)
    {
        _serviceProvider = serviceProvider;
        _logger = logger;
    }
    
    protected override async Task ExecuteAsync(CancellationToken stoppingToken)
    {
        _logger.LogInformation("랭킹 배경 작업 시작");
        
        while (!stoppingToken.IsCancellationRequested)
        {
            try
            {
                using var scope = _serviceProvider.CreateScope();
                var syncService = scope.ServiceProvider.GetRequiredService<RankingSyncService>();
                
                // 매 10분마다 Redis → MySQL 동기화
                await syncService.SyncFromRedisToMySQLAsync();
                
                // 매 1시간마다 데이터 검증
                var currentMinute = DateTime.UtcNow.Minute;
                if (currentMinute < 10)
                {
                    await syncService.ValidateAndFixDataAsync();
                }
                
                _logger.LogInformation("랭킹 동기화 작업 완료");
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "랭킹 배경 작업 오류");
            }
            
            // 10분 대기
            await Task.Delay(TimeSpan.FromMinutes(10), stoppingToken);
        }
        
        _logger.LogInformation("랭킹 배경 작업 종료");
    }
}
```

## 22.9 부정 행위 방지

랭킹 시스템에서 발생할 수 있는 부정 행위를 방지하는 검증 로직을 추가한다.

```csharp
// Services/RankingValidationService.cs
namespace GameAPIServer.Services;

public class RankingValidationService
{
    private readonly RedisConnection _redis;
    private readonly QueryFactory _queryFactory;
    private readonly ILogger<RankingValidationService> _logger;
    
    public RankingValidationService(
        RedisConnection redis,
        QueryFactory queryFactory,
        ILogger<RankingValidationService> logger)
    {
        _redis = redis;
        _queryFactory = queryFactory;
        _logger = logger;
    }
    
    // 점수 업데이트 검증
    public async Task<(bool IsValid, string Message)> ValidateScoreUpdateAsync(
        long userId,
        long newScore,
        long previousScore)
    {
        // 1. 점수 증가량 검증 (급격한 증가 방지)
        if (newScore > previousScore)
        {
            var increase = newScore - previousScore;
            var maxIncrease = 10000; // 한 번에 최대 증가 가능 점수
            
            if (increase > maxIncrease)
            {
                _logger.LogWarning("비정상적인 점수 증가 감지. UserId: {UserId}, Increase: {Increase}",
                    userId, increase);
                return (false, "비정상적인 점수 증가가 감지되었다");
            }
        }
        
        // 2. 업데이트 빈도 검증 (너무 자주 업데이트 방지)
        var redis = new RedisString<DateTime>(_redis, $"ranking:last_update:{userId}", null);
        var lastUpdate = await redis.GetAsync();
        
        if (lastUpdate.HasValue)
        {
            var timeSinceLastUpdate = DateTime.UtcNow - lastUpdate.Value;
            if (timeSinceLastUpdate.TotalSeconds < 5)
            {
                _logger.LogWarning("업데이트 빈도 제한 위반. UserId: {UserId}", userId);
                return (false, "업데이트가 너무 빈번하다");
            }
        }
        
        // 마지막 업데이트 시간 기록
        await redis.SetAsync(DateTime.UtcNow, TimeSpan.FromMinutes(5));
        
        // 3. 게임 플레이 기록 확인
        var hasPlayRecord = await CheckRecentGamePlayAsync(userId);
        if (!hasPlayRecord)
        {
            _logger.LogWarning("게임 플레이 기록 없이 점수 업데이트 시도. UserId: {UserId}", userId);
            return (false, "유효한 게임 플레이 기록이 필요하다");
        }
        
        return (true, "검증 성공");
    }
    
    // 최근 게임 플레이 기록 확인
    private async Task<bool> CheckRecentGamePlayAsync(long userId)
    {
        try
        {
            var recentPlay = await _queryFactory.Query("dungeon_clear_history")
                .Where("user_id", userId)
                .Where("created_at", ">", DateTime.UtcNow.AddMinutes(-10))
                .FirstOrDefaultAsync();
            
            return recentPlay != null;
        }
        catch
        {
            return true; // 에러 시 통과 (서비스 가용성 우선)
        }
    }
}
```

검증 서비스는 여러 측면에서 부정 행위를 방지한다. 점수의 급격한 증가, 비정상적인 업데이트 빈도, 게임 플레이 기록 부재 등을 감지하여 차단한다.

## 22.11 성능 최적화 팁

랭킹 시스템의 성능을 최적화하는 방법을 소개한다.

**캐시 활용**: 전체 랭킹 조회 결과를 Redis에 캐싱하여 반복적인 요청을 빠르게 처리한다. 1분 정도의 짧은 TTL을 설정하여 실시간성도 유지한다.

**페이지네이션**: 전체 랭킹 조회 시 한 번에 모든 데이터를 가져오지 않고 페이지 단위로 나누어 조회한다. 클라이언트가 스크롤할 때마다 다음 페이지를 요청하도록 구현한다.

**배치 처리**: MySQL 동기화나 보상 지급 같은 대량 작업은 배치로 처리하여 데이터베이스 부하를 줄인다.

**인덱스 최적화**: MySQL 테이블의 score 컬럼에 내림차순 인덱스를 생성하여 조회 성능을 향상시킨다.

**Redis 메모리 관리**: 랭킹 데이터가 계속 증가하면 메모리 부족이 발생할 수 있다. 일정 순위 이하의 데이터는 주기적으로 제거하는 정책을 적용한다.
