# 20장. 출석 및 미션 시스템 (실습)

이번 장에서는 플레이어의 일일 참여를 유도하는 출석 체크 시스템과 미션 시스템을 구현한다. 출석 체크는 매일 접속을 유도하는 기본적인 리텐션 시스템이며, 미션은 특정 행동을 유도하여 게임 플레이를 활성화한다. Redis를 활용한 일일 리셋 처리와 중복 보상 방지 로직을 함께 학습한다.

## 20.1 출석 체크 테이블 설계

출석 체크 시스템에 필요한 보상 마스터 데이터와 플레이어의 출석 기록 테이블을 설계한다.

### 출석 보상 마스터 테이블

```sql
-- 출석 보상 마스터 데이터
CREATE TABLE attendance_reward_master (
    day INT PRIMARY KEY,
    reward_type VARCHAR(20) NOT NULL, -- 'gold', 'gem', 'item', 'exp'
    reward_id INT NOT NULL DEFAULT 0, -- 아이템인 경우 item_id
    reward_amount INT NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 플레이어 출석 기록
CREATE TABLE player_attendance (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT NOT NULL,
    attendance_date DATE NOT NULL,
    attendance_day INT NOT NULL, -- 연속 출석 일수
    reward_received BOOLEAN DEFAULT FALSE,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    UNIQUE KEY unique_user_date (user_id, attendance_date),
    INDEX idx_user_id (user_id),
    INDEX idx_attendance_date (attendance_date)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 출석 연속 기록 (현재 연속 출석 일수 관리)
CREATE TABLE player_attendance_streak (
    user_id BIGINT PRIMARY KEY,
    current_streak INT NOT NULL DEFAULT 0,
    max_streak INT NOT NULL DEFAULT 0,
    last_attendance_date DATE,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

## 20.2 미션 시스템 테이블 설계

일일 미션과 주간 미션을 관리하는 테이블을 설계한다.

```sql
-- 미션 마스터 데이터
CREATE TABLE mission_master (
    mission_id INT PRIMARY KEY,
    mission_type VARCHAR(20) NOT NULL, -- 'daily', 'weekly', 'achievement'
    mission_name VARCHAR(100) NOT NULL,
    description TEXT,
    mission_category VARCHAR(30) NOT NULL, -- 'dungeon', 'login', 'purchase', 'social' 등
    target_count INT NOT NULL DEFAULT 1,
    reward_type VARCHAR(20) NOT NULL,
    reward_id INT NOT NULL DEFAULT 0,
    reward_amount INT NOT NULL,
    sort_order INT NOT NULL DEFAULT 0,
    is_active BOOLEAN DEFAULT TRUE,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 플레이어 미션 진행 상황
CREATE TABLE player_mission_progress (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT NOT NULL,
    mission_id INT NOT NULL,
    current_count INT NOT NULL DEFAULT 0,
    completed BOOLEAN DEFAULT FALSE,
    reward_received BOOLEAN DEFAULT FALSE,
    reset_date DATE NOT NULL, -- 일일/주간 미션 리셋 기준일
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    UNIQUE KEY unique_user_mission_reset (user_id, mission_id, reset_date),
    INDEX idx_user_id (user_id),
    INDEX idx_mission_id (mission_id),
    INDEX idx_reset_date (reset_date)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 미션 완료 이력
CREATE TABLE player_mission_history (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT NOT NULL,
    mission_id INT NOT NULL,
    completed_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    reward_type VARCHAR(20) NOT NULL,
    reward_amount INT NOT NULL,
    INDEX idx_user_id (user_id),
    INDEX idx_completed_at (completed_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

```csharp
namespace GameApiServer.Services;

public partial class MasterDataService
{
    private Dictionary<int, AttendanceRewardMaster> _attendanceRewardData;
    private Dictionary<int, MissionMaster> _missionMasterData;

    public async Task LoadAttendanceRewardDataAsync()
    {
        try
        {
            var rewards = await _queryFactory.Query("attendance_reward_master")
                .GetAsync<AttendanceRewardMaster>();
            
            _attendanceRewardData = rewards.ToDictionary(r => r.Day);
            
            _logger.LogInformation($"출석 보상 마스터 데이터 로딩 완료: {_attendanceRewardData.Count}개");
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "출석 보상 마스터 데이터 로딩 실패");
            throw;
        }
    }

    public async Task LoadMissionMasterDataAsync()
    {
        try
        {
            var missions = await _queryFactory.Query("mission_master")
                .Where("is_active", true)
                .GetAsync<MissionMaster>();
            
            _missionMasterData = missions.ToDictionary(m => m.MissionId);
            
            _logger.LogInformation($"미션 마스터 데이터 로딩 완료: {_missionMasterData.Count}개");
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "미션 마스터 데이터 로딩 실패");
            throw;
        }
    }

    public AttendanceRewardMaster GetAttendanceReward(int day)
    {
        return _attendanceRewardData.TryGetValue(day, out var reward) ? reward : null;
    }

    public MissionMaster GetMissionMaster(int missionId)
    {
        return _missionMasterData.TryGetValue(missionId, out var mission) ? mission : null;
    }

    public List<MissionMaster> GetMissionsByType(string missionType)
    {
        return _missionMasterData.Values
            .Where(m => m.MissionType == missionType)
            .OrderBy(m => m.SortOrder)
            .ToList();
    }

    public int GetMaxAttendanceDay()
    {
        return _attendanceRewardData.Keys.Max();
    }
}
```

### Services/AttendanceService.cs

```csharp
namespace GameApiServer.Services;

public class AttendanceService
{
    private readonly ILogger<AttendanceService> _logger;
    private readonly MasterDataService _masterDataService;
    private readonly RedisDb _redisDb;
    private readonly QueryFactory _queryFactory;

    public AttendanceService(
        ILogger<AttendanceService> logger,
        MasterDataService masterDataService,
        RedisDb redisDb,
        IDbConnection dbConnection)
    {
        _logger = logger;
        _masterDataService = masterDataService;
        _redisDb = redisDb;
        _queryFactory = new QueryFactory(dbConnection, new MySqlCompiler());
    }

    public async Task<(ErrorCode, AttendanceCheckResponse)> CheckAttendanceAsync(long userId)
    {
        try
        {
            var today = DateTime.UtcNow.Date;

            // 1. Redis에서 오늘 출석 체크 여부 확인
            var cacheKey = MakeAttendanceCacheKey(userId, today);
            var cached = await _redisDb.GetAsync<AttendanceCache>(cacheKey);

            if (cached != null)
            {
                return (ErrorCode.AttendanceAlreadyChecked, null);
            }

            // 2. DB에서 출석 연속 기록 조회
            var streak = await GetOrCreateStreakAsync(userId);

            // 3. 연속 출석 계산
            var newStreak = CalculateNewStreak(streak.LastAttendanceDate, today, streak.CurrentStreak);
            var maxAttendanceDay = _masterDataService.GetMaxAttendanceDay();
            var attendanceDay = (newStreak - 1) % maxAttendanceDay + 1;

            // 4. 출석 보상 정보 가져오기
            var rewardMaster = _masterDataService.GetAttendanceReward(attendanceDay);
            if (rewardMaster == null)
            {
                return (ErrorCode.AttendanceRewardNotFound, null);
            }

            // 5. 보상 지급
            var rewardResult = await GrantAttendanceRewardAsync(userId, rewardMaster);
            if (rewardResult != ErrorCode.None)
            {
                return (rewardResult, null);
            }

            // 6. 출석 기록 저장
            await SaveAttendanceRecordAsync(userId, today, attendanceDay);

            // 7. 연속 출석 기록 업데이트
            await UpdateStreakAsync(userId, newStreak, today);

            // 8. Redis 캐시 저장 (자정까지 유효)
            var cache = new AttendanceCache
            {
                UserId = userId,
                CheckDate = today,
                AttendanceDay = attendanceDay
            };

            var ttl = GetSecondsUntilMidnight();
            await _redisDb.SetAsync(cacheKey, cache, TimeSpan.FromSeconds(ttl));

            _logger.LogInformation($"출석 체크 완료: UserId={userId}, Day={attendanceDay}, Streak={newStreak}");

            return (ErrorCode.None, new AttendanceCheckResponse
            {
                Success = true,
                CurrentStreak = newStreak,
                MaxStreak = Math.Max(newStreak, streak.MaxStreak),
                AttendanceDay = attendanceDay,
                Reward = new RewardInfo
                {
                    RewardType = rewardMaster.RewardType,
                    RewardId = rewardMaster.RewardId,
                    RewardAmount = rewardMaster.RewardAmount
                }
            });
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, $"출석 체크 실패: UserId={userId}");
            return (ErrorCode.AttendanceCheckFailed, null);
        }
    }

    public async Task<AttendanceInfoResponse> GetAttendanceInfoAsync(long userId)
    {
        try
        {
            var today = DateTime.UtcNow.Date;

            // 1. Redis에서 오늘 출석 여부 확인
            var cacheKey = MakeAttendanceCacheKey(userId, today);
            var cached = await _redisDb.GetAsync<AttendanceCache>(cacheKey);
            var checkedToday = cached != null;

            // 2. 연속 출석 정보 조회
            var streak = await GetOrCreateStreakAsync(userId);

            // 3. 출석 보상 목록 생성
            var maxDay = _masterDataService.GetMaxAttendanceDay();
            var rewards = new List<AttendanceRewardInfo>();

            for (int day = 1; day <= maxDay; day++)
            {
                var rewardMaster = _masterDataService.GetAttendanceReward(day);
                if (rewardMaster != null)
                {
                    rewards.Add(new AttendanceRewardInfo
                    {
                        Day = day,
                        RewardType = rewardMaster.RewardType,
                        RewardId = rewardMaster.RewardId,
                        RewardAmount = rewardMaster.RewardAmount,
                        IsReceived = day < ((streak.CurrentStreak - 1) % maxDay + 1) || (day == ((streak.CurrentStreak - 1) % maxDay + 1) && checkedToday),
                        IsToday = day == ((streak.CurrentStreak - 1) % maxDay + 1) + (checkedToday ? 0 : 0)
                    });
                }
            }

            return new AttendanceInfoResponse
            {
                CheckedToday = checkedToday,
                CurrentStreak = streak.CurrentStreak,
                MaxStreak = streak.MaxStreak,
                LastAttendanceDate = streak.LastAttendanceDate,
                Rewards = rewards
            };
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, $"출석 정보 조회 실패: UserId={userId}");
            return new AttendanceInfoResponse
            {
                CheckedToday = false,
                CurrentStreak = 0,
                MaxStreak = 0,
                Rewards = new List<AttendanceRewardInfo>()
            };
        }
    }

    private async Task<dynamic> GetOrCreateStreakAsync(long userId)
    {
        var streak = await _queryFactory.Query("player_attendance_streak")
            .Where("user_id", userId)
            .FirstOrDefaultAsync<dynamic>();

        if (streak == null)
        {
            await _queryFactory.Query("player_attendance_streak")
                .InsertAsync(new
                {
                    user_id = userId,
                    current_streak = 0,
                    max_streak = 0,
                    last_attendance_date = (DateTime?)null
                });

            streak = await _queryFactory.Query("player_attendance_streak")
                .Where("user_id", userId)
                .FirstAsync<dynamic>();
        }

        return streak;
    }

    private int CalculateNewStreak(DateTime? lastAttendanceDate, DateTime today, int currentStreak)
    {
        if (lastAttendanceDate == null)
        {
            return 1;
        }

        var daysSinceLastAttendance = (today - lastAttendanceDate.Value).Days;

        if (daysSinceLastAttendance == 1)
        {
            // 연속 출석
            return currentStreak + 1;
        }
        else if (daysSinceLastAttendance > 1)
        {
            // 연속 끊김
            return 1;
        }

        // 같은 날 (발생하면 안 됨)
        return currentStreak;
    }

    private async Task<ErrorCode> GrantAttendanceRewardAsync(long userId, AttendanceRewardMaster rewardMaster)
    {
        try
        {
            switch (rewardMaster.RewardType.ToLower())
            {
                case "gold":
                    await _queryFactory.Query("game_user")
                        .Where("user_id", userId)
                        .IncrementAsync("gold", rewardMaster.RewardAmount);
                    break;

                case "gem":
                    await _queryFactory.Query("game_user")
                        .Where("user_id", userId)
                        .IncrementAsync("gem", rewardMaster.RewardAmount);
                    break;

                case "exp":
                    await _queryFactory.Query("game_user")
                        .Where("user_id", userId)
                        .IncrementAsync("exp", rewardMaster.RewardAmount);
                    break;

                case "item":
                    var existing = await _queryFactory.Query("player_item")
                        .Where("user_id", userId)
                        .Where("item_id", rewardMaster.RewardId)
                        .FirstOrDefaultAsync<dynamic>();

                    if (existing == null)
                    {
                        await _queryFactory.Query("player_item")
                            .InsertAsync(new
                            {
                                user_id = userId,
                                item_id = rewardMaster.RewardId,
                                item_count = rewardMaster.RewardAmount
                            });
                    }
                    else
                    {
                        await _queryFactory.Query("player_item")
                            .Where("user_id", userId)
                            .Where("item_id", rewardMaster.RewardId)
                            .IncrementAsync("item_count", rewardMaster.RewardAmount);
                    }
                    break;

                default:
                    _logger.LogWarning($"알 수 없는 보상 타입: {rewardMaster.RewardType}");
                    return ErrorCode.InvalidRewardType;
            }

            return ErrorCode.None;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, $"출석 보상 지급 실패: UserId={userId}");
            return ErrorCode.AttendanceRewardGrantFailed;
        }
    }

    private async Task SaveAttendanceRecordAsync(long userId, DateTime attendanceDate, int attendanceDay)
    {
        await _queryFactory.Query("player_attendance")
            .InsertAsync(new
            {
                user_id = userId,
                attendance_date = attendanceDate,
                attendance_day = attendanceDay,
                reward_received = true
            });
    }

    private async Task UpdateStreakAsync(long userId, int newStreak, DateTime today)
    {
        var streak = await _queryFactory.Query("player_attendance_streak")
            .Where("user_id", userId)
            .FirstAsync<dynamic>();

        var maxStreak = Math.Max(newStreak, (int)streak.max_streak);

        await _queryFactory.Query("player_attendance_streak")
            .Where("user_id", userId)
            .UpdateAsync(new
            {
                current_streak = newStreak,
                max_streak = maxStreak,
                last_attendance_date = today
            });
    }

    private string MakeAttendanceCacheKey(long userId, DateTime date)
    {
        return $"attendance:{userId}:{date:yyyyMMdd}";
    }

    private int GetSecondsUntilMidnight()
    {
        var now = DateTime.UtcNow;
        var midnight = now.Date.AddDays(1);
        return (int)(midnight - now).TotalSeconds;
    }
}
```

```csharp
namespace GameApiServer.Services;

public class MissionService
{
    private readonly ILogger<MissionService> _logger;
    private readonly MasterDataService _masterDataService;
    private readonly RedisDb _redisDb;
    private readonly QueryFactory _queryFactory;

    public MissionService(
        ILogger<MissionService> logger,
        MasterDataService masterDataService,
        RedisDb redisDb,
        IDbConnection dbConnection)
    {
        _logger = logger;
        _masterDataService = masterDataService;
        _redisDb = redisDb;
        _queryFactory = new QueryFactory(dbConnection, new MySqlCompiler());
    }

    public async Task<MissionListResponse> GetMissionListAsync(long userId, string missionType)
    {
        try
        {
            // 1. 미션 마스터 데이터 가져오기
            var missions = _masterDataService.GetMissionsByType(missionType);

            // 2. 리셋 날짜 계산
            var resetDate = GetResetDate(missionType);

            // 3. 플레이어 미션 진행 상황 조회
            var progressList = await _queryFactory.Query("player_mission_progress")
                .Where("user_id", userId)
                .Where("reset_date", resetDate)
                .GetAsync<dynamic>();

            var progressMap = progressList.ToDictionary(
                p => (int)p.mission_id,
                p => new
                {
                    CurrentCount = (int)p.current_count,
                    Completed = (bool)p.completed,
                    RewardReceived = (bool)p.reward_received
                }
            );

            // 4. 미션 정보 구성
            var missionInfoList = new List<MissionInfo>();

            foreach (var mission in missions)
            {
                var hasProgress = progressMap.TryGetValue(mission.MissionId, out var progress);

                missionInfoList.Add(new MissionInfo
                {
                    MissionId = mission.MissionId,
                    MissionName = mission.MissionName,
                    Description = mission.Description,
                    MissionCategory = mission.MissionCategory,
                    TargetCount = mission.TargetCount,
                    CurrentCount = hasProgress ? progress.CurrentCount : 0,
                    Completed = hasProgress && progress.Completed,
                    RewardReceived = hasProgress && progress.RewardReceived,
                    Reward = new RewardInfo
                    {
                        RewardType = mission.RewardType,
                        RewardId = mission.RewardId,
                        RewardAmount = mission.RewardAmount
                    }
                });
            }

            return new MissionListResponse
            {
                Missions = missionInfoList
            };
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, $"미션 목록 조회 실패: UserId={userId}, Type={missionType}");
            return new MissionListResponse { Missions = new List<MissionInfo>() };
        }
    }

    public async Task<ErrorCode> UpdateMissionProgressAsync(long userId, int missionId, int incrementCount)
    {
        try
        {
            // 1. 미션 마스터 데이터 확인
            var missionMaster = _masterDataService.GetMissionMaster(missionId);
            if (missionMaster == null)
            {
                return ErrorCode.MissionNotFound;
            }

            // 2. 리셋 날짜 계산
            var resetDate = GetResetDate(missionMaster.MissionType);

            // 3. 현재 진행 상황 조회
            var progress = await _queryFactory.Query("player_mission_progress")
                .Where("user_id", userId)
                .Where("mission_id", missionId)
                .Where("reset_date", resetDate)
                .FirstOrDefaultAsync<dynamic>();

            if (progress == null)
            {
                // 신규 진행 상황 생성
                await _queryFactory.Query("player_mission_progress")
                    .InsertAsync(new
                    {
                        user_id = userId,
                        mission_id = missionId,
                        current_count = incrementCount,
                        completed = incrementCount >= missionMaster.TargetCount,
                        reward_received = false,
                        reset_date = resetDate
                    });
            }
            else
            {
                // 기존 진행 상황 업데이트
                var newCount = (int)progress.current_count + incrementCount;
                var completed = newCount >= missionMaster.TargetCount;

                await _queryFactory.Query("player_mission_progress")
                    .Where("user_id", userId)
                    .Where("mission_id", missionId)
                    .Where("reset_date", resetDate)
                    .UpdateAsync(new
                    {
                        current_count = newCount,
                        completed = completed
                    });
            }

            _logger.LogInformation($"미션 진행 업데이트: UserId={userId}, MissionId={missionId}, Increment={incrementCount}");

            return ErrorCode.None;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, $"미션 진행 업데이트 실패: UserId={userId}, MissionId={missionId}");
            return ErrorCode.MissionProgressUpdateFailed;
        }
    }

    public async Task<(ErrorCode, MissionRewardResponse)> ReceiveMissionRewardAsync(long userId, int missionId)
    {
        try
        {
            // 1. 미션 마스터 데이터 확인
            var missionMaster = _masterDataService.GetMissionMaster(missionId);
            if (missionMaster == null)
            {
                return (ErrorCode.MissionNotFound, null);
            }

            // 2. 리셋 날짜 계산
            var resetDate = GetResetDate(missionMaster.MissionType);

            // 3. 미션 진행 상황 확인
            var progress = await _queryFactory.Query("player_mission_progress")
                .Where("user_id", userId)
                .Where("mission_id", missionId)
                .Where("reset_date", resetDate)
                .FirstOrDefaultAsync<dynamic>();

            if (progress == null)
            {
                return (ErrorCode.MissionNotStarted, null);
            }

            if (!(bool)progress.completed)
            {
                return (ErrorCode.MissionNotCompleted, null);
            }

            if ((bool)progress.reward_received)
            {
                return (ErrorCode.MissionRewardAlreadyReceived, null);
            }

            // 4. 보상 지급
            var rewardResult = await GrantMissionRewardAsync(userId, missionMaster);
            if (rewardResult != ErrorCode.None)
            {
                return (rewardResult, null);
            }

            // 5. 보상 수령 상태 업데이트
            await _queryFactory.Query("player_mission_progress")
                .Where("user_id", userId)
                .Where("mission_id", missionId)
                .Where("reset_date", resetDate)
                .UpdateAsync(new { reward_received = true });

            // 6. 미션 완료 이력 저장
            await _queryFactory.Query("player_mission_history")
                .InsertAsync(new
                {
                    user_id = userId,
                    mission_id = missionId,
                    reward_type = missionMaster.RewardType,
                    reward_amount = missionMaster.RewardAmount
                });

            _logger.LogInformation($"미션 보상 수령: UserId={userId}, MissionId={missionId}");

            return (ErrorCode.None, new MissionRewardResponse
            {
                Success = true,
                Reward = new RewardInfo
                {
                    RewardType = missionMaster.RewardType,
                    RewardId = missionMaster.RewardId,
                    RewardAmount = missionMaster.RewardAmount
                }
            });
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, $"미션 보상 수령 실패: UserId={userId}, MissionId={missionId}");
            return (ErrorCode.MissionRewardReceiveFailed, null);
        }
    }

    private async Task<ErrorCode> GrantMissionRewardAsync(long userId, MissionMaster missionMaster)
    {
        try
        {
            switch (missionMaster.RewardType.ToLower())
            {
                case "gold":
                    await _queryFactory.Query("game_user")
                        .Where("user_id", userId)
                        .IncrementAsync("gold", missionMaster.RewardAmount);
                    break;

                case "gem":
                    await _queryFactory.Query("game_user")
                        .Where("user_id", userId)
                        .IncrementAsync("gem", missionMaster.RewardAmount);
                    break;

                case "exp":
                    await _queryFactory.Query("game_user")
                        .Where("user_id", userId)
                        .IncrementAsync("exp", missionMaster.RewardAmount);
                    break;

                case "item":
                    var existing = await _queryFactory.Query("player_item")
                        .Where("user_id", userId)
                        .Where("item_id", missionMaster.RewardId)
                        .FirstOrDefaultAsync<dynamic>();

                    if (existing == null)
                    {
                        await _queryFactory.Query("player_item")
                            .InsertAsync(new
                            {
                                user_id = userId,
                                item_id = missionMaster.RewardId,
                                item_count = missionMaster.RewardAmount
                            });
                    }
                    else
                    {
                        await _queryFactory.Query("player_item")
                            .Where("user_id", userId)
                            .Where("item_id", missionMaster.RewardId)
                            .IncrementAsync("item_count", missionMaster.RewardAmount);
                    }
                    break;

                default:
                    _logger.LogWarning($"알 수 없는 보상 타입: {missionMaster.RewardType}");
                    return ErrorCode.InvalidRewardType;
            }

            return ErrorCode.None;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, $"미션 보상 지급 실패: UserId={userId}");
            return ErrorCode.MissionRewardGrantFailed;
        }
    }

    private DateTime GetResetDate(string missionType)
    {
        var now = DateTime.UtcNow;

        switch (missionType.ToLower())
        {
            case "daily":
                return now.Date;

            case "weekly":
                // 주의 시작을 월요일로 설정
                var dayOfWeek = (int)now.DayOfWeek;
                var daysToMonday = dayOfWeek == 0 ? 6 : dayOfWeek - 1;
                return now.Date.AddDays(-daysToMonday);

            default:
                return now.Date;
        }
    }
}
```

## 20.12 정리

이번 장에서는 출석 체크와 미션 시스템을 구현했다. 주요 내용은 다음과 같다.

- 출석 보상 및 미션 마스터 데이터 테이블 설계 및 생성
- Redis를 활용한 일일 출석 중복 체크 방지
- 연속 출석 추적 및 리셋 로직 구현
- 자정까지 유효한 Redis 캐시를 사용한 출석 상태 관리
- 일일/주간 미션의 자동 리셋 날짜 계산
- 미션 진행 상황 추적 및 완료 여부 판단
- 중복 보상 수령 방지 로직
- 다른 API에서 미션 진행도를 자동으로 업데이트하는 추적 시스템
- 다양한 보상 타입(골드, 젬, 경험치, 아이템) 지급 처리

출석 및 미션 시스템은 플레이어의 일일 참여를 유도하는 핵심 리텐션 기능이다. Redis를 활용한 일일 리셋과 중복 방지, 그리고 자동 미션 추적 시스템을 통해 효율적인 운영이 가능하다. 다음 장에서는 플레이어 간 소셜 기능인 친구 시스템을 구현한다.
