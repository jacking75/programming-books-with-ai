# 19장. 던전 및 스테이지 시스템 (실습)

이번 장에서는 수집형 RPG 게임의 핵심 콘텐츠인 던전 시스템을 구현한다. 던전은 플레이어가 캐릭터를 육성하고 아이템을 획득하는 주요 콘텐츠다. 던전 입장부터 진행, 클리어, 보상 지급까지의 전체 흐름을 구현하며, Redis를 활용한 실시간 상태 관리와 부정 방지 로직을 함께 학습한다.

## 19.1 던전 마스터 데이터 설계

던전 시스템에 필요한 마스터 데이터 테이블을 설계한다. 던전의 기본 정보, 스테이지 정보, 보상 정보를 관리하는 테이블을 생성한다.

### 던전 마스터 테이블

```sql
-- 던전 기본 정보
CREATE TABLE dungeon_master (
    dungeon_id INT PRIMARY KEY,
    dungeon_name VARCHAR(100) NOT NULL,
    description TEXT,
    required_level INT NOT NULL DEFAULT 1,
    entry_cost_type VARCHAR(20) NOT NULL, -- 'gold', 'gem', 'ticket'
    entry_cost_amount INT NOT NULL DEFAULT 0,
    max_clear_count INT NOT NULL DEFAULT 0, -- 0이면 무제한
    difficulty VARCHAR(20) NOT NULL, -- 'easy', 'normal', 'hard'
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 스테이지 정보
CREATE TABLE stage_master (
    stage_id INT PRIMARY KEY,
    dungeon_id INT NOT NULL,
    stage_number INT NOT NULL,
    stage_name VARCHAR(100) NOT NULL,
    monster_info JSON, -- 몬스터 정보 (id, level, count 등)
    clear_exp INT NOT NULL DEFAULT 0,
    clear_gold INT NOT NULL DEFAULT 0,
    FOREIGN KEY (dungeon_id) REFERENCES dungeon_master(dungeon_id),
    UNIQUE KEY unique_dungeon_stage (dungeon_id, stage_number)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 스테이지 보상 아이템
CREATE TABLE stage_reward_item (
    id INT AUTO_INCREMENT PRIMARY KEY,
    stage_id INT NOT NULL,
    item_id INT NOT NULL,
    item_count INT NOT NULL DEFAULT 1,
    drop_rate DECIMAL(5,2) NOT NULL, -- 드롭 확률 (0.00 ~ 100.00)
    FOREIGN KEY (stage_id) REFERENCES stage_master(stage_id),
    INDEX idx_stage_id (stage_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 플레이어 던전 클리어 기록
CREATE TABLE player_dungeon_clear (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT NOT NULL,
    dungeon_id INT NOT NULL,
    stage_id INT NOT NULL,
    clear_time INT NOT NULL, -- 클리어에 걸린 시간 (초)
    score INT NOT NULL DEFAULT 0,
    clear_date DATE NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (dungeon_id) REFERENCES dungeon_master(dungeon_id),
    FOREIGN KEY (stage_id) REFERENCES stage_master(stage_id),
    INDEX idx_user_dungeon (user_id, dungeon_id),
    INDEX idx_clear_date (clear_date)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 일일 던전 입장 횟수 관리
CREATE TABLE player_dungeon_entry (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT NOT NULL,
    dungeon_id INT NOT NULL,
    entry_date DATE NOT NULL,
    entry_count INT NOT NULL DEFAULT 0,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    UNIQUE KEY unique_user_dungeon_date (user_id, dungeon_id, entry_date),
    INDEX idx_user_date (user_id, entry_date)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

```csharp
namespace GameApiServer.Services;

public partial class MasterDataService
{
    private Dictionary<int, DungeonMaster> _dungeonMasterData;
    private Dictionary<int, StageMaster> _stageMasterData;
    private Dictionary<int, List<StageRewardItem>> _stageRewardData;

    public async Task LoadDungeonMasterDataAsync()
    {
        try
        {
            var dungeons = await _queryFactory.Query("dungeon_master")
                .GetAsync<DungeonMaster>();
            
            _dungeonMasterData = dungeons.ToDictionary(d => d.DungeonId);
            
            _logger.LogInformation($"던전 마스터 데이터 로딩 완료: {_dungeonMasterData.Count}개");
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "던전 마스터 데이터 로딩 실패");
            throw;
        }
    }

    public async Task LoadStageMasterDataAsync()
    {
        try
        {
            var stages = await _queryFactory.Query("stage_master")
                .GetAsync<StageMaster>();
            
            _stageMasterData = stages.ToDictionary(s => s.StageId);
            
            _logger.LogInformation($"스테이지 마스터 데이터 로딩 완료: {_stageMasterData.Count}개");
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "스테이지 마스터 데이터 로딩 실패");
            throw;
        }
    }

    public async Task LoadStageRewardDataAsync()
    {
        try
        {
            var rewards = await _queryFactory.Query("stage_reward_item")
                .GetAsync<StageRewardItem>();
            
            _stageRewardData = rewards
                .GroupBy(r => r.StageId)
                .ToDictionary(g => g.Key, g => g.ToList());
            
            _logger.LogInformation($"스테이지 보상 데이터 로딩 완료: {_stageRewardData.Count}개 스테이지");
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "스테이지 보상 데이터 로딩 실패");
            throw;
        }
    }

    public DungeonMaster GetDungeonMaster(int dungeonId)
    {
        return _dungeonMasterData.TryGetValue(dungeonId, out var dungeon) ? dungeon : null;
    }

    public StageMaster GetStageMaster(int stageId)
    {
        return _stageMasterData.TryGetValue(stageId, out var stage) ? stage : null;
    }

    public List<StageMaster> GetDungeonStages(int dungeonId)
    {
        return _stageMasterData.Values
            .Where(s => s.DungeonId == dungeonId)
            .OrderBy(s => s.StageNumber)
            .ToList();
    }

    public List<StageRewardItem> GetStageRewards(int stageId)
    {
        return _stageRewardData.TryGetValue(stageId, out var rewards) ? rewards : new List<StageRewardItem>();
    }
}
```

### Services/DungeonService.cs

```csharp
namespace GameApiServer.Services;

public class DungeonService
{
    private readonly ILogger<DungeonService> _logger;
    private readonly MasterDataService _masterDataService;
    private readonly RedisDb _redisDb;
    private readonly QueryFactory _queryFactory;
    private const int DungeonSessionExpirySeconds = 1800; // 30분

    public DungeonService(
        ILogger<DungeonService> logger,
        MasterDataService masterDataService,
        RedisDb redisDb,
        IDbConnection dbConnection)
    {
        _logger = logger;
        _masterDataService = masterDataService;
        _redisDb = redisDb;
        _queryFactory = new QueryFactory(dbConnection, new MySqlCompiler());
    }

    public async Task<(ErrorCode, DungeonEnterResponse)> EnterDungeonAsync(long userId, DungeonEnterRequest request)
    {
        try
        {
            // 1. 던전 마스터 데이터 확인
            var dungeonMaster = _masterDataService.GetDungeonMaster(request.DungeonId);
            if (dungeonMaster == null)
            {
                return (ErrorCode.DungeonNotFound, null);
            }

            // 2. 플레이어 레벨 확인
            var user = await _queryFactory.Query("game_user")
                .Where("user_id", userId)
                .FirstOrDefaultAsync<dynamic>();

            if (user == null)
            {
                return (ErrorCode.UserNotFound, null);
            }

            if (user.level < dungeonMaster.RequiredLevel)
            {
                return (ErrorCode.InsufficientLevel, null);
            }

            // 3. 일일 입장 횟수 확인
            if (dungeonMaster.MaxClearCount > 0)
            {
                var todayEntryCount = await GetTodayEntryCountAsync(userId, request.DungeonId);
                if (todayEntryCount >= dungeonMaster.MaxClearCount)
                {
                    return (ErrorCode.DungeonEntryLimitExceeded, null);
                }
            }

            // 4. 입장 비용 확인 및 차감
            var costResult = await DeductEntryCostAsync(userId, dungeonMaster);
            if (costResult != ErrorCode.None)
            {
                return (costResult, null);
            }

            // 5. 첫 번째 스테이지 가져오기
            var stages = _masterDataService.GetDungeonStages(request.DungeonId);
            if (stages.Count == 0)
            {
                return (ErrorCode.StageNotFound, null);
            }

            var firstStage = stages.First();

            // 6. 던전 토큰 생성
            var dungeonToken = GenerateDungeonToken(userId, request.DungeonId, firstStage.StageId);

            // 7. Redis에 던전 세션 저장
            var session = new DungeonSession
            {
                UserId = userId,
                DungeonId = request.DungeonId,
                StageId = firstStage.StageId,
                DungeonToken = dungeonToken,
                EnterTime = DateTime.UtcNow,
                Progress = 0,
                LastUpdateTime = DateTime.UtcNow
            };

            var redisKey = MakeRedisKey(userId);
            await _redisDb.SetAsync(redisKey, session, TimeSpan.FromSeconds(DungeonSessionExpirySeconds));

            // 8. 입장 횟수 기록
            await IncrementEntryCountAsync(userId, request.DungeonId);

            return (ErrorCode.None, new DungeonEnterResponse
            {
                DungeonId = request.DungeonId,
                StageId = firstStage.StageId,
                DungeonToken = dungeonToken,
                EnterTime = session.EnterTime
            });
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, $"던전 입장 실패: UserId={userId}, DungeonId={request.DungeonId}");
            return (ErrorCode.DungeonEnterFailed, null);
        }
    }

    private async Task<int> GetTodayEntryCountAsync(long userId, int dungeonId)
    {
        var today = DateTime.UtcNow.Date;
        
        var entry = await _queryFactory.Query("player_dungeon_entry")
            .Where("user_id", userId)
            .Where("dungeon_id", dungeonId)
            .Where("entry_date", today)
            .FirstOrDefaultAsync<dynamic>();

        return entry?.entry_count ?? 0;
    }

    private async Task<ErrorCode> DeductEntryCostAsync(long userId, DungeonMaster dungeonMaster)
    {
        if (dungeonMaster.EntryCostAmount == 0)
        {
            return ErrorCode.None;
        }

        var columnName = dungeonMaster.EntryCostType.ToLower() == "gold" ? "gold" : "gem";

        var user = await _queryFactory.Query("game_user")
            .Where("user_id", userId)
            .FirstOrDefaultAsync<dynamic>();

        var currentAmount = columnName == "gold" ? (int)user.gold : (int)user.gem;

        if (currentAmount < dungeonMaster.EntryCostAmount)
        {
            return columnName == "gold" ? ErrorCode.InsufficientGold : ErrorCode.InsufficientGem;
        }

        var affectedRows = await _queryFactory.Query("game_user")
            .Where("user_id", userId)
            .UpdateAsync(new Dictionary<string, object>
            {
                { columnName, currentAmount - dungeonMaster.EntryCostAmount }
            });

        return affectedRows > 0 ? ErrorCode.None : ErrorCode.DungeonEntryCostDeductFailed;
    }

    private async Task IncrementEntryCountAsync(long userId, int dungeonId)
    {
        var today = DateTime.UtcNow.Date;

        var existing = await _queryFactory.Query("player_dungeon_entry")
            .Where("user_id", userId)
            .Where("dungeon_id", dungeonId)
            .Where("entry_date", today)
            .FirstOrDefaultAsync<dynamic>();

        if (existing == null)
        {
            await _queryFactory.Query("player_dungeon_entry")
                .InsertAsync(new
                {
                    user_id = userId,
                    dungeon_id = dungeonId,
                    entry_date = today,
                    entry_count = 1
                });
        }
        else
        {
            await _queryFactory.Query("player_dungeon_entry")
                .Where("user_id", userId)
                .Where("dungeon_id", dungeonId)
                .Where("entry_date", today)
                .UpdateAsync(new { entry_count = (int)existing.entry_count + 1 });
        }
    }

    private string GenerateDungeonToken(long userId, int dungeonId, int stageId)
    {
        var data = $"{userId}|{dungeonId}|{stageId}|{DateTime.UtcNow.Ticks}";
        var bytes = System.Text.Encoding.UTF8.GetBytes(data);
        return Convert.ToBase64String(bytes);
    }

    private string MakeRedisKey(long userId)
    {
        return $"dungeon_session:{userId}";
    }
}
```

## 19.4 던전 진행 상태 저장

던전 진행 중 플레이어의 상태를 Redis에 저장하여 클라이언트와 동기화한다.

### DungeonService.cs (진행 상태 저장 메서드 추가)

```csharp
public async Task<ErrorCode> UpdateDungeonProgressAsync(long userId, DungeonProgressRequest request)
{
    try
    {
        // 1. Redis에서 던전 세션 가져오기
        var redisKey = MakeRedisKey(userId);
        var session = await _redisDb.GetAsync<DungeonSession>(redisKey);

        if (session == null)
        {
            return ErrorCode.DungeonSessionNotFound;
        }

        // 2. 토큰 검증
        if (session.DungeonToken != request.DungeonToken)
        {
            _logger.LogWarning($"던전 토큰 불일치: UserId={userId}, Expected={session.DungeonToken}, Actual={request.DungeonToken}");
            return ErrorCode.InvalidDungeonToken;
        }

        // 3. 던전 및 스테이지 ID 검증
        if (session.DungeonId != request.DungeonId || session.StageId != request.StageId)
        {
            return ErrorCode.DungeonSessionMismatch;
        }

        // 4. 진행률 검증 (0~100)
        if (request.Progress < 0 || request.Progress > 100)
        {
            return ErrorCode.InvalidProgress;
        }

        // 5. 진행률 업데이트
        session.Progress = request.Progress;
        session.LastUpdateTime = DateTime.UtcNow;

        await _redisDb.SetAsync(redisKey, session, TimeSpan.FromSeconds(DungeonSessionExpirySeconds));

        _logger.LogInformation($"던전 진행 상태 업데이트: UserId={userId}, DungeonId={request.DungeonId}, StageId={request.StageId}, Progress={request.Progress}%");

        return ErrorCode.None;
    }
    catch (Exception ex)
    {
        _logger.LogError(ex, $"던전 진행 상태 업데이트 실패: UserId={userId}");
        return ErrorCode.DungeonProgressUpdateFailed;
    }
}
```

```csharp
public class DungeonProgressResponse
```

```csharp
public async Task<(ErrorCode, DungeonClearResponse)> ClearDungeonAsync(long userId, DungeonClearRequest request)
{
    try
    {
        // 1. Redis에서 던전 세션 가져오기
        var redisKey = MakeRedisKey(userId);
        var session = await _redisDb.GetAsync<DungeonSession>(redisKey);

        if (session == null)
        {
            return (ErrorCode.DungeonSessionNotFound, null);
        }

        // 2. 토큰 검증
        if (session.DungeonToken != request.DungeonToken)
        {
            _logger.LogWarning($"던전 토큰 불일치: UserId={userId}");
            return (ErrorCode.InvalidDungeonToken, null);
        }

        // 3. 던전 및 스테이지 ID 검증
        if (session.DungeonId != request.DungeonId || session.StageId != request.StageId)
        {
            return (ErrorCode.DungeonSessionMismatch, null);
        }

        // 4. 부정 클리어 방지 로직
        var antiCheatResult = ValidateClearTime(session, request);
        if (antiCheatResult != ErrorCode.None)
        {
            return (antiCheatResult, null);
        }

        // 5. 스테이지 마스터 데이터 가져오기
        var stageMaster = _masterDataService.GetStageMaster(request.StageId);
        if (stageMaster == null)
        {
            return (ErrorCode.StageNotFound, null);
        }

        // 6. 보상 계산
        var rewards = await CalculateRewardsAsync(request.StageId);

        // 7. 보상 지급
        var rewardResult = await GrantRewardsAsync(userId, stageMaster, rewards);
        if (rewardResult != ErrorCode.None)
        {
            return (rewardResult, null);
        }

        // 8. 클리어 기록 저장
        await SaveClearRecordAsync(userId, request);

        // 9. Redis 세션 삭제
        await _redisDb.DeleteAsync<DungeonSession>(redisKey);

        _logger.LogInformation($"던전 클리어 완료: UserId={userId}, DungeonId={request.DungeonId}, StageId={request.StageId}");

        return (ErrorCode.None, new DungeonClearResponse
        {
            Success = true,
            Exp = stageMaster.ClearExp,
            Gold = stageMaster.ClearGold,
            Items = rewards.Select(r => new RewardItem
            {
                ItemId = r.ItemId,
                ItemName = _masterDataService.GetItemMaster(r.ItemId)?.ItemName ?? "Unknown",
                Count = r.ItemCount
            }).ToList()
        });
    }
    catch (Exception ex)
    {
        _logger.LogError(ex, $"던전 클리어 처리 실패: UserId={userId}");
        return (ErrorCode.DungeonClearFailed, null);
    }
}

private ErrorCode ValidateClearTime(DungeonSession session, DungeonClearRequest request)
{
    // 최소 클리어 시간 검증 (예: 10초 이상)
    if (request.ClearTime < 10)
    {
        _logger.LogWarning($"비정상적으로 빠른 클리어: UserId={session.UserId}, ClearTime={request.ClearTime}초");
        return ErrorCode.SuspiciousClearTime;
    }

    // 세션 시작 시간과 클리어 시간 비교
    var elapsedTime = (DateTime.UtcNow - session.EnterTime).TotalSeconds;
    var timeDifference = Math.Abs(elapsedTime - request.ClearTime);

    // 허용 오차 범위 (예: 5초)
    if (timeDifference > 5)
    {
        _logger.LogWarning($"클리어 시간 불일치: UserId={session.UserId}, Elapsed={elapsedTime}초, Claimed={request.ClearTime}초");
        return ErrorCode.ClearTimeValidationFailed;
    }

    // 최대 클리어 시간 검증 (세션 만료 시간 이내)
    if (request.ClearTime > DungeonSessionExpirySeconds)
    {
        return ErrorCode.ClearTimeExceeded;
    }

    return ErrorCode.None;
}

private async Task<List<StageRewardItem>> CalculateRewardsAsync(int stageId)
{
    var allRewards = _masterDataService.GetStageRewards(stageId);
    var grantedRewards = new List<StageRewardItem>();
    var random = new Random();

    foreach (var reward in allRewards)
    {
        var roll = random.NextDouble() * 100;
        if (roll <= (double)reward.DropRate)
        {
            grantedRewards.Add(reward);
        }
    }

    return grantedRewards;
}

private async Task<ErrorCode> GrantRewardsAsync(long userId, StageMaster stageMaster, List<StageRewardItem> rewards)
{
    try
    {
        // 1. 경험치 및 골드 지급
        await _queryFactory.Query("game_user")
            .Where("user_id", userId)
            .IncrementAsync("exp", stageMaster.ClearExp);

        await _queryFactory.Query("game_user")
            .Where("user_id", userId)
            .IncrementAsync("gold", stageMaster.ClearGold);

        // 2. 아이템 지급
        foreach (var reward in rewards)
        {
            var existing = await _queryFactory.Query("player_item")
                .Where("user_id", userId)
                .Where("item_id", reward.ItemId)
                .FirstOrDefaultAsync<dynamic>();

            if (existing == null)
            {
                await _queryFactory.Query("player_item")
                    .InsertAsync(new
                    {
                        user_id = userId,
                        item_id = reward.ItemId,
                        item_count = reward.ItemCount
                    });
            }
            else
            {
                await _queryFactory.Query("player_item")
                    .Where("user_id", userId)
                    .Where("item_id", reward.ItemId)
                    .IncrementAsync("item_count", reward.ItemCount);
            }
        }

        return ErrorCode.None;
    }
    catch (Exception ex)
    {
        _logger.LogError(ex, $"보상 지급 실패: UserId={userId}");
        return ErrorCode.RewardGrantFailed;
    }
}

    await _queryFactory.Query("player_dungeon_clear")

            dungeon_id = request.DungeonId,

            score = request.Score,
            clear_date = DateTime.UtcNow.Date
        });
}
```

```csharp
public async Task<DungeonListResponse> GetDungeonListAsync(long userId)
{
    try
    {
        // 1. 사용자 정보 가져오기
        var user = await _queryFactory.Query("game_user")
            .Where("user_id", userId)
            .FirstOrDefaultAsync<dynamic>();

        if (user == null)
        {
            return new DungeonListResponse { Dungeons = new List<DungeonInfo>() };
        }

        var userLevel = (int)user.level;
        var userGold = (int)user.gold;
        var userGem = (int)user.gem;

        // 2. 오늘의 입장 횟수 가져오기
        var today = DateTime.UtcNow.Date;
        var entryRecords = await _queryFactory.Query("player_dungeon_entry")
            .Where("user_id", userId)
            .Where("entry_date", today)
            .GetAsync<dynamic>();

        var entryCountMap = entryRecords
            .ToDictionary(e => (int)e.dungeon_id, e => (int)e.entry_count);

        // 3. 모든 던전 마스터 데이터 가져오기 및 조건 검증
        var dungeons = _masterDataService.GetAllDungeons();
        var dungeonInfoList = new List<DungeonInfo>();

        foreach (var dungeon in dungeons)
        {
            var todayEntryCount = entryCountMap.TryGetValue(dungeon.DungeonId, out var count) ? count : 0;
            
            var canEnter = true;

            // 레벨 확인
            if (userLevel < dungeon.RequiredLevel)
            {
                canEnter = false;
            }

            // 입장 횟수 확인
            if (dungeon.MaxClearCount > 0 && todayEntryCount >= dungeon.MaxClearCount)
            {
                canEnter = false;
            }

            // 입장 비용 확인
            if (dungeon.EntryCostAmount > 0)
            {
                var hasEnoughCost = dungeon.EntryCostType.ToLower() == "gold" 
                    ? userGold >= dungeon.EntryCostAmount 
                    : userGem >= dungeon.EntryCostAmount;

                if (!hasEnoughCost)
                {
                    canEnter = false;
                }
            }

            dungeonInfoList.Add(new DungeonInfo
            {
                DungeonId = dungeon.DungeonId,
                DungeonName = dungeon.DungeonName,
                Description = dungeon.Description,
                RequiredLevel = dungeon.RequiredLevel,
                EntryCostType = dungeon.EntryCostType,
                EntryCostAmount = dungeon.EntryCostAmount,
                MaxClearCount = dungeon.MaxClearCount,
                TodayEntryCount = todayEntryCount,
                Difficulty = dungeon.Difficulty,
                CanEnter = canEnter
            });
        }

        return new DungeonListResponse
        {
            Dungeons = dungeonInfoList.OrderBy(d => d.RequiredLevel).ToList()
        };
    }
    catch (Exception ex)
    {
        _logger.LogError(ex, $"던전 목록 조회 실패: UserId={userId}");
        return new DungeonListResponse { Dungeons = new List<DungeonInfo>() };
    }
}
```

## 19.10 정리

이번 장에서는 던전 및 스테이지 시스템을 구현했다. 주요 내용은 다음과 같다.

- 던전, 스테이지, 보상 마스터 데이터 테이블 설계 및 생성
- 마스터 데이터를 메모리에 로딩하여 빠른 조회 구현
- Redis를 활용한 던전 세션 관리로 실시간 입장 상태 추적
- 입장 조건 검증 (레벨, 입장 비용, 일일 입장 횟수)
- 던전 진행 상태를 Redis에 저장하여 클라이언트와 동기화
- 클리어 시간 검증을 통한 부정 방지 로직 구현
- 확률 기반 보상 계산 및 지급
- SqlKata를 사용한 클리어 기록 및 보상 지급 처리
- 던전 목록 조회 API로 입장 가능 여부 제공

던전 시스템은 수집형 RPG 게임의 핵심 콘텐츠로, Redis를 활용한 실시간 상태 관리와 부정 방지 로직이 중요하다. 다음 장에서는 플레이어 참여를 유도하는 출석 및 미션 시스템을 구현한다.
