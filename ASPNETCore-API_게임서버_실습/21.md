# 21장. 친구 및 소셜 기능 (실습)

이번 장에서는 플레이어 간 소셜 상호작용을 지원하는 친구 시스템을 구현한다. 친구 검색, 친구 추가/삭제, 친구 목록 조회, 친구에게 선물 보내기 등의 기능을 다룬다. Redis Set을 활용하여 친구 관계를 캐싱하고 빠른 조회가 가능하도록 구현한다.

## 21.1 친구 관리 테이블 설계

친구 관계 및 친구 요청을 관리하는 테이블을 설계한다.

### 친구 관계 테이블

```sql
-- 친구 관계 테이블
CREATE TABLE player_friend (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT NOT NULL,
    friend_user_id BIGINT NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    UNIQUE KEY unique_friendship (user_id, friend_user_id),
    INDEX idx_user_id (user_id),
    INDEX idx_friend_user_id (friend_user_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 친구 요청 테이블
CREATE TABLE friend_request (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    requester_user_id BIGINT NOT NULL,
    receiver_user_id BIGINT NOT NULL,
    status VARCHAR(20) NOT NULL DEFAULT 'pending', -- 'pending', 'accepted', 'rejected'
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    UNIQUE KEY unique_request (requester_user_id, receiver_user_id),
    INDEX idx_requester (requester_user_id),
    INDEX idx_receiver (receiver_user_id),
    INDEX idx_status (status)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 친구 선물 기록
CREATE TABLE friend_gift (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    sender_user_id BIGINT NOT NULL,
    receiver_user_id BIGINT NOT NULL,
    item_id INT NOT NULL,
    item_count INT NOT NULL DEFAULT 1,
    message TEXT,
    received BOOLEAN DEFAULT FALSE,
    sent_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    received_at DATETIME,
    INDEX idx_sender (sender_user_id),
    INDEX idx_receiver (receiver_user_id),
    INDEX idx_received (received)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 일일 선물 보내기 제한
CREATE TABLE friend_daily_gift_limit (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT NOT NULL,
    send_date DATE NOT NULL,
    send_count INT NOT NULL DEFAULT 0,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    UNIQUE KEY unique_user_date (user_id, send_date),
    INDEX idx_user_id (user_id),
    INDEX idx_send_date (send_date)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

## 21.3 친구 서비스 구현

친구 관련 비즈니스 로직을 처리하는 서비스를 구현한다.

### Services/FriendService.cs

```csharp
namespace GameApiServer.Services;

public class FriendService
{
    private readonly ILogger<FriendService> _logger;
    private readonly RedisDb _redisDb;
    private readonly QueryFactory _queryFactory;
    private readonly MasterDataService _masterDataService;
    private readonly MissionTrackerService _missionTracker;
    private const int MaxDailyGifts = 10;

    public FriendService(
        ILogger<FriendService> logger,
        RedisDb redisDb,
        IDbConnection dbConnection,
        MasterDataService masterDataService,
        MissionTrackerService missionTracker)
    {
        _logger = logger;
        _redisDb = redisDb;
        _queryFactory = new QueryFactory(dbConnection, new MySqlCompiler());
        _masterDataService = masterDataService;
        _missionTracker = missionTracker;
    }

    public async Task<FriendSearchResponse> SearchUsersAsync(long userId, string searchKeyword)
    {
        try
        {
            if (string.IsNullOrWhiteSpace(searchKeyword))
            {
                return new FriendSearchResponse { Results = new List<UserSearchResult>() };
            }

            // 1. 사용자 검색 (닉네임 또는 이메일)
            var users = await _queryFactory.Query("game_user")
                .Where(q => q
                    .WhereLike("email", $"%{searchKeyword}%")
                    .OrWhereLike("nickname", $"%{searchKeyword}%"))
                .Where("user_id", "!=", userId) // 자기 자신 제외
                .Limit(20)
                .GetAsync<dynamic>();

            // 2. 친구 관계 확인
            var friendIds = await GetFriendIdsAsync(userId);

            // 3. 대기 중인 친구 요청 확인
            var pendingRequests = await _queryFactory.Query("friend_request")
                .Where("requester_user_id", userId)
                .Where("status", "pending")
                .GetAsync<dynamic>();

            var pendingUserIds = pendingRequests.Select(r => (long)r.receiver_user_id).ToHashSet();

            // 4. 검색 결과 구성
            var results = new List<UserSearchResult>();

            foreach (var user in users)
            {
                var targetUserId = (long)user.user_id;

                results.Add(new UserSearchResult
                {
                    UserId = targetUserId,
                    Email = user.email,
                    Nickname = user.nickname,
                    Level = user.level,
                    IsFriend = friendIds.Contains(targetUserId),
                    HasPendingRequest = pendingUserIds.Contains(targetUserId)
                });
            }

            return new FriendSearchResponse { Results = results };
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, $"사용자 검색 실패: UserId={userId}, Keyword={searchKeyword}");
            return new FriendSearchResponse { Results = new List<UserSearchResult>() };
        }
    }

    public async Task<ErrorCode> SendFriendRequestAsync(long userId, long targetUserId)
    {
        try
        {
            // 1. 자기 자신에게 요청 방지
            if (userId == targetUserId)
            {
                return ErrorCode.CannotAddSelfAsFriend;
            }

            // 2. 대상 사용자 존재 확인
            var targetUser = await _queryFactory.Query("game_user")
                .Where("user_id", targetUserId)
                .FirstOrDefaultAsync<dynamic>();

            if (targetUser == null)
            {
                return ErrorCode.UserNotFound;
            }

            // 3. 이미 친구인지 확인
            var isFriend = await IsFriendAsync(userId, targetUserId);
            if (isFriend)
            {
                return ErrorCode.AlreadyFriend;
            }

            // 4. 기존 요청 확인
            var existingRequest = await _queryFactory.Query("friend_request")
                .Where("requester_user_id", userId)
                .Where("receiver_user_id", targetUserId)
                .Where("status", "pending")
                .FirstOrDefaultAsync<dynamic>();

            if (existingRequest != null)
            {
                return ErrorCode.FriendRequestAlreadySent;
            }

            // 5. 역방향 요청 확인 (상대방이 이미 요청을 보낸 경우 자동 수락)
            var reverseRequest = await _queryFactory.Query("friend_request")
                .Where("requester_user_id", targetUserId)
                .Where("receiver_user_id", userId)
                .Where("status", "pending")
                .FirstOrDefaultAsync<dynamic>();

            if (reverseRequest != null)
            {
                // 자동으로 친구 관계 생성
                await CreateFriendshipAsync(userId, targetUserId);
                
                // 역방향 요청 상태를 accepted로 변경
                await _queryFactory.Query("friend_request")
                    .Where("id", reverseRequest.id)
                    .UpdateAsync(new { status = "accepted" });

                _logger.LogInformation($"역방향 요청으로 자동 친구 추가: UserId={userId}, TargetUserId={targetUserId}");
                
                await _missionTracker.TrackAddFriendAsync(userId);
                await _missionTracker.TrackAddFriendAsync(targetUserId);

                return ErrorCode.None;
            }

            // 6. 친구 요청 생성
            await _queryFactory.Query("friend_request")
                .InsertAsync(new
                {
                    requester_user_id = userId,
                    receiver_user_id = targetUserId,
                    status = "pending"
                });

            _logger.LogInformation($"친구 요청 전송: UserId={userId}, TargetUserId={targetUserId}");

            return ErrorCode.None;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, $"친구 요청 실패: UserId={userId}, TargetUserId={targetUserId}");
            return ErrorCode.FriendRequestFailed;
        }
    }

    public async Task<ErrorCode> AcceptFriendRequestAsync(long userId, long requestId)
    {
        try
        {
            // 1. 친구 요청 조회
            var request = await _queryFactory.Query("friend_request")
                .Where("id", requestId)
                .Where("receiver_user_id", userId)
                .Where("status", "pending")
                .FirstOrDefaultAsync<dynamic>();

            if (request == null)
            {
                return ErrorCode.FriendRequestNotFound;
            }

            var requesterId = (long)request.requester_user_id;

            // 2. 친구 관계 생성
            await CreateFriendshipAsync(userId, requesterId);

            // 3. 요청 상태 업데이트
            await _queryFactory.Query("friend_request")
                .Where("id", requestId)
                .UpdateAsync(new { status = "accepted" });

            _logger.LogInformation($"친구 요청 수락: UserId={userId}, RequesterId={requesterId}");

            // 4. 미션 추적
            await _missionTracker.TrackAddFriendAsync(userId);
            await _missionTracker.TrackAddFriendAsync(requesterId);

            return ErrorCode.None;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, $"친구 요청 수락 실패: UserId={userId}, RequestId={requestId}");
            return ErrorCode.FriendRequestAcceptFailed;
        }
    }

    public async Task<ErrorCode> RejectFriendRequestAsync(long userId, long requestId)
    {
        try
        {
            // 1. 친구 요청 조회
            var request = await _queryFactory.Query("friend_request")
                .Where("id", requestId)
                .Where("receiver_user_id", userId)
                .Where("status", "pending")
                .FirstOrDefaultAsync<dynamic>();

            if (request == null)
            {
                return ErrorCode.FriendRequestNotFound;
            }

            // 2. 요청 상태 업데이트
            await _queryFactory.Query("friend_request")
                .Where("id", requestId)
                .UpdateAsync(new { status = "rejected" });

            _logger.LogInformation($"친구 요청 거절: UserId={userId}, RequestId={requestId}");

            return ErrorCode.None;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, $"친구 요청 거절 실패: UserId={userId}, RequestId={requestId}");
            return ErrorCode.FriendRequestRejectFailed;
        }
    }

    public async Task<FriendListResponse> GetFriendListAsync(long userId)
    {
        try
        {
            // 1. Redis에서 친구 ID 목록 가져오기
            var friendIds = await GetFriendIdsAsync(userId);

            if (friendIds.Count == 0)
            {
                return new FriendListResponse 
                { 
                    Friends = new List<FriendInfo>(),
                    TotalCount = 0
                };
            }

            // 2. 친구 정보 조회
            var friends = await _queryFactory.Query("game_user")
                .WhereIn("user_id", friendIds)
                .GetAsync<dynamic>();

            // 3. 친구 관계 생성 시간 조회
            var friendships = await _queryFactory.Query("player_friend")
                .Where("user_id", userId)
                .WhereIn("friend_user_id", friendIds)
                .GetAsync<dynamic>();

            var friendSinceMap = friendships.ToDictionary(
                f => (long)f.friend_user_id,
                f => (DateTime)f.created_at
            );

            // 4. 친구 목록 구성
            var friendList = new List<FriendInfo>();

            foreach (var friend in friends)
            {
                var friendUserId = (long)friend.user_id;

                friendList.Add(new FriendInfo
                {
                    UserId = friendUserId,
                    Email = friend.email,
                    Nickname = friend.nickname,
                    Level = friend.level,
                    FriendSince = friendSinceMap.TryGetValue(friendUserId, out var since) ? since : DateTime.UtcNow,
                    IsOnline = await IsUserOnlineAsync(friendUserId)
                });
            }

            return new FriendListResponse
            {
                Friends = friendList.OrderByDescending(f => f.IsOnline).ThenBy(f => f.Nickname).ToList(),
                TotalCount = friendList.Count
            };
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, $"친구 목록 조회 실패: UserId={userId}");
            return new FriendListResponse 
            { 
                Friends = new List<FriendInfo>(),
                TotalCount = 0
            };
        }
    }

    public async Task<FriendRequestListResponse> GetFriendRequestListAsync(long userId)
    {
        try
        {
            // 1. 받은 친구 요청 조회
            var requests = await _queryFactory.Query("friend_request as fr")
                .Join("game_user as gu", "fr.requester_user_id", "gu.user_id")
                .Where("fr.receiver_user_id", userId)
                .Where("fr.status", "pending")
                .Select("fr.id as request_id", "fr.requester_user_id", "gu.nickname", "gu.level", "fr.created_at")
                .GetAsync<dynamic>();

            var requestList = new List<FriendRequestInfo>();

            foreach (var req in requests)
            {
                requestList.Add(new FriendRequestInfo
                {
                    RequestId = req.request_id,
                    RequesterId = req.requester_user_id,
                    RequesterNickname = req.nickname,
                    RequesterLevel = req.level,
                    RequestedAt = req.created_at
                });
            }

            return new FriendRequestListResponse
            {
                Requests = requestList.OrderByDescending(r => r.RequestedAt).ToList()
            };
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, $"친구 요청 목록 조회 실패: UserId={userId}");
            return new FriendRequestListResponse { Requests = new List<FriendRequestInfo>() };
        }
    }

    public async Task<ErrorCode> DeleteFriendAsync(long userId, long friendUserId)
    {
        try
        {
            // 1. 친구 관계 확인
            var isFriend = await IsFriendAsync(userId, friendUserId);
            if (!isFriend)
            {
                return ErrorCode.NotFriend;
            }

            // 2. 양방향 친구 관계 삭제
            await _queryFactory.Query("player_friend")
                .Where(q => q
                    .Where(w => w.Where("user_id", userId).Where("friend_user_id", friendUserId))
                    .OrWhere(w => w.Where("user_id", friendUserId).Where("friend_user_id", userId)))
                .DeleteAsync();

            // 3. Redis 캐시 업데이트
            await RemoveFriendFromCacheAsync(userId, friendUserId);
            await RemoveFriendFromCacheAsync(friendUserId, userId);

            _logger.LogInformation($"친구 삭제: UserId={userId}, FriendUserId={friendUserId}");

            return ErrorCode.None;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, $"친구 삭제 실패: UserId={userId}, FriendUserId={friendUserId}");
            return ErrorCode.FriendDeleteFailed;
        }
    }

    private async Task CreateFriendshipAsync(long userId1, long userId2)
    {
        // 양방향 친구 관계 생성
        await _queryFactory.Query("player_friend")
            .InsertAsync(new[]
            {
                new { user_id = userId1, friend_user_id = userId2 },
                new { user_id = userId2, friend_user_id = userId1 }
            });

        // Redis 캐시에 추가
        await AddFriendToCacheAsync(userId1, userId2);
        await AddFriendToCacheAsync(userId2, userId1);
    }

    private async Task<bool> IsFriendAsync(long userId, long targetUserId)
    {
        var friendIds = await GetFriendIdsAsync(userId);
        return friendIds.Contains(targetUserId);
    }

    private async Task<HashSet<long>> GetFriendIdsAsync(long userId)
    {
        var redisKey = MakeFriendCacheKey(userId);
        var cached = await _redisDb.GetAsync<HashSet<long>>(redisKey);

        if (cached != null && cached.Count > 0)
        {
            return cached;
        }

        // DB에서 친구 목록 조회
        var friends = await _queryFactory.Query("player_friend")
            .Where("user_id", userId)
            .GetAsync<dynamic>();

        var friendIds = friends.Select(f => (long)f.friend_user_id).ToHashSet();

        // Redis에 캐싱 (1시간)
        if (friendIds.Count > 0)
        {
            await _redisDb.SetAsync(redisKey, friendIds, TimeSpan.FromHours(1));
        }

        return friendIds;
    }

    private async Task AddFriendToCacheAsync(long userId, long friendUserId)
    {
        var redisKey = MakeFriendCacheKey(userId);
        var friendIds = await GetFriendIdsAsync(userId);
        friendIds.Add(friendUserId);
        await _redisDb.SetAsync(redisKey, friendIds, TimeSpan.FromHours(1));
    }

    private async Task RemoveFriendFromCacheAsync(long userId, long friendUserId)
    {
        var redisKey = MakeFriendCacheKey(userId);
        var friendIds = await GetFriendIdsAsync(userId);
        friendIds.Remove(friendUserId);
        
        if (friendIds.Count > 0)
        {
            await _redisDb.SetAsync(redisKey, friendIds, TimeSpan.FromHours(1));
        }
        else
        {
            await _redisDb.DeleteAsync<HashSet<long>>(redisKey);
        }
    }

    private async Task<bool> IsUserOnlineAsync(long userId)
    {
        // 온라인 상태는 인증 토큰 존재 여부로 확인
        var authKey = $"auth:{userId}";
        var authData = await _redisDb.GetAsync<dynamic>(authKey);
        return authData != null;
    }

    private string MakeFriendCacheKey(long userId)
    {
        return $"friends:{userId}";
    }
}
```

## 21.4 선물 기능 구현

친구에게 아이템을 선물하는 기능을 구현한다.

### Services/FriendGiftService.cs

```csharp
namespace GameApiServer.Services;

public class FriendGiftService
{
    private readonly ILogger<FriendGiftService> _logger;
    private readonly QueryFactory _queryFactory;
    private readonly MasterDataService _masterDataService;
    private readonly FriendService _friendService;
    private readonly MissionTrackerService _missionTracker;
    private const int MaxDailyGifts = 10;

    public FriendGiftService(
        ILogger<FriendGiftService> logger,
        IDbConnection dbConnection,
        MasterDataService masterDataService,
        FriendService friendService,
        MissionTrackerService missionTracker)
    {
        _logger = logger;
        _queryFactory = new QueryFactory(dbConnection, new MySqlCompiler());
        _masterDataService = masterDataService;
        _friendService = friendService;
        _missionTracker = missionTracker;
    }

    public async Task<(ErrorCode, SendGiftResponse)> SendGiftAsync(long userId, SendGiftRequest request)
    {
        try
        {
            // 1. 친구 관계 확인
            var isFriend = await _friendService.IsFriendAsync(userId, request.FriendUserId);
            if (!isFriend)
            {
                return (ErrorCode.NotFriend, null);
            }

            // 2. 아이템 마스터 데이터 확인
            var itemMaster = _masterDataService.GetItemMaster(request.ItemId);
            if (itemMaster == null)
            {
                return (ErrorCode.ItemNotFound, null);
            }

            // 3. 일일 선물 제한 확인
            var today = DateTime.UtcNow.Date;
            var dailyLimit = await GetOrCreateDailyLimitAsync(userId, today);

            if (dailyLimit.SendCount >= MaxDailyGifts)
            {
                return (ErrorCode.DailyGiftLimitExceeded, null);
            }

            // 4. 플레이어 아이템 소유 확인
            var playerItem = await _queryFactory.Query("player_item")
                .Where("user_id", userId)
                .Where("item_id", request.ItemId)
                .FirstOrDefaultAsync<dynamic>();

            if (playerItem == null || (int)playerItem.item_count < request.ItemCount)
            {
                return (ErrorCode.InsufficientItem, null);
            }

            // 5. 아이템 차감
            await _queryFactory.Query("player_item")
                .Where("user_id", userId)
                .Where("item_id", request.ItemId)
                .DecrementAsync("item_count", request.ItemCount);

            // 6. 선물 기록 생성
            await _queryFactory.Query("friend_gift")
                .InsertAsync(new
                {
                    sender_user_id = userId,
                    receiver_user_id = request.FriendUserId,
                    item_id = request.ItemId,
                    item_count = request.ItemCount,
                    message = request.Message ?? ""
                });

            // 7. 일일 선물 카운트 증가
            await _queryFactory.Query("friend_daily_gift_limit")
                .Where("user_id", userId)
                .Where("send_date", today)
                .IncrementAsync("send_count", 1);

            var remainingGifts = MaxDailyGifts - (dailyLimit.SendCount + 1);

            _logger.LogInformation($"선물 전송: SenderId={userId}, ReceiverId={request.FriendUserId}, ItemId={request.ItemId}, Count={request.ItemCount}");

            // 8. 미션 추적
            await _missionTracker.TrackSendGiftAsync(userId);

            return (ErrorCode.None, new SendGiftResponse
            {
                Success = true,
                RemainingGiftsToday = remainingGifts
            });
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, $"선물 전송 실패: UserId={userId}");
            return (ErrorCode.SendGiftFailed, null);
        }
    }

    public async Task<ReceivedGiftListResponse> GetReceivedGiftListAsync(long userId)
    {
        try
        {
            // 1. 받은 선물 조회
            var gifts = await _queryFactory.Query("friend_gift as fg")
                .Join("game_user as gu", "fg.sender_user_id", "gu.user_id")
                .Where("fg.receiver_user_id", userId)
                .Where("fg.received", false)
                .Select(
                    "fg.id as gift_id",
                    "fg.sender_user_id",
                    "gu.nickname as sender_nickname",
                    "fg.item_id",
                    "fg.item_count",
                    "fg.message",
                    "fg.sent_at"
                )
                .OrderByDesc("fg.sent_at")
                .GetAsync<dynamic>();

            var giftList = new List<GiftInfo>();

            foreach (var gift in gifts)
            {
                var itemMaster = _masterDataService.GetItemMaster((int)gift.item_id);

                giftList.Add(new GiftInfo
                {
                    GiftId = gift.gift_id,
                    SenderId = gift.sender_user_id,
                    SenderNickname = gift.sender_nickname,
                    ItemId = gift.item_id,
                    ItemName = itemMaster?.ItemName ?? "Unknown",
                    ItemCount = gift.item_count,
                    Message = gift.message,
                    SentAt = gift.sent_at
                });
            }

            return new ReceivedGiftListResponse { Gifts = giftList };
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, $"받은 선물 목록 조회 실패: UserId={userId}");
            return new ReceivedGiftListResponse { Gifts = new List<GiftInfo>() };
        }
    }

    public async Task<ErrorCode> ReceiveGiftAsync(long userId, long giftId)
    {
        try
        {
            // 1. 선물 정보 조회
            var gift = await _queryFactory.Query("friend_gift")
                .Where("id", giftId)
                .Where("receiver_user_id", userId)
                .Where("received", false)
                .FirstOrDefaultAsync<dynamic>();

            if (gift == null)
            {
                return ErrorCode.GiftNotFound;
            }

            // 2. 아이템 지급
            var itemId = (int)gift.item_id;
            var itemCount = (int)gift.item_count;

            var existingItem = await _queryFactory.Query("player_item")
                .Where("user_id", userId)
                .Where("item_id", itemId)
                .FirstOrDefaultAsync<dynamic>();

            if (existingItem == null)
            {
                await _queryFactory.Query("player_item")
                    .InsertAsync(new
                    {
                        user_id = userId,
                        item_id = itemId,
                        item_count = itemCount
                    });
            }
            else
            {
                await _queryFactory.Query("player_item")
                    .Where("user_id", userId)
                    .Where("item_id", itemId)
                    .IncrementAsync("item_count", itemCount);
            }

            // 3. 선물 수령 처리
            await _queryFactory.Query("friend_gift")
                .Where("id", giftId)
                .UpdateAsync(new
                {
                    received = true,
                    received_at = DateTime.UtcNow
                });

            _logger.LogInformation($"선물 수령: UserId={userId}, GiftId={giftId}, ItemId={itemId}, Count={itemCount}");

            return ErrorCode.None;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, $"선물 수령 실패: UserId={userId}, GiftId={giftId}");
            return ErrorCode.ReceiveGiftFailed;
        }
    }

    private async Task<dynamic> GetOrCreateDailyLimitAsync(long userId, DateTime date)
    {
        var limit = await _queryFactory.Query("friend_daily_gift_limit")
            .Where("user_id", userId)
            .Where("send_date", date)
            .FirstOrDefaultAsync<dynamic>();

        if (limit == null)
        {
            await _queryFactory.Query("friend_daily_gift_limit")
                .InsertAsync(new
                {
                    user_id = userId,
                    send_date = date,
                    send_count = 0
                });

            limit = await _queryFactory.Query("friend_daily_gift_limit")
                .Where("user_id", userId)
                .Where("send_date", date)
                .FirstAsync<dynamic>();
        }

        return limit;
    }
}
```

## 21.10 정리

이번 장에서는 친구 및 소셜 기능을 구현했다. 주요 내용은 다음과 같다.

- 친구 관계, 친구 요청, 선물 관련 테이블 설계 및 생성
- Redis Set을 활용한 친구 ID 목록 캐싱으로 빠른 친구 관계 확인
- 닉네임 및 이메일 기반 사용자 검색 기능
- 친구 요청 전송, 수락, 거절 기능 구현
- 양방향 친구 관계 자동 처리 (역방향 요청 시 자동 수락)
- 친구 목록 조회 및 온라인 상태 표시
- 친구 삭제 시 양방향 관계 및 캐시 동시 삭제
- 친구에게 아이템 선물 보내기 기능
- 일일 선물 보내기 제한 관리
- 받은 선물 목록 조회 및 수령 처리
- 선물 보내기 시 아이템 소유 여부 검증
- 미션 시스템과 연동하여 친구 추가 및 선물 보내기 추적

친구 시스템은 소셜 상호작용을 통해 플레이어 리텐션을 높이는 중요한 기능이다. Redis를 활용한 캐싱과 효율적인 친구 관계 검증으로 성능을 최적화했다. 다음 장에서는 Redis Sorted Set을 활용한 실시간 랭킹 시스템을 구현한다.
