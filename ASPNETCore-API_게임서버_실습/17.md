# 17장. 우편함 시스템 (실습)

이번 장에서는 게임 운영에 필수적인 우편함 시스템을 구현한다. 우편함은 운영자가 사용자에게 보상을 지급하거나, 시스템에서 자동으로 아이템을 전달하는 용도로 활용된다. Redis를 이용한 알림 카운트 관리와 수신자 검증 로직을 중점적으로 학습한다.

## 17.1 우편함 시스템 개요

우편함 시스템은 비동기적으로 사용자에게 아이템이나 재화를 전달하는 메커니즘이다. 게임 내에서 다음과 같은 상황에 사용된다.

운영자가 전체 사용자나 특정 사용자에게 보상을 지급할 때 사용한다. 이벤트 참여 보상이나 서버 점검 보상 등이 이에 해당한다. 친구가 보낸 선물이나 길드 보상처럼 사용자 간 아이템 전달에도 활용된다. 시스템에서 자동으로 지급하는 일일 로그인 보상이나 랭킹 보상도 우편함을 통해 전달된다.

우편함 시스템의 핵심은 수신자 검증과 중복 수령 방지다. 또한 읽지 않은 우편 개수를 실시간으로 관리하여 사용자에게 알림을 제공해야 한다.

## 17.2 데이터베이스 테이블 설계

우편함 데이터를 저장할 테이블을 생성한다. 우편의 메타 정보와 첨부된 아이템 정보를 분리하여 설계한다.

```sql
CREATE TABLE user_mail (
    mail_id BIGINT PRIMARY KEY AUTO_INCREMENT,
    receiver_id BIGINT NOT NULL COMMENT '수신자 ID',
    sender_type TINYINT NOT NULL COMMENT '1:시스템, 2:운영자, 3:사용자',
    sender_id BIGINT DEFAULT NULL COMMENT '발신자 ID (사용자인 경우)',
    mail_title VARCHAR(100) NOT NULL,
    mail_content VARCHAR(500) NOT NULL,
    is_read BOOLEAN NOT NULL DEFAULT false,
    is_received BOOLEAN NOT NULL DEFAULT false COMMENT '아이템 수령 여부',
    expire_at DATETIME NOT NULL COMMENT '만료 시간',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    received_at DATETIME DEFAULT NULL COMMENT '수령 시간',
    INDEX idx_receiver (receiver_id, is_received, expire_at),
    INDEX idx_receiver_read (receiver_id, is_read),
    FOREIGN KEY (receiver_id) REFERENCES users(user_id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

우편에 첨부된 아이템 정보를 저장하는 테이블을 생성한다.

```sql
CREATE TABLE mail_item (
    mail_item_id BIGINT PRIMARY KEY AUTO_INCREMENT,
    mail_id BIGINT NOT NULL,
    item_type TINYINT NOT NULL COMMENT '1:아이템, 2:골드, 3:다이아',
    item_id INT DEFAULT NULL COMMENT '아이템 타입인 경우',
    item_count INT NOT NULL,
    INDEX idx_mail_id (mail_id),
    FOREIGN KEY (mail_id) REFERENCES user_mail(mail_id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

## 17.4 우편함 Repository 구현

우편 데이터를 MySQL에서 조회하고 수정하는 Repository를 구현한다.

```csharp
// Repositories/MailRepository.cs
using GameServer.Models;
using SqlKata.Execution;

namespace GameServer.Repositories
{
    public class MailRepository
    {
        private readonly QueryFactory _queryFactory;

        public MailRepository(QueryFactory queryFactory)
        {
            _queryFactory = queryFactory;
        }

        public async Task<long> CreateMailAsync(UserMail mail)
        {
            var mailId = await _queryFactory.Query("user_mail")
                .InsertGetIdAsync<long>(new
                {
                    receiver_id = mail.ReceiverId,
                    sender_type = mail.SenderType,
                    sender_id = mail.SenderId,
                    mail_title = mail.MailTitle,
                    mail_content = mail.MailContent,
                    is_read = false,
                    is_received = false,
                    expire_at = mail.ExpireAt,
                    created_at = DateTime.Now
                });

            return mailId;
        }

        public async Task<bool> CreateMailItemsAsync(long mailId, List<MailItem> items)
        {
            var insertData = items.Select(item => new
            {
                mail_id = mailId,
                item_type = item.ItemType,
                item_id = item.ItemId,
                item_count = item.ItemCount
            }).ToList();

            var affected = await _queryFactory.Query("mail_item")
                .InsertAsync(insertData);

            return affected > 0;
        }

        public async Task<List<UserMail>> GetUserMailListAsync(long userId, int pageNumber, int pageSize)
        {
            var offset = (pageNumber - 1) * pageSize;

            var mails = await _queryFactory.Query("user_mail")
                .Where("receiver_id", userId)
                .Where("expire_at", ">", DateTime.Now)
                .OrderByDesc("created_at")
                .Offset(offset)
                .Limit(pageSize)
                .GetAsync<UserMail>();

            return mails.ToList();
        }

        public async Task<int> GetUserMailCountAsync(long userId)
        {
            var count = await _queryFactory.Query("user_mail")
                .Where("receiver_id", userId)
                .Where("expire_at", ">", DateTime.Now)
                .CountAsync<int>();

            return count;
        }

        public async Task<int> GetUnreadMailCountAsync(long userId)
        {
            var count = await _queryFactory.Query("user_mail")
                .Where("receiver_id", userId)
                .Where("is_read", false)
                .Where("expire_at", ">", DateTime.Now)
                .CountAsync<int>();

            return count;
        }

        public async Task<UserMail> GetMailAsync(long mailId)
        {
            var mail = await _queryFactory.Query("user_mail")
                .Where("mail_id", mailId)
                .FirstOrDefaultAsync<UserMail>();

            return mail;
        }

        public async Task<List<MailItem>> GetMailItemsAsync(long mailId)
        {
            var items = await _queryFactory.Query("mail_item")
                .Where("mail_id", mailId)
                .GetAsync<MailItem>();

            return items.ToList();
        }

        public async Task<bool> MarkAsReadAsync(long mailId)
        {
            var affected = await _queryFactory.Query("user_mail")
                .Where("mail_id", mailId)
                .UpdateAsync(new { is_read = true });

            return affected > 0;
        }

        public async Task<bool> MarkAsReceivedAsync(long mailId)
        {
            var affected = await _queryFactory.Query("user_mail")
                .Where("mail_id", mailId)
                .UpdateAsync(new
                {
                    is_received = true,
                    received_at = DateTime.Now
                });

            return affected > 0;
        }

        public async Task<bool> DeleteMailAsync(long mailId)
        {
            var affected = await _queryFactory.Query("user_mail")
                .Where("mail_id", mailId)
                .DeleteAsync();

            return affected > 0;
        }

        public async Task<int> DeleteExpiredMailsAsync()
        {
            var affected = await _queryFactory.Query("user_mail")
                .Where("expire_at", "<=", DateTime.Now)
                .DeleteAsync();

            return affected;
        }
    }
}
```

## 17.5 Redis를 이용한 알림 카운트 관리

읽지 않은 우편 개수를 Redis에 캐싱하여 빠르게 조회할 수 있도록 구현한다.

```csharp
// Services/MailNotificationService.cs
using CloudStructures;
using CloudStructures.Structures;

namespace GameServer.Services
{
    public class MailNotificationService
    {
        private readonly RedisConnection _redisConnection;
        private readonly TimeSpan _cacheExpiry = TimeSpan.FromHours(1);

        public MailNotificationService(RedisConnection redisConnection)
        {
            _redisConnection = redisConnection;
        }

        private string GetUnreadCountKey(long userId)
        {
            return $"mail:unread:{userId}";
        }

        public async Task<int?> GetUnreadCountAsync(long userId)
        {
            var redis = new RedisString<int>(_redisConnection, GetUnreadCountKey(userId), _cacheExpiry);
            var result = await redis.GetAsync();

            return result.HasValue ? result.Value : null;
        }

        public async Task SetUnreadCountAsync(long userId, int count)
        {
            var redis = new RedisString<int>(_redisConnection, GetUnreadCountKey(userId), _cacheExpiry);
            await redis.SetAsync(count, _cacheExpiry);
        }

        public async Task IncrementUnreadCountAsync(long userId)
        {
            var redis = new RedisString<int>(_redisConnection, GetUnreadCountKey(userId), _cacheExpiry);
            await redis.IncrementAsync(1);
        }

        public async Task DecrementUnreadCountAsync(long userId)
        {
            var redis = new RedisString<int>(_redisConnection, GetUnreadCountKey(userId), _cacheExpiry);
            var currentValue = await redis.GetAsync();
            
            if (currentValue.HasValue && currentValue.Value > 0)
            {
                await redis.IncrementAsync(-1);
            }
        }

        public async Task InvalidateUnreadCountAsync(long userId)
        {
            var redis = new RedisString<int>(_redisConnection, GetUnreadCountKey(userId), _cacheExpiry);
            await redis.DeleteAsync();
        }
    }
}
```

## 17.6 우편함 서비스 구현

우편함 관련 비즈니스 로직을 처리하는 서비스를 구현한다.

```csharp
// Services/MailService.cs
using GameServer.DTOs;
using GameServer.Models;
using GameServer.Repositories;

namespace GameServer.Services
{
    public class MailService
    {
        private readonly MailRepository _mailRepository;
        private readonly MailNotificationService _notificationService;
        private readonly InventoryService _inventoryService;
        private readonly MasterDataService _masterDataService;

        public MailService(
            MailRepository mailRepository,
            MailNotificationService notificationService,
            InventoryService inventoryService,
            MasterDataService masterDataService)
        {
            _mailRepository = mailRepository;
            _notificationService = notificationService;
            _inventoryService = inventoryService;
            _masterDataService = masterDataService;
        }

        public async Task<SendMailResponse> SendMailAsync(
            long receiverId,
            byte senderType,
            long? senderId,
            string title,
            string content,
            List<MailItemDto> items,
            int expireDays)
        {
            // 우편 제목과 내용을 검증한다
            if (string.IsNullOrWhiteSpace(title) || string.IsNullOrWhiteSpace(content))
            {
                return new SendMailResponse { Result = 1 }; // 잘못된 입력
            }

            // 첨부 아이템을 검증한다
            if (items != null && items.Count > 0)
            {
                foreach (var item in items)
                {
                    if (item.ItemType == (byte)MailItemType.Item)
                    {
                        if (!item.ItemId.HasValue || !_masterDataService.IsValidItem(item.ItemId.Value))
                        {
                            return new SendMailResponse { Result = 2 }; // 잘못된 아이템
                        }
                    }

                    if (item.ItemCount <= 0)
                    {
                        return new SendMailResponse { Result = 3 }; // 잘못된 수량
                    }
                }
            }

            // 우편을 생성한다
            var mail = new UserMail
            {
                ReceiverId = receiverId,
                SenderType = senderType,
                SenderId = senderId,
                MailTitle = title,
                MailContent = content,
                ExpireAt = DateTime.Now.AddDays(expireDays)
            };

            var mailId = await _mailRepository.CreateMailAsync(mail);

            // 첨부 아이템을 저장한다
            if (items != null && items.Count > 0)
            {
                var mailItems = items.Select(item => new MailItem
                {
                    MailId = mailId,
                    ItemType = item.ItemType,
                    ItemId = item.ItemId,
                    ItemCount = item.ItemCount
                }).ToList();

                await _mailRepository.CreateMailItemsAsync(mailId, mailItems);
            }

            // 읽지 않은 우편 카운트를 증가시킨다
            await _notificationService.IncrementUnreadCountAsync(receiverId);

            return new SendMailResponse
            {
                Result = 0,
                MailId = mailId
            };
        }

        public async Task<GetMailListResponse> GetMailListAsync(long userId, int pageNumber, int pageSize)
        {
            // 우편 목록을 조회한다
            var mails = await _mailRepository.GetUserMailListAsync(userId, pageNumber, pageSize);
            var totalCount = await _mailRepository.GetUserMailCountAsync(userId);

            // 읽지 않은 우편 개수를 조회한다 (캐시 우선)
            var unreadCount = await _notificationService.GetUnreadCountAsync(userId);
            if (!unreadCount.HasValue)
            {
                unreadCount = await _mailRepository.GetUnreadMailCountAsync(userId);
                await _notificationService.SetUnreadCountAsync(userId, unreadCount.Value);
            }

            // 각 우편의 첨부 아이템을 조회한다
            var mailDtos = new List<MailDto>();
            foreach (var mail in mails)
            {
                var items = await _mailRepository.GetMailItemsAsync(mail.MailId);
                var itemDtos = items.Select(item => new MailItemDto
                {
                    ItemType = item.ItemType,
                    ItemId = item.ItemId,
                    ItemCount = item.ItemCount
                }).ToList();

                mailDtos.Add(new MailDto
                {
                    MailId = mail.MailId,
                    SenderType = mail.SenderType,
                    MailTitle = mail.MailTitle,
                    MailContent = mail.MailContent,
                    IsRead = mail.IsRead,
                    IsReceived = mail.IsReceived,
                    ExpireAt = mail.ExpireAt,
                    CreatedAt = mail.CreatedAt,
                    Items = itemDtos
                });
            }

            return new GetMailListResponse
            {
                Result = 0,
                Mails = mailDtos,
                TotalCount = totalCount,
                UnreadCount = unreadCount.Value
            };
        }

        public async Task<ReadMailResponse> ReadMailAsync(long userId, long mailId)
        {
            // 우편을 조회한다
            var mail = await _mailRepository.GetMailAsync(mailId);
            if (mail == null)
            {
                return new ReadMailResponse { Result = 1 }; // 우편을 찾을 수 없음
            }

            // 수신자를 검증한다
            if (mail.ReceiverId != userId)
            {
                return new ReadMailResponse { Result = 2 }; // 권한 없음
            }

            // 만료된 우편인지 확인한다
            if (mail.ExpireAt <= DateTime.Now)
            {
                return new ReadMailResponse { Result = 3 }; // 만료된 우편
            }

            // 이미 읽은 우편인지 확인한다
            if (mail.IsRead)
            {
                return new ReadMailResponse { Result = 0 }; // 이미 읽음
            }

            // 읽음 상태로 변경한다
            await _mailRepository.MarkAsReadAsync(mailId);

            // 읽지 않은 우편 카운트를 감소시킨다
            await _notificationService.DecrementUnreadCountAsync(userId);

            return new ReadMailResponse { Result = 0 };
        }

        public async Task<ReceiveMailItemResponse> ReceiveMailItemAsync(long userId, long mailId)
        {
            // 우편을 조회한다
            var mail = await _mailRepository.GetMailAsync(mailId);
            if (mail == null)
            {
                return new ReceiveMailItemResponse { Result = 1 }; // 우편을 찾을 수 없음
            }

            // 수신자를 검증한다
            if (mail.ReceiverId != userId)
            {
                return new ReceiveMailItemResponse { Result = 2 }; // 권한 없음
            }

            // 만료된 우편인지 확인한다
            if (mail.ExpireAt <= DateTime.Now)
            {
                return new ReceiveMailItemResponse { Result = 3 }; // 만료된 우편
            }

            // 이미 수령했는지 확인한다
            if (mail.IsReceived)
            {
                return new ReceiveMailItemResponse { Result = 4 }; // 이미 수령함
            }

            // 첨부 아이템을 조회한다
            var items = await _mailRepository.GetMailItemsAsync(mailId);
            if (items.Count == 0)
            {
                return new ReceiveMailItemResponse { Result = 5 }; // 첨부 아이템 없음
            }

            // 아이템을 지급한다
            foreach (var item in items)
            {
                if (item.ItemType == (byte)MailItemType.Item)
                {
                    await _inventoryService.AcquireItemAsync(userId, item.ItemId.Value, item.ItemCount);
                }
                else if (item.ItemType == (byte)MailItemType.Gold)
                {
                    // 골드 지급 로직 (여기서는 간략화)
                    // await _userRepository.AddGoldAsync(userId, item.ItemCount);
                }
                else if (item.ItemType == (byte)MailItemType.Diamond)
                {
                    // 다이아 지급 로직 (여기서는 간략화)
                    // await _userRepository.AddDiamondAsync(userId, item.ItemCount);
                }
            }

            // 수령 상태로 변경한다
            await _mailRepository.MarkAsReceivedAsync(mailId);

            // 읽지 않은 상태였다면 읽음 처리한다
            if (!mail.IsRead)
            {
                await _mailRepository.MarkAsReadAsync(mailId);
                await _notificationService.DecrementUnreadCountAsync(userId);
            }

            var itemDtos = items.Select(item => new MailItemDto
            {
                ItemType = item.ItemType,
                ItemId = item.ItemId,
                ItemCount = item.ItemCount
            }).ToList();

            return new ReceiveMailItemResponse
            {
                Result = 0,
                ReceivedItems = itemDtos
            };
        }

        public async Task<DeleteMailResponse> DeleteMailAsync(long userId, long mailId)
        {
            // 우편을 조회한다
            var mail = await _mailRepository.GetMailAsync(mailId);
            if (mail == null)
            {
                return new DeleteMailResponse { Result = 1 }; // 우편을 찾을 수 없음
            }

            // 수신자를 검증한다
            if (mail.ReceiverId != userId)
            {
                return new DeleteMailResponse { Result = 2 }; // 권한 없음
            }

            // 첨부 아이템이 있고 수령하지 않았다면 삭제를 막는다
            var items = await _mailRepository.GetMailItemsAsync(mailId);
            if (items.Count > 0 && !mail.IsReceived)
            {
                return new DeleteMailResponse { Result = 3 }; // 아이템을 수령하지 않음
            }

            // 우편을 삭제한다
            await _mailRepository.DeleteMailAsync(mailId);

            // 읽지 않은 상태였다면 카운트를 감소시킨다
            if (!mail.IsRead)
            {
                await _notificationService.DecrementUnreadCountAsync(userId);
            }

            return new DeleteMailResponse { Result = 0 };
        }
    }
```

## 17.10 만료된 우편 자동 삭제

일정 주기로 만료된 우편을 자동으로 삭제하는 백그라운드 서비스를 구현한다.

```csharp
// Services/MailCleanupService.cs
using GameServer.Repositories;

namespace GameServer.Services
{
    public class MailCleanupService : BackgroundService
    {
        private readonly IServiceProvider _serviceProvider;
        private readonly ILogger<MailCleanupService> _logger;
        private readonly TimeSpan _interval = TimeSpan.FromHours(1);

        public MailCleanupService(
            IServiceProvider serviceProvider,
            ILogger<MailCleanupService> logger)
        {
            _serviceProvider = serviceProvider;
            _logger = logger;
        }

        protected override async Task ExecuteAsync(CancellationToken stoppingToken)
        {
            _logger.LogInformation("[MailCleanup] Service started");

            while (!stoppingToken.IsCancellationRequested)
            {
                try
                {
                    using (var scope = _serviceProvider.CreateScope())
                    {
                        var mailRepository = scope.ServiceProvider.GetRequiredService<MailRepository>();
                        var deletedCount = await mailRepository.DeleteExpiredMailsAsync();

                        if (deletedCount > 0)
                        {
                            _logger.LogInformation($"[MailCleanup] Deleted {deletedCount} expired mails");
                        }
                    }
                }
                catch (Exception ex)
                {
                    _logger.LogError(ex, "[MailCleanup] Error occurred");
                }

                await Task.Delay(_interval, stoppingToken);
            }

            _logger.LogInformation("[MailCleanup] Service stopped");
        }
    }
}
```

Program.cs에 백그라운드 서비스를 등록한다.


## 17.12 수신자 검증의 중요성

우편함 시스템에서 가장 중요한 보안 요소는 수신자 검증이다. 우편을 읽거나 아이템을 수령할 때 반드시 요청한 사용자가 우편의 실제 수신자인지 확인해야 한다.

모든 우편 관련 API에서 userId와 mailId의 관계를 검증하지 않으면, 악의적인 사용자가 다른 사람의 mailId를 추측하여 부정하게 아이템을 탈취할 수 있다.

```csharp
// 올바른 방법 - 수신자를 확인한다
var mail = await _mailRepository.GetMailAsync(mailId);
if (mail.ReceiverId != userId)
{
    return new Response { Result = 2 }; // 권한 없음
}

// 잘못된 방법 - mailId만 확인한다 (보안 취약)
var mail = await _mailRepository.GetMailAsync(mailId);
// 수신자 검증 없이 처리
```

또한 우편 아이템 수령 시 중복 수령을 방지하기 위해 isReceived 플래그를 반드시 확인해야 한다. 동시 요청으로 인한 중복 수령을 막기 위해서는 데이터베이스 트랜잭션을 활용하거나 Redis 분산 락을 사용할 수 있다.
